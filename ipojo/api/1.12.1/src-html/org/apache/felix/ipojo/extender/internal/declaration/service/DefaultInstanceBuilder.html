020package org.apache.felix.ipojo.extender.internal.declaration.service;
021
022import static java.lang.String.format;
023
024import java.util.Dictionary;
025import java.util.Hashtable;
026
027import org.apache.felix.ipojo.Factory;
028import org.apache.felix.ipojo.extender.ConfigurationBuilder;
029import org.apache.felix.ipojo.extender.InstanceBuilder;
030import org.apache.felix.ipojo.extender.DeclarationHandle;
031import org.apache.felix.ipojo.extender.internal.declaration.DefaultInstanceDeclaration;
032import org.osgi.framework.BundleContext;
033
034/**
035 * User: guillaume
036 * Date: 13/02/2014
037 * Time: 09:36
038 */
039public class DefaultInstanceBuilder implements InstanceBuilder {
040
041    private BundleContext context;
042    private String name;
043    private String type;
044    private String version;
045
046    public DefaultInstanceBuilder(final BundleContext context, final String type) {
047        this.context(context);
048        this.type(type);
049    }
050
051    public InstanceBuilder name(final String name) {
052        this.name = name;
053        return this;
054    }
055
056    public InstanceBuilder type(final String type) {
057        if (type == null) {
058            throw new IllegalArgumentException(format("'type' parameter cannot be null (instance must be of a given type)"));
059        }
060        this.type = type;
061        return this;
062    }
063
064    public InstanceBuilder type(final Class<?> type) {
065        if (type == null) {
066            throw new IllegalArgumentException(format("'type' parameter cannot be null (instance must be of a given type)"));
067        }
068        return type(type.getName());
069    }
070
071    public InstanceBuilder version(final String version) {
072        this.version = version;
073        return this;
074    }
075
076    public InstanceBuilder context(final BundleContext context) {
077        if (context == null) {
078            throw new IllegalArgumentException(format("'context' parameter cannot be null"));
079        }
080        this.context = context;
081        return this;
082    }
083
084    public ConfigurationBuilder configure() {
085        return new DefaultConfigurationBuilder(this);
086    }
087
088    /**
089     * Only called through ConfigurationBuilder to apply the created configuration to this instance
090     * @param configuration
091     */
092    public DeclarationHandle build(Dictionary<String, Object> configuration) {
093
094        // Prepare the instance configuration
095        if (configuration == null) {
096            configuration = new Hashtable<String, Object>();
097        }
098
099        if (name != null) {
100            configuration.put(Factory.INSTANCE_NAME_PROPERTY, name);
101        }
102
103        if (version != null) {
104            configuration.put(Factory.FACTORY_VERSION_PROPERTY, version);
105        }
106
107        return new DefaultInstanceDeclaration(context, type, configuration);
108
109    }
110
111    public DeclarationHandle build() {
112        return build(new Hashtable<String, Object>());
113    }
114}
