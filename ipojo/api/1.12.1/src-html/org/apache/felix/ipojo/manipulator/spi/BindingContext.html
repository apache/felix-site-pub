020package org.apache.felix.ipojo.manipulator.spi;
021
022import org.apache.felix.ipojo.manipulator.Reporter;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.ComponentWorkbench;
024import org.objectweb.asm.Type;
025import org.objectweb.asm.tree.ClassNode;
026import org.objectweb.asm.tree.FieldNode;
027import org.objectweb.asm.tree.MethodNode;
028
029import java.lang.annotation.ElementType;
030
031/**
032 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
033 */
034public class BindingContext {
035
036    public static final int NO_INDEX = -1;
037
038    /**
039     *
040     */
041    private ComponentWorkbench workbench;
042    private FieldNode field;
043    private MethodNode method;
044    private ClassNode clazz;
045    private ElementType elementType;
046    private int parameterIndex;
047    private Reporter reporter;
048    private Type annotationType;
049    private Object visitor;
050
051
052    public BindingContext(final ComponentWorkbench workbench,
053                          final Reporter reporter,
054                          final Type annotationType,
055                          final FieldNode node,
056                          final ElementType elementType,
057                          final int parameterIndex,
058                          final Object visitor) {
059        this(workbench, reporter, annotationType, elementType, parameterIndex, visitor);
060        this.field = node;
061    }
062
063    public BindingContext(final ComponentWorkbench workbench,
064                          final Reporter reporter,
065                          final Type annotationType,
066                          final MethodNode node,
067                          final ElementType elementType,
068                          final int parameterIndex,
069                          final Object visitor) {
070        this(workbench, reporter, annotationType, elementType, parameterIndex, visitor);
071        this.method = node;
072    }
073
074    public BindingContext(final ComponentWorkbench workbench,
075                          final Reporter reporter,
076                          final Type annotationType,
077                          final ClassNode node,
078                          final ElementType elementType,
079                          final int parameterIndex,
080                          final Object visitor) {
081        this(workbench, reporter, annotationType, elementType, parameterIndex, visitor);
082        this.clazz = node;
083    }
084
085    private BindingContext(final ComponentWorkbench workbench,
086                           final Reporter reporter,
087                           final Type annotationType,
088                           final ElementType elementType,
089                           final int parameterIndex,
090                           final Object visitor) {
091        this.workbench = workbench;
092        this.reporter = reporter;
093        this.annotationType = annotationType;
094        this.elementType = elementType;
095        this.parameterIndex = parameterIndex;
096        this.visitor = visitor;
097    }
098
099    public ComponentWorkbench getWorkbench() {
100        return workbench;
101    }
102
103    public FieldNode getFieldNode() {
104        return field;
105    }
106
107    public MethodNode getMethodNode() {
108        return method;
109    }
110
111    public ClassNode getClassNode() {
112        return clazz;
113    }
114
115    public ElementType getElementType() {
116        return elementType;
117    }
118
119    public int getParameterIndex() {
120        return parameterIndex;
121    }
122
123    public Reporter getReporter() {
124        return reporter;
125    }
126
127    public Type getAnnotationType() {
128        return annotationType;
129    }
130
131    public Object getVisitor() {
132        return visitor;
133    }
134
135    /**
136     * This method is just to support the compatibility with the previous version. It returns an ASM node of the
137     * member on which the annotation is 'attached'. The type of the node depends on this element. It can be a {@link
138     * org.objectweb.asm.tree.FieldNode}, a {@link org.objectweb.asm.tree.MethodNode} or a
139     * {@link org.objectweb.asm.tree.ClassNode}. It actually checks the current node and returns the first set ones
140     * in this order: field, method, class.
141     * <p/>
142     * The type of the returned object can be determined using the {@link #getElementType} method. However it is
143     * recommended to use one of the following method: {@link #getFieldNode()}, {@link #getMethodNode()}  or
144     * {@link #getClassNode()}.
145     *
146     * @return the node.
147     * @deprecated
148     */
149    public Object getNode() {
150        if (field != null) {
151            return field;
152        }
153        if (method != null) {
154            return method;
155        }
156        if (clazz != null) {
157            return clazz;
158        }
159        // No node ?
160        return null;
161    }
162}
