020package org.apache.felix.ipojo.extender.internal.queue;
021
022import org.apache.felix.ipojo.extender.queue.Callback;
023import org.apache.felix.ipojo.extender.queue.Job;
024import org.apache.felix.ipojo.extender.queue.JobInfo;
025
026import java.util.concurrent.Callable;
027
028/**
029 * A callable computing job statistics. The job is given as another callable.
030 * The statistics are global, so must be used carefully.
031 */
032public class JobInfoCallable<T> implements Callable<T>, JobInfo {
033
034    /**
035     * Notifier helper for {@link org.apache.felix.ipojo.extender.queue.QueueListener}.
036     */
037    private final QueueNotifier m_queueNotifier;
038
039    /**
040     * The statistic object.
041     */
042    private final Statistic m_statistic;
043
044    /**
045     * The genuine job.
046     */
047    private final Job<T> m_delegate;
048
049    /**
050     * A callback notified when the job is processed.
051     */
052    private final Callback<T> m_callback;
053
054    /**
055     * The job description.
056     */
057    private final String m_description;
058
059    /**
060     * The date (in milli) when this object is created.
061     */
062    private long enlistmentTime = System.currentTimeMillis();
063
064    /**
065     * The date when the job processing started.
066     */
067    private long startTime = -1;
068
069    /**
070     * The date when the job processing is completed.
071     */
072    private long endTime = -1;
073
074    /**
075     * Creates the job info callable.
076     *
077     * @param queueNotifier notifier for QueueListeners
078     * @param statistic   the statistics that will be populated
079     * @param delegate    the real job
080     * @param callback    the callback notified when the job is completed
081     * @param description the job description
082     */
083    public JobInfoCallable(QueueNotifier queueNotifier,
084                           Statistic statistic,
085                           Job<T> delegate,
086                           Callback<T> callback,
087                           String description) {
088        m_queueNotifier = queueNotifier;
089        m_statistic = statistic;
090        m_delegate = delegate;
091        m_callback = callback;
092        m_description = description;
093        m_statistic.getWaiters().add(this);
094
095        // Assume that we will be enlisted in the next few cycles
096        m_queueNotifier.fireEnlistedJobInfo(this);
097    }
098
099    /**
100     * Executes the job.
101     * This method updates the statistics.
102     *
103     * @return the job result
104     * @throws Exception the job execution failed
105     */
106    public T call() throws Exception {
107        m_statistic.getWaiters().remove(this);
108        startTime = System.currentTimeMillis();
109        m_statistic.getCurrentsCounter().incrementAndGet();
110        T result = null;
111        Exception exception = null;
112        try {
113            m_queueNotifier.fireStartedJobInfo(this);
114            result = m_delegate.call();
115            endTime = System.currentTimeMillis();
116            return result;
117        } catch (Exception e) {
118            endTime = System.currentTimeMillis();
119            m_queueNotifier.fireFailedJobInfo(this, e);
120            if (m_callback != null) {
121                m_callback.error(this, e);
122            }
123            exception = e;
124            throw e;
125        } finally {
126            m_statistic.getCurrentsCounter().decrementAndGet();
127            m_statistic.getFinishedCounter().incrementAndGet();
128
129            // Only exec success callbacks when no error occurred
130            if (exception == null) {
131                m_queueNotifier.fireExecutedJobInfo(this, result);
132                if (m_callback != null) {
133                    m_callback.success(this, result);
134                }
135            }
136        }
137    }
138
139    /**
140     * @return the enlistment date.
141     */
142    public long getEnlistmentTime() {
143        return enlistmentTime;
144    }
145
146    /**
147     * @return the job start date.
148     */
149    public long getStartTime() {
150        return startTime;
151    }
152
153    /**
154     * @return the job completion date.
155     */
156    public long getEndTime() {
157        return endTime;
158    }
159
160    /**
161     * Computes the time spent in the waiting queue
162     *
163     * @return the waited time, if the job is still waiting, gets the current waited time
164     */
165    public long getWaitDuration() {
166        long end = startTime;
167        if (end == -1) {
168            // Not yet started
169            // Still waiting
170            end = System.currentTimeMillis();
171        }
172        return end - enlistmentTime;
173    }
174
175    /**
176     * Computes the time spent to execute the job (this does not include the waiting).
177     * If the job is not executed yet, or is still executing, {@literal -1} is returned
178     *
179     * @return the execution duration, or {@literal -1}.
180     */
181    public long getExecutionDuration() {
182        if ((startTime == -1) || (endTime == -1)) {
183            return -1;
184        }
185        return endTime - startTime;
186    }
187
188    /**
189     * @return the job description
190     */
191    public String getDescription() {
192        return m_description;
193    }
194
195    public String getJobType() {
196        return m_delegate.getJobType();
197    }
198}
