122    /**
123     * Manipulates an input bundle.
124     * This method creates an iPOJO bundle based on the given metadata file.
125     * The original and final bundles must be different.
126     * <p/>
127     * This method does not use the classloader parameter, and use the classloader having loaded the
128     * {@link org.apache.felix.ipojo.manipulator.Pojoization} class. It's definitely not recommended.
129     *
130     * @param in       the original bundle.
131     * @param out      the final bundle.
132     * @param metadata the iPOJO metadata input stream.
133     * @deprecated
134     */
135    public void pojoization(File in, File out, InputStream metadata) {
136        pojoization(in, out, metadata, this.getClass().getClassLoader());
137    }

139    /**
140     * Manipulates an input bundle.
141     * This method creates an iPOJO bundle based on the given metadata file.
142     * The original and final bundles must be different.
143     *
144     * @param in       the original bundle.
145     * @param out      the final bundle.
146     * @param metadata the iPOJO metadata input stream.
147     * @param loader   the classloader used to compute the bytecode frames.
148     */
149    public void pojoization(File in, File out, InputStream metadata, ClassLoader loader) {
150        StreamMetadataProvider provider = new StreamMetadataProvider(metadata, m_reporter);
151        provider.setValidateUsingLocalSchemas(m_useLocalXSD);
152
153        ResourceStore store;
154        try {
155            JarFile origin = new JarFile(in);
156            JarFileResourceStore jfrs = new JarFileResourceStore(origin, out);
157            jfrs.setClassLoader(loader);
158            if (in.getName().endsWith(".war")) {
159                // this is a war file, use the right mapper
160                jfrs.setResourceMapper(new WABResourceMapper());
161            }
162            jfrs.setManifest(origin.getManifest());
163            DefaultManifestBuilder dmb = new DefaultManifestBuilder();
164            dmb.setMetadataRenderer(new MetadataRenderer());
165            jfrs.setManifestBuilder(dmb);
166            store = jfrs;
167        } catch (IOException e) {
168            m_reporter.error("The input file " + in.getAbsolutePath() + " is not a Jar file");
169            return;
170        }
171
172        ManipulationVisitor visitor = createDefaultVisitorChain(store);
173
174        pojoization(store, provider, visitor, loader);
175    }
176
177    private ManipulationVisitor createDefaultVisitorChain(ResourceStore store) {
178        ManipulatedResourcesWriter writer = new ManipulatedResourcesWriter();
179        writer.setReporter(m_reporter);
180        writer.setResourceStore(store);
181
182        CheckFieldConsistencyVisitor checkFieldConsistencyVisitor = new CheckFieldConsistencyVisitor(writer);
183        checkFieldConsistencyVisitor.setReporter(m_reporter);
184        return checkFieldConsistencyVisitor;
185    }

204    /**
205     * Manipulates an input bundle.
206     * This method creates an iPOJO bundle based on the given metadata file.
207     * The original and final bundles must be different.
208     *
209     * @param in           the original bundle.
210     * @param out          the final bundle.
211     * @param metadataFile the iPOJO metadata file (XML).
212     * @param loader       the classloader used to compute the bytecode frames.
213     */
214    public void pojoization(File in, File out, File metadataFile, ClassLoader loader) {
215        MetadataProvider provider = new EmptyMetadataProvider();
216        if (metadataFile != null) {
217            FileMetadataProvider fileMetadataProvider = new FileMetadataProvider(metadataFile, m_reporter);
218            fileMetadataProvider.setValidateUsingLocalSchemas(m_useLocalXSD);
219            provider = fileMetadataProvider;
220        }
221
222        ResourceStore store;
223        try {
224            JarFile origin = new JarFile(in);
225            JarFileResourceStore jfrs = new JarFileResourceStore(origin, out);
226            jfrs.setClassLoader(loader);
227            if (in.getName().endsWith(".war")) {
228                // this is a war file, use the right mapper
229                jfrs.setResourceMapper(new WABResourceMapper());
230            }
231            jfrs.setManifest(origin.getManifest());
232            DefaultManifestBuilder dmb = new DefaultManifestBuilder();
233            dmb.setMetadataRenderer(new MetadataRenderer());
234            jfrs.setManifestBuilder(dmb);
235            store = jfrs;
236        } catch (IOException e) {
237            m_reporter.error("The input file " + in.getAbsolutePath() + " is not a Jar file");
238            return;
239        }
240
241        ManipulationVisitor visitor = createDefaultVisitorChain(store);
242
243        pojoization(store, provider, visitor, loader);
244    }

263    /**
264     * Manipulates an expanded bundles.
265     * Classes are in the specified directory.
266     * this method allows to update a customized manifest.
267     *
268     * @param directory    the directory containing classes
269     * @param metadataFile the metadata file
270     * @param manifestFile the manifest file. <code>null</code> to use directory/META-INF/MANIFEST.mf
271     * @param loader       the classloader used to compute the bytecode frames.
272     */
273    public void directoryPojoization(File directory, File metadataFile, File manifestFile, ClassLoader loader) {
274        // Get the metadata.xml location if not null
275        MetadataProvider provider = new EmptyMetadataProvider();
276        if (metadataFile != null) {
277            FileMetadataProvider fileMetadataProvider = new FileMetadataProvider(metadataFile, m_reporter);
278            fileMetadataProvider.setValidateUsingLocalSchemas(m_useLocalXSD);
279            provider = fileMetadataProvider;
280        }
281
282        ManifestProvider manifestProvider;
283        File selectedManifestFile;
284        if (manifestFile != null) {
285            if (manifestFile.isFile()) {
286                try {
287                    manifestProvider = new FileManifestProvider(manifestFile);
288                    selectedManifestFile = manifestFile;
289                } catch (IOException e) {
290                    m_reporter.error("Cannot read Manifest from '" + manifestFile.getAbsolutePath() + "'");
291                    return;
292                }
293            } else {
294                m_reporter.error("The manifest file " + manifestFile.getAbsolutePath() + " does not exist");
295                return;
296            }
297        } else {
298            // If the manifest is not specified, the m_dir/META-INF/MANIFEST.MF is used.
299            File metaInf = new File(directory, "META-INF");
300            File original = new File(metaInf, "MANIFEST.MF");
301            if (original.isFile()) {
302                try {
303                    manifestProvider = new FileManifestProvider(original);
304                    selectedManifestFile = original;
305                } catch (IOException e) {
306                    m_reporter.error("Cannot read Manifest from '" + original.getAbsolutePath() + "'");
307                    return;
308                }
309            } else {
310                m_reporter.error("The manifest file " + original.getAbsolutePath() + " does not exist");
311                return;
312            }
313        }
314
315        DirectoryResourceStore store;
316        if (directory.exists() && directory.isDirectory()) {
317            store = new DirectoryResourceStore(directory);
318            File webinf = new File(directory, "WEB-INF");
319            if (directory.getName().endsWith(".war") || webinf.isDirectory()) {
320                // this is a war file, use the right mapper
321                store.setResourceMapper(new WABResourceMapper());
322            }
323            store.setManifest(manifestProvider.getManifest());
324            store.setManifestFile(selectedManifestFile);
325            DefaultManifestBuilder dmb = new DefaultManifestBuilder();
326            dmb.setMetadataRenderer(new MetadataRenderer());
327            store.setManifestBuilder(dmb);
328        } else {
329            m_reporter.error("The directory " + directory.getAbsolutePath() + " does not exist or is not a directory.");
330            return;
331        }
332
333        ManipulationVisitor visitor = createDefaultVisitorChain(store);
334
335        pojoization(store, provider, visitor, loader);
336
337    }
338
339    public void pojoization(final ResourceStore store,
340                            final MetadataProvider metadata,
341                            final ManipulationVisitor visitor,
342                            final ClassLoader loader) {
343
344        ManipulationEngine engine = new ManipulationEngine(loader);
345        engine.setResourceStore(store);
346        engine.setReporter(m_reporter);
347        engine.setManipulationVisitor(visitor);
348
349        try {
350
351            // Creates a composite to store multiple metadata providers
352            CompositeMetadataProvider composite = new CompositeMetadataProvider(m_reporter);
353            composite.addMetadataProvider(metadata);
354
355            if (!m_ignoreAnnotations) {
356                composite.addMetadataProvider(new AnnotationMetadataProvider(store, m_moduleProvider, m_reporter));
357            }
358
359            // Get metadata
360            List<Element> metadatas = composite.getMetadatas();
361
362            // Construct ManipulationUnits
363            // And collect non-component metadata
364            for (Element meta : metadatas) {
365                String name = Metadatas.getComponentType(meta);
366                if (name != null) {
367                    // Only handler and component have a classname attribute
368                    String path = Strings.asResourcePath(name);
369                    engine.addManipulationUnit(new ManipulationUnit(path, meta));
370                } else {
371                    visitor.visitMetadata(meta);
372                }
373            }
374
375        } catch (IOException e) {
376            m_reporter.error("Cannot load metadata " + e.getMessage());
377            return;
378        }
379
380        // Start the manipulation
381        engine.generate();
382
383        // Tell the visitor that we have finished
384        visitor.visitEnd();
385
386    }
