020package org.apache.felix.ipojo.extender.internal.queue.debug;
021
022import java.util.ArrayList;
023import java.util.List;
024import java.util.concurrent.CopyOnWriteArrayList;
025
026import org.apache.felix.ipojo.extender.queue.JobInfo;
027import org.apache.felix.ipojo.extender.queue.QueueListener;
028import org.apache.felix.ipojo.extender.queue.debug.QueueEventProxy;
029
030/**
031 * User: guillaume
032 * Date: 02/10/13
033 * Time: 10:51
034 */
035public class ReplayQueueEventProxy implements QueueEventProxy, QueueListener {
036    /**
037     * Store QueueListeners.
038     */
039    private final List<QueueListener> m_listeners = new ArrayList<QueueListener>();
040
041    /**
042     * Store QueueEvent to be replayed.
043     */
044    private List<QueueEvent> m_events = new CopyOnWriteArrayList<QueueEvent>();
045
046    public void addQueueListener(final QueueListener listener) {
047        m_listeners.add(listener);
048        // replay all the existing events to the new listener
049        replay(listener);
050    }
051
052    public void removeQueueListener(final QueueListener listener) {
053        m_listeners.remove(listener);
054    }
055
056    public void enlisted(final JobInfo info) {
057        EnlistedQueueEvent event = new EnlistedQueueEvent(info);
058        m_events.add(event);
059        forward(event);
060    }
061
062    public void started(final JobInfo info) {
063        StartedQueueEvent event = new StartedQueueEvent(info);
064        m_events.add(event);
065        forward(event);
066    }
067
068    public void executed(final JobInfo info, final Object result) {
069        ExecutedQueueEvent event = new ExecutedQueueEvent(info, result);
070        m_events.add(event);
071        forward(event);
072    }
073
074    public void failed(final JobInfo info, final Throwable throwable) {
075        FailedQueueEvent event = new FailedQueueEvent(info, throwable);
076        m_events.add(event);
077        forward(event);
078    }
079
080    /**
081     * Replay all stored events to the given QueueListener.
082     */
083    private void replay(final QueueListener listener) {
084        for (QueueEvent event : m_events) {
085            event.replay(listener);
086        }
087    }
088
089    /**
090     * Forward the given QueueEvent to all the registered listeners.
091     */
092    private void forward(final QueueEvent event) {
093        for (QueueListener listener : m_listeners) {
094            event.replay(listener);
095        }
096    }
097
098    /**
099     * Encapsulate event forwarding logic.
100     */
101    private interface QueueEvent {
102        void replay(QueueListener listener);
103    }
104
105    private class EnlistedQueueEvent implements QueueEvent {
106        private final JobInfo m_info;
107
108        public EnlistedQueueEvent(final JobInfo info) {
109            m_info = info;
110        }
111
112        public void replay(final QueueListener listener) {
113            listener.enlisted(m_info);
114        }
115    }
116
117    private class StartedQueueEvent implements QueueEvent {
118        private final JobInfo m_info;
119
120        public StartedQueueEvent(final JobInfo info) {
121            m_info = info;
122        }
123
124        public void replay(final QueueListener listener) {
125            listener.started(m_info);
126        }
127    }
128
129    private class ExecutedQueueEvent implements QueueEvent {
130        private final JobInfo m_info;
131        private final Object m_result;
132
133        public ExecutedQueueEvent(final JobInfo info, final Object result) {
134            m_info = info;
135            m_result = result;
136        }
137
138        public void replay(final QueueListener listener) {
139            listener.executed(m_info, m_result);
140        }
141    }
142
143    private class FailedQueueEvent implements QueueEvent {
144        private final JobInfo m_info;
145        private final Throwable m_throwable;
146
147        public FailedQueueEvent(final JobInfo info, final Throwable throwable) {
148            m_info = info;
149            m_throwable = throwable;
150        }
151
152        public void replay(final QueueListener listener) {
153            listener.failed(m_info, m_throwable);
154        }
155    }
156
157}
