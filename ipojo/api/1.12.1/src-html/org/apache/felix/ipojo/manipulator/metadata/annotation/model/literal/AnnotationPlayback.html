022import static java.lang.String.format;
023import static org.objectweb.asm.Type.getType;
024
025import java.io.Serializable;
026import java.lang.annotation.Annotation;
027import java.lang.reflect.Method;
028import java.util.Arrays;
029import java.util.HashMap;
030import java.util.List;
031import java.util.Map;
032
033import org.apache.felix.ipojo.manipulator.metadata.annotation.model.AnnotationDiscovery;
034import org.apache.felix.ipojo.manipulator.metadata.annotation.model.Playback;
035import org.objectweb.asm.AnnotationVisitor;
036import org.objectweb.asm.ClassVisitor;
037import org.objectweb.asm.FieldVisitor;
038import org.objectweb.asm.MethodVisitor;
039import org.objectweb.asm.Type;

045 */
046public class AnnotationPlayback implements Playback {
047
048    public static final List<? extends Class<? extends Serializable>> BOXED_TYPES = Arrays.asList(Byte.class, Long.class, Character.class, Boolean.class, Double.class, Float.class, Integer.class, Short.class);
049    private final Annotation m_annotation;
050    private final Type m_annotationType;
051
052    public AnnotationPlayback(final Annotation annotation) {
053        m_annotation = annotation;
054        m_annotationType = Type.getType(annotation.annotationType());
055    }
056
057    private Map<String, Object> getValues() {
058        Map<String, Object> values = new HashMap<String, Object>();
059        for (Method method : m_annotation.annotationType().getDeclaredMethods()) {
060            try {
061                values.put(method.getName(), method.invoke(m_annotation));
062            } catch (Throwable t) {
063                throw new IllegalStateException(
064                        format("Cannot get value of the %s.%s attribute",
065                               m_annotation.annotationType().getSimpleName(),
066                               method.getName()),
067                        t
068                );
069            }
070        }
071        return values;
072    }
073    public void accept(final FieldVisitor visitor) {
074        AnnotationVisitor av = visitor.visitAnnotation(m_annotationType.getDescriptor(),
075                                                       true);
076        if (av != null) {
077            accept(av);
078        }
079    }
080
081    public void accept(final ClassVisitor visitor) {
082        AnnotationVisitor av = visitor.visitAnnotation(m_annotationType.getDescriptor(),
083                                                       true);
084        if (av != null) {
085            accept(av);
086        }
087    }
088
089    public void accept(final MethodVisitor visitor) {
090        AnnotationVisitor av = visitor.visitAnnotation(m_annotationType.getDescriptor(),
091                                                       true);
092        if (av != null) {
093            accept(av);
094        }
095    }
096
097    public void accept(final MethodVisitor visitor, final int index) {
098        AnnotationVisitor av = visitor.visitParameterAnnotation(index,
099                                                                m_annotationType.getDescriptor(),
100                                                                true);
101        if (av != null) {
102            accept(av);
103        }
104    }
105
106    public void accept(final AnnotationDiscovery visitor) {
107        AnnotationVisitor av = visitor.visitAnnotation(m_annotationType.getDescriptor());
108        if (av != null) {
109            accept(av);
110        }
111    }
112
113    private void accept(final AnnotationVisitor visitor) {
114        // As per the ASM doc, visit methods must be called in a given order:
115        // 1. visit()
116        // 2. visitEnum()
117        // 3. visitAnnotation()
118        // 4. visitArray()
119
120        // So values must be sorted
121        Map<String, Object> values = getValues();
122        accept(values, visitor);
123        acceptEnum(values, visitor);
124        acceptAnnotation(values, visitor);
125        acceptArray(values, visitor);
126
127        // Do not forget to visitEnd()
128        visitor.visitEnd();
129
130        // TODO This should disappear, only useful for testing
131        if (!values.isEmpty()) {
132            // We missed something during serialization
133            throw new IllegalStateException(
134                    format("Attributes of @%s could not be serialized: %s",
135                           m_annotation.annotationType().getSimpleName(),
136                           values.keySet())
137            );
138        }
139    }
140
141    private void acceptAnnotation(final Map<String, Object> values, final AnnotationVisitor visitor) {
142        Map<String, Object> copy = new HashMap<String, Object>(values);
143        for (Map.Entry<String, Object> entry : copy.entrySet()) {
144
145            Class<?> type = entry.getValue().getClass();
146            if (Annotation.class.isAssignableFrom(type)) {
147
148                Annotation annotation = (Annotation) entry.getValue();
149                AnnotationVisitor annotationVisitor = visitor.visitAnnotation(entry.getKey(),
150                                                                              getType(annotation.annotationType()).getDescriptor());
151                if (annotationVisitor != null) {
152                    AnnotationPlayback playback = new AnnotationPlayback(annotation);
153                    playback.accept(annotationVisitor);
154                }
155
156                values.remove(entry.getKey());
157            }
158        }
159    }
160
161    private void acceptEnum(final Map<String, Object> values, final AnnotationVisitor visitor) {
162
163        Map<String, Object> copy = new HashMap<String, Object>(values);
164        for (Map.Entry<String, Object> entry : copy.entrySet()) {
165
166            Class<?> type = entry.getValue().getClass();
167            if (type.isEnum()) {
168                Enum<?> enumValue = (Enum<?>) entry.getValue();
169                visitor.visitEnum(entry.getKey(),
170                                  getType(type).getDescriptor(),
171                                  enumValue.name());
172
173                values.remove(entry.getKey());
174            }
175        }
176    }
177
178    private void accept(final Map<String, Object> values, final AnnotationVisitor visitor) {
179
180        Map<String, Object> copy = new HashMap<String, Object>(values);
181        for (Map.Entry<String, Object> entry : copy.entrySet()) {
182
183            Class<?> type = entry.getValue().getClass();
184            if (isSimpleType(type)) {
185
186                // Accept Byte, Boolean, Character, Short, Integer, Long, Float, Double
187                // Accept String
188                // Accept Array of byte, boolean, char, short, int, long, float, double
189                visitor.visit(entry.getKey(), transform(entry.getValue()));
190
191                values.remove(entry.getKey());
192            }
193        }
194    }
195
196    private boolean isSimpleType(final Class<?> type) {
197        return isPrimitive(type) ||
198                String.class.equals(type) ||
199                Class.class.equals(type) ||
200                (type.isArray() && isPrimitive(type.getComponentType()));
201    }
202
203    private boolean isPrimitive(final Class<?> type) {
204        if (type.isPrimitive()) {
205            return true;
206        }
207
208        if (BOXED_TYPES.contains(type)) {
209            return true;
210        }
211
212        return false;
213    }
214
215    private void acceptArray(final Map<String, Object> values, final AnnotationVisitor visitor) {
216        Map<String, Object> copy = new HashMap<String, Object>(values);
217        for (Map.Entry<String, Object> entry : copy.entrySet()) {
218
219            Class<?> type = entry.getValue().getClass();
220            if (type.isArray()) {
221
222                // Simple arrays have been visited using AnnotationVisitor.visit(String, Object)
223
224                AnnotationVisitor arrayVisitor = visitor.visitArray(entry.getKey());
225                if (arrayVisitor != null) {
226                    Object[] array = (Object[]) entry.getValue();
227                    Class<?> componentType = array.getClass().getComponentType();
228                    Type asmType = Type.getType(componentType);
229
230                    if (componentType.isEnum()) {
231                        for (Object o : array) {
232                            Enum eValue = (Enum) o;
233                            arrayVisitor.visitEnum(null, asmType.getDescriptor(), eValue.name());
234                        }
235                    } else if (componentType.isAnnotation()) {
236                        for (Object o : array) {
237                            Annotation annotation = (Annotation) o;
238                            AnnotationVisitor annotationVisitor = arrayVisitor.visitAnnotation(null, asmType.getDescriptor());
239                            if (annotationVisitor != null) {
240                                AnnotationPlayback playback = new AnnotationPlayback(annotation);
241                                playback.accept(annotationVisitor);
242                            }
243                        }
244                    } else {
245                        for (Object o : array) {
246                            arrayVisitor.visit(null, transform(o));
247                        }
248                    }
249
250                    arrayVisitor.visitEnd();
251                }
252
253                values.remove(entry.getKey());
254            }
255        }
256    }
257
258    private Object transform(final Object value) {
259        if (value instanceof Class) {
260            return getType((Class) value);
261        }
262        return value;
263    }
