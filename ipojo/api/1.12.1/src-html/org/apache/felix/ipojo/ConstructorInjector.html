019package org.apache.felix.ipojo;
020
021/**
022 * Interface implemented to support constructor parameter injection.
023 * When a new POJO object has to be created, all constructor injectors are
024 * called to gets the objects to injects as well as the type (to discover
025 * the constructor).
026 * Handlers willing to inject constructor parameters must register themselves
027 * using {@link InstanceManager#register(int, ConstructorInjector)} where
028 * the integer is the parameter index. Only one injector can inject a value
029 * for a specific index. If several injectors are registered for the same index,
030 * the component type is declared as invalid.
031 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
032 */
033public interface ConstructorInjector {
034
035    /**
036     * Gets the object to inject in the constructor parameter.
037     * @param index the parameter index
038     * @return the object to be injected.
039     */
040        Object getConstructorParameter(int index);
041
042    /**
043     * Gets the type of the object to inject in the constructor parameter.
044     * This is the type looked into the Pojo class, so it must match.
045     * Returning <code>null</code> will try to get the class from the
046     * injected object, however this can be wrong (implementation instead of interface,
047     * boxed objects...) and error-prone.
048     * @param index the parameter index
049     * @return the Class object (must fit also for primitive type)
050     */
051        Class getConstructorParameterType(int index);
052
053}
