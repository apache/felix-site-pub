108    /**
109     * Creates a property.
110     * At least the method or the field need
111     * to be specified.
112     * @param name the name of the property (optional)
113     * @param field the name of the field
114     * @param method the method name
115     * @param value the initial value of the property (optional)
116     * @param type the the type of the property
117     * @param manager the instance manager
118     * @param handler the handler object which manage this property.
119     * @throws ConfigurationException if the property value cannot be set.
120     */
121    public Property(String name, String field, String method, String value, String type, InstanceManager manager, Handler handler) throws ConfigurationException {
122        m_handler = handler;
123        m_manager = manager;
124        m_field = field;
125
126        if (name == null) {
127            if (m_field == null) {
128                m_name = method;
129            } else {
130                m_name = field;
131            }
132        } else {
133            m_name = name;
134        }
135
136        m_type = computeType(type, manager.getGlobalContext());
137        if (value != null) {
138            m_value = create(m_type, value);
139            m_defaultValue = m_value;
140        }
141
142        if (method != null) {
143            m_method = new Callback(method, new String[] { m_type.getName() }, false, manager);
144        } else {
145            m_method = null;
146        }
147    }

150     * Creates a property.
151     * At least the method or the field need
152     * to be specified.
153     * @param name the name of the property (optional)
154     * @param field the name of the field
155     * @param method the method name
156     * @param value the initial value of the property (optional)
157     * @param manager the instance manager
158     * @param handler the handler object which manage this property.
159     * @throws ConfigurationException if the property value cannot be set.

161    public Property(String name, String field, String method, Object value, InstanceManager manager, Handler handler) throws ConfigurationException {
162        m_handler = handler;
163        m_manager = manager;
164        m_field = field;
165
166        if (value == null) {
167            throw new ConfigurationException("Cannot create properties without a value");
168        }
169
170        if (name == null) {
171            if (m_field == null) {
172                m_name = method;
173            } else {
174                m_name = field;
175            }
176        } else {
177            m_name = name;
178        }
179
180        m_type = value.getClass();
181        m_value = value;
182        m_defaultValue = m_value;
183
184        if (method != null) {
185            m_method = new Callback(method, new String[] { m_type.getName() }, false, manager);
186        } else {
187            m_method = null;
188        }

197    /**
198     * Computes and returns the property type according to the given type name.
199     * @param type the the type name
200     * @param context the bundle context (used to load classes)
201     * @return the class of the given type
202     * @throws ConfigurationException if an error occurs when loading the type class for non-primitive types.
203     */
204    public static Class computeType(String type, BundleContext context) throws ConfigurationException {
205        // Array :
206        if (type.endsWith("[]")) {
207            return computeArrayType(type, context);
208        } else {
209            // Syntactic sugar to avoid writing java.lang.String
210            if ("string".equals(type) || "String".equals(type)) {
211                return java.lang.String.class;
212            } else if ("boolean".equals(type)) {
213                return Boolean.TYPE;
214            } else if ("byte".equals(type)) {
215                return Byte.TYPE;
216            } else if ("short".equals(type)) {
217                return  Short.TYPE;
218            } else if ("int".equals(type)) {
219                return Integer.TYPE;
220            } else if ("long".equals(type)) {
221                return Long.TYPE;
222            } else if ("float".equals(type)) {
223                return Float.TYPE;
224            } else if ("double".equals(type)) {
225                return Double.TYPE;
226            } else if ("char".equals(type)) {
227                return Character.TYPE;
228            } else {
229                // Non array, complex type.
230                try {
231                    return context.getBundle().loadClass(type);
232                } catch (ClassNotFoundException e) {
233                    throw new ConfigurationException("Class not found exception in setValue on " + type, e);
234                } catch (SecurityException e) {
235                    throw new ConfigurationException("Security exception in setValue on " + type, e);
236                } catch (IllegalArgumentException e) {
237                    throw new ConfigurationException("Argument issue when calling the constructor of the type " + type, e);
238                }
239            }
240        }
241    }

243    /**
244     * Gets the Class object of a type array.
245     * @param type the string descriptor of the type (must end by [] )
246     * @param context the bundle context (used to load classes)
247     * @return the Class object of the given type array.
248     * @throws ConfigurationException if the class cannot be loaded
249     */
250    private static Class computeArrayType(String type, BundleContext context) throws ConfigurationException {
251        // Note: Harmony does't support the type[].class notation.
252        // An empty array has to be created to get the class object.
253        String internalType = type.substring(0, type.length() - 2);
254        if ("string".equals(internalType) || "String".equals(internalType)) {
255            return new String[0].getClass();
256        }
257        if ("boolean".equals(internalType)) {
258            return new boolean[0].getClass();
259        }
260        if ("byte".equals(internalType)) {
261            return new byte[0].getClass();
262        }
263        if ("short".equals(internalType)) {
264            return new short[0].getClass();
265        }
266        if ("int".equals(internalType)) {
267            return new int[0].getClass();
268        }
269        if ("long".equals(internalType)) {
270            return new long[0].getClass();
271        }
272        if ("float".equals(internalType)) {
273            return new float[0].getClass();
274        }
275        if ("double".equals(internalType)) {
276            return new double[0].getClass();
277        }
278        if ("char".equals(internalType)) {
279            return new char[0].getClass();
280        }
281
282        // Complex array type.
283        try {
284            Class clazz = context.getBundle().loadClass(internalType);
285            Object[] object = (Object[]) Array.newInstance(clazz, 0);
286            return object.getClass();
287        } catch (ClassNotFoundException e) {
288            throw new ConfigurationException("Class not found exception in setValue on " + internalType, e);
289        } catch (SecurityException e) {
290            throw new ConfigurationException("Security Exception in setValue on " + internalType, e);
291        } catch (IllegalArgumentException e) {
292            throw new ConfigurationException("Argument issue when calling the constructor of the type " + internalType, e);
293        }
294    }

356    /**
357     * Gets the NO VALUE Object.
358     * This method returns the object to inject when the property
359     * was not assigned to a value.
360     * @param type the type of the value.
361     * @return the object to inject when the property has no value.
362     */
363    private static Object getNoValue(Class type) {
364        if (Boolean.TYPE.equals(type)) { return Boolean.FALSE; }
365        if (Byte.TYPE.equals(type)) { return new Byte((byte) 0); }
366        if (Short.TYPE.equals(type)) { return new Short((short) 0); }
367        if (Integer.TYPE.equals(type)) { return new Integer(0); }
368        if (Long.TYPE.equals(type)) { return new Long(0); }
369        if (Float.TYPE.equals(type)) { return new Float(0); }
370        if (Double.TYPE.equals(type)) { return new Double(0); }
371        if (Character.TYPE.equals(type)) { return new Character((char) 0); }
372        // If all other case, return null.
373        return null;
374    }

376    /**
377     * Sets the value of the property.
378     * @param value the new value.
379     */
380    public void setValue(Object value) {
381        synchronized (this) {
382            // Is the object is directly assignable to the property, affect it.
383            if (isAssignable(m_type, value)) {
384                m_value = value;
385            } else {
386                // If the object is a String, we must recreate the object from the String form
387                if (value instanceof String) {
388                    try {
389                        m_value = create(m_type, (String) value);
390                    } catch (ConfigurationException e) {
391                        throw new ClassCastException("Incompatible type for the property " + m_name + " : " + e.getMessage());
392                    }
393                } else {
394                    // Error, the given property cannot be injected.
395                    throw new ClassCastException("Incompatible type for the property " + m_name + " " + m_type.getName() + " expected, "
396                                                 + value.getClass() + " found");
397                }
398            }
399            m_invoked = false;
400        }
401    }

403    /**
404     * Checks if the given value is assignable to the given type.
405     * @param type the class of the type
406     * @param value the object to check
407     * @return <code>true</code> if the object is assignable in the property of type 'type'.
408     */
409    public static boolean isAssignable(Class type, Object value) {
410        if (value == null || type.isInstance(value) || value == Property.NO_VALUE) { // When the value is null, the assign works necessary.
411            return true;
412        } else if (type.isPrimitive()) {
413            // Manage all boxing types.
414            if (value instanceof Boolean && Boolean.TYPE.equals(type)) { return true; }
415            if (value instanceof Byte && Byte.TYPE.equals(type)) { return true; }
416            if (value instanceof Short && Short.TYPE.equals(type)) { return true; }
417            if (value instanceof Integer && Integer.TYPE.equals(type)) { return true; }
418            if (value instanceof Long && Long.TYPE.equals(type)) { return true; }
419            if (value instanceof Float && Float.TYPE.equals(type)) { return true; }
420            if (value instanceof Double && Double.TYPE.equals(type)) { return true; }
421            if (value instanceof Character && Character.TYPE.equals(type)) { return true; }
422            return false;
423        } else {
424            // Else return false.
425            return false;
426        }
427    }

429    /**
430     * Creates an object of the given type with the given String value.
431     * @param type the type of the returned object
432     * @param strValue the String value.
433     * @return the object of type 'type' created from the String 'value'
434     * @throws ConfigurationException if the object cannot be created.
435     */
436    public static Object create(Class type, String strValue) throws ConfigurationException {
437        if (Boolean.TYPE.equals(type)) {
438            return Boolean.valueOf(strValue);
439        }
440        if (Byte.TYPE.equals(type)) { return new Byte(strValue); }
441        if (Short.TYPE.equals(type)) { return new Short(strValue); }
442        if (Integer.TYPE.equals(type)) { return new Integer(strValue); }
443        if (Long.TYPE.equals(type)) { return new Long(strValue); }
444        if (Float.TYPE.equals(type)) { return new Float(strValue); }
445        if (Double.TYPE.equals(type)) { return new Double(strValue); }
446        // Character is a bit tricky, it's a boxing type, but there is not creator taking a String as parameter.
447        if (Character.TYPE.equals(type)  || Character.class.equals(type)) { return strValue.charAt(0); }
448
449        // Array :
450        if (type.isArray()) {
451            return createArrayObject(type.getComponentType(), ParseUtils.parseArrays(strValue));
452        }
453
454        // Enum :
455        if (type.getSuperclass() != null  && type.getSuperclass().getName().equals("java.lang.Enum")) {
456            try {
457                Method valueOf = type.getMethod("valueOf", new Class[] {String.class});
458                if (! valueOf.isAccessible()) {
459                    valueOf.setAccessible(true);
460                }
461                 // Invoke the static method
462                return valueOf.invoke(null, new String[] {strValue});
463            } catch (InvocationTargetException e) {
464                throw new ConfigurationException("Cannot create an enumerated value for " + type
465                        + " with " + strValue, e.getTargetException());
466            } catch (Exception e) {
467                throw new ConfigurationException("Cannot create an enumerated value for " + type
468                        + " with " + strValue, e);
469            }
470        }
471
472        // Else it is a neither a primitive type neither a String -> create
473        // the object by calling a constructor with a string in argument.
474        try {
475            Constructor cst = type.getConstructor(new Class[] { String.class });
476            return cst.newInstance(new Object[] { strValue });
477        } catch (SecurityException e) {
478            throw new ConfigurationException("Security exception during the creation of " + type, e);
479        } catch (NoSuchMethodException e) {
480            throw new ConfigurationException("Constructor not found exception during the creation of " + type, e);
481        } catch (IllegalArgumentException e) {
482            throw new ConfigurationException("Argument issue when calling the constructor of the type " + type, e);
483        } catch (InstantiationException e) {
484            throw new ConfigurationException("Instantiation problem  " + type, e);
485        } catch (IllegalAccessException e) {
486            throw new ConfigurationException("Illegal Access " + type, e);
487        } catch (InvocationTargetException e) {
488            throw new ConfigurationException("Invocation problem during the creation of " + type, e.getTargetException());
489        }
490
491    }

493    /**
494     * Creates an array object containing the type component type from
495     * the String array 'values'.
496     * @param interntype the internal type of the array.
497     * @param values the String array
498     * @return the array containing objects created from the 'values' array
499     * @throws ConfigurationException if the array cannot be created correctly
500     */
501    public static Object createArrayObject(Class interntype, String[] values) throws ConfigurationException {
502        if (Boolean.TYPE.equals(interntype)) {
503            boolean[] bool = new boolean[values.length];
504            for (int i = 0; i < values.length; i++) {
505                bool[i] = Boolean.valueOf(values[i]).booleanValue();
506            }
507            return bool;
508        }
509        if (Byte.TYPE.equals(interntype)) {
510            byte[] byt = new byte[values.length];
511            for (int i = 0; i < values.length; i++) {
512                byt[i] = new Byte(values[i]).byteValue();
513            }
514            return byt;
515        }
516        if (Short.TYPE.equals(interntype)) {
517            short[] shor = new short[values.length];
518            for (int i = 0; i < values.length; i++) {
519                shor[i] = new Short(values[i]).shortValue();
520            }
521            return shor;
522        }
523        if (Integer.TYPE.equals(interntype)) {
524            int[] ints = new int[values.length];
525            for (int i = 0; i < values.length; i++) {
526                ints[i] = new Integer(values[i]).intValue();
527            }
528            return ints;
529        }
530        if (Long.TYPE.equals(interntype)) {
531            long[] longs = new long[values.length];
532            for (int i = 0; i < values.length; i++) {
533                longs[i] = new Long(values[i]).longValue();
534            }
535            return longs;
536        }
537        if (Float.TYPE.equals(interntype)) {
538            float[] floats = new float[values.length];
539            for (int i = 0; i < values.length; i++) {
540                floats[i] = new Float(values[i]).floatValue();
541            }
542            return floats;
543        }
544        if (Double.TYPE.equals(interntype)) {
545            double[] doubles = new double[values.length];
546            for (int i = 0; i < values.length; i++) {
547                doubles[i] = new Double(values[i]).doubleValue();
548            }
549            return doubles;
550        }
551        if (Character.TYPE.equals(interntype)) {
552            char[] chars = new char[values.length];
553            for (int i = 0; i < values.length; i++) {
554                chars[i] = values[i].toCharArray()[0];
555            }
556            return chars;
557        }
558
559        // Else it is a neither a primitive type -> create the
560        // object by calling a constructor with a string in argument.
561        try {
562            Constructor cst = interntype.getConstructor(new Class[] { String.class });
563            Object[] object = (Object[]) Array.newInstance(interntype, values.length);
564            for (int i = 0; i < values.length; i++) {
565                object[i] = cst.newInstance(new Object[] { values[i].trim() });
566            }
567            return object;
568        } catch (NoSuchMethodException e) {
569            throw new ConfigurationException("Constructor not found exception in setValue on " + interntype.getName(), e);
570        } catch (IllegalArgumentException e) {
571            throw new ConfigurationException("Argument issue when calling the constructor of the type " + interntype.getName(), e);
572        } catch (InstantiationException e) {
573            throw new ConfigurationException("Instantiation problem  " + interntype.getName(), e);
574        } catch (IllegalAccessException e) {
575            throw new ConfigurationException("Illegal Access Exception in  " + interntype.getName(), e);
576        } catch (InvocationTargetException e) {
577            throw new ConfigurationException("Invocation problem " + interntype.getName(), e.getTargetException());
578        }
579    }

590    /**
591     * Invokes the setter method on the given pojo object.
592     * If no specified pojo object, it calls on each created pojo object.
593     * @param instance the created object (could be <code>null</code>)
594     */
595    public synchronized void invoke(Object instance) {
596        if (m_invoked) {
597            return; // Already called.
598        }
599
600        if (m_value == NO_VALUE) {
601            // Don't call method if no value
602            return;
603        }
604
605        try {
606            if (instance == null) {
607                m_method.call(new Object[] { m_value });
608            } else {
609                m_method.call(instance, new Object[] { m_value });
610            }
611            m_invoked = true;
612        } catch (NoSuchMethodException e) {
613            m_handler.error("The method " + m_method + " does not exist in the implementation class " + m_manager.getClassName(), e);
614            m_manager.stop();
615        } catch (IllegalAccessException e) {
616            m_handler.error("The method " + m_method + " is not accessible in the implementation class " + m_manager.getClassName(), e);
617            m_manager.stop();
618        } catch (InvocationTargetException e) {
619            m_handler.error("The method " + m_method + " in the implementation class " + m_manager.getClassName() + "throws an exception : " + e.getTargetException().getMessage(), e.getTargetException());
620            m_manager.setState(ComponentInstance.INVALID);
621        }
622    }

624    /**
625     * A field value is required by the object 'pojo'.
626     * @param pojo the POJO object
627     * @param fieldName the field
628     * @param value the last value
629     * @return the value if the handler want to inject this value.
630     * @see org.apache.felix.ipojo.FieldInterceptor#onGet(java.lang.Object, java.lang.String, java.lang.Object)
631     */
632    public synchronized Object onGet(Object pojo, String fieldName, Object value) {
633        if (m_value  == NO_VALUE) {
634            return getNoValue(m_type);
635        }
636        return m_value;
637    }

639    /**
640     * The field 'field' receives a new value.
641     * @param pojo the pojo
642     * @param fieldName the field name
643     * @param value the new value
644     * @see org.apache.felix.ipojo.FieldInterceptor#onSet(java.lang.Object, java.lang.String, java.lang.Object)
645     */
646    public synchronized void onSet(Object pojo, String fieldName, Object value) {
647        if (m_value == null || ! m_value.equals(value)) {
648            setValue(value);
649        }
650    }

652    /**
653     * Gets the object to inject as constructor parameter.
654     * @param index the constructor parameter index
655     * @return the object to inject, so the property value.
656     * @see org.apache.felix.ipojo.ConstructorInjector#getConstructorParameter(int)
657     */
658    public Object getConstructorParameter(int index) {
659        if (m_index != index) {
660            return null;
661        }
662
663        if (m_value  == NO_VALUE) {
664           return getNoValue(m_type);
665        }
666        return m_value;
667    }
