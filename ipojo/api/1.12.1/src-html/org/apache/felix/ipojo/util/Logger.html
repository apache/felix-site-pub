019package org.apache.felix.ipojo.util;
020
021import org.apache.felix.ipojo.ComponentInstance;
022import org.apache.felix.ipojo.ErrorHandler;
023import org.apache.felix.ipojo.extender.internal.Extender;
024import org.osgi.framework.BundleContext;
025import org.osgi.framework.ServiceReference;
026import org.osgi.service.log.LogService;
027
028/**
029 * iPOJO Logger.
030 * This class is an helper class implementing a simple log system.
031 * This logger sends log messages to a log service if available.
032 *
033 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
034 */
035public class Logger implements Log {
036
037    /**
038     * The iPOJO default log level property.
039     */
040    public static final String IPOJO_LOG_LEVEL_PROP = "ipojo.log.level";
041    /**
042     * iPOJO log level manifest header.
043     * The uppercase 'I' is important as BND removes all headers that do not
044     * start with an uppercase are not added to the bundle.
045     * Use an upper case to support bnd.
046     */
047    public static final String IPOJO_LOG_LEVEL_HEADER = "IPOJO-log-level";
048    /**
049     * A shared log service implementation writing messages on the console.
050     */
051    private static final LogService m_defaultLogger = new ConsoleLogService();
052    private static final String DEBUG_HEADER = "[DEBUG]";
053    private static final String INFO_HEADER = "[INFO]";
054    private static final String WARNING_HEADER = "[WARNING]";
055    private static final String ERROR_HEADER = "[ERROR]";
056    private static final String UNKNOWN_HEADER = "[UNKNOWN]";
057    /**
058     * The Bundle Context used to get the
059     * log service.
060     */
061    private final BundleContext m_context;
062    /**
063     * The name of the logger.
064     */
065    private final String m_name;
066    /**
067     * The trace level of this logger.
068     */
069    private final int m_level;
070    /**
071     * The instance associated to the logger if any.
072     */
073    private ComponentInstance m_instance;
074
075    /**
076     * Creates a logger.
077     *
078     * @param context the bundle context
079     * @param name    the name of the logger
080     * @param level   the trace level
081     */
082    public Logger(BundleContext context, String name, int level) {
083        m_name = name;
084        m_level = level;
085        m_context = context;
086    }
087
088    /**
089     * Creates a logger.
090     *
091     * @param context  the bundle context
092     * @param instance the instance
093     * @param level    the trace level
094     */
095    public Logger(BundleContext context, ComponentInstance instance, int level) {
096        this(context, instance.getInstanceName(), level);
097        m_instance = instance;
098    }
099
100    /**
101     * Create a logger.
102     * Uses the default logger level.
103     *
104     * @param context the bundle context
105     * @param name    the name of the logger
106     */
107    public Logger(BundleContext context, String name) {
108        this(context, name, getDefaultLevel(context));
109    }
110
111    /**
112     * Create a logger.
113     * Uses the default logger level.
114     *
115     * @param context  the bundle context
116     * @param instance the instance
117     */
118    public Logger(BundleContext context, ComponentInstance instance) {
119        this(context, instance, getDefaultLevel(context));
120    }
121
122    /**
123     * Gets the default logger level.
124     * The property is searched inside the framework properties,
125     * the system properties, and in the manifest from the given
126     * bundle context. By default, set the level to {@link Logger#WARNING}.
127     *
128     * @param context the bundle context.
129     * @return the default log level.
130     */
131    private static int getDefaultLevel(BundleContext context) {
132        // First check in the framework and in the system properties
133        String level = context.getProperty(IPOJO_LOG_LEVEL_PROP);
134
135        // If null, look in the bundle manifest
136        if (level == null) {
137            String key = IPOJO_LOG_LEVEL_PROP.replace('.', '-');
138            level = (String) context.getBundle().getHeaders().get(key);
139        }
140
141        // if still null try the second header
142        if (level == null) {
143            level = (String) context.getBundle().getHeaders().get(IPOJO_LOG_LEVEL_HEADER);
144        }
145
146        if (level != null) {
147            if (level.equalsIgnoreCase("info")) {
148                return INFO;
149            } else if (level.equalsIgnoreCase("debug")) {
150                return DEBUG;
151            } else if (level.equalsIgnoreCase("warning")) {
152                return WARNING;
153            } else if (level.equalsIgnoreCase("error")) {
154                return ERROR;
155            }
156        }
157
158        // Either l is null, either the specified log level was unknown
159        // Set the default to WARNING
160        return WARNING;
161
162    }
163
164    private static String getLogHeaderForLevel(int level) {
165        switch (level) {
166            case DEBUG:
167                return DEBUG_HEADER;
168            case INFO:
169                return INFO_HEADER;
170            case WARNING:
171                return WARNING_HEADER;
172            case ERROR:
173                return ERROR_HEADER;
174            default:
175                return UNKNOWN_HEADER;
176        }
177    }
178
179    /**
180     * Logs a message.
181     *
182     * @param level the level of the message
183     * @param msg   the the message to log
184     */
185    public void log(int level, String msg) {
186        if (m_level >= level) {
187            dispatch(level, msg, null);
188        }
189        invokeErrorHandler(level, msg, null);
190    }
191
192    /**
193     * Logs a message with an exception.
194     *
195     * @param level     the level of the message
196     * @param msg       the message to log
197     * @param exception the exception attached to the message
198     */
199    public void log(int level, String msg, Throwable exception) {
200        if (m_level >= level) {
201            dispatch(level, msg, exception);
202        }
203        invokeErrorHandler(level, msg, exception);
204    }
205
206    /**
207     * Internal log method.
208     *
209     * @param level     the level of the message.
210     * @param msg       the message to log
211     * @param exception the exception attached to the message
212     */
213    private void dispatch(int level, String msg, Throwable exception) {
214        LogService log = null;
215        ServiceReference ref = null;
216        try {
217            // Security Check
218            if (SecurityHelper.hasPermissionToGetService(LogService.class.getName(), m_context)) {
219                ref = m_context.getServiceReference(LogService.class.getName());
220            } else {
221                Extender.getIPOJOBundleContext().getServiceReference(LogService.class.getName());
222            }
223
224            if (ref != null) {
225                log = (LogService) m_context.getService(ref);
226            }
227        } catch (IllegalStateException e) {
228            // Handle the case where the iPOJO bundle is stopping, or the log service already ran away.
229        }
230
231        if (log == null) {
232            log = m_defaultLogger;
233        }
234
235        String name = m_name;
236        if (name == null) {
237            name = "";
238        }
239
240        String message = String.format("%s %s : %s", getLogHeaderForLevel(level), name, msg);
241        switch (level) {
242            case DEBUG:
243                log.log(LogService.LOG_DEBUG, message, exception);
244                break;
245            case INFO:
246                log.log(LogService.LOG_INFO, message, exception);
247                break;
248            case WARNING:
249                log.log(LogService.LOG_WARNING, message, exception);
250                break;
251            case ERROR:
252                log.log(LogService.LOG_ERROR, message, exception);
253                break;
254            default:
255                log.log(LogService.LOG_INFO, message, exception);
256                break;
257        }
258
259        if (ref != null) {
260            m_context.ungetService(ref);
261        }
262    }
263
264    /**
265     * Invokes the error handler service is present.
266     *
267     * @param level the log level
268     * @param msg   the message
269     * @param error the error
270     */
271    private void invokeErrorHandler(int level, String msg, Throwable error) {
272        // First check the level
273        if (level > WARNING) {
274            return; // Others levels are not supported.
275        }
276        // Try to get the error handler service
277        try {
278            ServiceReference ref = m_context.getServiceReference(ErrorHandler.class.getName());
279            if (ref != null) {
280                ErrorHandler handler = (ErrorHandler) m_context.getService(ref);
281                if (level == ERROR) {
282                    handler.onError(m_instance, msg, error);
283                } else if (level == WARNING) {
284                    handler.onWarning(m_instance, msg, error);
285                } // The others case are not supported
286                m_context.ungetService(ref);
287            } // Else do nothing...
288        } catch (IllegalStateException e) {
289            // Ignore
290            // The iPOJO bundle is stopping.
291        }
292    }
293
294    /**
295     * A simple log service implementation writing the messages and stack trace on System.err.
296     */
297    private static class ConsoleLogService implements LogService {
298
299        public void log(int i, String s) {
300            log(i, s, null);
301        }
302
303        public void log(int level, String message, Throwable exception) {
304            System.err.println(message);
305            if (exception != null) {
306                exception.printStackTrace();
307            }
308        }
309
310        public void log(ServiceReference serviceReference, int i, String s) {
311            // not used.
312        }
313
314        public void log(ServiceReference serviceReference, int i, String s, Throwable throwable) {
315            // not used.
316        }
317    }
