032/**
033 * Constructor Adapter.
034 * This class adds an instance manager argument (so switch variable index).
035 * Moreover, it adapts field accesses to delegate accesses to the instance
036 * manager if needed.
037 *
038 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
039 */
040public class ConstructorCodeAdapter extends GeneratorAdapter implements Opcodes {
041
042    /**
043     * The class containing the field.
044     * m_owner : String
045     */
046    private String m_owner;
047
048    /**
049     * Is the super call detected ?
050     */
051    private boolean m_superDetected;
052
053    /**
054     * The super class.
055     */
056    private String m_superClass;
057
058    /**
059     * Set of contained fields.
060     */
061    private Set<String> m_fields;
062
063
064    /**
065     * PropertyCodeAdapter constructor.
066     * A new FiledCodeAdapter should be create for each method visit.
067     *
068     * @param mv     the MethodVisitor
069     * @param owner  the name of the class
070     * @param fields the list of contained fields
071     * @param access the constructor access
072     * @param desc   the constructor descriptor
073     * @param name   the name
074     */
075    public ConstructorCodeAdapter(final MethodVisitor mv, final String owner, Set<String> fields, int access, String name, String desc, String superClass) {
076        super(Opcodes.ASM5, mv, access, name, desc);
077        m_owner = owner;
078        m_superDetected = false;
079        m_fields = fields;
080        m_superClass = superClass;
081    }
082
083    /**
084     * Visits an annotation.
085     * If the annotation is visible, the annotation is removed. In fact
086     * the annotation was already moved to the method replacing this one.
087     * If the annotation is not visible, this annotation is kept on this method.
088     *
089     * @param name    the name of the annotation
090     * @param visible the annotation visibility
091     * @return the <code>null</code> if the annotation is visible, otherwise returns
092     * {@link GeneratorAdapter#visitAnnotation(String, boolean)}
093     * @see org.objectweb.asm.commons.GeneratorAdapter#visitAnnotation(java.lang.String, boolean)
094     */
095    public AnnotationVisitor visitAnnotation(String name, boolean visible) {
096        // Annotations are moved to the injected constructor.
097        if (visible) {
098            return null;
099        } else {
100            return super.visitAnnotation(name, false);
101        }
102    }
103
104    /**
105     * Visits a parameter annotation.
106     * Parameter annotations are moved to replacing constructor except
107     * they are injection annotations(-@Property and -@Requires).
108     * Because injection annotations shouldn't be copied to generated one
109     * in case of re-manipulation, since this is caused to wrong type resolution
110     * of injected parameters.
111     *
112     * @param parameter parameter index
113     * @param desc      annotation description(annotation name)
114     * @param visible   is parameter annotation visible
115     * @return @AnnotationVisitor
116     */
117    public AnnotationVisitor visitParameterAnnotation(
118            final int parameter,
119            final String desc,
120            final boolean visible) {
121
122        /*
123         * Generated constructor shouldn't inherit injection annotations
124         */
125        if (desc.equals("Lorg/apache/felix/ipojo/annotations/Property;")
126                || desc.equals("Lorg/apache/felix/ipojo/annotations/Requires;")
127                || Names.isCustomAnnotation(desc)) {
128            return null;
129        } else {
130            return super.visitParameterAnnotation(parameter, desc, visible);
131        }
132
133    }
134
135
136    /**
137     * Adapts field accesses.
138     * If the field is owned by the visited class:
139     * <ul>
140     * <li><code>GETFIELD</code> are changed to a <code>__getX</code> invocation.</li>
141     * <li><code>SETFIELD</code> are changed to a <code>__setX</code> invocation.</li>
142     * </ul>
143     *
144     * @param opcode the visited operation code
145     * @param owner  the owner of the field
146     * @param name   the name of the field
147     * @param desc   the descriptor of the field
148     * @see org.objectweb.asm.MethodVisitor#visitFieldInsn(int, String, String, String)
149     */
150    public void visitFieldInsn(
151            final int opcode,
152            final String owner,
153            final String name,
154            final String desc) {
155        if (m_fields.contains(name) && m_owner.equals(owner)) {
156            if (opcode == GETFIELD) {
157                String gDesc = "()" + desc;
158                mv.visitMethodInsn(INVOKEVIRTUAL, owner, "__get" + name, gDesc, false);
159                return;
160            } else if (opcode == PUTFIELD) {
161                String sDesc = "(" + desc + ")V";
162                mv.visitMethodInsn(INVOKEVIRTUAL, owner, "__set" + name, sDesc, false);
163                return;
164            }
165        }
166        super.visitFieldInsn(opcode, owner, name, desc);
167    }
168
169    /**
170     * Visits a method invocation instruction.
171     * After the super constructor invocation, insert the _setComponentManager invocation.
172     * Otherwise, the method invocation doesn't change
173     *
174     * @param opcode the opcode
175     * @param owner  the class owning the invoked method
176     * @param name   the method name
177     * @param desc   the method descriptor
178     * @param itf    if the method's owner class is an interface
179     * @see org.objectweb.asm.commons.GeneratorAdapter#visitMethodInsn(int, java.lang.String, java.lang.String, java.lang.String)
180     */
181    public void visitMethodInsn(int opcode, String owner, String name, String desc, boolean itf) {
182
183        // A method call is detected, check if it is the super call :
184        // the first init is not necessary the super call, so check that it is really the super class.
185        if (!m_superDetected && name.equals("<init>") && owner.equals(m_superClass)) {
186            m_superDetected = true;
187            // The first invocation is the super call
188            // 1) Visit the super constructor :
189
190            //mv.visitVarInsn(ALOAD, 0); The ALOAD 0 was already visited. This previous visit allows
191            // Super constructor parameters.
192            mv.visitMethodInsn(opcode, owner, name, desc, false); // Super constructor invocation
193
194            // 2) Load the object and the component manager argument
195            mv.visitVarInsn(ALOAD, 0);
196            //mv.visitVarInsn(ALOAD, Type.getArgumentTypes(m_constructorDesc).length);
197            mv.visitVarInsn(ALOAD, 1);  // CM is always the first argument
198            // 3) Initialize the field
199            mv.visitMethodInsn(INVOKESPECIAL, m_owner, "_setInstanceManager",
200                    "(Lorg/apache/felix/ipojo/InstanceManager;)V", false);
201
202        } else {
203            if (opcode == INVOKEINTERFACE) {
204                mv.visitMethodInsn(opcode, owner, name, desc, true);
205            } else {
206                mv.visitMethodInsn(opcode, owner, name, desc, false);
207            }
208        }
209    }

211    /**
212     * Visits a variable instruction.
213     * This method increments the variable index if
214     * it is not <code>this</code> (i.e. 0). This increment
215     * is due to the instance manager parameter added in the method
216     * signature.
217     *
218     * @param opcode the opcode
219     * @param var    the variable index
220     * @see org.objectweb.asm.commons.GeneratorAdapter#visitVarInsn(int, int)
221     */
222    public void visitVarInsn(int opcode, int var) {
223        if (var == 0) {
224            mv.visitVarInsn(opcode, var); // ALOAD 0 (THIS)
225        } else {
226            mv.visitVarInsn(opcode, var + 1); // All other variable index must be incremented (due to
227            // the instance manager argument
228        }
229
230    }
231
232    /**
233     * Visits an increment instruction.
234     * This method increments the variable index if
235     * it is not <code>this</code> (i.e. 0). This increment
236     * is due to the instance manager parameter added in the method
237     * signature.
238     *
239     * @param var       the variable index
240     * @param increment the increment
241     * @see org.objectweb.asm.commons.GeneratorAdapter#visitIincInsn(int, int)
242     */
243    public void visitIincInsn(int var, int increment) {
244        if (var != 0) {
245            mv.visitIincInsn(var + 1, increment);
246        } else {
247            mv.visitIincInsn(var, increment); // Increment the current object ???
248        }
249    }
250
251    /**
252     * Visits a local variable.
253     * Adds _manager and increment others variable indexes.
254     * This variable has the same scope than <code>this</code> and
255     * has the <code>1</code> index.
256     *
257     * @param name      the variable name
258     * @param desc      the variable descriptor
259     * @param signature the variable signature
260     * @param start     the beginning label
261     * @param end       the ending label
262     * @param index     the variable index
263     * @see org.objectweb.asm.commons.GeneratorAdapter#visitLocalVariable(java.lang.String, java.lang.String, java.lang.String, org.objectweb.asm.Label, org.objectweb.asm.Label, int)
264     */
265    public void visitLocalVariable(String name, String desc, String signature, Label start, Label end, int index) {
266        if (index == 0) {
267            mv.visitLocalVariable(name, desc, signature, start, end, index);
268            mv.visitLocalVariable("_manager", "Lorg/apache/felix/ipojo/InstanceManager;", null, start, end, 1);
269        } else {
270            mv.visitLocalVariable(name, desc, signature, start, end, index + 1);
271        }
272    }
273
274    /**
275     * Visit max method.
276     * The stack size is incremented of 1. The
277     * local variable count is incremented of 2.
278     *
279     * @param maxStack  the stack size.
280     * @param maxLocals the local variable count.
281     * @see org.objectweb.asm.commons.GeneratorAdapter#visitMaxs(int, int)
282     */
283    public void visitMaxs(int maxStack, int maxLocals) {
284        mv.visitMaxs(maxStack + 1, maxLocals + 2);
285    }
