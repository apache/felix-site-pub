139    /**
140     * Creates a provided service object.
141     *
142     * @param handler               the the provided service handler.
143     * @param specification         the specifications provided by this provided service
144     * @param factoryPolicy         the service providing policy
145     * @param creationStrategyClass the customized service object creation strategy.
146     * @param conf                  the instance configuration.
147     */
148    public ProvidedService(ProvidedServiceHandler handler, String[] specification, int factoryPolicy, Class creationStrategyClass, Dictionary conf) {
149        CreationStrategy strategy;
150        m_handler = handler;
151
152        m_serviceSpecifications = specification;
153
154        if (creationStrategyClass == null) {
155            m_policy = factoryPolicy;
156        } else {
157            m_policy = CUSTOM_STRATEGY;
158        }
159
160        // Add instance name, factory name and factory version is set.
161        try {
162            addProperty(new Property(Factory.INSTANCE_NAME_PROPERTY, null, null, handler.getInstanceManager().getInstanceName(), String.class.getName(), handler.getInstanceManager(), handler));
163            addProperty(new Property("factory.name", null, null, handler.getInstanceManager().getFactory().getFactoryName(), String.class.getName(), handler.getInstanceManager(), handler));
164
165            if (handler.getInstanceManager().getFactory().getVersion() != null) {
166                addProperty(new Property(Factory.FACTORY_VERSION_PROPERTY, null, null, handler.getInstanceManager().getFactory().getVersion(), String.class.getName(), handler.getInstanceManager(), handler));
167            }
168
169            // Add the service.* if defined
170            if (conf.get(Constants.SERVICE_PID) != null) {
171                addProperty(new Property(Constants.SERVICE_PID, null, null, (String) conf.get(Constants.SERVICE_PID), String.class.getName(), handler.getInstanceManager(), handler));
172            }
173            if (conf.get(Constants.SERVICE_RANKING) != null) {
174                addProperty(new Property(Constants.SERVICE_RANKING, null, null, (String) conf.get(Constants.SERVICE_RANKING), "int", handler.getInstanceManager(), handler));
175            }
176            if (conf.get(Constants.SERVICE_VENDOR) != null) {
177                addProperty(new Property(Constants.SERVICE_VENDOR, null, null, (String) conf.get(Constants.SERVICE_VENDOR), String.class.getName(), handler.getInstanceManager(), handler));
178            }
179            if (conf.get(Constants.SERVICE_DESCRIPTION) != null) {
180                addProperty(new Property(Constants.SERVICE_DESCRIPTION, null, null, (String) conf.get(Constants.SERVICE_DESCRIPTION), String.class.getName(), handler.getInstanceManager(), handler));
181            }
182
183        } catch (ConfigurationException e) {
184            m_handler.error("An exception occurs when adding instance.name and factory.name property : " + e.getMessage());
185        }
186
187        if (creationStrategyClass != null) {
188            try {
189                strategy = (CreationStrategy) creationStrategyClass.newInstance();
190            } catch (IllegalAccessException e) {
191                strategy = null;
192                m_handler.error("["
193                        + m_handler.getInstanceManager().getInstanceName()
194                        + "] The customized service object creation policy "
195                        + "(" + creationStrategyClass.getName() + ") is not accessible: "
196                        + e.getMessage(), e);
197                getInstanceManager().stop();
198            } catch (InstantiationException e) {
199                strategy = null;
200                m_handler.error("["
201                        + m_handler.getInstanceManager().getInstanceName()
202                        + "] The customized service object creation policy "
203                        + "(" + creationStrategyClass.getName() + ") cannot be instantiated: "
204                        + e.getMessage(), e);
205                getInstanceManager().stop();
206            }
207        } else {
208            switch (factoryPolicy) {
209                case SINGLETON_STRATEGY:
210                    strategy = new SingletonStrategy();
211                    break;
212                case SERVICE_STRATEGY:
213                case STATIC_STRATEGY:
214                    // In this case, we need to try to create a new pojo object,
215                    // the factory method will handle the creation.
216                    strategy = new FactoryStrategy();
217                    break;
218                case INSTANCE_STRATEGY:
219                    strategy = new PerInstanceStrategy();
220                    break;
221                // Other policies:
222                // Thread : one service object per asking thread
223                // Consumer : one service object per consumer
224                default:
225                    strategy = null;
226                    List specs = Arrays.asList(m_serviceSpecifications);
227                    m_handler.error("["
228                            + m_handler.getInstanceManager().getInstanceName()
229                            + "] Unknown creation policy for " + specs + " : "
230                            + factoryPolicy);
231                    getInstanceManager().stop();
232                    break;
233            }
234        }
235        m_strategy = strategy;
236    }

313    /**
314     * Registers the service. The service object must be able to serve this
315     * service.
316     * This method also notifies the creation strategy of the publication.
317     */
318    public void registerService() {
319        ServiceRegistration reg = null;
320        Properties serviceProperties = null;
321        // Do not have to be in the synchronized block, immutable.
322        final BundleContext bc = m_handler.getInstanceManager().getContext();
323
324        synchronized (this) {
325            if (m_serviceRegistration != null) {
326                return;
327            } else {
328                if (m_handler.getInstanceManager().getState() == ComponentInstance.VALID && isAtLeastAServiceControllerValid()) {
329                    // Security check
330                    if (SecurityHelper.hasPermissionToRegisterServices(
331                            m_serviceSpecifications, bc) && SecurityHelper.canRegisterService(bc)) {
332                        serviceProperties = getServiceProperties();
333                    } else {
334                        throw new SecurityException("The bundle "
335                                + bc.getBundle().getBundleId()
336                                + " does not have the"
337                                + " permission to register the services "
338                                + Arrays.asList(m_serviceSpecifications));
339                    }
340                } else {
341                    // We don't have to do anything.
342                    return;
343                }
344            }
345        }
346
347        // Registration must be done outside of the synchronized block.
348        m_strategy.onPublication(getInstanceManager(),
349                getServiceSpecificationsToRegister(),
350                serviceProperties);
351        reg = bc.registerService(
352                getServiceSpecificationsToRegister(), this,
353                (Dictionary) serviceProperties);
354
355        boolean update = false;
356        synchronized (this) {
357            if (m_serviceRegistration != null) {
358                // Oh oh the service was registered twice. Unregister the last one
359                reg.unregister();
360                return;
361            } else {
362                m_serviceRegistration = reg;
363            }
364
365            // An update may happen during the registration, re-check and apply.
366            // This must be call outside the synchronized block.
367            // If the registration is null, the security helper returns false.
368            if (m_wasUpdated && SecurityHelper.canUpdateService(reg)) {
369                serviceProperties = getServiceProperties();
370                update = true;
371            }
372        }
373
374        if (update) {
375            reg.setProperties(serviceProperties);
376        }
377
378        synchronized (this) {
379            m_publishedProperties = serviceProperties;
380            m_wasUpdated = false;
381
382            // Call the post-registration callback in the same thread holding
383            // the monitor lock.
384            // This allows to be sure that the callback is called once per
385            // registration.
386            // But the callback must take care to not create a deadlock
387            if (m_postRegistration != null) {
388                try {
389                    m_postRegistration
390                            .call(new Object[]{m_serviceRegistration
391                                    .getReference()});
392                } catch (Exception e) {
393                    m_handler.error(
394                            "Cannot invoke the post-registration callback "
395                                    + m_postRegistration.getMethod(), e);
396                }
397            }
398        }
399
400        // Notify: ProvidedServiceListeners.serviceRegistered()
401        notifyListeners(+1);
402
403    }

405    /**
406     * Withdraws the service from the service registry.
407     */
408    public void unregisterService() {
409        ServiceReference ref = null;
410        synchronized (this) {
411            // Create a copy of the service reference in the case we need
412            // to inject it to the post-unregistration callback.
413            if (m_serviceRegistration != null) {
414                ref = m_serviceRegistration.getReference();
415                m_serviceRegistration.unregister();
416                m_serviceRegistration = null;
417            }
418
419            m_strategy.onUnpublication();
420
421            // Call the post-unregistration callback in the same thread holding the monitor lock.
422            // This allows to be sure that the callback is called once per unregistration.
423            // But the callback must take care to not create a deadlock
424            if (m_postUnregistration != null && ref != null) {
425                try {
426                    m_postUnregistration.call(new Object[]{ref});
427                } catch (Exception e) {
428                    m_handler.error("Cannot invoke the post-unregistration callback " + m_postUnregistration.getMethod(), e);
429                }
430            }
431        }
432
433        // Notify: ProvidedServiceListeners.serviceUnregistered()
434        if (ref != null) {
435            notifyListeners(-1);
436        }
437
438    }

459     * contains only property where a value are assigned.

461     * This method is called while holding the monitor lock.
462     *

465    private Properties getServiceProperties() {
466        // Build the service properties list
467        Properties serviceProperties = new Properties();
468        for (Property p : m_properties.values()) {
469            final Object value = p.getValue();
470            if (value != null && value != Property.NO_VALUE) {
471                serviceProperties.put(p.getName(), value);
472            }
473        }
474        return serviceProperties;
475    }

486    /**
487     * Update the service properties. The new list of properties is sent to
488     * the service registry.
489     */
490    public void update() {
491        boolean doCallListener = false;
492        synchronized (this) {
493            // Update the service registration
494            if (m_serviceRegistration != null) {
495                Properties updated = getServiceProperties();
496                Dictionary oldProps = (Dictionary) ((Properties) m_publishedProperties).clone();
497                Dictionary newProps = (Dictionary) (updated.clone());
498
499                // Remove keys that must not be compared
500                newProps.remove(Factory.INSTANCE_NAME_PROPERTY);
501                oldProps.remove(Factory.INSTANCE_NAME_PROPERTY);
502                newProps.remove(Constants.SERVICE_ID);
503                oldProps.remove(Constants.SERVICE_ID);
504                newProps.remove(Constants.SERVICE_PID);
505                oldProps.remove(Constants.SERVICE_PID);
506                newProps.remove("factory.name");
507                oldProps.remove("factory.name");
508                newProps.remove(ConfigurationAdmin.SERVICE_FACTORYPID);
509                oldProps.remove(ConfigurationAdmin.SERVICE_FACTORYPID);
510
511                // Trigger the update only if the properties have changed.
512
513                // First check, are the size equals
514                if (oldProps.size() != newProps.size()) {
515                    if (SecurityHelper.canUpdateService(m_serviceRegistration)) {
516                        m_handler.info("Updating Registration : " + oldProps.size() + " / " + newProps.size());
517                        m_publishedProperties = updated;
518                        m_serviceRegistration.setProperties(updated);
519                        doCallListener = true;
520                    }
521                } else {
522                    // Check changes
523                    Enumeration keys = oldProps.keys();
524                    boolean hasChanged = false;
525                    while (!hasChanged && keys.hasMoreElements()) {
526                        String k = (String) keys.nextElement();
527                        Object val = oldProps.get(k);
528                        if (!val.equals(updated.get(k))) {
529                            hasChanged = true;
530                        }
531                    }
532                    if (hasChanged && SecurityHelper.canUpdateService(m_serviceRegistration)) {
533                        m_handler.info("Updating Registration : " + updated);
534                        m_publishedProperties = updated;
535                        m_serviceRegistration.setProperties(updated);
536                        doCallListener = true;
537                    }
538                }
539            } else {
540                // Need to be updated later.
541                m_wasUpdated = true;
542            }
543        }
544        if (doCallListener) {
545            // Notify: ProvidedServiceListeners.serviceUpdated()
546            notifyListeners(0);
547        }
548    }

550    /**
551     * Add properties to the list.
552     *
553     * @param props : properties to add
554     */
555    protected void addProperties(Dictionary props) {
556        Enumeration keys = props.keys();
557        boolean updated = false;
558        while (keys.hasMoreElements()) {
559            String key = (String) keys.nextElement();
560            Object value = props.get(key);
561
562            // m_properties can be modified by another thread, we need to make a stack confinement
563            Property property;
564            synchronized (this) {
565                property = m_properties.get(key);
566            }
567
568            if (property != null) {
569                // Already existing property
570                if (property.getValue() == null || !value.equals(property.getValue())) {
571                    property.setValue(value);
572                    updated = true;
573                }
574            } else {
575                try {
576                    // Create the property.
577                    property = new Property(key, null, null, value, getInstanceManager(), m_handler);
578                    addProperty(property);
579                    updated = true;
580                } catch (ConfigurationException e) {
581                    m_handler.error("The propagated property " + key + " cannot be created correctly : " + e.getMessage());
582                }
583            }
584        }
585
586        if (updated) {
587            m_handler.info("Update triggered by adding properties " + props);
588            update();
589        }
590    }

592    /**
593     * Remove properties from the list.
594     *
595     * @param props : properties to remove
596     */
597    protected void deleteProperties(Dictionary props) {
598        Enumeration keys = props.keys();
599        boolean mustUpdate = false;
600        while (keys.hasMoreElements()) {
601            String key = (String) keys.nextElement();
602            mustUpdate = mustUpdate || removeProperty(key);
603        }
604
605        if (mustUpdate) {
606            m_handler.info("Update triggered when removing properties : " + props);
607            update();
608        }
609    }

674    /**
675     * Checks if at least one service controller is valid.
676     *
677     * @return <code>true</code> if one service controller at least
678     * is valid.
679     */
680    private boolean isAtLeastAServiceControllerValid() {
681        // No controller
682        if (m_controllers.isEmpty()) {
683            return true;
684        }
685
686        for (ServiceController controller : m_controllers.values()) {
687            if (controller.getValue()) {
688                return true;
689            }
690        }
691        return false;
692    }

694    private String[] getServiceSpecificationsToRegister() {
695        if (m_controllers.isEmpty()) {
696            return m_serviceSpecifications;
697        }
698
699        ArrayList<String> l = new ArrayList<String>();
700        if (m_controllers.containsKey(ALL_SPECIFICATIONS_FOR_CONTROLLERS)) {
701            ServiceController ctrl = m_controllers.get(ALL_SPECIFICATIONS_FOR_CONTROLLERS);
702            if (ctrl.m_value) {
703                l.addAll(Arrays.asList(m_serviceSpecifications));
704            }
705        }
706
707        for (String spec : m_controllers.keySet()) {
708            ServiceController ctrl = m_controllers.get(spec);
709            if (ctrl.m_value) {
710                if (!ALL_SPECIFICATIONS_FOR_CONTROLLERS.equals(spec)) { // Already added.
711                    if (!l.contains(spec)) {
712                        l.add(spec);
713                    }
714                }
715            } else {
716                l.remove(spec);
717            }
718        }
719
720        return l.toArray(new String[l.size()]);
721
722    }

755    /**
756     * Remove the given listener from the provided service handler's list of listeners.
757     *
758     * @param listener the {@code ProvidedServiceListener} object to be removed
759     * @throws NullPointerException   if {@code listener} is {@code null}
760     * @throws NoSuchElementException if {@code listener} wasn't present the in provided service handler's list of listeners
761     */
762    public void removeListener(ProvidedServiceListener listener) {
763        if (listener == null) {
764            throw new NullPointerException("null listener");
765        }
766        synchronized (m_listeners) {
767            // We definitely cannot rely on listener's equals method...
768            // ...so we need to manually search for the listener, using ==.
769            int i = -1;
770            for (int j = m_listeners.size() - 1; j >= 0; j--) {
771                if (m_listeners.get(j) == listener) {
772                    // Found!
773                    i = j;
774                    break;
775                }
776            }
777            if (i != -1) {
778                m_listeners.remove(i);
779            } else {
780                throw new NoSuchElementException("no such listener");
781            }
782        }
783    }
784
785    /**
786     * Notify all listeners that a change has occurred in this provided service.
787     *
788     * @param direction the "direction" of the change (+1:registration, 0:update, -1:unregistration)
789     */
790    private void notifyListeners(int direction) {
791        // Get a snapshot of the listeners
792        List<ProvidedServiceListener> tmp;
793        synchronized (m_listeners) {
794            tmp = new ArrayList<ProvidedServiceListener>(m_listeners);
795        }
796        // Do notify, outside the m_listeners lock
797        for (ProvidedServiceListener l : tmp) {
798            try {
799                if (direction > 0) {
800                    l.serviceRegistered(m_handler.getInstanceManager(), this);
801                } else if (direction < 0) {
802                    l.serviceUnregistered(m_handler.getInstanceManager(), this);
803                } else {
804                    l.serviceModified(m_handler.getInstanceManager(), this);
805                }
806            } catch (Throwable e) {
807                // Put a warning on the logger, and continue
808                m_handler.warn(
809                        String.format(
810                                "[%s] A ProvidedServiceListener has failed: %s",
811                                m_handler.getInstanceManager().getInstanceName(),
812                                e.getMessage())
813                        , e);
814            }
815        }
816    }
817
818    /**
819     * Removes all the listeners from this provided service before it gets disposed.
820     */
821    public void cleanup() {
822        synchronized (m_listeners) {
823            m_listeners.clear();
824        }
825    }

997    /**
998     * Service object creation policy creating a service object per asking iPOJO component
999     * instance. This creation policy follows the iPOJO Service Factory interaction pattern
1000     * and does no support 'direct' invocation.
1001     */
1002    private class PerInstanceStrategy extends CreationStrategy implements IPOJOServiceFactory, InvocationHandler {
1003        /**
1004         * Map [ComponentInstance->ServiceObject] storing created service objects.
1005         */
1006        private Map/*<ComponentInstance, ServiceObject>*/ m_instances = new HashMap();
1007
1008        /**
1009         * A method is invoked on the proxy object.
1010         * If the method is the {@link IPOJOServiceFactory#getService(ComponentInstance)}
1011         * method, this method creates a service object if not already created for the asking
1012         * component instance.
1013         * If the method is {@link IPOJOServiceFactory#ungetService(ComponentInstance, Object)}
1014         * the service object is unget (i.e. removed from the map and deleted).
1015         * In all other cases, a {@link UnsupportedOperationException} is thrown as this policy
1016         * requires to use  the {@link IPOJOServiceFactory} interaction pattern.
1017         *
1018         * @param arg0 the proxy object
1019         * @param arg1 the called method
1020         * @param arg2 the arguments
1021         * @return the service object attached to the asking instance for 'get',
1022         * <code>null</code> for 'unget',
1023         * a {@link UnsupportedOperationException} for all other methods.
1024         * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
1025         */
1026        public Object invoke(Object arg0, Method arg1, Object[] arg2) {
1027            if (isGetServiceMethod(arg1)) {
1028                return getService((ComponentInstance) arg2[0]);
1029            }
1030
1031            if (isUngetServiceMethod(arg1)) {
1032                ungetService((ComponentInstance) arg2[0], arg2[1]);
1033                return null;
1034            }
1035
1036            // Regular methods from java.lang.Object : equals and hashCode
1037            if (arg1.getName().equals("equals") && arg2 != null && arg2.length == 1) {
1038                return this.equals(arg2[0]);
1039            }
1040
1041            if (arg1.getName().equals("hashCode")) {
1042                return this.hashCode();
1043            }
1044
1045            throw new UnsupportedOperationException("This service requires an advanced creation policy. "
1046                    + "Before calling the service, call the getService(ComponentInstance) method to get "
1047                    + "the service object. - Method called: " + arg1.getName());
1048        }
1049
1050        /**
1051         * A service object is required.
1052         * This policy returns a service object per asking instance.
1053         *
1054         * @param instance the instance requiring the service object
1055         * @return the service object for this instance
1056         * @see org.apache.felix.ipojo.IPOJOServiceFactory#getService(org.apache.felix.ipojo.ComponentInstance)
1057         */
1058        public Object getService(ComponentInstance instance) {
1059            Object obj = m_instances.get(instance);
1060            if (obj == null) {
1061                obj = m_handler.getInstanceManager().createPojoObject();
1062                m_instances.put(instance, obj);
1063            }
1064            return obj;
1065        }
1066
1067        /**
1068         * A service object is unget.
1069         * The service object is removed from the map and deleted.
1070         *
1071         * @param instance  the instance releasing the service
1072         * @param svcObject the service object
1073         * @see org.apache.felix.ipojo.IPOJOServiceFactory#ungetService(org.apache.felix.ipojo.ComponentInstance, java.lang.Object)
1074         */
1075        public void ungetService(ComponentInstance instance, Object svcObject) {
1076            Object pojo = m_instances.remove(instance);
1077            m_handler.getInstanceManager().deletePojoObject(pojo);
1078        }
1079
1080        /**
1081         * The service is going to be registered.
1082         *
1083         * @param instance   the instance manager
1084         * @param interfaces the published interfaces
1085         * @param props      the properties
1086         * @see org.apache.felix.ipojo.handlers.providedservice.CreationStrategy#onPublication(InstanceManager, java.lang.String[], java.util.Properties)
1087         */
1088        public void onPublication(InstanceManager instance, String[] interfaces,
1089                                  Properties props) {
1090        }
1091
1092        /**
1093         * The service is going to be unregistered.
1094         * The instance map is cleared. Created object are disposed.
1095         *
1096         * @see org.apache.felix.ipojo.handlers.providedservice.CreationStrategy#onUnpublication()
1097         */
1098        public void onUnpublication() {
1099            Collection col = m_instances.values();
1100            Iterator it = col.iterator();
1101            while (it.hasNext()) {
1102                m_handler.getInstanceManager().deletePojoObject(it.next());
1103            }
1104            m_instances.clear();
1105        }
1106
1107        /**
1108         * OSGi Service Factory getService method.
1109         *
1110         * @param arg0 the asking bundle
1111         * @param arg1 the service registration
1112         * @return a proxy implementing the {@link IPOJOServiceFactory}
1113         * @see org.osgi.framework.ServiceFactory#getService(org.osgi.framework.Bundle, org.osgi.framework.ServiceRegistration)
1114         */
1115        public Object getService(Bundle arg0, ServiceRegistration arg1) {
1116            Object proxy = Proxy.newProxyInstance(getInstanceManager().getClazz().getClassLoader(),
1117                    getSpecificationsWithIPOJOServiceFactory(m_serviceSpecifications, m_handler.getInstanceManager().getContext()), this);
1118            return proxy;
1119        }
1120
1121        /**
1122         * OSGi Service factory unget method.
1123         * Does nothing.
1124         *
1125         * @param arg0 the asking bundle
1126         * @param arg1 the service registration
1127         * @param arg2 the service object created for this bundle.
1128         * @see org.osgi.framework.ServiceFactory#ungetService(org.osgi.framework.Bundle, org.osgi.framework.ServiceRegistration, java.lang.Object)
1129         */
1130        public void ungetService(Bundle arg0, ServiceRegistration arg1,
1131                                 Object arg2) {
1132        }
1133
1134        /**
1135         * Utility method returning the class array of provided service
1136         * specification and the {@link IPOJOServiceFactory} interface.
1137         *
1138         * @param specs the published service interface
1139         * @param bc    the bundle context, used to load classes
1140         * @return the class array containing provided service specification and
1141         * the {@link IPOJOServiceFactory} class.
1142         */
1143        private Class[] getSpecificationsWithIPOJOServiceFactory(String[] specs, BundleContext bc) {
1144            Class[] classes = new Class[specs.length + 1];
1145            int i = 0;
1146            for (i = 0; i < specs.length; i++) {
1147                try {
1148                    classes[i] = bc.getBundle().loadClass(specs[i]);
1149                } catch (ClassNotFoundException e) {
1150                    // Should not happen.
1151                }
1152            }
1153            classes[i] = IPOJOServiceFactory.class;
1154            return classes;
1155        }
