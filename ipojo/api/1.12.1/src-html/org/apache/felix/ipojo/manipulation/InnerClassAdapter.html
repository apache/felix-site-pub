022import org.objectweb.asm.*;
023import org.objectweb.asm.commons.GeneratorAdapter;
024import org.objectweb.asm.tree.LocalVariableNode;

026import java.util.List;
027import java.util.Map;

030/**
031 * Adapts a inner class in order to allow accessing outer class fields.
032 * A manipulated inner class has access to the managed field of the outer class.
033 *
034 * Only non-static inner classes are manipulated, others are not.
035 *
036 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
037 */
038public class InnerClassAdapter extends ClassVisitor implements Opcodes {
039
040    /**
041     * The manipulator having manipulated the outer class.
042     * We add method descriptions to this manipulator.
043     */
044    private final Manipulator m_manipulator;
045
046    /**
047     * The name of the inner class. This name is only define in the outer class.
048     */
049    private final String m_name;
050
051    /**
052     * The ismple name of the class.
053     */
054    private final String m_simpleName;
055
056    /**
057     * Implementation class name.
058     */
059    private String m_outer;
060    /**
061     * List of fields of the implementation class.
062     */
063    private Set<String> m_fields;
064
065    /**
066     * Creates the inner class adapter.
067     *
068     * @param name      the inner class name (internal name)
069     * @param visitor       parent class visitor
070     * @param outerClassName outer class (implementation class)
071     * @param manipulator the manipulator having manipulated the outer class.
072     */
073    public InnerClassAdapter(String name, ClassVisitor visitor, String outerClassName,
074                             Manipulator manipulator) {
075        super(Opcodes.ASM5, visitor);
076        m_name = name;
077        m_simpleName = m_name.substring(m_name.indexOf("$") + 1);
078        m_outer = outerClassName;
079        m_manipulator = manipulator;
080        m_fields = manipulator.getFields().keySet();
081    }
082
083    /**
084     * Visits a method.
085     * This methods create a code visitor manipulating outer class field accesses.
086     *
087     * @param access     method visibility
088     * @param name       method name
089     * @param desc       method descriptor
090     * @param signature  method signature
091     * @param exceptions list of exceptions thrown by the method
092     * @return a code adapter manipulating field accesses
093     * @see org.objectweb.asm.ClassVisitor#visitMethod(int, java.lang.String, java.lang.String, java.lang.String,
094     * java.lang.String[])
095     */
096    public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
097        // Do nothing on static methods, should not happen in non-static inner classes.
098        if ((access & ACC_STATIC) == ACC_STATIC) {
099            return super.visitMethod(access, name, desc, signature, exceptions);
100        }
101
102        // Do nothing on native methods
103        if ((access & ACC_NATIVE) == ACC_NATIVE) {
104            return super.visitMethod(access, name, desc, signature, exceptions);
105        }
106
107
108        // Do not re-manipulate.
109        if (! m_manipulator.isAlreadyManipulated()) {
110
111            if (name.equals("<init>")) {
112                // We change the field access from the constructor, but we don't generate the wrapper.
113                MethodVisitor mv = super.visitMethod(access, name, desc, signature, exceptions);
114                return new MethodCodeAdapter(mv, m_outer, access, name, desc, m_fields);
115            }
116
117            // For all non constructor methods
118
119            MethodDescriptor md = getMethodDescriptor(name, desc);
120            if (md == null) {
121                generateMethodWrapper(access, name, desc, signature, exceptions, null, null,
122                        null);
123            } else {
124                generateMethodWrapper(access, name, desc, signature, exceptions,
125                        md.getArgumentLocalVariables(),
126                        md.getAnnotations(), md.getParameterAnnotations());
127            }
128
129            // The new name is the method name prefixed by the PREFIX.
130            MethodVisitor mv = super.visitMethod(ACC_PRIVATE, ClassManipulator.PREFIX + name, desc, signature,
131                    exceptions);
132            return new MethodCodeAdapter(mv, m_outer, ACC_PRIVATE,  ClassManipulator.PREFIX + name, desc, m_fields);
133        } else {
134            return super.visitMethod(access, name, desc, signature, exceptions);
135        }
136    }
137
138    private String getMethodFlagName(String name, String desc) {
139        return ClassManipulator.METHOD_FLAG_PREFIX + getMethodId(name, desc);
140    }
141
142    private String getMethodId(String name, String desc) {
143        StringBuilder id = new StringBuilder(m_simpleName);
144        id.append("___"); // Separator
145        id.append(name);
146
147        Type[] args = Type.getArgumentTypes(desc);
148        for (Type type : args) {
149            String arg = type.getClassName();
150            if (arg.endsWith("[]")) {
151                // We have to replace all []
152                String acc = "";
153                while (arg.endsWith("[]")) {
154                    arg = arg.substring(0, arg.length() - 2);
155                    acc += "__";
156                }
157                id.append("$").append(arg.replace('.', '_')).append(acc);
158            } else {
159                id.append("$").append(arg.replace('.', '_'));
160            }
161        }
162        return id.toString();
163    }

165    /**
166     * Generate the method header of a POJO method.
167     * This method header encapsulate the POJO method call to
168     * signal entry exit and error to the container.
169     *
170     * The instance manager and flag are accessed using method calls.
171     * @param access : access flag.
172     * @param name : method name.
173     * @param desc : method descriptor.
174     * @param signature : method signature.
175     * @param exceptions : declared exceptions.
176     * @param localVariables : the local variable nodes.
177     * @param annotations : the annotations to move to this method.
178     * @param paramAnnotations : the parameter annotations to move to this method.
179     */
180    private void generateMethodWrapper(int access, String name, String desc, String signature, String[] exceptions,
181                                       List<LocalVariableNode> localVariables, List<ClassChecker.AnnotationDescriptor> annotations,
182                                       Map<Integer, List<ClassChecker.AnnotationDescriptor>> paramAnnotations) {
183        GeneratorAdapter mv = new GeneratorAdapter(cv.visitMethod(access, name, desc, signature, exceptions), access, name, desc);
184
185        // If we have variables, we wraps the code within labels. The `lifetime` of the variables are bound to those
186        // two variables.
187        boolean hasArgumentLabels = localVariables != null && !localVariables.isEmpty();
188        Label start = null;
189        if (hasArgumentLabels) {
190            start = new Label();
191            mv.visitLabel(start);
192        }
193
194        mv.visitCode();
195
196        Type returnType = Type.getReturnType(desc);
197
198        // Compute result and exception stack location
199        int result = -1;
200        int exception;
201
202        //int arguments = mv.newLocal(Type.getType((new Object[0]).getClass()));
203
204        if (returnType.getSort() != Type.VOID) {
205            // The method returns something
206            result = mv.newLocal(returnType);
207            exception = mv.newLocal(Type.getType(Throwable.class));
208        } else {
209            exception = mv.newLocal(Type.getType(Throwable.class));
210        }
211
212        Label l0 = new Label();
213        Label l1 = new Label();
214        Label l2 = new Label();
215
216        mv.visitTryCatchBlock(l0, l1, l2, "java/lang/Throwable");
217
218        // Access the flag from the outer class
219        mv.visitVarInsn(ALOAD, 0);
220        mv.visitFieldInsn(GETFIELD, m_name, "this$0", "L" + m_outer + ";");
221        mv.visitFieldInsn(GETFIELD, m_outer, getMethodFlagName(name, desc), "Z");
222        mv.visitJumpInsn(IFNE, l0);
223
224        mv.visitVarInsn(ALOAD, 0);
225        mv.loadArgs();
226        mv.visitMethodInsn(INVOKESPECIAL, m_name, ClassManipulator.PREFIX + name, desc, false);
227        mv.visitInsn(returnType.getOpcode(IRETURN));
228
229        // end of the non intercepted method invocation.
230
231        mv.visitLabel(l0);
232
233        mv.visitVarInsn(ALOAD, 0);
234        mv.visitFieldInsn(GETFIELD, m_name, "this$0", "L" + m_outer + ";");
235        mv.visitFieldInsn(GETFIELD, m_outer, ClassManipulator.IM_FIELD, "Lorg/apache/felix/ipojo/InstanceManager;");
236        mv.visitVarInsn(ALOAD, 0);
237        mv.visitLdcInsn(getMethodId(name, desc));
238        mv.loadArgArray();
239        mv.visitMethodInsn(INVOKEVIRTUAL, "org/apache/felix/ipojo/InstanceManager", ClassManipulator.ENTRY,
240                "(Ljava/lang/Object;Ljava/lang/String;[Ljava/lang/Object;)V", false);
241
242        mv.visitVarInsn(ALOAD, 0);
243
244        // Do not allow argument modification : just reload arguments.
245        mv.loadArgs();
246        mv.visitMethodInsn(INVOKESPECIAL, m_name, ClassManipulator.PREFIX + name, desc, false);
247
248        if (returnType.getSort() != Type.VOID) {
249            mv.visitVarInsn(returnType.getOpcode(ISTORE), result);
250        }
251
252        mv.visitVarInsn(ALOAD, 0);
253        mv.visitFieldInsn(GETFIELD, m_name, "this$0", "L" + m_outer + ";");
254        mv.visitFieldInsn(GETFIELD, m_outer, ClassManipulator.IM_FIELD, "Lorg/apache/felix/ipojo/InstanceManager;");
255        mv.visitVarInsn(ALOAD, 0);
256        mv.visitLdcInsn(getMethodId(name, desc));
257        if (returnType.getSort() != Type.VOID) {
258            mv.visitVarInsn(returnType.getOpcode(ILOAD), result);
259            mv.box(returnType);
260        } else {
261            mv.visitInsn(ACONST_NULL);
262        }
263        mv.visitMethodInsn(INVOKEVIRTUAL, "org/apache/felix/ipojo/InstanceManager",
264                ClassManipulator.EXIT, "(Ljava/lang/Object;Ljava/lang/String;Ljava/lang/Object;)V", false);
265
266        mv.visitLabel(l1);
267        Label l7 = new Label();
268        mv.visitJumpInsn(GOTO, l7);
269        mv.visitLabel(l2);
270
271        mv.visitVarInsn(ASTORE, exception);
272        mv.visitVarInsn(ALOAD, 0);
273        mv.visitFieldInsn(GETFIELD, m_name, "this$0", "L" + m_outer + ";");
274        mv.visitFieldInsn(GETFIELD, m_outer, ClassManipulator.IM_FIELD, "Lorg/apache/felix/ipojo/InstanceManager;");
275        mv.visitVarInsn(ALOAD, 0);
276        mv.visitLdcInsn(getMethodId(name, desc));
277        mv.visitVarInsn(ALOAD, exception);
278        mv.visitMethodInsn(INVOKEVIRTUAL, "org/apache/felix/ipojo/InstanceManager", ClassManipulator.ERROR,
279                "(Ljava/lang/Object;Ljava/lang/String;Ljava/lang/Throwable;)V", false);
280        mv.visitVarInsn(ALOAD, exception);
281        mv.visitInsn(ATHROW);
282
283        mv.visitLabel(l7);
284        if (returnType.getSort() != Type.VOID) {
285            mv.visitVarInsn(returnType.getOpcode(ILOAD), result);
286        }
287        mv.visitInsn(returnType.getOpcode(IRETURN));
288
289        // If we had arguments, we mark the end of the lifetime.
290        Label end = null;
291        if (hasArgumentLabels) {
292            end = new Label();
293            mv.visitLabel(end);
294        }
295
296        // Move annotations
297        if (annotations != null) {
298            for (ClassChecker.AnnotationDescriptor ad : annotations) {
299                ad.visitAnnotation(mv);
300            }
301        }
302
303        // Move parameter annotations
304        if (paramAnnotations != null  && ! paramAnnotations.isEmpty()) {
305            for (Integer id : paramAnnotations.keySet()) {
306                List<ClassChecker.AnnotationDescriptor> ads = paramAnnotations.get(id);
307                for (ClassChecker.AnnotationDescriptor ad : ads) {
308                    ad.visitParameterAnnotation(id, mv);
309                }
310            }
311        }
312
313        // Write the arguments name.
314        if (hasArgumentLabels) {
315            for (LocalVariableNode var : localVariables) {
316                mv.visitLocalVariable(var.name, var.desc, var.signature, start, end, var.index);
317            }
318        }
319
320        mv.visitMaxs(0, 0);
321        mv.visitEnd();
322    }
323
324    /**
325     * Gets the method descriptor for the specified name and descriptor.
326     * The method descriptor is looked inside the
327     * {@link ClassManipulator#m_visitedMethods}
328     * @param name the name of the method
329     * @param desc the descriptor of the method
330     * @return the method descriptor or <code>null</code> if not found.
331     */
332    private MethodDescriptor getMethodDescriptor(String name, String desc) {
333        for (MethodDescriptor md : m_manipulator.getMethodsFromInnerClass(m_name)) {
334            if (md.getName().equals(name) && md.getDescriptor().equals(desc)) {
335                return md;
336            }
337        }
338        return null;
339    }
