019package org.apache.felix.ipojo.xml.parser;
020
021import java.io.IOException;
022import java.net.URL;
023
024import org.xml.sax.EntityResolver;
025import org.xml.sax.InputSource;
026import org.xml.sax.SAXException;
027
028
029/**
030 * Entity Resolver loading embedded XML Schemas.
031 * This resolver avoid using a network connection to get schemas as they
032 * are loaded from the manipulator jar file.
033 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
034 */
035public class SchemaResolver implements EntityResolver {
036    
037    /**
038     * Directory where embedded schemas are copied.
039     */
040    public static final String XSD_PATH = "xsd";
041
042    /**
043     * Resolves systemIds to use embedded schemas. The schemas are loaded from
044     * the {@link SchemaResolver#XSD_PATH} directory with the current classloader.
045     * @param publicId the publicId
046     * @param systemId the systemId (Schema URL)
047     * @return the InputSource to load the schemas or <code>null</code> if the schema
048     * cannot be loaded (not embedded)
049     * @throws SAXException cannot happen 
050     * @throws IOException when the embedded resource cannot be read correctly
051     * @see org.xml.sax.EntityResolver#resolveEntity(java.lang.String, java.lang.String)
052     */
053    public InputSource resolveEntity(String publicId, String systemId)
054        throws SAXException, IOException {
055        
056        URL url = getURL(systemId);
057        if (url == null) {
058            // Cannot found the resource
059            return null;
060        } else {
061            return new InputSource(url.openStream());
062        }
063        
064    }
065    
066    /**
067     * Computes the local URL of the given system Id.
068     * This URL is computed by trying to load the resource from
069     * the current classloader. First, the last fragment (file name) of the system id
070     * url is extracted and the file is loaded from the {@link SchemaResolver#XSD_PATH}
071     * directory ('xsd/extracted') 
072     * @param id the systemId to load
073     * @return the URL to the resources or <code>null</code> if the resource cannot be found.
074     */
075    private URL getURL(String id) {
076        int index = id.lastIndexOf('/');
077        String fragment = id.substring(index);
078        return this.getClass().getClassLoader().getResource(XSD_PATH + fragment);
079    }
080
081}
