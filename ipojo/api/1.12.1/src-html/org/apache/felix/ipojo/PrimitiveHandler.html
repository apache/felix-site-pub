019package org.apache.felix.ipojo;
020
021import java.lang.reflect.Member;
022import java.lang.reflect.Method;
023
024import org.apache.felix.ipojo.parser.PojoMetadata;
025import org.apache.felix.ipojo.util.Logger;
026
027
028/**
029* This class defines common mechanisms of primitive handlers.
030* Primitive handlers are handler composing the container of primitive
031* component instances (declared by the 'component' element inside the
032* iPOJO descriptor).
033* Note that this class also defines default method implementation.
034* Classes overriding this class can change the behavior of those methods.
035* @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
036*/
037public abstract class PrimitiveHandler extends Handler implements FieldInterceptor, MethodInterceptor,
038    ConstructorInjector {
039
040    /**
041     * The "Primitive" Handler type (value).
042     */
043    public static final String HANDLER_TYPE = "primitive";
044
045    /**
046     * The reference on the instance manager.
047     */
048    private InstanceManager m_manager;
049
050
051    /**
052     * The factory of the instance manager.
053     */
054    private ComponentFactory m_factory;
055
056    /**
057     * Instance Logger used by the handler.
058     */
059    private Logger m_instanceLogger;
060
061    /**
062     * Attaches the current handler to the given instance.
063     * This method must be called only once, and should not be overridden.
064     * @param manager the instance on which the current handler will be attached.
065     * @see org.apache.felix.ipojo.Handler#attach(org.apache.felix.ipojo.ComponentInstance)
066     */
067    protected void attach(ComponentInstance manager) {
068        m_manager = (InstanceManager) manager;
069        m_instanceLogger = m_manager.getLogger();
070    }
071
072    /**
073     * Sets the factory of the managed instance.
074     * @param factory the factory
075     * @see org.apache.felix.ipojo.Handler#setFactory(org.apache.felix.ipojo.Factory)
076     */
077    public final void setFactory(Factory factory) {
078        m_factory = (ComponentFactory) factory;
079    }
080
081    /**
082     * Gets the logger of the managed instance.
083     * IF no instance attached yet, use the factory logger.
084     * @return the logger to use to log messages.
085     * @see org.apache.felix.ipojo.Handler#getLogger()
086     */
087    public Logger getLogger() {
088        if (m_instanceLogger == null) {
089            return m_factory.getLogger();
090        }
091        return m_instanceLogger;
092    }
093
094    /**
095     * Gets the instance manager managing the instance.
096     * @return the instance manager
097     */
098    public InstanceManager getInstanceManager() {
099        return m_manager;
100    }
101
102    /**
103     * Gets the factory which creates the managed instance.
104     * @return the factory which creates the managed instance.
105     */
106    public ComponentFactory getFactory() {
107        return m_factory;
108    }
109
110    /**
111     * Gets the PojoMetadata of the content of the managed
112     * instance. This method allows getting manipulation
113     * metadata.
114     * @return the information about the content of the
115     * managed instance.
116     */
117    public PojoMetadata getPojoMetadata() {
118        return m_factory.getPojoMetadata();
119    }
120
121    /**
122     * Gets a plugged handler of the same container.
123     * This method must be called only in the start method (or after).
124     * In the configure method, this method can be inconsistent
125     * as all handlers are not initialized.
126     * @param name the name of the handler to find (class name or qualified
127     * handler name (<code>ns:name</code>)).
128     * @return the handler object or <code>null</code> if the handler is not found.
129     */
130    public final Handler getHandler(String name) {
131        return m_manager.getHandler(name);
132    }
133
134    /**
135     * Callback method called when a managed field
136     * receives a new value. The given pojo can be
137     * null if the new value is set by another handler.
138     * This default implementation does nothing.
139     * @param pojo the pojo object setting the value
140     * @param fieldName the field name
141     * @param value the value received by the field
142     * @see FieldInterceptor#onSet(Object, String, Object)
143     */
144    public void onSet(Object pojo, String fieldName, Object value) {
145        // Nothing to do in the default implementation
146    }
147
148    /**
149     * Callback method called when a managed field
150     * asks for a value.
151     * The default implementation returned the previously
152     * injected value.
153     * @param pojo the pojo object requiring the value
154     * @param fieldName the field name
155     * @param value the value passed to the field (by the previous call)
156     * @return the value to inject, of the previous value if the handler
157     * don't want to inject a value.
158     * @see FieldInterceptor#onGet(Object, String, Object)
159     */
160    public Object onGet(Object pojo, String fieldName, Object value) {
161        return value;
162    }
163
164    /**
165     * Gets the object to inject as a constructor parameter
166     * @param index the index of the parameter
167     * @return the object to inject, or <code>null</code> if no
168     * objects are injected. This implementation returns <code>null</code>
169     * @see org.apache.felix.ipojo.ConstructorInjector#getConstructorParameter(int)
170     */
171    public Object getConstructorParameter(int index) {
172        return null;
173    }
174
175    /**
176     * Gets the type of the object to inject in the constructor parameter.
177     * This is the type looked into the Pojo class, so it must match.
178     * Returning <code>null</code> will try to get the class from the
179     * injected object, however this can be wrong (implementation instead of interface,
180     * boxed objects...) and error-prone.
181     * @param index the parameter index
182     * @return the Class object (must fit for primitive type), this implementation
183     * just returns <code>null</code>
184     * @see org.apache.felix.ipojo.ConstructorInjector#getConstructorParameterType(int)
185     */
186    public Class getConstructorParameterType(int index) {
187        return null;
188    }
189
190    /**
191     * Callback method called when a method will be invoked.
192     * This default implementation does nothing.
193     * @param pojo the pojo on which the method is called.
194     * @param method the method invoked.
195     * @param args the arguments array.
196     * @see MethodInterceptor#onEntry(Object, java.lang.reflect.Member, Object[])
197     */
198    public void onEntry(Object pojo, Member method, Object[] args) {
199        // Nothing to do in the default implementation
200    }
201
202    /**
203     * Callback method called when a method ends.
204     * This method is called when a thread exits a method (before a return or a throw).
205     * If the given returned object is <code>null</code>, either the method is
206     * <code>void</code>, or it returns <code>null</code>.
207     * You must not modified the returned object.
208     * The default implementation does nothing.
209     * @param pojo the pojo on which the method exits.
210     * @param method the exiting method.
211     * @param returnedObj the returned object (boxed for primitive type)
212     * @see MethodInterceptor#onExit(Object, java.lang.reflect.Member, Object)
213     */
214    public void onExit(Object pojo, Member method, Object returnedObj) {
215        // Nothing to do in the default implementation
216    }
217
218    /**
219     * Callback method called when an error occurs.
220     * This method is called when the execution throws an exception
221     * in the given method.
222     * The default implementation does nothing.
223     * @param pojo the pojo on which the method was accessed.
224     * @param method the invoked method.
225     * @param throwable the thrown exception
226     * @see org.apache.felix.ipojo.MethodInterceptor#onError(Object, java.lang.reflect.Member, Throwable)
227     */
228    public void onError(Object pojo, Member method, Throwable throwable) {
229        // Nothing to do in the default implementation
230    }
231
232    /**
233     * Callback method called when the execution of a method will terminate :
234     * just before to throw an exception or before to return. This callback is called after
235     * {@link MethodInterceptor#onExit(Object, java.lang.reflect.Member, Object)} and
236     * {@link MethodInterceptor#onError(Object, java.lang.reflect.Member, Throwable)}
237     * This default implementation does nothing.
238     * @param pojo the pojo on which the method was accessed.
239     * @param method the invoked method.
240     */
241    public void onFinally(Object pojo, Member method) {
242        // Nothing to do in the default implementation
243    }
244
245    /**
246     * Callback method called when an instance of the component is created, but
247     * before someone can use it.
248     * The default implementation does nothing.
249     * @param instance the created instance
250     */
251    public void onCreation(Object instance) {
252        // Nothing to do in the default implementation
253    }
