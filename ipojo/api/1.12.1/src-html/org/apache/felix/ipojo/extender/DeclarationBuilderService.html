020package org.apache.felix.ipojo.extender;
021
022import org.apache.felix.ipojo.extender.builder.FactoryBuilder;
023import org.apache.felix.ipojo.metadata.Element;
024
025/**
026 * This service provides a way for users to manage declarations through code.
027 *
028 * Notice that produced declarations are immutable once build. But it is possible to re-use a
029 * builder to share some configuration through instances.
030 *
031 * <pre>
032 *     // Obtain the service through the service registry
033 *     DeclarationBuilderService service = ...
034 *
035 *     // Get a fresh instance builder
036 *     DeclarationBuilder builder = service.newInstance("the.full.name.of.the.component.to.instantiate");
037 *
038 *     DeclarationHandle handle = builder.name("a-unique-name") // Make sure name is unique for the expected type
039 *                                       .configure()
040 *                                           .property("a-property", "a-value")
041 *                                           .property("another-property", "another-value")
042 *                                           .build();
043 *
044 *     // Push the InstanceDeclaration service in the registry
045 *     handle.publish();
046 * </pre>
047 *
048 * Except the {@link InstanceBuilder#build()} call, all methods are optional:
049 * <ul>
050 *     <li>{@link InstanceBuilder#name(String)}: if no name is provided,
051 *     a default one will be generated by iPOJO.</li>
052 *     <li>{@link InstanceBuilder#version(String)}: if no version is provided,
053 *     the first un-versioned type will be used.</li>
054 *     <li>{@link InstanceBuilder#configure()}: if no configuration is required, can be omitted.</li>
055 *     <li>{@link InstanceBuilder#type(String)}: can be used to change the <bold>required</bold> component type.</li>
056 *     <li>{@link InstanceBuilder#context(org.osgi.framework.BundleContext)}: by default, the bundle context
057 *     used to register the created {@link org.apache.felix.ipojo.extender.Declaration} is the one of the
058 *     caller. It can be changed though this method.</li>
059 * </ul>
060 *
061 * Once an instance handle has been created, its configuration (name, type, version and properties) is immutable. It can
062 * only be {@linkplain DeclarationHandle#publish() published} (so that the framework will try to instantiate the
063 * instance) or {@linkplain DeclarationHandle#retract() retracted} (framework will remove the instance).
064 *
065 * Notice that all created instances will appear as "coming from" the bundle that requires the
066 * {@link DeclarationBuilderService} service. Just like having a {@literal metadata.xml} file in your
067 * bundle that declares your instances. It is possible to override this default behavior using
068 * {@link InstanceBuilder#context(org.osgi.framework.BundleContext)}.
069 *
070 * @see org.apache.felix.ipojo.extender.InstanceDeclaration
071 * @see org.apache.felix.ipojo.extender.TypeDeclaration
072 * @see org.apache.felix.ipojo.extender.ExtensionDeclaration
073 * @see org.apache.felix.ipojo.extender.InstanceBuilder
074 * @see org.apache.felix.ipojo.extender.ConfigurationBuilder
075 * @see org.apache.felix.ipojo.extender.DeclarationHandle
076 *
077 * @since 1.11.2
078 */
079public interface DeclarationBuilderService {
080
081    /**
082     * Declares a new anonymous instance of a given type.
083     * Invoking this method is equivalent to invoking <code>{@linkplain #newInstance(String, String) newInstance(type, null)}</code>.
084     * @param type name of the component to be instantiated (cannot be null).
085     * @return a handle usable to publish / retract the declaration.
086     * @see org.apache.felix.ipojo.extender.InstanceDeclaration
087     */
088    InstanceBuilder newInstance(String type);
089
090    /**
091     * Declares a new instance of a given type.
092     * Invoking this method is equivalent to invoking <code>{@linkplain #newInstance(String, String, String) newInstance(type, name, null)}</code>.
093     * @param type name of the component to be instantiated.
094     * @param name name of the new instance (can be null)
095     * @return a handle usable to publish / retract the declaration.
096     * @see org.apache.felix.ipojo.extender.InstanceDeclaration
097     */
098    InstanceBuilder newInstance(String type, String name);
099
100    /**
101     * Declares a new instance of a given type.
102     * @param type name of the component to be instantiated.
103     * @param name name of the new instance (can be null)
104     * @param version version of the expected type (can be null)
105     * @return a handle usable to publish / retract the declaration.
106     * @see org.apache.felix.ipojo.extender.InstanceDeclaration
107     */
108    InstanceBuilder newInstance(String type, String name, String version);
109
110    /**
111     * Declares a new extension (supports new types like {@literal component}, {@literal composite}, {@literal handler}).
112     * @param name name of the type to support (no namespace to be provided)
113     * @param builder associated factory builder
114     * @return a handle usable to publish / retract the declaration.
115     * @see org.apache.felix.ipojo.IPojoFactory
116     * @see org.apache.felix.ipojo.extender.ExtensionDeclaration
117     */
118    DeclarationHandle newExtension(String name, FactoryBuilder builder);
119
120    /**
121     * Declares a new type using the given element description.
122     * @param description description of the component type
123     * @return a handle usable to publish / retract the declaration.
124     * @see org.apache.felix.ipojo.extender.TypeDeclaration
125     */
126    DeclarationHandle newType(Element description);
127}
