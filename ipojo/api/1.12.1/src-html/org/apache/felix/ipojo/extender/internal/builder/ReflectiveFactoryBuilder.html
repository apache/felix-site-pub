020package org.apache.felix.ipojo.extender.internal.builder;
021
022import org.apache.felix.ipojo.IPojoFactory;
023import org.apache.felix.ipojo.extender.builder.FactoryBuilder;
024import org.apache.felix.ipojo.extender.builder.FactoryBuilderException;
025import org.apache.felix.ipojo.metadata.Element;
026import org.osgi.framework.BundleContext;
027
028import java.lang.reflect.Constructor;
029import java.lang.reflect.InvocationTargetException;
030
031/**
032 * An Factory builder using a reflective call to build the factory.
033 * <p/>
034 * This builder is compatible with the original iPOJO method consisting in calling a constructor receiving the bundle
035 * context and the metadata as parameters.
036 * <p/>
037 * This factory builder need the constructor to be called.
038 */
039public class ReflectiveFactoryBuilder implements FactoryBuilder {
040
041    /**
042     * The constructor to call.
043     */
044    private final Constructor<? extends IPojoFactory> m_constructor;
045
046    /**
047     * Creates the factory builder.
048     *
049     * @param constructor the constructor that will be called when a new component factory will be created.
050     */
051    public ReflectiveFactoryBuilder(Constructor<? extends IPojoFactory> constructor) {
052        m_constructor = constructor;
053    }
054
055    /**
056     * Calls the wrapped constructor to create an iPOJO factory.
057     *
058     * @param bundleContext the bundle context of the bundle declaring the component type
059     * @param metadata      the metadata of the component type (<code>component</code> element).
060     * @return the created iPOJO factory
061     * @throws FactoryBuilderException if the constructor cannot be called or throws an error.
062     */
063    public IPojoFactory build(BundleContext bundleContext, Element metadata) throws FactoryBuilderException {
064        try {
065            return m_constructor.newInstance(bundleContext, metadata);
066        } catch (InstantiationException e) {
067            throw new FactoryBuilderException("Cannot create instance of " + m_constructor.getDeclaringClass(), e);
068        } catch (IllegalAccessException e) {
069            throw new FactoryBuilderException(m_constructor.getDeclaringClass() + " constructor is not " +
070                    "accessible (not public)", e);
071        } catch (InvocationTargetException e) {
072            throw new FactoryBuilderException("Cannot create instance of " + m_constructor.getDeclaringClass(), e);
073        }
074    }
075}
