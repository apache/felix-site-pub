020package org.apache.felix.ipojo.manipulator.metadata.annotation.registry;
021
022import org.apache.felix.ipojo.manipulator.Reporter;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.ComponentWorkbench;
024import org.apache.felix.ipojo.manipulator.spi.BindingContext;
025import org.apache.felix.ipojo.manipulator.util.ChainedAnnotationVisitor;
026import org.objectweb.asm.AnnotationVisitor;
027import org.objectweb.asm.ClassVisitor;
028import org.objectweb.asm.FieldVisitor;
029import org.objectweb.asm.MethodVisitor;
030import org.objectweb.asm.Type;
031import org.objectweb.asm.tree.ClassNode;
032import org.objectweb.asm.tree.FieldNode;
033import org.objectweb.asm.tree.MethodNode;
034
035import java.lang.annotation.ElementType;
036import java.util.ArrayList;
037import java.util.List;
038
039/**
040 * A {@link Selection} is used to select a subset of all supported {@link AnnotationVisitor}.
041 * It's a query DSL.
042 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
043 */
044public class Selection {
045
046    private BindingRegistry registry;
047    private ComponentWorkbench workbench;
048    private Reporter reporter;
049    private FieldNode field;
050    private MethodNode method;
051    private ClassNode clazz;
052    private int index = BindingContext.NO_INDEX;
053    private String annotation;
054    private ElementType elementType = null;
055    private Object visitor;
056
057    public Selection(BindingRegistry registry, ComponentWorkbench workbench, Reporter reporter) {
058        this.registry = registry;
059        this.workbench = workbench;
060        this.reporter = reporter;
061    }
062
063    public Selection field(FieldVisitor visitor, FieldNode node) {
064        this.visitor = visitor;
065        this.field = node;
066        this.elementType = ElementType.FIELD;
067        return this;
068    }
069
070    public Selection method(MethodVisitor visitor, MethodNode node) {
071        this.visitor = visitor;
072        this.method = node;
073        this.elementType = ElementType.METHOD;
074        return this;
075    }
076
077    public Selection type(ClassVisitor visitor, ClassNode node) {
078        this.visitor = visitor;
079        this.clazz = node;
080        this.elementType = ElementType.TYPE;
081        return this;
082    }
083
084    public Selection parameter(MethodVisitor visitor, MethodNode node, int index) {
085        this.visitor = visitor;
086        this.index = index;
087        this.method = node;
088        this.elementType = ElementType.PARAMETER;
089        return this;
090    }
091
092    public Selection annotatedWith(String desc) {
093        this.annotation = desc;
094        return this;
095    }
096
097    public AnnotationVisitor get() {
098        List<AnnotationVisitor> visitors = list();
099
100        if (visitors.isEmpty()) {
101            return null;
102        }
103
104        if (visitors.size() == 1) {
105            return visitors.get(0);
106        }
107
108        ChainedAnnotationVisitor chained = new ChainedAnnotationVisitor();
109        chained.getVisitors().addAll(visitors);
110        return chained;
111    }
112
113    private List<AnnotationVisitor> list() {
114        BindingContext context;
115
116        if (elementType == ElementType.FIELD) {
117            context = new BindingContext(workbench, reporter, Type.getType(annotation), field,
118                    elementType, index, visitor);
119        } else if (elementType == ElementType.TYPE) {
120            context = new BindingContext(workbench, reporter, Type.getType(annotation), clazz,
121                    elementType, index, visitor);
122        } else {
123            // Parameter of method.
124            context = new BindingContext(workbench, reporter, Type.getType(annotation), method,
125                    elementType, index, visitor);
126        }
127
128
129        List<Binding> predicates = registry.getBindings(annotation);
130
131        List<AnnotationVisitor> visitors = new ArrayList<AnnotationVisitor>();
132        if (predicates != null && !predicates.isEmpty()) {
133            collectMatchingVisitors(predicates, context, visitors);
134        }
135        return visitors;
136    }
137
138
139    private void collectMatchingVisitors(List<Binding> bindings, BindingContext context, List<AnnotationVisitor> visitors) {
140        for (Binding binding : bindings) {
141            if (binding.getPredicate().matches(context)) {
142                AnnotationVisitor visitor = binding.getFactory().newAnnotationVisitor(context);
143                if (visitor != null) {
144                    visitors.add(visitor);
145                }
146            }
147        }
148    }
149
150}
