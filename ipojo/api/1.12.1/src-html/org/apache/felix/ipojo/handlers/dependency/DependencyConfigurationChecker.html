032/**
033 * Utility class checking the configuration of a dependency.
034 */
035public class DependencyConfigurationChecker {
036
037    public static void ensure(Dependency dependency, Element metadata, PojoMetadata manipulation) throws
038            ConfigurationException {
039        ensureThatAtLeastOneInjectionIsSpecified(dependency);
040        ensureThatTheFieldIsInComponentClass(dependency, manipulation);
041        ensureThatTheConstructorParameterIsCoherent(dependency, manipulation);
042        ensureThatCallbacksAreCoherent(dependency, manipulation);
043        deduceAggregationFromTheInjectionPoints(dependency, manipulation);
044        deduceTheServiceSpecification(dependency, manipulation);
045        checkTheServiceUnavailableAction(dependency, metadata);
046        checkTheConsistencyOfTheFromAttribute(dependency, metadata);
047        disableProxyForInconsistentTypes(dependency);
048    }
049
050    /**
051     * Disables the proxy settings for types that does not support it: Vector, Array, and non interface specification.
052     * If the dependency is disabled, check that we are no constructor injection.
053     * @param dependency the dependency
054     */
055    private static void disableProxyForInconsistentTypes(Dependency dependency) throws ConfigurationException {
056        if (! dependency.getSpecification().isInterface()
057                || dependency.isAggregate()  && dependency.getAggregateType() == AggregateDependencyInjectionType.ARRAY
058                || dependency.isAggregate()  && dependency.getAggregateType() == AggregateDependencyInjectionType.VECTOR) {
059            dependency.setProxy(false);
060            if (dependency.getConstructorParameterIndex() != -1) {
061                throw new ConfigurationException("The dependency " + DependencyHandler.getDependencyIdentifier
062                        (dependency) + " has an inconsistent configuration. - reason: the service specification " +
063                        "or container do not support proxy, which is required for constructor injection");
064            }
065            dependency.getHandler().info("Proxy disabled for " + DependencyHandler.getDependencyIdentifier
066                    (dependency) + " - the service specification or container do not support proxy");
067        }
068    }

070    /**
071     * Checks that the dependency callbacks are consistent:
072     * <ul>
073     *     <li>have a supported 'type'</li>
074     *     <li>have a supported signature</li>
075     * </ul>
076     * If the method is not in the component class, a message is logged, as this verification cannot be used.
077     * If the method is found in the manipulation metadata, the callback parameters are set.
078     * @param dependency the dependency
079     * @param manipulation the manipulation
080     * @throws ConfigurationException if the methods do not obey to the previously mentioned rules.
081     */
082    private static void ensureThatCallbacksAreCoherent(Dependency dependency, PojoMetadata manipulation) throws
083            ConfigurationException {
084        DependencyCallback[] callbacks = dependency.getCallbacks();
085        if (callbacks != null) {
086            for (DependencyCallback callback : callbacks) {
087                MethodMetadata metadata = manipulation.getMethod(callback.getMethodName());
088                if (metadata == null) {
089                    dependency.getHandler().debug("A dependency callback " + callback.getMethodName() + " of " +
090                            DependencyHandler.getDependencyIdentifier(dependency) + " does not " +
091                            "exist in the implementation class, will try the parent classes");
092                } else {
093                    String[] parameters = metadata.getMethodArguments();
094                   switch(parameters.length) {
095                       case 0 : // Just a notification method.
096                            callback.setArgument(parameters);
097                            break;
098                       case 1 :
099                           // Can be the service object, service reference or properties
100                           callback.setArgument(parameters);
101                           break;
102                       case 2 :
103                           // Constraints on the second argument, must be a service reference, a dictionary or a map
104                           if (!ServiceReference.class.getName().equals(parameters[1])
105                               && ! Dictionary.class.getName().equals(parameters[1])
106                               && ! Map.class.getName().equals(parameters[1])) {
107                                throw new ConfigurationException("The method " + callback.getMethodName() + " of " +
108                                        DependencyHandler.getDependencyIdentifier(dependency) + " is not a valid " +
109                                        "dependency callback - reason: the second argument (" + parameters[1] +
110                                        ")  must be a service reference, a dictionary or a map.");
111                                }
112                           callback.setArgument(parameters);
113                           break;
114                       default:
115                           // Invalid signature.
116                           throw new ConfigurationException("The method " + callback.getMethodName() + " of " +
117                                   DependencyHandler.getDependencyIdentifier(dependency) + " is not a valid " +
118                                   "dependency callback - reason: the signature is invalid");
119                    }
120                }
121            }
122        }
123    }

125    /**
126     * Checks whether the constructor parameter injection is suitable. this check verified that the constructor has
127     * enough parameter.
128     * @param dependency the dependency
129     * @param manipulation the manipulation metadata
130     * @throws ConfigurationException if the constructor is not suitable
131     */
132    private static void ensureThatTheConstructorParameterIsCoherent(Dependency dependency,
133                                                                    PojoMetadata manipulation) throws
134            ConfigurationException {
135        if (dependency.getConstructorParameterIndex() != -1) {
136            MethodMetadata[] constructors = manipulation.getConstructors();
137            if (constructors == null  || constructors.length == 0) {
138                throw new ConfigurationException("The constructor parameter attribute of " + DependencyHandler
139                        .getDependencyIdentifier(dependency) + " is inconsistent - reason: there is no constructor in" +
140                        " the component class (" + dependency.getHandler().getInstanceManager().getClassName() + ")");
141            }
142
143            //TODO Consider only the first constructor. This is a limitation we should think about,
144            // how to determine which constructor to use. Only one constructor should have annotations,
145            // it could be use as hint.
146            MethodMetadata constructor = constructors[0];
147            if (! (constructor.getMethodArguments().length > dependency.getConstructorParameterIndex())) {
148                throw new ConfigurationException("The constructor parameter attribute of " + DependencyHandler
149                        .getDependencyIdentifier(dependency) + " is inconsistent - reason: the constructor with the " +
150                        "signature " + Arrays.toString(constructor.getMethodArguments()) + " has not enough " +
151                        "parameters");
152            }
153
154        }
155    }

157    /**
158     * Checks that the field used to inject the dependency is in the component class. If the dependency has no field,
159     * this method does nothing.
160     * @param dependency the dependency
161     * @param manipulation the manipulation metadata
162     * @throws ConfigurationException if the field used to inject the given dependency is not in the component class.
163     */
164    private static void ensureThatTheFieldIsInComponentClass(Dependency dependency, PojoMetadata manipulation) throws ConfigurationException {
165        if (dependency.getField() != null) {
166            FieldMetadata field = manipulation.getField(dependency.getField());
167            if (field == null) {
168                throw new ConfigurationException("Incorrect field injection for " + DependencyHandler
169                        .getDependencyIdentifier(dependency) + " - reason: the field " + dependency.getField() + " is" +
170                        " not in the component class (" + dependency.getHandler().getInstanceManager().getClassName()
171                      + ")");
172            }
173        }
174    }

176    /**
177     * Determines if the dependency is aggregate from the field or constructor parameter used to inject the dependency.
178     * If the dependency just uses methods, this method does nothing. This method also check that dependencies set to
179     * aggregate have a valid injection type.
180     * @param dependency the dependency
181     * @param manipulation the manipulation metadata
182     * @throws ConfigurationException if the type of the field or constructor parameter used to inject the dependency
183     * is not suitable for aggregate dependencies.
184     */
185    private static void deduceAggregationFromTheInjectionPoints(Dependency dependency, PojoMetadata manipulation) throws ConfigurationException {
186        if (dependency.getField() != null) {
187            FieldMetadata field = manipulation.getField(dependency.getField());
188            String type = field.getFieldType();
189            if (type.endsWith("[]")) {
190                dependency.setAggregateType(AggregateDependencyInjectionType.ARRAY);
191            } else if (Collection.class.getName().equals(type)  || List.class.getName().equals(type)) {
192                dependency.setAggregateType(AggregateDependencyInjectionType.LIST);
193            } else if (Set.class.getName().equals(type)) {
194                dependency.setAggregateType(AggregateDependencyInjectionType.SET);
195            } else if (Vector.class.getName().equals(type)) {
196                dependency.setAggregateType(AggregateDependencyInjectionType.VECTOR);
197            } else if (dependency.isAggregate()) {
198                // Something wrong. The dependency has a field that is not suitable for aggregate dependencies
199                throw new ConfigurationException("The dependency " + DependencyHandler.getDependencyIdentifier
200                        (dependency) + " cannot be an aggregate dependency - reason: the type " + field.getFieldType
201                        () + " of the field " + field.getFieldName() + " is not suitable for aggregate " +
202                        "dependencies. Compatible types are array, vector, list, set and collection.");
203            }
204        }
205        if (dependency.getConstructorParameterIndex() != -1) {
206            String type = manipulation.getConstructors()[0].getMethodArguments()[dependency
207                    .getConstructorParameterIndex()];
208            if (type.endsWith("[]")) {
209                dependency.setAggregateType(AggregateDependencyInjectionType.ARRAY);
210            } else if (Collection.class.getName().equals(type)  || List.class.getName().equals(type)) {
211                dependency.setAggregateType(AggregateDependencyInjectionType.LIST);
212            } else if (Set.class.getName().equals(type)) {
213                dependency.setAggregateType(AggregateDependencyInjectionType.SET);
214            } else if (Vector.class.getName().equals(type)) {
215                dependency.setAggregateType(AggregateDependencyInjectionType.VECTOR);
216            } else if (dependency.isAggregate()) {
217                // Something wrong. The dependency has a field that is not suitable for aggregate dependencies
218                throw new ConfigurationException("The dependency " + DependencyHandler.getDependencyIdentifier
219                        (dependency) + " cannot be an aggregate dependency - reason: the type " + type
220                        + " of the constructor parameter " + dependency.getConstructorParameterIndex() + " is not suitable for aggregate " +
221                        "dependencies. Compatible types are array, vector, list, set and collection.");
222            }
223        }
224        //TODO We may not cover some cases such as inconsistency between the constructor and the field. However this
225        // should be very rare.
226
227    }

229    /**
230     * Checks that the dependency has at least one injection point.
231     * @param dependency the dependency
232     * @throws ConfigurationException if the dependency has no injection point
233     */
234    private static void ensureThatAtLeastOneInjectionIsSpecified(Dependency dependency) throws ConfigurationException {
235        if (dependency.getField() == null
236                && (dependency.getCallbacks() == null  || dependency.getCallbacks().length == 0)
237                && dependency.getConstructorParameterIndex() == -1) {
238            throw new ConfigurationException("The dependency " + DependencyHandler.getDependencyIdentifier
239                    (dependency) + " is invalid - reason: no injection specified, at least a field, " +
240                    "a method or a constructor parameter index must be set");
241        }
242    }

244    /**
245     * Checks that the `from` attribute is used consistently:
246     * <ul>
247     * <li>Rule 1 : it cannot be used on aggregate dependency</li>
248     * <li>Rule 2 : it cannot be used in combination with the `comparator` attribute</li>
249     * <li>Rule 3 : it cannot be used in combination with the `dynamic-priority` binding policy</li>
250     * </ul>
251     *
252     * @param dependency the dependency
253     * @param metadata the dependency metadata
254     * @throws ConfigurationException if the `from` attribute is used inconsistently.
255     */
256    private static void checkTheConsistencyOfTheFromAttribute(Dependency dependency,
257                                                              Element metadata) throws ConfigurationException {
258        // Check if we have a from attribute.
259        if (metadata.getAttribute("from") != null) {
260            final String message = "The `from` attribute is not usable in " + DependencyHandler
261                    .getDependencyIdentifier(dependency) + " - reason: ";
262            // Rule 1
263            if (dependency.isAggregate()) {
264                throw new ConfigurationException(message + "the dependency is " +
265                        "aggregate");
266            }
267            // Rule 2
268            String comparator = metadata.getAttribute("comparator");
269            if (comparator != null) {
270                throw new ConfigurationException(message + "the dependency uses a comparator");
271            }
272            // Rule 3
273            if (dependency.getBindingPolicy() == DependencyModel.DYNAMIC_PRIORITY_BINDING_POLICY) {
274                throw new ConfigurationException(message + "the dependency uses the dynamic-priority " +
275                        "binding policy");
276            }
277        }
278    }

280    /**
281     * Checks that service unavailable actions are consistent.
282     * <ul>
283     * <li>Rule 1: Nullable, Exception, Default-Implementation... can only be used for scalar optional dependency</li>
284     * <li>Rule 2: Only one can be used</li>
285     * <li>Rule 3: Timeout can only be used on optional dependency</li>
286     * </ul>
287     *
288     * @param dependency the dependency
289     * @throws ConfigurationException if the dependency used inconsistent attributes
290     */
291    private static void checkTheServiceUnavailableAction(Dependency dependency,
292                                                         Element metadata) throws ConfigurationException {
293        if (metadata.containsAttribute("nullable") || dependency.getDefaultImplementation() != null  || dependency
294                .getException() != null) {
295            // Rule 1:
296            String message = "The `nullable`, `default-implementation` and `exception` attributes are not " +
297                    "usable in " + DependencyHandler.getDependencyIdentifier(dependency) + " - reason: ";
298            if (dependency.isAggregate()) {
299                throw  new ConfigurationException(message + "the dependency is aggregate");
300            }
301            if (! dependency.isOptional()) {
302                throw  new ConfigurationException(message + "the dependency is mandatory");
303            }
304
305            // At this point, we know that the dependency is scalar and optional, and at least one attribute is set
306
307            // Rule 2:
308            message = "Inconsistent use of the `nullable`, `default-implementation` and `exception` attributes are " +
309                    "not usable in " + DependencyHandler.getDependencyIdentifier(dependency) + " - reason: ";
310            if (metadata.containsAttribute("nullable")  && dependency.getDefaultImplementation() != null) {
311                throw new ConfigurationException(message + "`nullable` and `default-implementation` cannot be " +
312                        "combined");
313            }
314            if (metadata.containsAttribute("nullable") && dependency.getException() != null) {
315                throw new ConfigurationException(message + "`nullable` and `exception` cannot be combined");
316            }
317            if (dependency.getDefaultImplementation() != null  && dependency.getException() != null) {
318                throw new ConfigurationException(message + "`exception` and `default-implementation` cannot be " +
319                        "combined");
320            }
321        }
322
323        // Rule 3:
324        if (dependency.getTimeout() != 0  && ! dependency.isOptional()) {
325            throw new ConfigurationException("The `timeout` attribute is not usable in " + DependencyHandler
326                    .getDependencyIdentifier(dependency) + " - reason: the dependency is not optional");
327        }

334    /**
335     * Tries to determine the service specification to inject in the dependency.
336     * If the specification is already checked by the dependency, just checks the consistency.
337     *
338     * @param dependency   the dependency
339     * @param manipulation the manipulation metadata
340     * @throws ConfigurationException if the specification cannot be deduced, or when the set specification is not
341     *                                consistent.
342     */
343    private static void deduceTheServiceSpecification(Dependency dependency, PojoMetadata manipulation) throws
344            ConfigurationException {
345
346        // Deduction algorithm
347        String fieldType = null;
348        String callbackType = null;
349        String constructorType = null;
350        // First check the field
351        if (dependency.getField() != null) {
352           fieldType = extractSpecificationFromField(dependency.getField(), manipulation);
353        }
354        if (dependency.getCallbacks() != null  && dependency.getCallbacks().length != 0) {
355            callbackType = extractSpecificationFromMethods(dependency, dependency.getCallbacks(), manipulation);
356        }
357        if (dependency.getConstructorParameterIndex() != -1) {
358            constructorType = extractSpecificationFromConstructor(dependency.getConstructorParameterIndex(),
359                    manipulation);
360        }
361
362        if (dependency.getSpecification() == null
363                && fieldType == null  && callbackType == null  && constructorType == null) {
364            throw new ConfigurationException("The deduction of the service specification for " + DependencyHandler
365                    .getDependencyIdentifier(dependency) + " has failed - reason: when neither the field, " +
366                    "methods and constructor parameter have provided the service specification, " +
367                    "the `specification` attribute must be set");
368
369        }
370
371        // The Dependency.setSpecification method check whether the specification coming from the different sources
372        // are consistent.
373        if (fieldType != null) {
374            setSpecification(dependency, fieldType);
375        }
376        if (callbackType != null) {
377            setSpecification(dependency, callbackType);
378        }
379        if (constructorType != null) {
380            setSpecification(dependency, constructorType);
381        }
382    }
383
384    private static String extractSpecificationFromMethods(Dependency dependency, DependencyCallback[] callbacks,
385                                                          PojoMetadata manipulation) throws ConfigurationException {
386        String type = null;
387        for (DependencyCallback callback : callbacks) {
388            MethodMetadata metadata = manipulation.getMethod(callback.getMethodName());
389            if (metadata != null) {
390                String[] parameters = metadata.getMethodArguments();
391                if (parameters.length == 1  || parameters.length == 2) {
392                    if (! ServiceReference.class.getName().equals(parameters[0])
393                        && ! Dictionary.class.getName().equals(parameters[0])
394                        && ! Map.class.getName().equals(parameters[0])) {
395                        if (type == null) {
396                            type = parameters[0];
397                        } else {
398                            if (! type.equals(parameters[0])) {
399                                throw new ConfigurationException("The callbacks of " + DependencyHandler
400                                        .getDependencyIdentifier(dependency) + " have inconsistent parameters");
401                            }
402                        }
403                    }
404                }
405            }
406        }
407        return type;
408    }
409
410    private static String extractSpecificationFromConstructor(int index, PojoMetadata manipulation) {
411        // We can write the following instructions as everything was previously checked.
412        String type = manipulation.getConstructors()[0].getMethodArguments()[index];
413        if (type.endsWith("[]")) {
414            return type.substring(0, type.length() - 2);
415        }
416        if (AggregateDependencyInjectionType.AGGREGATE_TYPES.contains(type)) {
417            return null; // It's an aggregate
418        }
419        return type;
420    }

422    /**
423     * Extracts the service specification from the field.
424     * When this method is called, we know that the field is containing in the component class.
425     * @param field the field
426     * @param manipulation the manipulation metadata
427     * @return the service specification, or {@code null} if is cannot be extracted.
428     */
429    private static String extractSpecificationFromField(String field, PojoMetadata manipulation) {
430        FieldMetadata metadata = manipulation.getField(field);
431        if (metadata.getFieldType().endsWith("[]")) {
432            return metadata.getFieldType().substring(0, metadata.getFieldType().length() - 2);
433        }
434        if (AggregateDependencyInjectionType.AGGREGATE_TYPES.contains(metadata.getFieldType())) {
435            return null; // It's an aggregate
436        }
437        return metadata.getFieldType();
438    }

440    /**
441     * Sets the dependency specification. If the dependency has already a specification set,
442     * throw an error if the current specification and the given one are not equal.
443     * @param dep the dependency
444     * @param className the service specification
445     * @throws ConfigurationException if the given specification is not loadable or if the dependency has already a
446     * specification set that is not the given one.
447     */
448    private static void setSpecification(Dependency dep, String className) throws ConfigurationException {
449        if (dep.getSpecification() != null  && ! dep.getSpecification().getName().equals(className)) {
450            throw new ConfigurationException("Inconsistent service specification for " + DependencyHandler
451                    .getDependencyIdentifier(dep) + " - reason: mismatch between the current specification (" + dep
452                    .getSpecification().getName() + ") and the discovered specification (" + className + ")");
453        } else if (dep.getSpecification() == null) {
454            // Set the specification
455            try {
456                dep.setSpecification(dep.getBundleContext().getBundle().loadClass(className));
457            } catch (ClassNotFoundException e) {
458                throw new ConfigurationException("Cannot set the service specification of " + DependencyHandler
459                        .getDependencyIdentifier(dep) + " - reason: the class " + className + " cannot be loaded from" +
460                        " the bundle " + dep.getBundleContext().getBundle().getBundleId(), e);
461            }
462        }
463    }
