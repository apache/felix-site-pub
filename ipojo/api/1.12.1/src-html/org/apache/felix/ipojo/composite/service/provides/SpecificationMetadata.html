019package org.apache.felix.ipojo.composite.service.provides;
020
021import java.lang.reflect.Method;
022import java.util.ArrayList;
023import java.util.List;
024
025import org.osgi.framework.BundleContext;
026
027/**
028 * Represent a service specification.
029 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
030 */
031public class SpecificationMetadata {
032
033    /**
034     * Name of the specification, i.e. name of the interface.
035     */
036    private String m_name;
037
038    /**
039     * List of the method contained in the specification.
040     */
041    private List/* <MethodMetadata> */m_methods = new ArrayList/* <MethodMetadata> */();
042
043    /**
044     * Is the specification an aggregate?
045     */
046    private boolean m_isAggregate;
047
048    /**
049     * Is the specification optional?
050     */
051    private boolean m_isOptional = false;
052
053    /**
054     * Is the specification an interface?
055     */
056    private boolean m_isInterface = true;
057
058    /**
059     * Component Type.
060     */
061    private String m_componentType = null;
062
063    /**
064     * Reference on the handler.
065     */
066    private ProvidedServiceHandler m_handler;
067
068    /**
069     * Constructor.
070     * @param name : specification name.
071     * @param context : bundle context.
072     * @param isAggregate : is the specification aggregate.
073     * @param isOptional : is the specification optional.
074     * @param psd : the handler.
075     */
076    public SpecificationMetadata(String name, BundleContext context, boolean isAggregate, boolean isOptional, ProvidedServiceHandler psd) {
077        m_name = name;
078        m_handler = psd;
079
080        // Populate methods :
081        try {
082            Class clazz = context.getBundle().loadClass(name);
083            Method[] methods = clazz.getMethods();
084            for (int i = 0; i < methods.length; i++) {
085                MethodMetadata method = new MethodMetadata(methods[i]);
086                m_methods.add(method);
087            }
088        } catch (ClassNotFoundException e) {
089            m_handler.error("Cannot open " + name + " : " + e.getMessage());
090            return;
091        }
092
093        m_isAggregate = isAggregate;
094        m_isOptional = isOptional;
095    }
096
097    /**
098     * Constructor.
099     * @param clazz : class
100     * @param type : component type
101     * @param psd : the parent handler
102     */
103    public SpecificationMetadata(Class clazz, String type, ProvidedServiceHandler psd) {
104        m_handler = psd;
105        m_isAggregate = false;
106        m_isOptional = false;
107        m_componentType = type;
108        m_name = clazz.getName();
109        Method[] methods = clazz.getMethods();
110        for (int i = 0; i < methods.length; i++) {
111            MethodMetadata method = new MethodMetadata(methods[i]);
112            m_methods.add(method);
113        }
114        m_isInterface = false;
115    }
116
117    public String getName() {
118        return m_name;
119    }
120
121    public List/* <MethodMetadata> */getMethods() {
122        return m_methods;
123    }
124
125    /**
126     * Get a method by its name.
127     * @param name : method name
128     * @return the method metadata contained in the current specification with the given name. Null if the method is not found.
129     */
130    public MethodMetadata getMethodByName(String name) {
131        for (int i = 0; i < m_methods.size(); i++) {
132            MethodMetadata met = (MethodMetadata) m_methods.get(i);
133            if (met.getMethod().getName().equals(name)) { return met; }
134        }
135        return null;
136    }
137
138    public boolean isAggregate() {
139        return m_isAggregate;
140    }
141
142    public boolean isOptional() {
143        return m_isOptional;
144    }
145
146    public boolean isInterface() {
147        return m_isInterface;
148    }
149
150    public void setIsOptional(boolean optional) {
151        m_isOptional = optional;
152    }
153
154    public String getComponentType() {
155        return m_componentType;
156    }
157
158}
