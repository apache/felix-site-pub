020package org.apache.felix.ipojo.manipulator.spi;
021
022import static org.apache.felix.ipojo.manipulator.spi.helper.Predicates.onlySupportedElements;
023
024import java.lang.annotation.Annotation;
025import java.util.ArrayList;
026import java.util.Iterator;
027import java.util.List;
028
029import org.apache.felix.ipojo.manipulator.metadata.annotation.model.AnnotationType;
030import org.apache.felix.ipojo.manipulator.metadata.annotation.model.literal.AnnotationPlayback;
031import org.apache.felix.ipojo.manipulator.metadata.annotation.registry.Binding;
032import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.GenericVisitorFactory;
033import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.ignore.NullBinding;
034import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.stereotype.StereotypeVisitorFactory;
035import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util.Elements;
036import org.apache.felix.ipojo.metadata.Element;
037import org.objectweb.asm.Type;
038
039/**
040 * All provided {@link Module}s have to inherit from this class.
041 * It provides a simple to use DSL to express annotation bindings.
042 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
043 */
044public abstract class AbsBindingModule implements Module {
045
046    /**
047     * Build bindings.
048     */
049    private final List<Binding> bindings = new ArrayList<Binding>();
050
051    private boolean loaded = false;
052
053    public synchronized void load() {
054        if (!loaded) {
055            configure();
056            loaded = true;
057        }
058    }
059
060    /**
061     * Configure the bindings provided by this module.
062     */
063    protected abstract void configure();
064
065
066    public Iterator<Binding> iterator() {
067        return bindings.iterator();
068    }
069
070    /**
071     * Initiate an annotation binding.
072     * Examples:
073     * <pre>
074     *     AnnotationVisitorFactory factory = new CompositeVisitorFactory();
075     *     bind(Composite.class).to(factory);
076     *     bind(Composite.class).when(.. some condition ..)
077     *                          .to(factory);
078     * </pre>
079     * @param annotationType the annotation that will be bound to the {@link AnnotationVisitorFactory}
080     */
081    protected AnnotationBindingBuilder bind(Class<? extends Annotation> annotationType) {
082        return new AnnotationBindingBuilder(bindings, annotationType);
083    }
084
085    protected StereotypeBindingBuilder bindStereotype(Class<? extends Annotation> annotationType) {
086        return new StereotypeBindingBuilder(bindings, annotationType);
087    }
088
089    protected HandlerBindingBuilder bindHandlerBinding(Class<? extends Annotation> annotationType) {
090        return new HandlerBindingBuilder(bindings, annotationType);
091    }
092
093    protected void bindIgnore(Class<? extends Annotation> annotationType) {
094        bindings.add(new NullBinding(Type.getType(annotationType)));
095    }
096
097    /**
098     * DSL helper class.
099     */
100    public class AnnotationBindingBuilder {
101        private Class<? extends Annotation> annotationType;
102        private AnnotationVisitorFactory factory;
103        private List<Binding> registry;
104
105        public AnnotationBindingBuilder(List<Binding> registry,
106                                        Class<? extends Annotation> annotationType) {
107            this.registry = registry;
108            this.annotationType = annotationType;
109        }
110
111        /**
112         * Declares a {@link Predicate} that will add a condition to the annotation binding.
113         * @see org.apache.felix.ipojo.manipulator.spi.helper.Predicates
114         * @param predicate the predicate to use
115         */
116        public ConditionalBindingBuilder when(Predicate predicate) {
117            return new ConditionalBindingBuilder(this, predicate);
118        }
119
120        /**
121         * Complete the annotation binding with the {@link AnnotationVisitorFactory} to be executed
122         * when the annotation is found.
123         * @param factory to be executed when the annotation is found.
124         */
125        public void to(AnnotationVisitorFactory factory) {
126            this.factory = factory;
127            registry.add(build());
128        }
129
130        /**
131         * Creates the Binding.
132         */
133        private Binding build() {
134            Binding binding = new Binding();
135            binding.setAnnotationType(Type.getType(annotationType));
136            binding.setPredicate(onlySupportedElements(annotationType));
137            binding.setFactory(factory);
138            return binding;
139        }
140
141    }
142
143    public class ConditionalBindingBuilder {
144        private AnnotationBindingBuilder parent;
145        private Predicate predicate;
146        private AnnotationVisitorFactory factory;
147
148        public ConditionalBindingBuilder(AnnotationBindingBuilder parent, Predicate predicate) {
149            this.parent = parent;
150            this.predicate = predicate;
151        }
152
153        /**
154         * Complete the annotation binding with the {@link AnnotationVisitorFactory} to be executed
155         * when the annotation is found.
156         * @param factory to be executed when the annotation is found.
157         */
158        public AnnotationBindingBuilder to(AnnotationVisitorFactory factory) {
159            this.factory = factory;
160            bindings.add(build());
161
162            return parent;
163        }
164
165        /**
166         * Creates the Binding.
167         */
168        private Binding build() {
169            Binding binding = parent.build();
170            binding.setPredicate(predicate);
171            binding.setFactory(factory);
172            return binding;
173        }
174    }
175
176    public class StereotypeBindingBuilder {
177        private final AnnotationType m_annotationType;
178
179        public StereotypeBindingBuilder(final List<Binding> bindings, final Class<? extends Annotation> type) {
180            m_annotationType = new AnnotationType(Type.getType(type));
181            Binding binding = new Binding();
182            binding.setAnnotationType(m_annotationType.getType());
183            binding.setPredicate(onlySupportedElements(type));
184            binding.setFactory(new StereotypeVisitorFactory(m_annotationType));
185            bindings.add(binding);
186        }
187
188        public StereotypeBindingBuilder with(AnnotationLiteral<?> literal) {
189            m_annotationType.getPlaybacks().add(new AnnotationPlayback(literal));
190            return this;
191        }
192    }
193
194    public class HandlerBindingBuilder {
195
196        private final Binding m_binding;
197
198        public HandlerBindingBuilder(final List<Binding> bindings, final Class<? extends Annotation> annotationType) {
199            m_binding = new Binding();
200            Type type = Type.getType(annotationType);
201            m_binding.setAnnotationType(type);
202            m_binding.setPredicate(onlySupportedElements(annotationType));
203            Element e = Elements.buildElement(type);
204            m_binding.setFactory(new GenericVisitorFactory(e.getName(), e.getNameSpace()));
205            bindings.add(m_binding);
206        }
207
208        public void to(String namespace, String name) {
209            m_binding.setFactory(new GenericVisitorFactory(name, namespace));
210        }
211    }
212}
