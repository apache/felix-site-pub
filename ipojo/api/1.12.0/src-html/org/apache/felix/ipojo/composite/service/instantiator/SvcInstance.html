041/**
042 * Manage a service instantiation. This service create component instance providing the required service specification.
043 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
044 */
045public class SvcInstance extends DependencyModel {
046
047    /**
048     * Configuration to push to the instance.
049     */
050    private Dictionary m_configuration;
051
052    /**
053     * Handler creating the service instance.
054     */
055    private ServiceDependencyHandler m_handler;
056
057    /**
058     * Map of matching factories Service Reference => instance or null (null if the service reference is not actually used).
059     */
060    private Map /* <ServiceReference, Instance> */m_factories = new HashMap();
061
062    /**
063     * Required specification.
064     */
065    private String m_specification;
066
067    /**
068     * Is the service provider frozen ? (Is used for static biding policy)
069     */
070    private boolean m_isFrozen;
071
072    /**
073     * Constructor.
074     * @param handler : the handler.
075     * @param spec : required specification.
076     * @param conf : instance configuration.
077     * @param isAgg : is the service instance an aggregate service ?
078     * @param isOpt : is the service instance optional ?
079     * @param filt : LDAP filter
080     * @param cmp : comparator to use for the tracking
081     * @param policy : binding policy
082     * @throws ConfigurationException : an attribute cannot be parsed correctly, or is incorrect.
083     */
084    public SvcInstance(ServiceDependencyHandler handler, String spec, Dictionary conf, boolean isAgg, boolean isOpt, Filter filt, Comparator cmp, int policy) throws ConfigurationException {
085        super(Factory.class, isAgg, isOpt, filt, cmp, policy, null, handler, handler.getCompositeManager());
086
087        m_specification = spec;
088
089        m_handler = handler;
090        setBundleContext(m_handler.getCompositeManager().getServiceContext());
091
092        m_configuration = conf;
093    }
094
095    /**
096     * Stop the service instance.
097     */
098    public void stop() {
099        super.stop();
100
101        Set keys = m_factories.keySet();
102        Iterator iterator = keys.iterator();
103        while (iterator.hasNext()) {
104            ServiceReference ref = (ServiceReference) iterator.next();
105            Object object = m_factories.get(ref);
106            if (object != null) {
107                m_handler.info("Dispose a service instance when stopping the handler " + ((ComponentInstance) object).getInstanceName());
108                ((ComponentInstance) object).dispose();
109            }
110        }
111
112        m_factories.clear();
113
114    }
115
116    public boolean isFrozen() {
117        return m_isFrozen;
118    }

120    /**
121     * Freeze the set of used provider.
122     * This method is when the static binding policy is applied.
123     */
124    public void freeze() {
125        m_isFrozen = true;
126    }
127    
128    /**
129     * Unfreezes.
130     */
131    public void unfreeze() {
132        m_isFrozen = false;
133    }

136    /**
137     * Create an instance for the given reference. The instance is not added inside the map.
138     * @param factory : the factory from which we need to create the instance.
139     * @return the created component instance.
140     * @throws ConfigurationException : the instance cannot be configured correctly.
141     * @throws MissingHandlerException : the factory is invalid.
142     * @throws UnacceptableConfiguration : the given configuration is invalid for the given factory.
143     */
144    private ComponentInstance createInstance(Factory factory) throws UnacceptableConfiguration, MissingHandlerException, ConfigurationException {
145        // Recreate the configuration to avoid sharing.
146        Properties props = new Properties();
147        Enumeration keys = m_configuration.keys();
148        while (keys.hasMoreElements()) {
149            String key = (String) keys.nextElement();
150            props.put(key, m_configuration.get(key));
151        }
152        ComponentInstance instance = null;
153        instance = factory.createComponentInstance(props);
154        m_handler.info("Creation of a service instance " + instance.getInstanceName());
155        return instance;
156    }
157
158    /**
159     * Does the service instance match with the given factory ?
160     * @param fact : the factory to test.
161     * @return true if the factory match, false otherwise.
162     */
163    public boolean match(ServiceReference fact) {
164        // Check if the factory can provide the specification
165        ComponentTypeDescription desc = (ComponentTypeDescription) fact.getProperty("component.description");
166        if (desc == null) { 
167            return false; // No component type description.
168        }
169
170        String[] provides = desc.getprovidedServiceSpecification();
171        for (int i = 0; provides != null && i < provides.length; i++) {
172            if (provides[i].equals(m_specification)) {
173                // Check that the factory needs every properties contained in
174                // the configuration
175                PropertyDescription[] props = desc.getProperties();
176                Properties conf = new Properties();
177                Enumeration keys = m_configuration.keys();
178                while (keys.hasMoreElements()) {
179                    String key = (String) keys.nextElement();
180                    if (!containsProperty(key, props)) { return false; }
181                    conf.put(key, m_configuration.get(key));
182                }
183
184                Factory factory = (Factory) getService(fact);
185                return factory.isAcceptable(conf);
186            }
187        }
188        return false;
189    }
190
191    /**
192     * Does the factory support the given property ? This method check if the property is contained in the given property description array.
193     * @param name : name of the property
194     * @param props : list of property description
195     * @return true if the factory support this property
196     */
197    private boolean containsProperty(String name, org.apache.felix.ipojo.architecture.PropertyDescription[] props) {
198        for (int i = 0; props != null && i < props.length; i++) {
199            if (props[i].getName().equalsIgnoreCase(name)) { return true; }
200        }
201        if (name.equalsIgnoreCase("name")) { return true; } // Skip the name property
202        return false;
203    }

209    public String getServiceSpecification() {
210        return m_specification;
211    }

213    /**
214     * Get the map of used references [reference, component instance].
215     * @return the map of used references.
216     */
217    protected Map getMatchingFactories() {
218        return m_factories;
219    }
220
221    /**
222     * On Dependency Reconfiguration notification method.
223     * @param departs : leaving service references.
224     * @param arrivals : new injected service references.
225     * @see org.apache.felix.ipojo.util.DependencyModel#onDependencyReconfiguration(org.osgi.framework.ServiceReference[], org.osgi.framework.ServiceReference[])
226     */
227    public void onDependencyReconfiguration(ServiceReference[] departs, ServiceReference[] arrivals) {
228        for (int i = 0; departs != null && i < departs.length; i++) {
229            onServiceDeparture(departs[i]);
230        }
231        
232        for (int i = 0; arrivals != null && i < arrivals.length; i++) {
233            onServiceArrival(arrivals[i]);
234        }
235    }
236
237    /**
238     * A new service is injected.
239     * This method create the sub-service instance in the composite.
240     * @param ref : service reference.
241     * @see org.apache.felix.ipojo.util.DependencyModel#onServiceArrival(org.osgi.framework.ServiceReference)
242     */
243    public void onServiceArrival(ServiceReference ref) {
244        // The given factory matches.
245        try {
246            Factory fact = (Factory) getService(ref);
247            if (m_factories.get(ref) == null) {
248                ComponentInstance instance = createInstance(fact);
249                m_factories.put(ref, instance);
250            } else {
251                m_handler
252                        .info("An arriving factory is already used, ignore the creation : "
253                                + fact.getName());
254            }
255            
256        } catch (UnacceptableConfiguration e) {
257            m_handler.error("A matching factory refuses the actual configuration : " + e.getMessage());
258            m_handler.getCompositeManager().stop();
259        } catch (MissingHandlerException e) {
260            m_handler.error("A matching factory is no more valid : " + e.getMessage());
261            m_handler.getCompositeManager().stop();
262        } catch (ConfigurationException e) {
263            m_handler.error("A matching configuration is refused by the instance : " + e.getMessage());
264            m_handler.getCompositeManager().stop();
265        }
266
267    }
268
269    
270    /**
271     * A used service is leaving.
272     * This method dispose the created instance.
273     * @param ref : leaving service reference.
274     * @see org.apache.felix.ipojo.util.DependencyModel#onServiceDeparture(org.osgi.framework.ServiceReference)
275     */
276    public void onServiceDeparture(ServiceReference ref) {
277        // Remove the reference is contained
278        Object instance = m_factories.remove(ref);
279        if (instance != null) {
280            m_handler.info("Dispose the instance (departure of the factory) "
281                    + ((ComponentInstance) instance).getInstanceName());
282            ((ComponentInstance) instance).dispose();
283        }
284    }
285
286    /**
287     * A factory is modified. This should not happen.
288     * @param arg0 the service reference
289     * @see org.apache.felix.ipojo.util.DependencyModel#onServiceModification(org.osgi.framework.ServiceReference)
290     */
291    public void onServiceModification(ServiceReference arg0) {
292        // Nothing to do.
293    }
