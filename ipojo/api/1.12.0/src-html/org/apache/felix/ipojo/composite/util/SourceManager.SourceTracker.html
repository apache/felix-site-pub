039/**
040 * This class manages context-source management.
041 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
042 */
043public class SourceManager implements ContextListener {
044
045    /**
046     * Source Name service property.
047     */
048    public static final String SOURCE_NAME = "source.name";
049
050    /**
051     * Managed dependency.
052     */
053    private DependencyModel m_dependency;
054
055    /**
056     * List of monitored context sources.
057     */
058    private List/* <ContextSource> */m_sources = new ArrayList(1);
059
060    /**
061     * PRoperties contained in the original filter.
062     */
063    private String[] m_properties;
064
065    /**
066     * Original filter (containing variables). 
067     */
068    private String m_filter;
069
070    /**
071     * Bundle context.
072     */
073    private BundleContext m_context;
074
075    /**
076     * Service Tracker List.
077     */
078    private List/*<SourceTracker>*/m_trackers = new ArrayList(1);
079
080    /**
081     * Constructor.
082     * @param sources : context-source attribute from the dependency metadata
083     * @param depfilter : original dependency filter
084     * @param dependency : dependency object
085     * @param manager : composite manager
086     * @throws ConfigurationException : the sources are incorrect.
087     */
088    public SourceManager(String sources, String depfilter, DependencyModel dependency, CompositeManager manager) throws ConfigurationException {
089        m_filter = depfilter;
090        m_properties = getProperties(depfilter);
091        m_dependency = dependency;
092        m_context = manager.getGlobalContext();
093        if (manager.getParentServiceContext() == null) {
094            // The parent is the global context
095            parseSources(sources, manager.getGlobalContext(), manager.getGlobalContext(), manager.getServiceContext());
096        } else {
097            parseSources(sources, manager.getGlobalContext(), manager.getParentServiceContext(), manager.getServiceContext());
098        }
099    }
100
101    /**
102     * Start the context management.
103     */
104    public void start() {
105        for (int i = 0; i < m_trackers.size(); i++) {
106            ((SourceTracker) m_trackers.get(i)).open();
107        }
108        computeFilter();
109    }
110
111    /**
112     * Stop the context management.
113     */
114    public void stop() {
115        for (int i = 0; i < m_trackers.size(); i++) {
116            ((SourceTracker) m_trackers.get(i)).close();
117        }
118        setFilter(m_filter);
119        m_sources.clear();
120    }
121
122    /**
123     * Get the state of this source manager.
124     * @return the state of this source manager.
125     */
126    public int getState() {
127        if (m_sources.isEmpty()) {
128            return DependencyModel.UNRESOLVED;
129        } else {
130            return DependencyModel.RESOLVED;
131        }
132    }
133
134    /**
135     * Set the filter of the managed dependency.
136     * @param filter : the new filter to apply
137     */
138    private void setFilter(String filter) {
139        if (!filter.equals(m_dependency.getFilter())) {
140            // Reconfigure
141            try {
142                m_dependency.setFilter(m_context.createFilter(filter));
143            } catch (InvalidSyntaxException e) {
144                throw new IllegalStateException("A context filter is invalid : " + filter, e);
145            }
146        }
147    }
148
149    /**
150     * Compute the new filter.
151     */
152    private void computeFilter() {
153        String fil = m_filter;
154        synchronized (this) {
155            for (int i = 0; i < m_sources.size(); i++) {
156                Dictionary props = ((ContextSource) m_sources.get(i)).getContext();
157                fil = substitute(fil, props); //NOPMD
158            }
159        }
160        if (!fil.equals(m_dependency.getFilter())) {
161            setFilter(fil);
162        }
163    }
164
165    /**
166     * This method substitute ${var} substring by values stored in a map.
167     * @param str : string with variables
168     * @param values : dictionary containing the variable name and the value.
169     * @return resulted string
170     */
171    public static String substitute(String str, Dictionary values) {       
172        int len = str.length();
173        StringBuffer buffer = new StringBuffer(len);
174
175        int prev = 0;
176        int start = str.indexOf("${");
177        int end = str.indexOf('}', start);
178        while (start != -1 && end != -1) {
179            String key = str.substring(start + 2, end);
180            Object value = values.get(key);
181            if (value == null) {
182                buffer.append(str.substring(prev, end + 1));
183            } else {
184                buffer.append(str.substring(prev, start));
185                buffer.append(value);
186            }
187            prev = end + 1;
188            if (prev >= str.length()) {
189                break;
190            }
191
192            start = str.indexOf("${", prev);
193            if (start != -1) {
194                end = str.indexOf('}', start);
195            }
196        }
197
198        buffer.append(str.substring(prev));
199
200        return buffer.toString();
201    }

203    /**
204     * Compute the properties (${name}) from the given filter.
205     * @param str : string form of the filter.
206     * @return the list of found properties.
207     */
208    public static String[] getProperties(String str) {
209        List list = new ArrayList();
210        int prev = 0;
211        int start = str.indexOf("${");
212        int end = str.indexOf('}', start);
213        while (start != -1 && end != -1) {
214            String key = str.substring(start + 2, end);
215            list.add(key);
216            prev = end + 1;
217            if (prev >= str.length()) {
218                break;
219            }
220
221            start = str.indexOf("${", prev);
222            if (start != -1) {
223                end = str.indexOf('}', start);
224            }
225        }
226
227        return (String[]) list.toArray(new String[list.size()]);
228    }

230    /**
231     * A context source has modified a monitored property. 
232     * @param source : source
233     * @param property : modified property
234     * @param value : new value.
235     * @see org.apache.felix.ipojo.ContextListener#update(org.apache.felix.ipojo.ContextSource, java.lang.String, java.lang.Object)
236     */
237    public synchronized void update(ContextSource source, String property, Object value) {
238        computeFilter();
239    }

241    /**
242     * Parse the context-source attribute in order to create source tracker object.
243     * @param sourceAtt : context-source attribute.
244     * @param global : global bundle context.
245     * @param parent : parent bundle context.
246     * @param local : local bundle context.
247     * @throws ConfigurationException : the context-source attribute is invalid.
248     */
249    private void parseSources(String sourceAtt, BundleContext global, BundleContext parent, BundleContext local) throws ConfigurationException {
250        String[] sources = ParseUtils.split(sourceAtt, ",");
251        for (int i = 0; i < sources.length; i++) {
252            String[] srcs = ParseUtils.split(sources[i], ":");
253            if (srcs.length == 1) {
254                // No prefix use local. //TODO choose default case.
255                SourceTracker tracker = new SourceTracker(srcs[0], local);
256                m_trackers.add(tracker);
257            } else if (srcs.length == 2) {
258                // According to prefix add the source in the good list.
259                if (srcs[0].equalsIgnoreCase("parent")) {
260                    SourceTracker tracker = new SourceTracker(srcs[1], parent);
261                    m_trackers.add(tracker);
262                } else if (srcs[0].equalsIgnoreCase("local")) {
263                    SourceTracker tracker = new SourceTracker(srcs[1], local);
264                    m_trackers.add(tracker);
265                } else if (srcs[0].equalsIgnoreCase("global")) {
266                    SourceTracker tracker = new SourceTracker(srcs[1], global);
267                    m_trackers.add(tracker);
268                } else {
269                    throw new ConfigurationException("Unknowns context scope : " + srcs[0]);
270                }
271            } else {
272                throw new ConfigurationException("Malformed context source : " + sources[i]);
273            }
274        }
275    }
276
277    /**
278     * A context source appears.
279     * @param source : new context source.
280     */
281    private void addContextSource(ContextSource source) {
282        m_sources.add(source);
283        computeFilter();
284        source.registerContextListener(this, m_properties);
285    }
286
287    /**
288     * A context source disappears.
289     * @param source : leaving context source.
290     */
291    private void removeContextSource(ContextSource source) {
292        m_sources.remove(source);
293        computeFilter();
294    }
295
296    private class SourceTracker implements TrackerCustomizer {
297
298        /**
299         * Service tracker.
300         */
301        private Tracker m_tracker;
302
303        /**
304         * Constructor.
305         * @param name : name of the required context-source.
306         * @param countext : bundle context to use.
307         * @throws ConfigurationException : the context-source name is invalid.
308         */
309        public SourceTracker(String name, BundleContext countext) throws ConfigurationException {
310            String fil = "(&(" + Constants.OBJECTCLASS + "=" + ContextSource.class.getName() + ")(" + SOURCE_NAME + "=" + name + "))";
311            try {
312                Filter filter = countext.createFilter(fil);
313                m_tracker = new Tracker(countext, filter, this);
314            } catch (InvalidSyntaxException e) {
315                throw new ConfigurationException("A Context source filter is invalid " + fil, e);
316            }
317        }
318
319        /**
320         * Open the tracker.
321         */
322        public void open() {
323            m_tracker.open();
324        }
325
326        /**
327         * Close the tracker.
328         */
329        public void close() {
330            m_tracker.close();
331        }
332
333        /**
334         * A new context-source was added.
335         * This method inject the context-source object in the source manager.
336         * @param reference : service reference.
337         * @see org.apache.felix.ipojo.util.TrackerCustomizer#addedService(org.osgi.framework.ServiceReference)
338         */
339        public void addedService(ServiceReference reference) {
340            addContextSource((ContextSource) m_tracker.getService(reference));
341        }
342
343        /**
344         * A new context-source is adding in the tracker.. 
345         * @param reference : service reference
346         * @return true.
347         * @see org.apache.felix.ipojo.util.TrackerCustomizer#addingService(org.osgi.framework.ServiceReference)
348         */
349        public boolean addingService(ServiceReference reference) {
350            return true;
351        }
352
353        /**
354         * A used context-source is modified.
355         * @param reference : service reference.
356         * @param service : service object.
357         * @see org.apache.felix.ipojo.util.TrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, java.lang.Object)
358         */
359        public void modifiedService(ServiceReference reference, Object service) {
360            // Nothing to do.
361        }
362
363        /**
364         * A used context-source disappears.
365         * This method notify the Source Manager in order to manage this departure. 
366         * @param reference : service reference.
367         * @param service : service object.
368         * @see org.apache.felix.ipojo.util.TrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, java.lang.Object)
369         */
370        public void removedService(ServiceReference reference, Object service) {
371            removeContextSource((ContextSource) service);
372        }
373
374    }
