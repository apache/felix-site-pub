020package org.apache.felix.ipojo.manipulation;
021
022import java.util.Set;
023
024import org.objectweb.asm.AnnotationVisitor;
025import org.objectweb.asm.MethodVisitor;
026import org.objectweb.asm.Opcodes;
027import org.objectweb.asm.commons.GeneratorAdapter;
028
029/**
030 * Insert code calling callbacks at the entry and before the exit of a method.
031 * Moreover it replaces all GETFIELD and SETFIELD by getter and setter invocation.
032 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
033 */
034public class MethodCodeAdapter extends GeneratorAdapter implements Opcodes {
035
036    /**
037     * The owner class of the field. m_owner : String
038     */
039    private String m_owner;
040
041    /**
042     * Contained fields.
043     */
044    private Set<String> m_fields;
045
046    /**
047     * MethodCodeAdapter constructor.
048     * @param mv : MethodVisitor
049     * @param owner : Name of the class
050     * @param access : Method access
051     * @param name : Method name
052     * @param desc : Method descriptor
053     * @param fields : Contained fields
054     */
055    public MethodCodeAdapter(final MethodVisitor mv, final String owner, int access, String name, String desc, Set<String> fields) {
056        super(Opcodes.ASM5, mv, access, name, desc);
057        m_owner = owner;
058        m_fields = fields;
059    }
060
061    /**
062     * Visit an instruction modifying a method (GETFIELD/PUTFIELD).
063     * @see org.objectweb.asm.MethodVisitor#visitFieldInsn(int, String, String, String)
064     * @param opcode : visited operation code
065     * @param owner : owner of the field
066     * @param name : name of the field
067     * @param desc : descriptor of the field
068     */
069    public void visitFieldInsn(final int opcode, final String owner, final String name, final String desc) {
070        if (owner.equals(m_owner) && m_fields.contains(name)) {
071            if (opcode == GETFIELD) {
072                String gDesc = "()" + desc;
073                visitMethodInsn(INVOKEVIRTUAL, owner, "__get" + name, gDesc, false);
074                return;
075            } else if (opcode == PUTFIELD) {
076                String sDesc = "(" + desc + ")V";
077                visitMethodInsn(INVOKEVIRTUAL, owner, "__set" + name, sDesc, false);
078                return;
079            }
080        }
081        super.visitFieldInsn(opcode, owner, name, desc);
082    }
083
084    /**
085     * Visits an annotation.
086     * If the annotation is visible, the annotation is removed. In fact
087     * the annotation was already moved to the method replacing this one.
088     * If the annotation is not visible, this annotation is kept on this method.
089     * @param name the name of the annotation
090     * @param visible the annotation visibility
091     * @return the <code>null</code> if the annotation is visible, otherwise returns
092     * {@link GeneratorAdapter#visitAnnotation(String, boolean)}
093     * @see org.objectweb.asm.commons.GeneratorAdapter#visitAnnotation(java.lang.String, boolean)
094     */
095    public AnnotationVisitor visitAnnotation(String name, boolean visible) {
096        // Annotations are moved to the injected constructor.
097        if (visible) {
098            return null;
099        } else {
100            return super.visitAnnotation(name, false);
101        }
102    }
103
104    /**
105     * Visits a parameter annotation.
106     * @param id the parameter number.
107     * @param name the annotation name
108     * @param visible if te annotation visibility
109     * @return the <code>null</code> if the annotation is visible, otherwise returns
110     * {@link GeneratorAdapter#visitAnnotation(String, boolean)}
111     * @see org.objectweb.asm.commons.GeneratorAdapter#visitParameterAnnotation(int, java.lang.String, boolean)
112     */
113    public AnnotationVisitor visitParameterAnnotation(int id, String name,
114            boolean visible) {
115        if (visible) {
116            return null;
117        } else {
118            return super.visitParameterAnnotation(id, name, false);
119        }
120    }
121
122}
