020package org.apache.felix.ipojo.extender.internal.declaration;
021
022import org.apache.felix.ipojo.extender.Declaration;
023import org.apache.felix.ipojo.extender.DeclarationHandle;
024import org.apache.felix.ipojo.extender.Status;
025import org.apache.felix.ipojo.extender.internal.AbstractService;
026import org.osgi.framework.BundleContext;
027
028/**
029 * Common code to all Declaration objects.
030 */
031public abstract class AbstractDeclaration extends AbstractService implements Declaration, DeclarationHandle, Status {
032
033    /**
034     * The message used when a declaration is bound.
035     */
036
037    public static final String DECLARATION_BOUND_MESSAGE = "Declaration bound";
038    /**
039     * Is the declaration bound (i.e. successfully fulfilled), or not.
040     */
041    private boolean m_bound = false;
042
043    /**
044     * When not bound, the reason, otherwise {@link #DECLARATION_BOUND_MESSAGE}
045     */
046    private String m_message;
047
048    /**
049     * If an error was thrown during the binding, the error.
050     */
051    private Throwable m_throwable;
052
053    protected AbstractDeclaration(BundleContext bundleContext, Class<?> type) {
054        super(bundleContext, type);
055    }
056
057    /**
058     * @return whether the declaration is bound or not.
059     */
060    public boolean isBound() {
061        return m_bound;
062    }
063
064    /**
065     * @return the message. The message contains the not-bound reason.
066     */
067    public String getMessage() {
068        return m_message;
069    }
070
071    /**
072     * @return the error if something went wrong during the binding process.
073     */
074    public Throwable getThrowable() {
075        return m_throwable;
076    }
077
078    /**
079     * Gets the status of the declaration. This method returns an immutable object.
080     *
081     * @return the declaration status.
082     */
083    public Status getStatus() {
084        // We return an immutable object, created on the fly.
085        return new Status() {
086            final boolean m_bound = AbstractDeclaration.this.m_bound;
087            final String m_message = AbstractDeclaration.this.m_message;
088            final Throwable m_throwable = AbstractDeclaration.this.m_throwable;
089
090            public boolean isBound() {
091                return this.m_bound;
092            }
093
094            public String getMessage() {
095                return this.m_message;
096            }
097
098            public Throwable getThrowable() {
099                return this.m_throwable;
100            }
101        };
102    }
103
104    /**
105     * Binds the declaration.
106     * This method just set the status, message and error.
107     */
108    public void bind() {
109        m_bound = true;
110        m_message = DECLARATION_BOUND_MESSAGE;
111        m_throwable = null;
112    }
113
114    /**
115     * Unbinds the declaration.
116     *
117     * @param message an explanation
118     * @see #unbind(String, Throwable)
119     */
120    public void unbind(String message) {
121        unbind(message, null);
122    }
123
124    /**
125     * Unbinds the declaration.
126     * The flag, message and error are set.
127     *
128     * @param message   an explanation
129     * @param throwable an error
130     */
131    public void unbind(String message, Throwable throwable) {
132        m_bound = false;
133        m_message = message;
134        m_throwable = throwable;
135    }
136
137    public void publish() {
138        if (!isRegistered()) {
139            start();
140        }
141    }
142
143    public void retract() {
144        if (isRegistered()) {
145            stop();
146        }
147    }
148}
