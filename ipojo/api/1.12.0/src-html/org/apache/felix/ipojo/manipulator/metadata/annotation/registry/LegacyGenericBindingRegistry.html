020package org.apache.felix.ipojo.manipulator.metadata.annotation.registry;
021
022import org.apache.felix.ipojo.manipulator.Reporter;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.FieldGenericVisitor;
024import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.MethodGenericVisitor;
025import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.ParameterGenericVisitor;
026import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.TypeGenericVisitor;
027import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util.Elements;
028import org.apache.felix.ipojo.manipulator.spi.AnnotationVisitorFactory;
029import org.apache.felix.ipojo.manipulator.spi.BindingContext;
030import org.objectweb.asm.AnnotationVisitor;
031import org.objectweb.asm.Type;
032
033import java.util.List;
034import java.util.regex.Pattern;
035
036import static java.util.Collections.emptyList;
037import static java.util.Collections.singletonList;
038import static org.apache.felix.ipojo.manipulator.spi.helper.Predicates.alwaysTrue;
039
040/**
041 * User: guillaume
042 * Date: 11/07/13
043 * Time: 16:09
044 */
045public class LegacyGenericBindingRegistry extends CompletableBindingRegistry {
046    public static final Pattern CUSTOM_HANDLER_PATTERN = Pattern.compile("(.*\\.ipojo\\..*)|(.*\\.handler\\..*)");
047
048    public LegacyGenericBindingRegistry(final BindingRegistry delegate, final Reporter reporter) {
049        super(delegate, reporter);
050    }
051
052    @Override
053    protected List<Binding> createBindings(final Type type) {
054        if (CUSTOM_HANDLER_PATTERN.matcher(type.getClassName()).matches()) {
055            Binding binding = new Binding();
056            binding.setAnnotationType(type);
057            binding.setPredicate(alwaysTrue());
058            binding.setFactory(new AnnotationVisitorFactory() {
059                // Need to build a new Element instance for each created visitor
060                public AnnotationVisitor newAnnotationVisitor(BindingContext context) {
061                    if (context.getClassNode() != null) {
062                        return new TypeGenericVisitor(context.getWorkbench(),
063                                Elements.buildElement(type));
064                    } else if (context.getFieldNode() != null) {
065                        return new FieldGenericVisitor(context.getWorkbench(),
066                                Elements.buildElement(type),
067                                context.getFieldNode());
068
069                    } else if ((context.getMethodNode() != null) &&
070                            (context.getParameterIndex() == BindingContext.NO_INDEX)) {
071                        return new MethodGenericVisitor(context.getWorkbench(),
072                                Elements.buildElement(type),
073                                context.getMethodNode());
074                    } else {
075                        // last case: method parameter annotation
076                        return new ParameterGenericVisitor(context.getWorkbench(),
077                                Elements.buildElement(type),
078                                context.getMethodNode(),
079                                context.getParameterIndex());
080                    }
081                }
082
083                @Override
084                public String toString() {
085                    return "LegacyGenericVisitorFactory";
086                }
087            });
088
089            // Return the produced generic binding
090            return singletonList(binding);
091        }
092
093        return emptyList();
094    }
095}
