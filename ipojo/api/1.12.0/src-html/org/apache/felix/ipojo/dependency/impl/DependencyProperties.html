020package org.apache.felix.ipojo.dependency.impl;
021
022import org.apache.felix.ipojo.Factory;
023import org.apache.felix.ipojo.dependency.interceptors.ServiceTrackingInterceptor;
024import org.apache.felix.ipojo.util.DependencyModel;
025import org.apache.felix.ipojo.util.Log;
026import org.osgi.framework.*;
027
028import java.util.ArrayList;
029import java.util.Dictionary;
030import java.util.Hashtable;
031import java.util.List;
032
033/**
034 * Builds the properties used to checks if an interceptor matches a specific dependency.
035 */
036public class DependencyProperties {
037
038    //TODO Externalize and use constants
039    // TODO Cache dependency properties.
040
041    public static  Dictionary<String, ?> getDependencyProperties(DependencyModel dependency) {
042        Dictionary<String, Object> properties = new Hashtable<String, Object>();
043
044        // Instance, and Factory and Bundle (name, symbolic name, version)
045        properties.put(Factory.INSTANCE_NAME_PROPERTY, dependency.getComponentInstance().getInstanceName());
046        properties.put("instance.state", dependency.getComponentInstance().getState());
047        properties.put("factory.name", dependency.getComponentInstance().getFactory().getFactoryName());
048        final Bundle bundle = dependency.getBundleContext().getBundle();
049        properties.put("bundle.symbolicName", bundle.getSymbolicName());
050        if (bundle.getVersion() != null) {
051            properties.put("bundle.version", bundle.getVersion().toString());
052        }
053
054        // Dependency specification, and id
055        final Class specification = dependency.getSpecification();
056        properties.put("dependency.specification", specification.getName());
057        properties.put("dependency.id", dependency.getId());
058        properties.put("dependency.state", dependency.getState());
059
060        // We also provide the objectclass property, and to be compliant with the osgi specification,
061        // all interfaces are collected to this array.
062        List<String> classes = new ArrayList<String>();
063        classes.add(specification.getName());
064        for (Class clazz : specification.getInterfaces()) {
065            classes.add(clazz.getName());
066        }
067        properties.put(Constants.OBJECTCLASS, classes.toArray(new String[classes.size()]));
068
069        return properties;
070    }
071
072
073    /**
074     * Checks that the 'target' property of the service reference matches the dependency.
075     * @param reference the reference
076     * @param dependency the dependency
077     * @param context a bundle context used to build the filter
078     * @return {@literal true} if the target's property of reference matches the dependency.
079     */
080    public static boolean match(ServiceReference reference, DependencyModel dependency, BundleContext context) {
081        Object v = reference.getProperty(ServiceTrackingInterceptor.TARGET_PROPERTY);
082        Filter filter = null;
083        if (v == null) {
084            return false; // Invalid interceptor
085        }
086        if (v instanceof Filter) {
087            filter = (Filter) v;
088        } else if (v instanceof String) {
089            try {
090                filter = context.createFilter((String) v);
091            } catch (InvalidSyntaxException e) {
092                dependency.getComponentInstance().getFactory().getLogger().log(Log.ERROR,
093                        "Cannot build filter from the target property : " + v, e);
094            }
095        }
096
097        if (filter == null) {
098            return false; // Invalid interceptor.
099        }
100
101        Dictionary<String, ?> properties = getDependencyProperties(dependency);
102
103        return filter.match(properties);
104    }
105
106    public static boolean match(ServiceReference reference, DependencyModel dependency) {
107        return match(reference, dependency, dependency.getBundleContext());
108    }
109}
