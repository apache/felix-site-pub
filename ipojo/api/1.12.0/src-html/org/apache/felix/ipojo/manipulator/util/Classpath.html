019package org.apache.felix.ipojo.manipulator.util;
020
021import java.io.File;
022import java.net.MalformedURLException;
023import java.net.URL;
024import java.util.*;
025
026/**
027 * An ordered list of classpath elements with set behaviour. A Classpath is immutable and thread safe.
028 * This class let retrieve an isolated classloader to load classes from a set of jars.
029 */
030public class Classpath implements Iterable<String> {
031
032    private final List<String> unmodifiableElements;
033
034    public static Classpath join(Classpath firstClasspath, Classpath secondClasspath) {
035        LinkedHashSet<String> accumulated = new LinkedHashSet<String>();
036        if (firstClasspath != null) firstClasspath.addTo(accumulated);
037        if (secondClasspath != null) secondClasspath.addTo(accumulated);
038        return new Classpath(accumulated);
039    }
040
041
042    private void addTo(Collection<String> c) {
043        c.addAll(unmodifiableElements);
044    }
045
046    private Classpath() {
047        this.unmodifiableElements = Collections.emptyList();
048    }
049
050
051    public Classpath(Classpath other, String additionalElement) {
052        ArrayList<String> elems = new ArrayList<String>(other.unmodifiableElements);
053        elems.add(additionalElement);
054        this.unmodifiableElements = Collections.unmodifiableList(elems);
055    }
056
057    public Classpath(Iterable<String> paths) {
058        List<String> newCp = new ArrayList<String>();
059        for (String element : paths) {
060            newCp.add(element);
061        }
062        this.unmodifiableElements = Collections.unmodifiableList(newCp);
063    }
064
065    public static Classpath emptyClasspath() {
066        return new Classpath();
067    }
068
069    public Classpath addClassPathElementUrl(String path) {
070        if (path == null) {
071            throw new IllegalArgumentException("Null is not a valid class path element url.");
072        }
073        return !unmodifiableElements.contains(path) ? new Classpath(this, path) : this;
074    }
075
076    public List<String> getClassPath() {
077        return unmodifiableElements;
078    }
079
080    public List<URL> getAsUrlList()
081            throws MalformedURLException {
082        List<URL> urls = new ArrayList<URL>();
083        for (String path : unmodifiableElements) {
084            File f = new File(path);
085            urls.add(f.toURI().toURL());
086        }
087        return urls;
088    }
089
090    public void writeToSystemProperty(String propertyName) {
091        StringBuilder sb = new StringBuilder();
092        for (String element : unmodifiableElements) {
093            sb.append(element).append(File.pathSeparatorChar);
094        }
095        System.setProperty(propertyName, sb.toString());
096    }
097
098    public boolean equals(Object o) {
099        if (this == o) {
100            return true;
101        }
102        if (o == null || getClass() != o.getClass()) {
103            return false;
104        }
105
106        Classpath classpath = (Classpath) o;
107
108        return !(unmodifiableElements
109                != null ? !unmodifiableElements.equals(classpath.unmodifiableElements) : classpath.unmodifiableElements != null);
110
111    }
112
113    public ClassLoader createClassLoader(ClassLoader parent, boolean childDelegation)
114            throws RuntimeException {
115        try {
116            List urls = getAsUrlList();
117            IsolatedClassLoader classLoader = new IsolatedClassLoader(parent, childDelegation);
118            for (Object url1 : urls) {
119                URL url = (URL) url1;
120                classLoader.addURL(url);
121            }
122            return classLoader;
123        } catch (MalformedURLException e) {
124            throw new RuntimeException("When creating classloader", e);
125        }
126    }
127
128    public ClassLoader createClassLoader()
129            throws RuntimeException {
130        try {
131            List urls = getAsUrlList();
132            IsolatedClassLoader classLoader = new IsolatedClassLoader();
133            for (Object url1 : urls) {
134                URL url = (URL) url1;
135                classLoader.addURL(url);
136            }
137            return classLoader;
138        } catch (MalformedURLException e) {
139            throw new RuntimeException("When creating classloader", e);
140        }
141    }
142
143
144    public int hashCode() {
145        return unmodifiableElements != null ? unmodifiableElements.hashCode() : 0;
146    }
147
148    public String getLogMessage(String descriptor) {
149        StringBuilder result = new StringBuilder();
150        result.append(descriptor).append(" classpath:");
151        for (String element : unmodifiableElements) {
152            result.append("  ").append(element);
153        }
154        return result.toString();
155    }
156
157    public String getCompactLogMessage(String descriptor) {
158        StringBuilder result = new StringBuilder();
159        result.append(descriptor).append(" classpath:");
160        for (String element : unmodifiableElements) {
161            result.append("  ");
162            if (element != null) {
163                int pos = element.lastIndexOf(File.separatorChar);
164                if (pos >= 0) {
165                    result.append(element.substring(pos + 1));
166                } else {
167                    result.append(element);
168                }
169
170            } else {
171                result.append(element);
172            }
173        }
174        return result.toString();
175    }
176
177    public Iterator<String> iterator() {
178        return unmodifiableElements.iterator();
179    }
180}
