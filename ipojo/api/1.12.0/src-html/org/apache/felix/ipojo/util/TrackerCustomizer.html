019package org.apache.felix.ipojo.util;
020
021import org.osgi.framework.ServiceReference;
022
023/**
024 * Tracker Customizer.
025 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
026 */
027public interface TrackerCustomizer {
028
029    /**
030     * A service is being added to the Tracker object.
031     * This method is called before a service which matched the search parameters of the Tracker object is added to
032     * it. This method should return the service object to be tracked for this ServiceReference object.
033     * If this method returns {@code true}, the service object becomes available from the {@link Tracker#getService()} and
034     * {@link Tracker#getServices()} methods. However notice that the service is still not accessible at that time.
035     * @param reference the Reference to service being added to the Tracker object.
036     * @return {@code true} if the service reference must be tracked. {@code false} ff the service reference must be
037     * ignored (un-tracked)
038     */
039    boolean addingService(ServiceReference reference);
040    
041    /**
042     * A service tracked by the Tracker object has been added in the list.
043     * This method is called when a service has been added in the managed list (after addingService) and if the
044     * service has not disappeared before during the callback.
045     * In this method, the service object is accessible from the {@link Tracker#getService()} and
046     * {@link Tracker#getServices()} method.
047     * @param reference the added reference.
048     */
049    void addedService(ServiceReference reference);
050
051    /**
052     * A service tracked by the Tracker object has been modified.
053     * This method is called when a service tracked by the Tracker has its properties modified.
054     * @param reference the reference to service that has been modified.
055     * @param service The service object for the modified service.
056     */
057    void modifiedService(ServiceReference reference, Object service);
058
059    /**
060     * A service tracked by the Tracker object has been removed.
061     * This method is called when a tracked service is no longer being tracked by the Tracker object.
062     * Notice that some OSGi implementations does not support getting the service object from the bundle context in
063     * this method. The cached (given) object must be used.
064     * @param reference the reference to service that has been removed.
065     * @param service The service object for the removed service.
066     */
067    void removedService(ServiceReference reference, Object service);
068
069}
