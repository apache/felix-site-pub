020package org.apache.felix.ipojo.extender;
021
022import org.osgi.framework.BundleContext;
023
024/**
025 * Support class for fluent instance declaration building.
026 * This class can be used to specify instance details like naming, component's type, component's version.
027 * The {@link #context(org.osgi.framework.BundleContext)} method can be used to override the default
028 * {@link org.osgi.framework.BundleContext} used for declaration service registration.
029 *
030 * @since 1.11.2
031 */
032public interface InstanceBuilder {
033
034    /**
035     * Specify the instance name. Using {@literal null} will result in an anonymous instance.
036     * @param name instance name or {@literal null}
037     * @return this builder
038     */
039    InstanceBuilder name(String name);
040
041    /**
042     * Specify the component's type of this instance.
043     * @param type type of the instance (cannot be {@literal null}).
044     * @return this builder
045     */
046    InstanceBuilder type(String type);
047
048    /**
049     * Specify the component's type of this instance.
050     * @param type type of the instance (cannot be {@literal null}).
051     * @return this builder
052     */
053    InstanceBuilder type(Class<?> type);
054
055    /**
056     * Specify the component's version of this instance. If {@literal null} is used,
057     * the factory without any version attribute will be used.
058     * @param version component's version (can be {@literal null}).
059     * @return this builder
060     */
061    InstanceBuilder version(String version);
062
063    /**
064     * Override the default BundleContext used for declaration service registration.
065     * Use this with <bold>caution</bold>, for example, if the bundle does not import the
066     * <code>{@link org.apache.felix.ipojo.extender}</code> package, declarations may not be
067     * linked and activated properly.
068     * @param context new context to be used.
069     * @return this builder
070     */
071    InstanceBuilder context(BundleContext context);
072
073    /**
074     * Access the dedicated builder for configuration (properties).
075     * Notice that when this method is used, the called must use {@link ConfigurationBuilder#build()}
076     * to build the instance declaration configured with the right set of properties. Any attempt
077     * to use {@link #build()} will result in a new declaration with no properties associated.
078     *
079     * Good usage (produced declaration has the associated properties):
080     * <pre>
081     *     DeclarationHandle handle = builder.configure()
082     *                                            .property("hello", "world")
083     *                                            .build();
084     * </pre>
085     *
086     * Bad usage (produced declaration does not have the associated properties):
087     * <pre>
088     *     builder.configure()
089     *                .property("hello", "world");
090     *
091     *     DeclarationHandle handle = builder.build();
092     * </pre>
093     *
094     * @return the instance configuration builder
095     */
096    ConfigurationBuilder configure();
097
098    /**
099     * Build the declaration handle (never contains any configuration).
100     * Notice that the declaration is not yet published (no automatic activation).
101     * The client has to do it through {@link DeclarationHandle#publish()}
102     * @return the handle to the declaration
103     */
104    DeclarationHandle build();
105}
