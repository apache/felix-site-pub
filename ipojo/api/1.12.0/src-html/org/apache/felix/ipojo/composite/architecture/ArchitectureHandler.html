019package org.apache.felix.ipojo.composite.architecture;
020
021import java.util.Dictionary;
022import java.util.Hashtable;
023
024import org.apache.felix.ipojo.ComponentInstance;
025import org.apache.felix.ipojo.Factory;
026import org.apache.felix.ipojo.InstanceStateListener;
027import org.apache.felix.ipojo.architecture.Architecture;
028import org.apache.felix.ipojo.architecture.InstanceDescription;
029import org.apache.felix.ipojo.composite.CompositeHandler;
030import org.apache.felix.ipojo.metadata.Element;
031import org.osgi.framework.ServiceRegistration;
032
033/**
034 * Composite Architecture Handler.
035 * 
036 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
037 */
038public class ArchitectureHandler extends CompositeHandler implements Architecture, InstanceStateListener {
039
040    /**
041     * Name of the component.
042     */
043    private String m_name;
044
045    /**
046     * The Architecture service registration.
047     */
048    private ServiceRegistration m_serviceRegistration;
049
050    /**
051     * Configure the handler.
052     *
053     * @param metadata      : the metadata of the component
054     * @param configuration : the instance configuration
055     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
056     */
057    public void configure(Element metadata, Dictionary configuration) {
058        m_name = (String) configuration.get(Factory.INSTANCE_NAME_PROPERTY);
059        Dictionary<String, String> dict = new Hashtable<String, String>();
060        dict.put(ARCHITECTURE_INSTANCE, m_name);
061
062        debug("Registering architecture service for " + m_name);
063        m_serviceRegistration = getCompositeManager().getContext().registerService(Architecture.class.getName(), this, dict);
064
065        // We can't use the regular handler stateChanged method as this method is not called when the instance is
066        // disposed. This handler stays actives until the instance disposal.
067        getCompositeManager().addInstanceStateListener(this);
068    }
069
070    /**
071     * Stop method.
072     *
073     * @see org.apache.felix.ipojo.Handler#stop()
074     */
075    public void stop() {
076        // Nothing do do when stopping.
077    }
078
079    /**
080     * Start method.
081     *
082     * @see org.apache.felix.ipojo.Handler#start()
083     */
084    public void start() {
085        // Nothing to do.
086    }
087
088    /**
089     * The instance lifecycle listener callback.
090     * When we receive the DISPOSED state, the architecture is unregistered from the service registry.
091     * @param instance the changing instance the instance, meaningless in our case.
092     * @param newState the new instance state the new instance state.
093     */
094    public void stateChanged(ComponentInstance instance, int newState) {
095        if (newState == ComponentInstance.DISPOSED && m_serviceRegistration != null) {
096            debug("Withdrawing the architecture service of " + m_name + " due to instance disposal");
097            m_serviceRegistration.unregister();
098            m_serviceRegistration = null;
099        }
100    }
101
102    /**
103     * Get the instance description.
104     * @return the instance description
105     * @see org.apache.felix.ipojo.architecture.Architecture#getInstanceDescription() ()
106     */
107    public InstanceDescription getInstanceDescription() {
108        return getCompositeManager().getInstanceDescription();
109    }
110
111}
