020package org.apache.felix.ipojo.manipulation;
021
022import org.apache.felix.ipojo.metadata.Attribute;
023import org.apache.felix.ipojo.metadata.Element;
024import org.objectweb.asm.ClassReader;
025import org.objectweb.asm.ClassWriter;
026import org.objectweb.asm.Opcodes;
027import org.objectweb.asm.util.CheckClassAdapter;
028
029import java.io.ByteArrayInputStream;
030import java.io.IOException;
031import java.io.InputStream;
032import java.util.*;
033
034/**
035 * iPOJO Byte code Manipulator.
036 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
037 *
038 */
039public class Manipulator {
040    /**
041     * A classloader used to compute frames.
042     */
043    private final ClassLoader m_classLoader;
044    /**
045     * Store the visited fields : [name of the field, type of the field].
046     */
047    private Map<String, String> m_fields;
048
049    /**
050     * Store the interface implemented by the class.
051     */
052    private List<String> m_interfaces;
053
054    /**
055     * Store the methods list.
056     */
057    private List<MethodDescriptor> m_methods;
058
059    /**
060     * Pojo super class.
061     */
062    private String m_superClass;
063
064    /**
065     * List of owned inner classes and internal methods.
066     */
067    private Map<String, List<MethodDescriptor>> m_inners;
068
069    /**
070     * Java byte code version.
071     */
072    private int m_version;
073
074    /**
075     * Was the class already manipulated.
076     */
077    private boolean m_alreadyManipulated;
078
079    /**
080     * The manipulated class name.
081     */
082    private String m_className;
083
084    public Manipulator(ClassLoader loader) {
085        // No classloader set, use current one.
086        m_classLoader = loader;
087    }
088
089    /**
090     * Checks the given bytecode, determines if the class was already manipulated, and collect the metadata about the
091     * class.
092     * @param origin the bytecode
093     */
094    public void prepare(byte[] origin) throws IOException {
095        InputStream is = new ByteArrayInputStream(origin);
096
097        // First check if the class is already manipulated :
098        ClassReader ckReader = new ClassReader(is);
099        ClassChecker ck = new ClassChecker();
100        ckReader.accept(ck, ClassReader.SKIP_FRAMES);
101        is.close();
102
103        m_fields = ck.getFields(); // Get visited fields (contains only POJO fields)
104        m_className = ck.getClassName();
105
106        // Get interfaces and super class.
107        m_interfaces = ck.getInterfaces();
108        m_superClass = ck.getSuperClass();
109
110        // Get the methods list
111        m_methods = ck.getMethods();
112
113        // Methods are not yet collected, but the structure is ready.
114        m_inners = ck.getInnerClassesAndMethods();
115
116        m_version = ck.getClassVersion();
117
118        m_alreadyManipulated = ck.isAlreadyManipulated();
119    }
120
121    /**
122     * Manipulate the given byte array.
123     * @param origin : original class.
124     * @return the manipulated class, if the class is already manipulated, the original class.
125     * @throws IOException : if an error occurs during the manipulation.
126     */
127    public byte[] manipulate(byte[] origin) throws IOException {
128        if (!m_alreadyManipulated) {
129            InputStream is2 = new ByteArrayInputStream(origin);
130            ClassReader reader = new ClassReader(is2);
131            ClassWriter writer = new ClassLoaderAwareClassWriter(ClassWriter.COMPUTE_FRAMES, m_className, m_superClass,
132                    m_classLoader);
133            ClassManipulator process = new ClassManipulator(new CheckClassAdapter(writer, false), this);
134            if (m_version >= Opcodes.V1_6) {
135                reader.accept(process, ClassReader.EXPAND_FRAMES);
136            } else {
137                reader.accept(process, 0);
138            }
139            is2.close();
140            return writer.toByteArray();
141        } else {
142            return origin;
143        }
144    }
145
146    /**
147     * Checks whether the class was already manipulated.
148     * @return {@code true} if the class was already manipulated, {@code false} otherwise
149     */
150    public boolean isAlreadyManipulated() {
151        return m_alreadyManipulated;
152    }
153
154    public static String toQualifiedName(String clazz) {
155        return clazz.replace("/", ".");
156    }
157
158    /**
159     * Compute component type manipulation metadata.
160     * @return the manipulation metadata of the class.
161     */
162    public Element getManipulationMetadata() {
163        Element elem = new Element("Manipulation", "");
164
165        elem.addAttribute(new Attribute("className", toQualifiedName(m_className)));
166
167        if (m_superClass != null) {
168            elem.addAttribute(new Attribute("super", m_superClass));
169        }
170
171        for (String m_interface : m_interfaces) {
172            Element itf = new Element("Interface", "");
173            Attribute att = new Attribute("name", m_interface);
174            itf.addAttribute(att);
175            elem.addElement(itf);
176        }
177
178        for (Map.Entry<String, String> f : m_fields.entrySet()) {
179            Element field = new Element("Field", "");
180            Attribute attName = new Attribute("name", f.getKey());
181            Attribute attType = new Attribute("type", f.getValue());
182            field.addAttribute(attName);
183            field.addAttribute(attType);
184            elem.addElement(field);
185        }
186
187        for (MethodDescriptor method : m_methods) {
188            elem.addElement(method.getElement());
189        }
190
191        for (Map.Entry<String, List<MethodDescriptor>> inner : m_inners.entrySet()) {
192            Element element = new Element("Inner", "");
193            Attribute name = new Attribute("name", extractInnerClassName(toQualifiedName(inner.getKey())));
194            element.addAttribute(name);
195
196            for (MethodDescriptor method : inner.getValue()) {
197                element.addElement(method.getElement());
198            }
199            elem.addElement(element);
200        }
201
202        return elem;
203    }
204
205    /**
206     * Extracts the inner class simple name from the qualified name. It extracts the part after the `$` character.
207     * @param clazz the qualified class name
208     * @return the simple inner class name
209     */
210    public static String extractInnerClassName(String clazz) {
211        if (!clazz.contains("$")) {
212            return clazz;
213        } else {
214            return clazz.substring(clazz.indexOf("$") +1);
215        }
216    }
217
218    public Map<String, String> getFields() {
219        return m_fields;
220    }
221
222    public List<MethodDescriptor> getMethods() {
223        return m_methods;
224    }
225
226    public Collection<String> getInnerClasses() {
227        return new ArrayList<String>(m_inners.keySet());
228    }
229
230    public int getClassVersion() {
231        return m_version;
232    }
233
234    /**
235     * Adds a method to an inner class.
236     * @param name the inner class name
237     * @param md the method descriptor to add
238     */
239    public void addMethodToInnerClass(String name, MethodDescriptor md) {
240        List<MethodDescriptor> list = m_inners.get(name);
241        if (list == null) {
242            list = new ArrayList<MethodDescriptor>();
243            m_inners.put(name, list);
244        }
245        list.add(md);
246    }
247
248    /**
249     * Analyzes the given inner class.
250     * @param inner the inner class name
251     * @param bytecode the bytecode of the inner class
252     */
253    public void prepareInnerClass(String inner, byte[] bytecode) throws IOException {
254        InputStream is = new ByteArrayInputStream(bytecode);
255        ClassReader ckReader = new ClassReader(is);
256        InnerClassChecker ck = new InnerClassChecker(inner, this);
257        ckReader.accept(ck, ClassReader.SKIP_FRAMES);
258        is.close();
259        // The metadata are collected during the visit.
260    }
261
262    /**
263     * Manipulates the inner class. If the outer class was already manipulated does not re-manipulate the inner class.
264     * We consider that the manipulation cycle of the outer and inner classes are the same.
265     * @param inner the inner class name
266     * @param bytecode input (i.e. original) class
267     * @return the manipulated class
268     * @throws IOException the class cannot be read correctly
269     */
270    public byte[] manipulateInnerClass(String inner, byte[] bytecode) throws IOException {
271        if (!m_alreadyManipulated) {
272            InputStream is1 = new ByteArrayInputStream(bytecode);
273
274            ClassReader cr = new ClassReader(is1);
275            ClassWriter cw = new ClassLoaderAwareClassWriter(ClassWriter.COMPUTE_FRAMES, inner, null, m_classLoader);
276            InnerClassAdapter adapter = new InnerClassAdapter(inner, cw, m_className, this);
277            if (m_version >= Opcodes.V1_6) {
278                cr.accept(adapter, ClassReader.EXPAND_FRAMES);
279            } else {
280                cr.accept(adapter, 0);
281            }
282            is1.close();
283
284            return cw.toByteArray();
285        } else {
286            // Return the unchanged inner class
287            return bytecode;
288        }
289    }
290
291    public List<MethodDescriptor> getMethodsFromInnerClass(String innerClassInternalName) {
292        return m_inners.get(innerClassInternalName);
293    }
294
295    public Map<String, List<MethodDescriptor>> getInnerClassesAndMethods() {
296        // Transform the map to use the simple name of the inner classes.
297        Map<String, List<MethodDescriptor>> map = new HashMap<String, List<MethodDescriptor>>();
298        for (Map.Entry<String, List<MethodDescriptor>> entry : m_inners.entrySet()) {
299            String name = extractInnerClassName(toQualifiedName(entry.getKey()));
300            map.put(name, entry.getValue());
301        }
302        return map;
303    }
