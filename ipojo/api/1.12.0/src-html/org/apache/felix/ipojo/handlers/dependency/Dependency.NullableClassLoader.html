135    public Dependency(DependencyHandler handler, String field, Class spec, Filter filter, boolean isOptional,
136                      boolean isAggregate, boolean nullable, boolean isProxy, String identity, BundleContext context,
137                      int policy, Comparator cmp, String defaultImplementation, String exception) {
138        super(spec, isAggregate, isOptional, filter, cmp, policy, context, handler, handler.getInstanceManager());
139        m_handler = handler;
140        m_field = field;
141        m_isProxy = isProxy;
142
143        if (field != null) {
144            m_usage = new ServiceUsage();
145        } else {
146            m_usage = null;
147        }
148
149        m_supportNullable = nullable;
150        m_di = defaultImplementation;
151        m_exception = exception;
152
153        if (identity == null) {
154            if (spec != null) {
155                m_id = spec.getName();
156            }
157        } else {
158            m_id = identity;
159        }
160
161        // Else wait the setSpecification call.
162    }

185     * Add a callback to the dependency.
186     * This method is called during the configuration.
187     *
188     * @param callback : callback to add
189     */
190    protected void addDependencyCallback(DependencyCallback callback) {
191        if (m_callbacks == null) {
192            m_callbacks = new DependencyCallback[]{callback};
193        } else {
194            DependencyCallback[] newCallbacks = new DependencyCallback[m_callbacks.length + 1];
195            System.arraycopy(m_callbacks, 0, newCallbacks, 0, m_callbacks.length);
196            newCallbacks[m_callbacks.length] = callback;
197            m_callbacks = newCallbacks;
198        }
199    }

247    /**
248     * Call the bind method.
249     *
250     * @param pojo : pojo instance on which calling the bind method.
251     */
252    protected void onObjectCreation(Object pojo) {
253
254        ServiceReference[] refs;
255        try {
256            acquireWriteLockIfNotHeld();
257            if (!m_isStarted) {
258                return;
259            }
260
261            // We are notified of an instance creation, we have to freeze when the static policy is used
262            if (getBindingPolicy() == STATIC_BINDING_POLICY) {
263                m_isFrozen = true;
264            }
265
266            // Check optional case : nullable object case : do not call bind on nullable object
267            if (isOptional() && getSize() == 0) {
268                return;
269            }
270
271            refs = getServiceReferences(); // Stack confinement.
272        } finally {
273            releaseWriteLockIfHeld();
274        }
275
276        // This is a pretty strange case, but we don't have any service.
277        // This may happen during refresh.
278        // So we just return.
279        if (refs == null) {
280            return;
281        }
282
283        // Call bind callback.
284        for (int j = 0; m_callbacks != null && j < m_callbacks.length; j++) { // The array is constant.
285            if (m_callbacks[j].getMethodType() == DependencyCallback.BIND) {
286                if (isAggregate()) {
287                    for (ServiceReference ref : refs) {
288                        Object svc = getService(ref);
289                        if (svc != null) {
290                            invokeCallback(m_callbacks[j], ref, svc, pojo);
291                        } else {
292                            // The service left already, or the service object cannot be created.
293                            // We consider it as a departure.
294                            m_serviceReferenceManager.removedService(ref, null);
295                        }
296                    }
297                } else {
298                    // Take the first reference.
299                    Object svc = getService(refs[0]);
300                    if (svc != null) {
301                        invokeCallback(m_callbacks[j], refs[0], svc, pojo);
302                    } else {
303                        // The service left already, or the service object cannot be created.
304                        // We consider it as a departure.
305                        m_serviceReferenceManager.removedService(refs[0], null);
306                    }
307                }
308            }
309        }
310    }

312    /**
313     * Call unbind callback method.
314     *
315     * @param ref : reference to send (if accepted) to the method
316     */
317    private void callUnbindMethod(ServiceReference ref) {
318        if (m_handler.getInstanceManager().getState() > InstanceManager.STOPPED && m_handler.getInstanceManager().getPojoObjects() != null) {
319            for (int i = 0; m_callbacks != null && i < m_callbacks.length; i++) {
320                if (m_callbacks[i].getMethodType() == DependencyCallback.UNBIND) {
321                    invokeCallback(m_callbacks[i], ref, getService(ref, false), null); // Call on each created pojo objects.
322                }
323            }
324        }
325    }

327    /**
328     * Helper method calling the given callback.
329     *
330     * @param callback  : callback to call.
331     * @param ref       : service reference.
332     * @param svcObject : the service object
333     * @param pojo      : pojo on which calling the callback, if null call on each created pojo objects.
334     */
335    private void invokeCallback(DependencyCallback callback, ServiceReference ref, Object svcObject, Object pojo) {
336        try {
337            if (pojo == null) {
338                callback.call(ref, svcObject);
339            } else {
340                callback.callOnInstance(pojo, ref, svcObject);
341            }
342        } catch (NoSuchMethodException e) {
343            m_handler.error("The method " + callback.getMethodName() + " does not exist in the implementation class " + m_handler.getInstanceManager().getClassName(), e);
344            m_handler.getInstanceManager().stop();
345        } catch (IllegalAccessException e) {
346            m_handler.error("The method " + callback.getMethodName() + " is not accessible in the implementation class " + m_handler.getInstanceManager().getClassName(), e);
347            m_handler.getInstanceManager().stop();
348        } catch (InvocationTargetException e) {
349            m_handler.error("The method " + callback.getMethodName() + " in the implementation class " + m_handler.getInstanceManager().getClassName() + " throws an exception : " + e.getTargetException().getMessage(), e.getTargetException());
350            m_handler.getInstanceManager().stop();
351        }
352
353    }

355    /**
356     * Call 'modify' method with the service reference in parameter (if accepted).
357     *
358     * @param ref : the service reference of the modified service
359     */
360    private void callModifyMethod(ServiceReference ref) {
361        if (m_handler.getInstanceManager().getState() > InstanceManager.STOPPED && m_handler.getInstanceManager().getPojoObjects() != null) {
362            for (int i = 0; m_callbacks != null && i < m_callbacks.length; i++) {
363                if (m_callbacks[i].getMethodType() == DependencyCallback.MODIFIED) {
364                    invokeCallback(m_callbacks[i], ref, getService(ref), null); // Call on each created pojo objects.
365                }
366            }
367        }
368    }

370    /**
371     * Call  method with the service reference in parameter (if accepted).
372     *
373     * @param ref : the service reference of the new service
374     */
375    private void callBindMethod(ServiceReference ref) {
376        // call bind method :
377        // if (m_handler.getInstanceManager().getState() == InstanceManager.VALID) {
378        if (m_handler.getInstanceManager().getState() > InstanceManager.STOPPED && m_handler.getInstanceManager().getPojoObjects() != null) {
379            for (int i = 0; m_callbacks != null && i < m_callbacks.length; i++) {
380                if (m_callbacks[i].getMethodType() == DependencyCallback.BIND) {
381                    Object svc = getService(ref);
382                    if (svc != null) {
383                        invokeCallback(m_callbacks[i], ref, svc, null);
384                    } else {
385                        // We can't get the service object (https://issues.apache.org/jira/browse/FELIX-3896).
386                        // This is probably because the service is leaving.
387                        // We consider it as a departure.
388                        m_serviceReferenceManager.removedService(ref, null);
389                    }
390                }
391            }
392        }
393    }

395    private RuntimeException createExceptionToThrow() {
396        final String message = "No service available for " + DependencyHandler.getDependencyIdentifier(this);
397        if (m_exception == null) {
398            // Should never happen, but let's see.
399            return new RuntimeException(message);
400        }
401        try {
402            Class<RuntimeException> exceptionClass = (Class<RuntimeException>) getBundleContext()
403                    .getBundle().loadClass(m_exception);
404            // Check constructor
405            final Constructor<RuntimeException> constructor = exceptionClass.getConstructor(new Class[]{String.class});
406            if (constructor != null) {
407                return constructor.newInstance(message);
408            } else {
409                return exceptionClass.newInstance();
410            }
411        } catch (Exception e) {
412            m_handler.getLogger().log(Log.ERROR, "Cannot create the exception object for dependency " +
413                    DependencyHandler.getDependencyIdentifier(this) + " : " + e.getMessage(), e);
414        }
415
416        return new RuntimeException(message);
417    }

419    private Object createNullableObject() {
420        // To load the proxy we use the POJO class loader. Indeed, this classloader imports iPOJO (so can access to Nullable) and has
421        // access to the service specification.
422        if ( ! getSpecification().isInterface()) {
423            getHandler().getLogger().log(Log.INFO, "Cannot create the nullable object for " + getSpecification()
424                    .getName() + " - the specification is not an interface");
425            return null;
426        }
427
428        try {
429            ClassLoader cl = new NullableClassLoader(
430                    getHandler().getInstanceManager().getClazz().getClassLoader(),
431                    getSpecification().getClassLoader());
432
433            m_nullable =
434                    Proxy.newProxyInstance(cl, new Class[]{
435                            getSpecification(), Nullable.class}, new NullableObject()); // NOPMD
436
437        } catch (NoClassDefFoundError e) {
438            // A NoClassDefFoundError is thrown if the specification uses a class not accessible by the actual instance.
439            // It generally comes from a missing import.
440            throw new IllegalStateException("Cannot create the Nullable object, a referenced class cannot be loaded", e);
441        } catch (Throwable e) { // Catch any other exception that can occurs
442            throw new IllegalStateException("Cannot create the Nullable object, an unexpected error occurs", e);
443        }
444
445        return m_nullable;
446    }

448    /**
449     * Start the dependency.
450     */
451    public void start() {
452
453        if (isOptional() && !isAggregate()) {
454            if (m_di == null && m_exception == null) {
455                // If nullable are supported, create the nullable object.
456                if (m_supportNullable) {
457                    createNullableObject();
458                }
459            } else if (m_di != null) {
460                // Create the default-implementation object.
461                try {
462                    Class clazz = getHandler().getInstanceManager().getContext().getBundle().loadClass(m_di);
463                    m_nullable = clazz.newInstance();
464                } catch (IllegalAccessException e) {
465                    throw new IllegalStateException("Cannot load the default-implementation " + m_di, e);
466                } catch (InstantiationException e) {
467                    throw new IllegalStateException("Cannot load the default-implementation " + m_di, e);
468                } catch (ClassNotFoundException e) {
469                    throw new IllegalStateException("Cannot load the default-implementation " + m_di, e);
470                } catch (Throwable e) { // Catch any other exception
471                    throw new IllegalStateException("Cannot load the default-implementation (unexpected exception) " + m_di, e);
472                }
473            }
474        }
475
476        if (m_isProxy) {
477            if (isAggregate()) {
478                m_proxyObject = new ServiceCollection(this);
479            } else {
480                // Can we really proxy ? We can proxy only interfaces.
481                if (getSpecification().isInterface()) {
482                    String type = getHandler().getInstanceManager().getContext().getProperty(DependencyHandler.PROXY_TYPE_PROPERTY);
483
484                    // If it's null we should check on the System directly, Felix delegates to it,
485                    // but not other frameworks
486                    if (type == null) {
487                        type = System.getProperty(DependencyHandler.PROXY_TYPE_PROPERTY);
488                    }
489
490                    if (type == null || type.equals(DependencyHandler.SMART_PROXY)) {
491                        SmartProxyFactory proxyFactory = new SmartProxyFactory(this.getClass().getClassLoader());
492                        m_proxyObject = proxyFactory.getProxy(this);
493                    } else {
494                        DynamicProxyFactory proxyFactory = new DynamicProxyFactory();
495                        m_proxyObject = proxyFactory.getProxy(getSpecification());
496                    }
497                } else {
498                    m_handler.warn("Cannot create a proxy for a service dependency which is not an interface " +
499                            "- disabling proxy for " + getId());
500                }
501            }
502        }
503
504        super.start();
505
506        // Once the dependency is started, access to fields must be protected.
507        acquireWriteLockIfNotHeld();
508        if (getBindingPolicy() == STATIC_BINDING_POLICY && m_handler.getInstanceManager().getPojoObjects() != null) {
509            m_isFrozen = true;
510        }
511        m_isStarted = true;
512        releaseWriteLockIfHeld();
513    }

569     * The dependency has been reconfigured.
570     * Call unbind method and then bind methods. If the dependency cache is not reset,
571     * the thread continues to get older services.
572     *
573     * @param departs  : no more matching services.
574     * @param arrivals : new services
575     * @see org.apache.felix.ipojo.util.DependencyModel#onDependencyReconfiguration(org.osgi.framework.ServiceReference[], org.osgi.framework.ServiceReference[])

577    public void onDependencyReconfiguration(ServiceReference[] departs, ServiceReference[] arrivals) {
578        for (int i = 0; departs != null && i < departs.length; i++) {
579            callUnbindMethod(departs[i]);
580        }
581
582        for (int i = 0; arrivals != null && i < arrivals.length; i++) {
583            callBindMethod(arrivals[i]);
584        }
585    }

624    /**
625     * Called by the proxy to get  service objects to delegate a method.
626     * On aggregate dependencies, it returns a list.
627     *
628     * @return a service object or a nullable/default-implementation object.
629     *         For aggregate dependencies it returns a list or an empty list.
630     */
631    public Object getService() {
632        // Check that we're in proxy mode.
633        if (!m_isProxy) {
634            throw new IllegalStateException("The dependency has not enabled the `proxy` mode.");
635        }
636
637        Usage usage = (Usage) m_usage.get();
638        if (usage.m_stack == 0) { // uninitialized usage.
639            if (usage.m_componentStack > 0) {
640                // We comes from the component who didn't touch the service.
641                // So we initialize the usage.
642                createServiceObject(usage);
643                usage.inc(); // Start the caching, so set the stack level to 1
644                m_usage.set(usage); // Required by Dalvik.
645                if (isAggregate()) {
646                    Object obj = usage.m_object;
647                    if (obj instanceof Set) {
648                        List<Object> list = new ArrayList<Object>();
649                        list.addAll((Set) obj);
650                        return list;
651                    } else {
652                        // We already have a list
653                        return obj;
654                    }
655                } else {
656                    return usage.m_object;
657                }
658            } else {
659                // External access => Immediate get.
660                if (isAggregate()) {
661                    ServiceReference[] refs = getServiceReferences();
662                    if (refs == null) {
663                        return new ArrayList(0); // Create an empty list.
664                    } else {
665                        List<Object> objs = new ArrayList<Object>(refs.length);
666                        for (ServiceReference ref : refs) {
667                            objs.add(getService(ref));
668                        }
669                        return objs;
670                    }
671                } else { // Scalar dependency.
672                    ServiceReference ref = getServiceReference();
673                    if (ref != null) {
674                        return getService(ref);
675                    } else {
676                        // No service available.
677                        // TODO Decide what we have to do.
678                        throw new RuntimeException("Service " + getSpecification() + " unavailable");
679                    }
680                }
681            }
682        } else {
683            // Use the copy.
684            // if the copy is a set, transform to a list
685            if (isAggregate()) {
686                Object obj = usage.m_object;
687                if (obj instanceof Set) {
688                    List<Object> list = new ArrayList<Object>();
689                    list.addAll((Set) obj);
690                    return list;
691                } else {
692                    // We already have a list
693                    return obj;
694                }
695            } else {
696                return usage.m_object;
697            }
698
699        }
700    }

702    /**
703     * This method is called by the replaced code in the component
704     * implementation class. Construct the service object list is necessary.
705     *
706     * @param pojo      : POJO object.
707     * @param fieldName : field
708     * @param value     : last value.
709     * @return the service object or a nullable / default implementation if defined.
710     * @see org.apache.felix.ipojo.FieldInterceptor#onGet(java.lang.Object, java.lang.String, java.lang.Object)
711     */
712    public Object onGet(Object pojo, String fieldName, Object value) {
713
714        // Initialize the thread local object is not already touched.
715        Usage usage = m_usage.get();
716        if (usage.m_stack == 0) { // uninitialized usage.
717            createServiceObject(usage);
718            usage.inc(); // Start the caching, so set the stack level to 1
719            m_usage.set(usage); // Required by Dalvik
720        }
721        if (!m_isProxy) {
722            return usage.m_object;
723        } else {
724            return m_proxyObject;
725        }
726
727    }

729    /**
730     * Creates the object to store in the given Thread Local.
731     * This object will be injected inside the POJO field.
732     *
733     * @param usage : Thread Local to populate.
734     */
735    private void createServiceObject(Usage usage) {
736        ServiceReference[] refs = getServiceReferences();
737
738        // manage timeout
739        if (refs == null) {
740            waitForServiceUntilTimeout();
741        }
742
743        refs = getServiceReferences();
744
745        if (!isAggregate()) {
746            if (refs == null) {
747                if (m_exception != null) {
748                    // Throw the exception.
749                    throw createExceptionToThrow();
750                }
751
752                if (m_nullable == null && m_supportNullable) {
753                    m_handler.warn("[" + m_handler.getInstanceManager().getInstanceName() + "] The dependency is not optional, however no service object can be injected in " + m_field + " -> " + getSpecification().getName());
754                    createNullableObject();
755                }
756                usage.m_object = m_nullable; // Add null if the Nullable pattern is disabled.
757            } else {
758                ServiceReference ref = getServiceReference();
759                usage.m_object = getService(ref);
760            }
761        } else {
762            switch(m_type) {
763                case ARRAY:
764                    try {
765                        if (refs == null) {
766                            usage.m_object = (Object[]) Array.newInstance(getSpecification(), 0); // Create an empty array.
767                        } else {
768                            //  Use a reflective construction to avoid class cast exception. This method allows setting the component type.
769                            Object[] objs = (Object[]) Array.newInstance(getSpecification(), refs.length);
770                            for (int i = 0; i < refs.length; i++) {
771                                ServiceReference ref = refs[i];
772                                objs[i] = getService(ref);
773                            }
774                            usage.m_object = objs;
775                        }
776                    } catch (ArrayStoreException e) {
777                        throw new RuntimeException("Cannot create the array - Check that the bundle can access the service interface", e);
778                    }
779                    break;
780                case LIST:
781                    if (refs == null) {
782                        usage.m_object = Collections.emptyList();
783                    } else {
784                        // Use a list to store service objects
785                        List<Object> objs = new ArrayList<Object>(refs.length);
786                        for (ServiceReference ref : refs) {
787                            objs.add(getService(ref));
788                        }
789                        usage.m_object = objs;
790                    }
791                    break;
792                case SET:
793                    if (refs == null) {
794                        usage.m_object = Collections.emptySet();
795                    } else {
796                        // Use a vector to store service objects
797                        Set<Object> objs = new HashSet<Object>(refs.length);
798                        for (ServiceReference ref : refs) {
799                            objs.add(getService(ref));
800                        }
801                        usage.m_object = objs;
802                    }
803                    break;
804                case VECTOR:
805                    if (refs == null) {
806                        usage.m_object = new Vector(0); // Create an empty vector.
807                    } else {
808                        // Use a vector to store service objects
809                        Vector<Object> objs = new Vector<Object>(refs.length);
810                        for (ServiceReference ref : refs) {
811                            objs.add(getService(ref));
812                        }
813                        usage.m_object = objs;
814                    }
815                    break;
816            }
817        }
818    }

820    /**
821     * Waits a service providers. The wait stops when the timeout is reached.
822     */
823    private void waitForServiceUntilTimeout() {
824        // Begin to wait ...
825        long enter = System.currentTimeMillis();
826        boolean exhausted = false;
827
828        // We used a synchronized block here because we must hold the monitor lock during the 'wait'
829        synchronized (this) {
830            while (getServiceReference() == null && !exhausted) {
831                try {
832                    wait(1);
833                } catch (InterruptedException e) {
834                    // We was interrupted ....
835                } finally {
836                    long end = System.currentTimeMillis();
837                    exhausted = (end - enter) > m_timeout;
838                }
839            }
840        }
841        // When this method exit, the check will be done...
842    }

901    /**
902     * A POJO method is finished.
903     *
904     * @param pojo   : POJO object.
905     * @param method : Method object.
906     * @see org.apache.felix.ipojo.MethodInterceptor#onFinally(java.lang.Object, java.lang.reflect.Member)
907     */
908    public void onFinally(Object pojo, Member method) {
909        if (m_usage != null) {
910            Usage usage = m_usage.get();
911            usage.decComponentStack();
912            if (usage.m_stack > 0) {
913                if (usage.dec()) {
914                    // Exit the method flow => Release all objects
915                    usage.clear();
916                    // Also remove the thread local object.
917                    m_usage.remove();
918                }
919            }
920        }

1075    /**
1076     * Creates smart proxy object for proxied scalar dependencies.
1077     */
1078    private class SmartProxyFactory extends ClassLoader {
1079
1080        /**
1081         * Handler classloader, used to load the temporal dependency class.
1082         */
1083        private ClassLoader m_handlerCL;
1084
1085        /**
1086         * Creates the proxy classloader.
1087         *
1088         * @param parent the handler classloader.
1089         */
1090        public SmartProxyFactory(ClassLoader parent) {
1091            super(getHandler().getInstanceManager().getFactory().getBundleClassLoader());
1092            m_handlerCL = parent;
1093        }
1094
1095        /**
1096         * Loads a proxy class generated for the given (interface) class.
1097         *
1098         * @param clazz the service specification to proxy
1099         * @return the Class object of the proxy.
1100         */
1101        protected Class getProxyClass(Class clazz) {
1102            byte[] clz = ProxyGenerator.dumpProxy(clazz); // Generate the proxy.
1103            // Turn around the VM changes (FELIX-2716) about java.* classes.
1104            String cn = clazz.getName();
1105            if (cn.startsWith("java.")) {
1106                cn = "$" + cn;
1107            }
1108            return defineClass(cn + "$$Proxy", clz, 0, clz.length);
1109        }
1110
1111        /**
1112         * Create a proxy object for the given specification. The proxy
1113         * uses the given dependency to get the service object.
1114         *
1115         * @param dep  the dependency used to get the service
1116         * @return the proxy object.
1117         */
1118        public Object getProxy(Dependency dep) {
1119            try {
1120                Class clazz = getProxyClass(getSpecification());
1121                Constructor constructor = clazz.getConstructor(
1122                        new Class[]{clazz.getClassLoader().loadClass(Dependency.class.getName())});
1123                return constructor.newInstance(new Object[]{dep});
1124            } catch (Throwable e) {
1125                m_handler.error("Cannot create the proxy object", e);
1126                m_handler.getInstanceManager().stop();
1127                return null;
1128            }
1129        }
1130
1131        /**
1132         * Loads the given class.
1133         * This method uses the classloader of the specification class
1134         * or the handler class loader.
1135         *
1136         * @param name the class name
1137         * @return the class object
1138         * @throws ClassNotFoundException if the class is not found by the two classloaders.
1139         * @see java.lang.ClassLoader#loadClass(java.lang.String)
1140         */
1141        public Class loadClass(String name) throws ClassNotFoundException {
1142            try {
1143                return getHandler().getInstanceManager().getContext().getBundle().loadClass(name);
1144            } catch (ClassNotFoundException e) {
1145                return m_handlerCL.loadClass(name);
1146            }
1147        }
1148    }

1150    /**
1151     * Creates java dynamic proxy object for proxied scalar dependencies.
1152     */
1153    private class DynamicProxyFactory implements InvocationHandler {
1154
1155        /**
1156         * HashCode method.
1157         */
1158        private Method m_hashCodeMethod;
1159        /**
1160         * Equals method.
1161         */
1162        private Method m_equalsMethod;
1163        /**
1164         * toStirng method.
1165         */
1166        private Method m_toStringMethod;
1167
1168        /**
1169         * Creates a DynamicProxyFactory.
1170         */
1171        public DynamicProxyFactory() {
1172            try {
1173                m_hashCodeMethod = Object.class.getMethod("hashCode", null);
1174                m_equalsMethod = Object.class
1175                        .getMethod("equals", new Class[]{Object.class});
1176                m_toStringMethod = Object.class.getMethod("toString", null);
1177            } catch (NoSuchMethodException e) {
1178                throw new NoSuchMethodError(e.getMessage());
1179            }
1180        }
1181
1182        /**
1183         * Creates a proxy object for the given specification. The proxy
1184         * uses the given dependency to get the service object.
1185         *
1186         * @param spec the service specification (interface)
1187         * @return the proxy object.
1188         */
1189        public Object getProxy(Class spec) {
1190            return java.lang.reflect.Proxy.newProxyInstance(
1191                    getHandler().getInstanceManager().getClazz().getClassLoader(),
1192                    new Class[]{spec},
1193                    this);
1194        }
1195
1196        /**
1197         * Invocation Handler delegating invocation on the
1198         * service object.
1199         *
1200         * @param proxy  the proxy object
1201         * @param method the method
1202         * @param args   the arguments
1203         * @return a proxy object.
1204         * @throws Exception if the invocation throws an exception
1205         * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
1206         */
1207        public Object invoke(Object proxy, Method method, Object[] args) throws Exception {
1208            Object svc = getService();
1209            Class declaringClass = method.getDeclaringClass();
1210            if (declaringClass == Object.class) {
1211                if (method.equals(m_hashCodeMethod)) {
1212                    return this.hashCode();
1213                } else if (method.equals(m_equalsMethod)) {
1214                    return proxy == args[0] ? Boolean.TRUE : Boolean.FALSE;
1215                } else if (method.equals(m_toStringMethod)) {
1216                    return this.toString();
1217                } else {
1218                    throw new InternalError(
1219                            "Unexpected Object method dispatched: " + method);
1220                }
1221            }
1222
1223            return method.invoke(svc, args);
1224        }
1225
1226    }
