020package org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util;
021
022import java.util.ArrayList;
023import java.util.Arrays;
024import java.util.Dictionary;
025import java.util.List;
026import java.util.Map;
027
028import org.apache.felix.ipojo.manipulation.MethodCreator;
029import org.objectweb.asm.Type;
030import org.objectweb.asm.tree.MethodNode;
031
032/**
033 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
034 */
035public class Names {
036
037    /**
038     * Excluded types when searching for a specification interface in method's arguments.
039     */
040    private static List<Type> EXCLUSIONS = new ArrayList<Type>();
041
042    static {
043        EXCLUSIONS.add(Type.getType(Map.class));
044        EXCLUSIONS.add(Type.getType(Dictionary.class));
045        EXCLUSIONS.add(Type.getType("Lorg/osgi/framework/ServiceReference;"));
046    }
047
048    /**
049     * Computes the real method name. This method is useful when the annotation is collected on an manipulated method
050     * (prefixed by <code>__M_</code>). This method just removes the prefix if found.
051     * @param name the collected method name
052     * @return the effective method name, can be the collected method name if the method name does not start with
053     * the prefix.
054     */
055    public static String computeEffectiveMethodName(String name) {
056        if (name != null && name.startsWith(MethodCreator.PREFIX)) {
057            return name.substring(MethodCreator.PREFIX.length());
058        } else {
059            return name;
060        }
061    }
062
063    /**
064     * Extract an identifier from the given method name.
065     * It removes some pre-defined prefixes ({@literal bind}, {@literal unbind},
066     * {@literal set}, {@literal unset}, {@literal modified}).
067     *
068     * @param method method's name
069     * @return the method's identifier
070     */
071    public static String getMethodIdentifier(final MethodNode method) {
072
073        String effectiveName = computeEffectiveMethodName(method.name);
074
075        if (effectiveName.startsWith("bind")) {
076            return effectiveName.substring("bind".length());
077        }
078
079        if (effectiveName.startsWith("set")) {
080            return effectiveName.substring("set".length());
081        }
082
083        if (effectiveName.startsWith("unbind")) {
084            return effectiveName.substring("unbind".length());
085        }
086
087        if (effectiveName.startsWith("unset")) {
088            return effectiveName.substring("unset".length());
089        }
090
091        if (effectiveName.startsWith("modified")) {
092            return effectiveName.substring("modified".length());
093        }
094
095        if (effectiveName.startsWith("add")) {
096            return effectiveName.substring("add".length());
097        }
098
099        if (effectiveName.startsWith("remove")) {
100            return effectiveName.substring("remove".length());
101        }
102
103        // Try to discover the specification's type from method's parameters' type
104        Type[] arguments = Type.getArgumentTypes(method.desc);
105        return findSpecification(Arrays.asList(arguments));
106
107    }
108
109    /**
110     * Find the first type that was not excluded and consider it as the specification
111     * @param types method parameter's type
112     * @return the first non-excluded type or {@literal null} if no specification can be found
113     */
114    private static String findSpecification(final List<Type> types) {
115
116        // Find first non-excluded specification
117        // May return null if no specification is provided (likely a user error)
118        for (Type type : types) {
119            if (!EXCLUSIONS.contains(type)) {
120                return type.getClassName();
121            }
122        }
123        return null;
124    }
125
126    /**
127     * Check if the given annotation descriptor is an iPOJO custom annotation.
128     * A valid iPOJO custom annotation must contains 'ipojo' or 'handler' in its qualified name.
129     * @param desc annotation descriptor
130     * @return {@literal true} if the given descriptor is an iPOJO custom annotation
131     */
132    public static boolean isCustomAnnotation(final String desc) {
133        String lowerCase = desc.toLowerCase();
134        return lowerCase.contains("ipojo") || lowerCase.contains("handler");
135    }
136}
