020package org.apache.felix.ipojo.extender.internal.linker;
021
022import static java.lang.String.format;
023
024import org.apache.felix.ipojo.*;
025import org.apache.felix.ipojo.extender.ExtensionDeclaration;
026import org.apache.felix.ipojo.extender.InstanceDeclaration;
027import org.apache.felix.ipojo.extender.TypeDeclaration;
028import org.apache.felix.ipojo.extender.builder.FactoryBuilderException;
029import org.apache.felix.ipojo.extender.internal.Lifecycle;
030import org.apache.felix.ipojo.extender.internal.ReferenceableCallable;
031import org.apache.felix.ipojo.extender.queue.QueueService;
032import org.osgi.framework.Bundle;
033import org.osgi.framework.BundleContext;
034import org.osgi.framework.InvalidSyntaxException;
035import org.osgi.framework.ServiceReference;
036import org.osgi.util.tracker.ServiceTracker;
037import org.osgi.util.tracker.ServiceTrackerCustomizer;
038
039import java.util.concurrent.ExecutionException;
040import java.util.concurrent.Future;
041
042/**
043 * This class is responsible to create the factory for a given type declaration. It also instructs the factories to
044 * create
045 * the instance for each instance declaration targeting the managed factory.
046 */
047public class ManagedType implements FactoryStateListener, Lifecycle {
048    /**
049     * The bundle context
050     */
051    private final BundleContext m_bundleContext;
052    /**
053     * The queue service used for the creation.
054     */
055    private final QueueService m_queueService;
056    /**
057     * The type declaration that we have to handle.
058     */
059    private final TypeDeclaration m_declaration;
060    /**
061     * The service tracker tracking {@link ExtensionDeclaration} services.
062     */
063    private ServiceTracker m_extensionTracker;
064
065    /**
066     * The service tracker tracking the {@link InstanceDeclaration} services.
067     */
068    private ServiceTracker m_instanceTracker;
069
070    /**
071     * The job used to instantiate the factory.
072     */
073    private Future<IPojoFactory> m_future;
074
075    /**
076     * If the Managed Type cannot be initializes, sets this flag to true and no links will be created.
077     */
078    private boolean m_frozen;
079
080    /**
081     * Constructs a Managed Type object for the given type declaration.
082     *
083     * @param bundleContext the bundle context
084     * @param queueService  the queue service
085     * @param declaration   the declaration
086     */
087    public ManagedType(BundleContext bundleContext, QueueService queueService, TypeDeclaration declaration) {
088        m_bundleContext = bundleContext;
089        m_queueService = queueService;
090        m_declaration = declaration;
091        try {
092            initExtensionTracker();
093            initInstanceTracker();
094        } catch (InvalidSyntaxException e) {
095            // Error during filter creation, freeze the declaration and add a meaningful message
096            m_frozen = true;
097            m_declaration.unbind("Filter creation error", e);
098        }
099    }
100
101    /**
102     * Initializes the extension declaration tracker.
103     *
104     * @throws InvalidSyntaxException cannot happen
105     */
106    private void initExtensionTracker() throws InvalidSyntaxException {
107        String filter = format(
108                "(&(objectclass=%s)(%s=%s))",
109                ExtensionDeclaration.class.getName(),
110                ExtensionDeclaration.EXTENSION_NAME_PROPERTY,
111                m_declaration.getExtension()
112        );
113        m_extensionTracker = new ServiceTracker(m_bundleContext, m_bundleContext.createFilter(filter), new ExtensionSupport());
114    }
115
116    /**
117     * Initializes the instance declaration tracker.
118     *
119     * @throws InvalidSyntaxException cannot happen
120     */
121    private void initInstanceTracker() throws InvalidSyntaxException {
122
123        String filter;
124        String version = m_declaration.getComponentVersion();
125        if (version != null) {
126            // Track instance for:
127            // * this component AND
128            // * this component's version OR no version
129            filter = format(
130                    "(&(objectClass=%s)(%s=%s)(|(%s=%s)(!(%s=*))))",
131                    InstanceDeclaration.class.getName(),
132                    InstanceDeclaration.COMPONENT_NAME_PROPERTY,
133                    m_declaration.getComponentName(),
134                    InstanceDeclaration.COMPONENT_VERSION_PROPERTY,
135                    version,
136                    InstanceDeclaration.COMPONENT_VERSION_PROPERTY
137            );
138        } else {
139            // Track instance for:
140            // * this component AND no version
141            filter = format(
142                    "(&(objectClass=%s)(%s=%s)(!(%s=*)))",
143                    InstanceDeclaration.class.getName(),
144                    InstanceDeclaration.COMPONENT_NAME_PROPERTY,
145                    m_declaration.getComponentName(),
146                    InstanceDeclaration.COMPONENT_VERSION_PROPERTY
147            );
148        }
149        m_instanceTracker = new ServiceTracker(m_bundleContext, m_bundleContext.createFilter(filter), new InstanceSupport());
150    }
151
152    /**
153     * Starting the management.
154     * We open only the extension tracker.
155     */
156    public void start() {
157        if (!m_frozen) {
158            m_extensionTracker.open(true);
159        }
160    }
161
162    /**
163     * Stopping the management.
164     */
165    public void stop() {
166        m_instanceTracker.close();
167        m_extensionTracker.close();
168    }
169
170    /**
171     * The factory we have built has a state in his change.
172     *
173     * @param factory  the changing factory
174     * @param newState the new factory state
175     */
176    public void stateChanged(Factory factory, int newState) {
177        if (Factory.VALID == newState) {
178            // Start tracking instances
179            m_instanceTracker.open(true);
180        } else {
181            // Un-track all instances
182            m_instanceTracker.close();
183        }
184    }
185
186    /**
187     * The service tracker customizer for extension declaration.
188     * It submits a factory building job when an extension matching our type declaration is found.
189     */
190    private class ExtensionSupport implements ServiceTrackerCustomizer {
191        public Object addingService(ServiceReference reference) {
192            final Object service = m_bundleContext.getService(reference);
193            if (service instanceof ExtensionDeclaration) {
194                m_future = m_queueService.submit(new ReferenceableCallable<IPojoFactory>(reference.getBundle()) {
195
196                    /**
197                     * The factory creation job.
198                     * @return the IPojoFactory
199                     * @throws Exception the factory cannot be created
200                     */
201                    public IPojoFactory call() throws Exception {
202                        ExtensionDeclaration declaration = (ExtensionDeclaration) service;
203                        try {
204                            // Build and start the factory instance
205                            IPojoFactory factory = declaration.getFactoryBuilder().build(m_bundleContext, m_declaration.getComponentMetadata());
206                            factory.addFactoryStateListener(ManagedType.this);
207                            factory.start();
208
209                            // Change the status
210                            m_declaration.bind();
211
212                            return factory;
213                        } catch (FactoryBuilderException e) {
214                            m_declaration.unbind(format("Cannot build '%s' factory instance", m_declaration.getExtension()), e);
215                        } catch (Throwable t) {
216                            m_declaration.unbind(format("Error during '%s' factory instance creation", m_declaration.getExtension()), t);
217                        }
218
219                        return null;
220                    }
221                });
222                // Return something, otherwise, ServiceTracker think that we're not interested
223                // in this service and never call us back on disposal.
224                return service;
225            }
226
227            return null;
228        }
229
230        public void modifiedService(ServiceReference reference, Object o) {
231        }
232
233        public void removedService(ServiceReference reference, Object o) {
234
235            ExtensionDeclaration extensionDeclaration = (ExtensionDeclaration) o;
236            // Then stop the factory
237            try {
238                IPojoFactory factory = m_future.get();
239                // It is possible that the factory couldn't be created
240                if (factory != null) {
241                    factory.removeFactoryStateListener(ManagedType.this);
242                    factory.dispose();
243                    m_declaration.unbind(format("Extension '%s' is missing",
244                                                extensionDeclaration.getExtensionName()));
245                }
246            } catch (InterruptedException e) {
247                m_declaration.unbind("Could not create Factory", e);
248            } catch (ExecutionException e) {
249                m_declaration.unbind("Factory creation throw an Exception", e);
250            }
251            m_future = null;
252        }
253    }
254
255    private class InstanceSupport implements ServiceTrackerCustomizer {
256        public Object addingService(final ServiceReference reference) {
257            Object service = m_bundleContext.getService(reference);
258            if (service instanceof InstanceDeclaration) {
259                final InstanceDeclaration instanceDeclaration = (InstanceDeclaration) service;
260
261                // Check that instance is not already bound
262                if (instanceDeclaration.getStatus().isBound()) {
263                    return null;
264                }
265
266                // Handle visibility (private/public factories)
267                if (!m_declaration.isPublic()) {
268                    if (!reference.getBundle().equals(m_bundleContext.getBundle())) {
269                        Bundle origin = m_bundleContext.getBundle();
270                        instanceDeclaration.unbind(
271                                format("Component '%s/%s' is private. It only accept instances " +
272                                               "from bundle %s/%s [%d] (instance bundle origin: %d)",
273                                       m_declaration.getComponentName(),
274                                       m_declaration.getComponentVersion(),
275                                       origin.getSymbolicName(),
276                                       origin.getVersion(),
277                                       origin.getBundleId(),
278                                       reference.getBundle().getBundleId())
279                        );
280                        return null;
281                    }
282                }
283
284                return m_queueService.submit(new ReferenceableCallable<ComponentInstance>(reference.getBundle()) {
285                    public ComponentInstance call() throws Exception {
286                        try {
287                            // Create the component's instance
288                            // It is automatically started
289                            // Future.get should never be null since this tracker is started when the factory has been created
290                            ComponentInstance instance = m_future.get().createComponentInstance(instanceDeclaration.getConfiguration());
291
292                            // Notify the declaration that everything is fine
293                            instanceDeclaration.bind();
294
295                            return instance;
296                        } catch (UnacceptableConfiguration c) {
297                            instanceDeclaration.unbind(format("Instance configuration is invalid (component:%s/%s, bundle:%d)",
298                                                              m_declaration.getComponentName(),
299                                                              m_declaration.getComponentVersion(),
300                                                              reference.getBundle().getBundleId()),
301                                    c);
302                        } catch (MissingHandlerException e) {
303                            instanceDeclaration.unbind(
304                                    format(
305                                            "Component '%s/%s' (required for instance creation) is missing some handlers",
306                                            m_declaration.getComponentName(),
307                                            m_declaration.getComponentVersion()
308                                    ),
309                                    e);
310                        } catch (ConfigurationException e) {
311                            instanceDeclaration.unbind(
312                                    format(
313                                            "Instance configuration is incorrect for component '%s/%s'",
314                                            m_declaration.getComponentName(),
315                                            m_declaration.getComponentVersion()),
316                                    e);
317                        }
318
319                        return null;
320                    }
321                });
322            }
323
324            return null;
325        }
326
327        public void modifiedService(ServiceReference reference, Object o) {
328        }
329
330        public void removedService(ServiceReference reference, Object o) {
331            InstanceDeclaration instanceDeclaration = (InstanceDeclaration) m_bundleContext.getService(reference);
332            Future<ComponentInstance> future = (Future<ComponentInstance>) o;
333            ComponentInstance instance;
334            try {
335                instance = future.get();
336                // It is possible that the instance couldn't be created
337                if (instance != null) {
338                    String message = format("Factory for Component '%s/%s' is missing",
339                                            instance.getFactory().getName(),
340                                            m_declaration.getComponentVersion());
341                    instanceDeclaration.unbind(message);
342
343                    instance.stop();
344                    instance.dispose();
345                }
346
347            } catch (InterruptedException e) {
348                instanceDeclaration.unbind("Could not create ComponentInstance", e);
349            } catch (ExecutionException e) {
350                instanceDeclaration.unbind("ComponentInstance creation throw an Exception", e);
351            }
352        }
353    }
