101    /**
102     * Initialize the component type.
103     *
104     * @param desc     : component type description to populate.
105     * @param metadata : component type metadata.
106     * @throws ConfigurationException : metadata are incorrect.
107     * @see org.apache.felix.ipojo.Handler#initializeComponentFactory(org.apache.felix.ipojo.architecture.ComponentTypeDescription, org.apache.felix.ipojo.metadata.Element)
108     */
109    public void initializeComponentFactory(ComponentTypeDescription desc, Element metadata) throws ConfigurationException {
110        Element[] confs = metadata.getElements("Properties", "");
111        if (confs == null) {
112            return;
113        }
114        Element[] configurables = confs[0].getElements("Property");
115        for (int i = 0; configurables != null && i < configurables.length; i++) {
116            String fieldName = configurables[i].getAttribute("field");
117            String methodName = configurables[i].getAttribute("method");
118            String paramIndex = configurables[i].getAttribute("constructor-parameter");
119
120            if (fieldName == null && methodName == null && paramIndex == null) {
121                throw new ConfigurationException("Malformed property : The property needs to contain" +
122                        " at least a field, a method or a constructor-parameter");
123            }
124
125            String name = configurables[i].getAttribute("name");
126            if (name == null) {
127                if (fieldName == null && methodName != null) {
128                    name = methodName;
129                } else if (fieldName == null && paramIndex != null) {
130                    name = paramIndex;
131                } else {
132                    name = fieldName;
133                }
134                configurables[i].addAttribute(new Attribute("name", name)); // Add the type to avoid configure checking
135            }
136
137            String value = configurables[i].getAttribute("value");
138
139            // Detect the type of the property
140            PojoMetadata manipulation = getFactory().getPojoMetadata();
141            String type = null;
142            if (methodName != null) {
143                MethodMetadata[] method = manipulation.getMethods(methodName);
144                if (method.length == 0) {
145                    type = configurables[i].getAttribute("type");
146                    if (type == null) {
147                        throw new ConfigurationException("Malformed property : The type of the property cannot be discovered, add a 'type' attribute");
148                    }
149                } else {
150                    if (method[0].getMethodArguments().length != 1) {
151                        throw new ConfigurationException("Malformed property :  The method " + methodName + " does not have one argument");
152                    }
153                    type = method[0].getMethodArguments()[0];
154                    configurables[i].addAttribute(new Attribute("type", type)); // Add the type to avoid configure checking
155                }
156            } else if (fieldName != null) {
157                FieldMetadata field = manipulation.getField(fieldName);
158                if (field == null) {
159                    throw new ConfigurationException("Malformed property : The field " + fieldName + " does not exist in the implementation class");
160                }
161                type = field.getFieldType();
162                configurables[i].addAttribute(new Attribute("type", type)); // Add the type to avoid configure checking
163            } else if (paramIndex != null) {
164                int index = Integer.parseInt(paramIndex);
165                type = configurables[i].getAttribute("type");
166                MethodMetadata[] cts = manipulation.getConstructors();
167                // If we don't have a type, try to get the first constructor and get the type of the parameter
168                // we the index 'index'.
169                if (type == null && cts.length > 0 && cts[0].getMethodArguments().length > index) {
170                    type = cts[0].getMethodArguments()[index];
171                } else if (type == null) { // Applied only if type was not determined.
172                    throw new ConfigurationException("Cannot determine the type of the property " + index +
173                            ", please use the type attribute");
174                }
175                configurables[i].addAttribute(new Attribute("type", type));
176            }
177
178            // Is the property set to immutable
179            boolean immutable = false;
180            String imm = configurables[i].getAttribute("immutable");
181            immutable = imm != null && imm.equalsIgnoreCase("true");
182
183            boolean mandatory = false;
184            String man = configurables[i].getAttribute("mandatory");
185            mandatory = man != null && man.equalsIgnoreCase("true");
186
187            PropertyDescription pd = null;
188            if (value == null) {
189                pd = new PropertyDescription(name, type, null, false); // Cannot be immutable if we have no value.
190            } else {
191                pd = new PropertyDescription(name, type, value, immutable);
192            }
193
194            if (mandatory) {
195                pd.setMandatory();
196            }
197
198            desc.addProperty(pd);
199        }
200
201    }

203    /**
204     * Configures the handler.
205     * Access to field does not require synchronization as this method is executed
206     * before any thread access to this object.
207     *
208     * @param metadata      the metadata of the component
209     * @param configuration the instance configuration
210     * @throws ConfigurationException one property metadata is not correct
211     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
212     */
213    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
214        // Build the map
215        Element[] confs = metadata.getElements("Properties", "");
216        Element[] configurables = confs[0].getElements("Property");
217
218        // Check if the component is dynamically configurable
219        // Propagation enabled by default.
220        m_mustPropagate = true;
221        // We must create a copy as the Config Admin dictionary has some limitations
222        m_toPropagate = new Hashtable<String, Object>();
223        if (configuration != null) {
224            Enumeration keys = configuration.keys();
225            while (keys.hasMoreElements()) {
226                String key = (String) keys.nextElement();
227                // To conform with 'Property Propagation 104.4.4 (Config Admin spec)',
228                // we don't propagate properties starting with .
229                if (!key.startsWith(".")) {
230                    m_toPropagate.put(key, configuration.get(key));
231                }
232            }
233        }
234
235        String propa = confs[0].getAttribute("propagation");
236        if (propa != null && propa.equalsIgnoreCase("false")) {
237            m_mustPropagate = false;
238            m_toPropagate = null;
239        }
240
241        // Check if the component support ConfigurationADmin reconfiguration
242        m_managedServicePID = confs[0].getAttribute("pid"); // Look inside the component type description
243        String instanceMSPID = (String) configuration.get("managed.service.pid"); // Look inside the instance configuration.
244        if (instanceMSPID != null) {
245            m_managedServicePID = instanceMSPID;
246        }
247
248        // updated method
249        String upd = confs[0].getAttribute("updated");
250        if (upd != null) {
251            MethodMetadata method = getPojoMetadata().getMethod(upd);
252            if (method == null) {
253                throw new ConfigurationException("The updated method is not found in the class "
254                        + getInstanceManager().getClassName());
255            } else if (method.getMethodArguments().length == 0) {
256                m_updated = new Callback(upd, new Class[0], false, getInstanceManager());
257            } else if (method.getMethodArguments().length == 1
258                    && method.getMethodArguments()[0].equals(Dictionary.class.getName())) {
259                m_updated = new Callback(upd, new Class[]{Dictionary.class}, false, getInstanceManager());
260            } else {
261                throw new ConfigurationException("The updated method is found in the class "
262                        + getInstanceManager().getClassName() + " must have either no argument or a Dictionary");
263            }
264        }
265
266        for (int i = 0; configurables != null && i < configurables.length; i++) {
267            String fieldName = configurables[i].getAttribute("field");
268            String methodName = configurables[i].getAttribute("method");
269            String paramIndex = configurables[i].getAttribute("constructor-parameter");
270            int index = -1;
271
272            String name = configurables[i].getAttribute("name"); // The initialize method has fixed the property name.
273            String value = configurables[i].getAttribute("value");
274
275            String type = configurables[i].getAttribute("type"); // The initialize method has fixed the property name.
276
277            Property prop;
278            if (paramIndex == null) {
279                prop = new Property(name, fieldName, methodName, value, type, getInstanceManager(), this);
280            } else {
281                index = Integer.parseInt(paramIndex);
282                prop = new Property(name, fieldName, methodName, index,
283                        value, type, getInstanceManager(), this);
284            }
285            addProperty(prop);
286
287            // Check if the instance configuration contains value for the current property :
288            if (configuration.get(name) == null) {
289                if (fieldName != null && configuration.get(fieldName) != null) {
290                    prop.setValue(configuration.get(fieldName));
291                }
292            } else {
293                prop.setValue(configuration.get(name));
294            }
295
296            if (fieldName != null) {
297                FieldMetadata field = new FieldMetadata(fieldName, type);
298                getInstanceManager().register(field, prop);
299            }
300
301            if (index != -1) {
302                getInstanceManager().register(index, prop);
303            }
304        }
305
306        m_description = new ConfigurationHandlerDescription(this, m_configurableProperties, m_managedServicePID);
307
308    }

324    /**
325     * Start method.
326     * This method is synchronized to avoid the config admin pushing a configuration before ending the method.
327     * Propagate properties if the propagation is activated.
328     *
329     * @see org.apache.felix.ipojo.Handler#start()
330     */
331    public synchronized void start() {
332        // Get the provided service handler :
333        m_providedServiceHandler = (ProvidedServiceHandler) getHandler(HandlerFactory.IPOJO_NAMESPACE + ":provides");
334
335
336        // Propagation
337        if (m_mustPropagate) {
338            for (Property prop : m_configurableProperties) {
339                if (prop.getValue() != Property.NO_VALUE && prop.getValue() != null) { // No injected value, or null
340                    m_toPropagate.put(prop.getName(), prop.getValue());
341                }
342            }
343            reconfigure(m_toPropagate);
344        }
345
346
347        // Give initial values and reset the 'invoked' flag.
348        for (Property prop : m_configurableProperties) {
349            prop.reset(); // Clear the invoked flag.
350            if (prop.hasField() && prop.getValue() != Property.NO_VALUE && prop.getValue() != null) {
351                getInstanceManager().onSet(null, prop.getField(), prop.getValue());
352            }
353        }
354
355        if (m_managedServicePID != null && m_sr == null) {
356            Hashtable<String, Object> props = new Hashtable<String, Object>();
357            props.put(Constants.SERVICE_PID, m_managedServicePID);
358            props.put(Factory.INSTANCE_NAME_PROPERTY, getInstanceManager().getInstanceName());
359            props.put("factory.name", getInstanceManager().getFactory().getFactoryName());
360
361            // Security Check
362            if (SecurityHelper.hasPermissionToRegisterService(ManagedService.class.getName(),
363                    getInstanceManager().getContext())  && SecurityHelper.canRegisterService
364                    (getInstanceManager().getContext())) {
365                m_sr = getInstanceManager().getContext().registerService(ManagedService.class.getName(), this, props);
366            } else {
367                error("Cannot register the ManagedService - The bundle "
368                        + getInstanceManager().getContext().getBundle().getBundleId()
369                        + " does not have the permission to register the service");
370            }
371        }
372    }

383    /**
384     * Reconfigure the component instance.
385     * Check if the new configuration modifies the current configuration.
386     * Invokes the updated method is needed.
387     *
388     * @param configuration : the new configuration
389     * @see org.apache.felix.ipojo.Handler#reconfigure(java.util.Dictionary)
390     */
391    public void reconfigure(Dictionary configuration) {
392        Map<String, Object> map = new LinkedHashMap<String, Object>();
393        synchronized (this) {
394            info(getInstanceManager().getInstanceName() + " is reconfiguring the properties : " + configuration);
395            Properties props = reconfigureProperties(configuration);
396            propagate(props, m_propagatedFromInstance);
397            m_propagatedFromInstance = props;
398
399            if (getInstanceManager().getPojoObjects() != null) {
400                try {
401                    notifyUpdated(null);
402                } catch (Throwable e) {
403                    error("Cannot call the updated method : " + e.getMessage(), e);
404                }
405            }
406            // Make a snapshot of the current configuration
407            for (Property p : m_configurableProperties) {
408                map.put(p.getName(), p.getValue());
409            }
410        }
411        notifyListeners(map);
412    }

414    /**
415     * Reconfigured configuration properties and returns non matching properties.
416     * When called, it must hold the monitor lock.
417     *
418     * @param configuration : new configuration
419     * @return the properties that does not match with configuration properties
420     */
421    private Properties reconfigureProperties(Dictionary configuration) {
422        Properties toPropagate = new Properties();
423        Enumeration keysEnumeration = configuration.keys();
424        while (keysEnumeration.hasMoreElements()) {
425            String name = (String) keysEnumeration.nextElement();
426            Object value = configuration.get(name);
427            boolean found = false;
428            // Check if the name is a configurable property
429            for (Property prop : m_configurableProperties) {
430                if (prop.getName().equals(name)) {
431                    reconfigureProperty(prop, value);
432                    found = true;
433                    break; // Exit the search loop
434                }
435            }
436            if (!found) {
437                // The property is not a configurable property, add it to the toPropagate list.
438                toPropagate.put(name, value);
439            }
440        }
441
442        // Every removed configurable property gets reset to its default value
443        for (Property prop : m_configurableProperties) {
444            if (configuration.get(prop.getName()) == null) {
445                reconfigureProperty(prop, prop.getDefaultValue());
446            }
447        }
448        return toPropagate;
449
450    }

452    /**
453     * Reconfigures the given property with the given value.
454     * This methods handles {@link org.apache.felix.ipojo.InstanceManager#onSet(Object, String, Object)}
455     * call and the callback invocation.
456     * The reconfiguration occurs only if the value changes.
457     *
458     * @param prop  the property object to reconfigure
459     * @param value the new value.
460     */
461    public void reconfigureProperty(Property prop, Object value) {
462        if (prop.getValue() == null || !prop.getValue().equals(value)) {
463            prop.setValue(value);
464            if (prop.hasField()) {
465                getInstanceManager().onSet(null, prop.getField(), prop.getValue()); // Notify other handler of the field value change.
466            }
467            if (prop.hasMethod()) {
468                if (getInstanceManager().getPojoObjects() != null) {
469                    prop.invoke(null); // Call on all created pojo objects.
470                }
471            }
472        }
473    }

475    /**
476     * Removes the old properties from the provided services and propagate new properties.
477     *
478     * @param newProps : new properties to propagate
479     * @param oldProps : old properties to remove
480     */
481    private void propagate(Dictionary newProps, Dictionary oldProps) {
482        if (m_mustPropagate && m_providedServiceHandler != null) {
483            if (oldProps != null) {
484                m_providedServiceHandler.removeProperties(oldProps);
485            }
486
487            if (newProps != null) {
488                // Remove the name, the pid and the managed service pid props
489                newProps.remove("name");
490                newProps.remove("managed.service.pid");
491                newProps.remove(Constants.SERVICE_PID);
492
493                // Remove all properties starting with . (config admin specification)
494                Enumeration<String> keys = newProps.keys();
495                List<String> propertiesStartingWithDot = new ArrayList<String>();
496                while (keys.hasMoreElements()) {
497                    String key = keys.nextElement();
498                    if (key.startsWith(".")) {
499                        propertiesStartingWithDot.add(key);
500                    }
501                }
502                for (String k : propertiesStartingWithDot) {
503                    newProps.remove(k);
504                }
505
506                // Propagation of the properties to service registrations :
507                m_providedServiceHandler.addProperties(newProps);
508            }
509        }
510    }

512    /**
513     * Handler createInstance method.
514     * This method is override to allow delayed callback invocation.
515     * Invokes the updated method is needed.
516     *
517     * @param instance : the created object
518     * @see org.apache.felix.ipojo.PrimitiveHandler#onCreation(Object)
519     */
520    public void onCreation(Object instance) {
521        Map<String, Object> map = new LinkedHashMap<String, Object>();
522        for (Property prop : m_configurableProperties) {
523            if (prop.hasMethod()) {
524                prop.invoke(instance);
525            }
526            // Fill the snapshot copy while calling callbacks
527            map.put(prop.getName(), prop.getValue());
528        }
529
530        try {
531            notifyUpdated(instance);
532        } catch (Throwable e) {
533            error("Cannot call the updated method : " + e.getMessage(), e);
534        }
535        notifyListeners(map);
536    }

538    /**
539     * Invokes the updated method.
540     * This method build the dictionary containing all valued properties,
541     * as well as properties propagated to the provided service handler (
542     * only if the propagation is enabled).
543     *
544     * @param instance the instance on which the callback must be called.
545     *                 If <code>null</code> the callback is called on all the existing
546     *                 object.
547     */
548    private void notifyUpdated(Object instance) {
549        if (m_updated == null) {
550            return;
551        }
552
553        if (m_updated.getArguments().length == 0) {
554            // We don't have to compute the properties,
555            // we just call the callback.
556            try {
557                if (instance == null) {
558                    m_updated.call(new Object[0]);
559                } else {
560                    m_updated.call(instance, new Object[0]);
561                }
562            } catch (Exception e) {
563                error("Cannot call the updated method " + m_updated.getMethod() + " : " + e.getMessage());
564            }
565            return;
566        }
567
568        // Else we must compute the properties.
569        Properties props = new Properties();
570        for (Property property : m_configurableProperties) {
571            String n = property.getName();
572            Object v = property.getValue();
573            if (v != Property.NO_VALUE) {
574                props.put(n, v);
575            }
576        }
577        // add propagate properties to the list if propagation is enabled
578        if (m_mustPropagate) {
579            // Start by properties from the configuration admin,
580            if (m_propagatedFromCA != null) {
581
582                Enumeration e = m_propagatedFromCA.keys();
583                while (e.hasMoreElements()) {
584                    String k = (String) e.nextElement();
585                    if (!k.equals(Factory.INSTANCE_NAME_PROPERTY)) {
586                        props.put(k, m_propagatedFromCA.get(k));
587                    }
588                }
589            }
590            // Do also the one from the instance configuration
591            if (m_propagatedFromInstance != null) {
592                Enumeration e = m_propagatedFromInstance.keys();
593                while (e.hasMoreElements()) {
594                    String k = (String) e.nextElement();
595                    if (!k.equals(Factory.INSTANCE_NAME_PROPERTY)) { // Skip instance.name
596                        props.put(k, m_propagatedFromInstance.get(k));
597                    }
598                }
599            }
600        }
601
602        try {
603            if (instance == null) {
604                m_updated.call(new Object[]{props});
605            } else {
606                m_updated.call(instance, new Object[]{props});
607            }
608        } catch (Exception e) {
609            error("Cannot call the updated method " + m_updated.getMethod() + " : " + e.getMessage());
610        }
611    }

613    /**
614     * Managed Service method.
615     * This method is called when the instance is reconfigured by the ConfigurationAdmin.
616     * When called, it must hold the monitor lock.
617     *
618     * @param conf : pushed configuration.
619     * @throws org.osgi.service.cm.ConfigurationException
620     *          the reconfiguration failed.
621     * @see org.osgi.service.cm.ManagedService#updated(java.util.Dictionary)
622     */
623    public void updated(Dictionary conf) throws org.osgi.service.cm.ConfigurationException {
624        Map<String, Object> map = new LinkedHashMap<String, Object>();
625        synchronized (this) {
626            if (conf == null && !m_configurationAlreadyPushed) {
627                return; // First call
628            } else if (conf != null) { // Configuration push
629                Properties props = reconfigureProperties(conf);
630                propagate(props, m_propagatedFromCA);
631                m_propagatedFromCA = props;
632                m_configurationAlreadyPushed = true;
633            } else if (m_configurationAlreadyPushed) { // Configuration deletion
634                propagate(null, m_propagatedFromCA);
635                m_propagatedFromCA = null;
636                m_configurationAlreadyPushed = false;
637            }
638
639            if (getInstanceManager().getPojoObjects() != null) {
640                try {
641                    notifyUpdated(null);
642                } catch (Throwable e) {
643                    error("Cannot call the updated method : " + e.getMessage(), e);
644                }
645            }
646            // Make a snapshot of the current configuration
647            for (Property p : m_configurableProperties) {
648                map.put(p.getName(), p.getValue());
649            }
650        }
651        notifyListeners(map);
652    }

680     * Remove the given listener from the configuration handler's list of listeners.
681     *
682     * @param listener the {@code ConfigurationListener} object to be removed
683     * @throws NullPointerException if {@code listener} is {@code null}
684     * @throws NoSuchElementException if {@code listener} wasn't present the in configuration handler's list of listeners
685     */
686    public void removeListener(ConfigurationListener listener) {
687        if (listener == null) {
688            throw new NullPointerException("null listener");
689        }
690        synchronized (m_listeners) {
691            // We definitely cannot rely on listener's equals method...
692            // ...so we need to manually search for the listener, using ==.
693            int i = -1;
694            for(int j = m_listeners.size() -1; j>=0 ; j--) {
695                if (m_listeners.get(j) == listener) {
696                    // Found!
697                    i = j;
698                    break;
699                }
700            }
701            if (i != -1) {
702                m_listeners.remove(i);
703            } else {
704                throw new NoSuchElementException("no such listener");
705            }
706        }
707    }
708
709    /**
710     * Notify all listeners that a reconfiguration has occurred.
711     *
712     * @param map the new configuration of the component instance.
713     */
714    private void notifyListeners(Map<String, Object> map) {
715        // Get a snapshot of the listeners
716        List<ConfigurationListener> tmp;
717        synchronized (m_listeners) {
718            tmp = new ArrayList<ConfigurationListener>(m_listeners);
719        }
720        // Protect the map.
721        map = Collections.unmodifiableMap(map);
722        // Do notify, outside any lock
723        for (ConfigurationListener l : tmp) {
724            try {
725                l.configurationChanged(getInstanceManager(), map);
726            } catch (Throwable e) {
727                // Failure inside a listener: put a warning on the logger, and continue
728                warn(String.format(
729                            "[%s] A ConfigurationListener has failed: %s",
730                            getInstanceManager().getInstanceName(),
731                            e.getMessage())
732                        , e);
733            }
734        }
735    }
736
737    @Override
738    public void stateChanged(int state) {
739        if (state == ComponentInstance.DISPOSED) {
740            // Clean up the list of listeners
741            synchronized (m_listeners) {
742                m_listeners.clear();
743            }
744        }
745        super.stateChanged(state);
746    }
