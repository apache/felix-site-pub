019package org.apache.felix.ipojo.architecture;
020
021import org.apache.felix.ipojo.ComponentInstance;
022import org.apache.felix.ipojo.InstanceStateListener;
023import org.apache.felix.ipojo.metadata.Attribute;
024import org.apache.felix.ipojo.metadata.Element;
025
026/**
027 * Instance Description.
028 * 
029 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
030 */
031public class InstanceDescription implements InstanceStateListener {
032   
033    /**
034     * The list of handlers plugged on the component instance.
035     */
036    protected HandlerDescription[] m_handlers;
037    
038    /**
039     * The Underlying component instance.
040     */
041    protected final ComponentInstance m_instance;
042
043    /**
044     * Component Type of the instance.
045     */
046    protected final ComponentTypeDescription m_type;
047
048    /**
049     * Creates the instance description.
050     * @param ci  the state of the instance.
051     * @param desc  the component type description of this instance.
052     */
053    public InstanceDescription(ComponentTypeDescription desc, ComponentInstance ci) {
054        m_handlers = new HandlerDescription[0];
055        m_type = desc;
056        m_instance = ci;
057        m_instance.addInstanceStateListener(this);
058    }
059
060    /**
061     * Gets the underlying component instance
062     * @return the component instance.
063     */
064    public ComponentInstance getInstance() {
065        return m_instance;
066    }
067
068    /**
069     * Gets the instance name.
070     * @return the name of the instance.
071     */
072    public String getName() {
073        return m_instance.getInstanceName();
074    }
075
076    /**
077     * Gets the component type description of the described instance.
078     * @return : the component type description of this instance.
079     */
080    public ComponentTypeDescription getComponentDescription() {
081        return m_type;
082    }
083
084    /**
085     * Gets the plugged handler list.
086     * @return the live handler list
087     */
088    public HandlerDescription[] getHandlers() {
089        return m_handlers;
090    }
091
092    /**
093     * Adds an handler description to the list.
094     * @param desc : the handler description to add
095     */
096    public void addHandler(HandlerDescription desc) {
097        // Verify that the dependency description is not already in the array.
098        for (HandlerDescription handler : m_handlers) {
099            if (handler == desc) {
100                return; // NOTHING TO DO, the description is already in the
101                // array
102            }
103        }
104        // The component Description is not in the array, add it
105        HandlerDescription[] newHd = new HandlerDescription[m_handlers.length + 1];
106        System.arraycopy(m_handlers, 0, newHd, 0, m_handlers.length);
107        newHd[m_handlers.length] = desc;
108        m_handlers = newHd;
109    }
110    
111    /**
112     * Gets a handler description by specifying the handler qualified name.
113     * This method ignores case of the given handler name.
114     * @param handler the handler name
115     * @return the handler description or <code>null</code> if not found
116     */
117    public HandlerDescription getHandlerDescription(String handler) {
118        for (HandlerDescription description : m_handlers) {
119            if (description.getHandlerName().equalsIgnoreCase(handler)) {
120                return description;
121            }
122        }
123        return null;
124    }
125
126    /**
127     * Gets the state of the described instance.
128     * @return the state of the instance.
129     */
130    public int getState() {
131        waitForStability();
132        return m_instance.getState();
133    }
134
135    /**
136     * Gets the bundle id of the bundle containing the component type of the instance.
137     * @return the bundle id owning the component implementation class.
138     */
139    public long getBundleId() {
140        return m_instance.getFactory().getBundleContext().getBundle().getBundleId();
141    }
142
143    /**
144     * Gets the instance description.
145     * @return the instance description
146     */
147    public Element getDescription() {
148        Element instance = new Element("Instance", "");
149        instance.addAttribute(new Attribute("name", getName())); // Name
150        
151        int state = getState();
152        if (state == ComponentInstance.STOPPED) {
153            instance.addAttribute(new Attribute("state", "stopped"));
154        }
155        if (state == ComponentInstance.VALID) {
156            instance.addAttribute(new Attribute("state", "valid"));
157        }
158        if (state == ComponentInstance.INVALID) {
159            instance.addAttribute(new Attribute("state", "invalid"));
160        }
161        if (state == ComponentInstance.DISPOSED) {
162            instance.addAttribute(new Attribute("state", "disposed"));
163        }
164        // Bundle
165        instance.addAttribute(new Attribute("bundle", Long.toString(getBundleId())));
166
167        // Component Type
168        instance.addAttribute(new Attribute("component.type", m_type.getName()));
169
170        // Handlers
171        for (HandlerDescription description : m_handlers) {
172            instance.addElement(description.getHandlerInfo());
173        }
174
175        return instance;
176
177    }
178
179    /**
180     * Waits for state stability before returning results.
181     */
182    private synchronized void waitForStability() {
183        while (m_instance.getState() == -2) { // Transition
184            try {
185                wait();
186            } catch (InterruptedException e) {
187               // We're interrupted, will re-check the condition.
188            }
189        }
190        
191    }
192
193    /**
194     * The underlying instance state changes.
195     * @param instance the instance
196     * @param newState the new state
197     * @see org.apache.felix.ipojo.InstanceStateListener#stateChanged(org.apache.felix.ipojo.ComponentInstance, int)
198     */
199    public synchronized void stateChanged(ComponentInstance instance, int newState) {
200        notifyAll(); // if we were in a transition, the transition has completed.
201    }
202
203}
