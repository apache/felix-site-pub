019package org.apache.felix.ipojo.handlers.lifecycle.callback;
020
021import java.lang.reflect.InvocationTargetException;
022import java.util.Dictionary;
023
024import org.apache.felix.ipojo.ComponentInstance;
025import org.apache.felix.ipojo.ConfigurationException;
026import org.apache.felix.ipojo.InstanceManager;
027import org.apache.felix.ipojo.PrimitiveHandler;
028import org.apache.felix.ipojo.metadata.Element;
029import org.apache.felix.ipojo.parser.MethodMetadata;
030import org.apache.felix.ipojo.parser.PojoMetadata;
031
032/**
033 * Lifecycle callback handler.
034 * 
035 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
036 */
037public class LifecycleCallbackHandler extends PrimitiveHandler {
038
039    /**
040     * The list of the callback of the component.
041     */
042    private LifecycleCallback[] m_callbacks = new LifecycleCallback[0];
043
044    /**
045     * State of the instance manager (unresolved at the beginning).
046     */
047    private int m_state = InstanceManager.INVALID;
048    /**
049     * Does a POJO object be created at starting.
050     */
051    private boolean m_immediate = false;
052
053    /**
054     * Add the given callback to the callback list.
055     * 
056     * @param callback : the element to add
057     */
058    private void addCallback(LifecycleCallback callback) {
059        for (int i = 0; (m_callbacks != null) && (i < m_callbacks.length); i++) {
060            if (m_callbacks[i] == callback) {
061                return;
062            }
063        }
064
065        if (m_callbacks != null && m_callbacks.length > 0) { //TODO check here if we can improve the test
066            LifecycleCallback[] newHk = new LifecycleCallback[m_callbacks.length + 1];
067            System.arraycopy(m_callbacks, 0, newHk, 0, m_callbacks.length);
068            newHk[m_callbacks.length] = callback;
069            m_callbacks = newHk;
070        } else {
071            m_callbacks = new LifecycleCallback[] { callback };
072        }
073    }
074
075    /**
076     * Configure the handler.
077     * @param metadata : the component type metadata
078     * @param configuration : the instance configuration
079     * @throws ConfigurationException : one callback metadata is not correct (either the transition or the method are not correct).
080     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.InstanceManager, org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
081     */
082    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
083        m_callbacks = new LifecycleCallback[0];
084
085        String imm = metadata.getAttribute("immediate");
086        m_immediate = imm != null && imm.equalsIgnoreCase("true");
087        
088        PojoMetadata meta = getFactory().getPojoMetadata();
089
090        Element[] hooksMetadata = metadata.getElements("callback");
091        for (int i = 0; hooksMetadata != null && i < hooksMetadata.length; i++) {
092            String method = hooksMetadata[i].getAttribute("method");
093            if (method == null) {
094                throw new ConfigurationException("Lifecycle callback : A callback needs to contain a method attribute");
095            }
096            
097            MethodMetadata met = meta.getMethod(method, new String[0]);
098            
099            int transition = -1;
100            String trans = hooksMetadata[i].getAttribute("transition");
101            if (trans == null) {
102                throw new ConfigurationException("Lifecycle callback : the transition attribute is missing");
103            } else {
104                if (trans.equalsIgnoreCase("validate")) {
105                    transition = LifecycleCallback.VALIDATE;
106                } else if (trans.equalsIgnoreCase("invalidate")) {
107                    transition = LifecycleCallback.INVALIDATE; 
108                } else {
109                    throw new ConfigurationException("Lifecycle callback : Unknown or malformed transition : " + trans);
110                }
111            }
112            
113            LifecycleCallback callback = null;
114            if (met == null) {
115                callback = new LifecycleCallback(this, transition, method);
116            } else {
117                callback = new LifecycleCallback(this, transition, met);
118            }
119            addCallback(callback);
120        }
121    }
122
123    /**
124     * Start the handler. 
125     * @see org.apache.felix.ipojo.Handler#start()
126     */
127    public void start() {
128     // Do nothing during the start
129    } 
130
131    /**
132     * Stop the handler.
133     * @see org.apache.felix.ipojo.Handler#stop()
134     */
135    public void stop() {
136        m_state = InstanceManager.INVALID;
137    }
138
139    /**
140     * When the state change call the associated callback.
141     * 
142     * @param state : the new instance state.
143     * @see org.apache.felix.ipojo.Handler#stateChanged(int)
144     */
145    public void stateChanged(int state) {
146        int transition = -1;
147        if (m_state == ComponentInstance.INVALID && state == ComponentInstance.VALID) {
148            transition = LifecycleCallback.VALIDATE;
149        }
150        if (m_state == ComponentInstance.VALID && state == ComponentInstance.INVALID) {
151            transition = LifecycleCallback.INVALIDATE;
152        }
153        
154        // Manage immediate component
155        if (m_immediate && transition == LifecycleCallback.VALIDATE && getInstanceManager().getPojoObjects() == null) {
156            getInstanceManager().getPojoObject();
157        }
158
159        for (int i = 0; i < m_callbacks.length; i++) {
160            if (m_callbacks[i].getTransition() == transition) {
161                try {
162                    m_callbacks[i].call();
163                } catch (NoSuchMethodException e) {
164                    error("[" + getInstanceManager().getInstanceName() + "] The callback method " + m_callbacks[i].getMethod() + " is not found");
165                    throw new IllegalStateException(e);
166                } catch (IllegalAccessException e) {
167                    error("[" + getInstanceManager().getInstanceName() + "] The callback method " + m_callbacks[i].getMethod() + " is not accessible");
168                    throw new IllegalStateException(e);
169                } catch (InvocationTargetException e) {
170                    error("[" + getInstanceManager().getInstanceName() + "] The callback method " + m_callbacks[i].getMethod() + " has thrown an exception : " + e.getTargetException().getMessage(), e.getTargetException());
171                    throw new IllegalStateException(e.getTargetException());
172                }
173            }
174        }
175        // Update to internal state
176        m_state = state;
177    }
178}
