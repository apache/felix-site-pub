020package org.apache.felix.ipojo.dependency.interceptors;
021
022import org.apache.felix.ipojo.util.DependencyModel;
023import org.osgi.framework.BundleContext;
024import org.osgi.framework.ServiceReference;
025
026/**
027 * A service to influence the visibility of services within a service dependency.
028 * This service is called to determine which services from the tracker (base set) is going to the matching set.
029 *
030 * Several tracking interceptors can be plugged to the same service dependency. In this case,
031 * a chain is created where all interceptor can influence the next one. If the dependency has a filter,
032 * a tracking interceptor using this filter is the last interceptor of the chain.
033 *
034 * Obviously an interceptor can be plugged to several interceptors.
035 */
036public interface ServiceTrackingInterceptor extends DependencyInterceptor {
037
038    /**
039     * Does the interceptor accepts the reference of not ?
040     * This methods has two goals. It can filter out undesirable services by returning {@literal null}. In addition,
041     * it can <em>transform</em> the service reference to add / remove service properties. In this case,
042     * it must return the <strong>same</strong> instance of {@link TransformedServiceReference},
043     * but with the new set of properties.
044     *
045     * So to filter out the service, return {@literal null}. To accept the service,
046     * return the reference as it is. To transform the service update the service reference and return it.
047     *
048     * When several interceptors are collaborating on the same dependency, a chain is created. The received reference
049     * is the reference modified by the preceding interceptor. Notice that once an interceptor returns {@literal
050     * null} the chain is interrupted and the service rejected.
051     *
052     * @param dependency the dependency
053     * @param context the context of the dependency
054     * @param ref the reference
055     * @param <S> the type of service
056     * @return {@literal null} to filter out the service, the, optionally updated, reference to accept it.
057     */
058    public <S> TransformedServiceReference<S> accept(DependencyModel dependency, BundleContext context,
059                                                     TransformedServiceReference<S> ref);
060
061}
