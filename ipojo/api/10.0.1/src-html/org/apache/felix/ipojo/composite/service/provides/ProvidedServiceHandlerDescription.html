019package org.apache.felix.ipojo.composite.service.provides;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import org.apache.felix.ipojo.architecture.HandlerDescription;
025import org.apache.felix.ipojo.composite.CompositeHandler;
026import org.apache.felix.ipojo.metadata.Attribute;
027import org.apache.felix.ipojo.metadata.Element;
028import org.apache.felix.ipojo.util.DependencyModel;
029
030/**
031 * Provided Service Handler Description for composite.
032 * 
033 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
034 */
035public class ProvidedServiceHandlerDescription extends HandlerDescription {
036
037    /**
038     * Provided Service Description list.
039     */
040    private List m_services = new ArrayList();
041    
042    /**
043     * List of exports.
044     */
045    private List m_exports;
046
047    /**
048     * Constructor.
049     * 
050     * @param handler : composite handler.
051     * @param services : The list of Provided Service.
052     * @param exporters : list of managed exports
053     */
054    public ProvidedServiceHandlerDescription(CompositeHandler handler, List services, List exporters) {
055        super(handler);
056        m_services = services;
057        m_exports = exporters;
058    }
059
060    /**
061     * Get the handler description.
062     * @return the provided service handler description
063     * @see org.apache.felix.ipojo.architecture.HandlerDescription#getHandlerInfo()
064     */
065    public Element getHandlerInfo() {
066        Element services = super.getHandlerInfo();
067        for (int i = 0; i < m_services.size(); i++) {
068            ProvidedService svc = (ProvidedService) m_services.get(i);
069            Element service = new Element("service", "");
070            String state = "unregistered";
071            if (svc.isRegistered()) {
072                state = "registered";
073            }
074            String spec = "[" + svc.getSpecification() + "]";
075            service.addAttribute(new Attribute("Specification", spec));
076            service.addAttribute(new Attribute("State", state));
077            services.addElement(service);
078        }
079        
080        for (int i = 0; i < m_exports.size(); i++) {
081            ServiceExporter exp = (ServiceExporter) m_exports.get(i);
082            Element expo = new Element("Exports", "");
083            expo.addAttribute(new Attribute("Specification", exp.getSpecification().getName()));
084            expo.addAttribute(new Attribute("Filter", exp.getFilter()));
085            if (exp.getState() == DependencyModel.RESOLVED) {
086                expo.addAttribute(new Attribute("State", "resolved"));
087            } else {
088                expo.addAttribute(new Attribute("State", "unresolved"));
089            }
090            services.addElement(expo);
091        }
092        
093        return services;
094    }
095
096}
