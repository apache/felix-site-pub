136    /**
137     * Check the validity of the dependencies.
138     */
139    protected void checkContext() {
140        if (!m_started) {
141            return;
142        }
143        synchronized (m_dependencies) {
144            // Store the initial state
145            boolean initialState = getValidity();
146
147            boolean valid = true;
148            for (Dependency dep : m_dependencies) {
149                if (dep.getState() != Dependency.RESOLVED) {
150                    valid = false;
151                    break;
152                }
153            }
154
155            // Check the component dependencies
156            if (valid) {
157                // The dependencies are valid
158                if (!initialState) {
159                    // There is a state change
160                    setValidity(true);
161                }
162                // Else do nothing, the component state stay VALID
163            } else {
164                // The dependencies are not valid
165                if (initialState) {
166                    // There is a state change
167                    setValidity(false);
168                }
169                // Else do nothing, the component state stay UNRESOLVED
170            }
171
172        }
173    }

175    /**
176     * Check if the dependency given is valid in the sense that metadata are consistent.
177     * @param dep : the dependency to check
178     * @param manipulation : the component-type manipulation metadata
179     * @return true if the dependency is valid
180     * @throws ConfigurationException : the checked dependency is not correct
181     */
182    private boolean checkDependency(Dependency dep, PojoMetadata manipulation) throws ConfigurationException {
183        // Check the internal type of dependency
184        String field = dep.getField();
185        DependencyCallback[] callbacks = dep.getCallbacks();
186        int index = dep.getConstructorParameterIndex();
187
188        if (callbacks == null && field == null  && index == -1) {
189            throw new ConfigurationException("A service requirement requires at least binding methods, " +
190                        "a field or a constructor parameter");
191        }
192
193        for (int i = 0; callbacks != null && i < callbacks.length; i++) {
194            MethodMetadata[] mets = manipulation.getMethods(callbacks[i].getMethodName());
195            if (mets.length == 0) {
196                debug("A requirement callback " + callbacks[i].getMethodName() + " does not exist in the implementation class, will try the super classes");
197            } else {
198                if (mets[0].getMethodArguments().length > 2) {
199                    throw new ConfigurationException("Requirement Callback : A requirement callback "
200                            + callbacks[i].getMethodName()
201                            + " must have 0, 1 or 2 arguments");
202                }
203
204                callbacks[i].setArgument(mets[0].getMethodArguments());
205
206                if (mets[0].getMethodArguments().length == 1) {
207                    if (!mets[0].getMethodArguments()[0].equals(ServiceReference.class.getName())) {
208                        // The callback receives the service object.
209                        setSpecification(dep, mets[0].getMethodArguments()[0], false); // Just warn if a mismatch is discovered.
210                    }
211                } else if (mets[0].getMethodArguments().length == 2) {
212                    // The callback receives service object, service reference. Check that the second argument is a service reference
213                    if (!(mets[0].getMethodArguments()[1].equals(ServiceReference.class.getName()) // callback with (service object, service reference)
214                           || mets[0].getMethodArguments()[1].equals(Dictionary.class.getName()) // callback with (service object, service properties in a dictionary)
215                           || mets[0].getMethodArguments()[1].equals(Map.class.getName()))) { // callback with (service object, service properties in a map)
216                        String message =
217                                "The requirement callback " + callbacks[i].getMethodName() + " must have a ServiceReference, a Dictionary or a Map as the second argument";
218                        throw new ConfigurationException(message);
219                    }
220                    setSpecification(dep, mets[0].getMethodArguments()[0], false); // Just warn if a mismatch is discovered.
221                }
222            }
223
224        }
225
226        if (field != null) {
227            FieldMetadata meta = manipulation.getField(field);
228            if (meta == null) {
229                throw new ConfigurationException("Requirement Callback : A requirement field "
230                        + field
231                        + " does not exist in the implementation class");
232            }
233            String type = meta.getFieldType();
234            if (type.endsWith("[]")) {
235                if (dep.isProxy()) {
236                    info("Arrays cannot be used for proxied dependencies - Disabling the proxy mode");
237                    dep.setProxy(false);
238                }
239                // Set the dependency to multiple
240                dep.setAggregate(true);
241                type = type.substring(0, type.length() - 2);
242            } else if (type.equals(List.class.getName()) || type.equals(Collection.class.getName())) {
243                dep.setType(LIST);
244                type = null;
245            } else if (type.equals(Vector.class.getName())) {
246                dep.setType(VECTOR);
247                if (dep.isProxy()) {
248                    warn("Vectors cannot be used for proxied dependencies - Disabling the proxy mode");
249                    dep.setProxy(false);
250                }
251                type = null;
252            } else if (type.equals(Set.class.getName())) {
253                dep.setType(SET);
254                type = null;
255            } else {
256                if (dep.isAggregate()) {
257                    throw new ConfigurationException("A required service is not correct : the field "
258                            + meta.getFieldName()
259                            + " must be an array to support aggregate injections");
260                }
261            }
262            setSpecification(dep, type, true); // Throws an exception if the field type mismatch.
263        }
264
265        // Constructor parameter
266        if (index != -1) {
267                if (! dep.isProxy()) {
268                        throw new ConfigurationException("Services injected into constructor must be proxied");
269                }
270
271                MethodMetadata[] cts = manipulation.getConstructors();
272                // If we don't have a type, try to get the first constructor and get the type of the parameter
273                // we the index 'index'.
274                if (cts.length > 0  && cts[0].getMethodArguments().length > index) {
275                        String type = cts[0].getMethodArguments()[index];
276                        if (type.endsWith("[]")) {
277                    throw new ConfigurationException("Services injected into constructor cannot be arrays");
278                } else if (type.equals(List.class.getName()) || type.equals(Collection.class.getName())) {
279                    dep.setType(LIST);
280                    type = null;
281                } else if (type.equals(Vector.class.getName())) {
282                        throw new ConfigurationException("Services injected into constructor cannot be Vectors");
283                } else if (type.equals(Set.class.getName())) {
284                    dep.setType(SET);
285                    type = null;
286                } else {
287                    if (dep.isAggregate()) {
288                        throw new ConfigurationException("A required service is not correct : the constructor parameter "
289                                + index
290                                + " must be an aggregate type to support aggregate injections");
291                    }
292                }
293                setSpecification(dep, type, true); // Throws an exception if the field type mismatch.
294                } else {
295                        throw new ConfigurationException("Cannot determine the specification of the dependency " + index +
296                                        ", please use the specification attribute");
297                }
298        }
299
300        // At this point we must have discovered the specification, it it's null, throw a ConfiguraitonException
301        if (dep.getSpecification() == null) {
302            String id = dep.getId();
303            if (id == null) {
304                dep.getField();
305            }
306            if (id == null  && dep.getCallbacks() != null  && dep.getCallbacks().length > 0) {
307                id = dep.getCallbacks()[0].getMethodName();
308            }
309            throw new ConfigurationException("Cannot determine the targeted service specification for the dependency '" +
310                    id + "'");
311        }
312
313        // Disable proxy on scalar dependency targeting non-interface specification
314        if (! dep.isAggregate()  && dep.isProxy()) {
315                if (! dep.getSpecification().isInterface()) {
316                        warn("Proxies cannot be used on service dependency targeting non interface " +
317                                        "service specification " + dep.getSpecification().getName());
318                        dep.setProxy(false);
319                }
320        }
321
322        // Disables proxy on null (nullable=false)
323//        if (dep.isProxy()  && dep.isOptional() && ! dep.supportsNullable()) {
324//            dep.setProxy(false);
325//            warn("Optional Null Dependencies do not support proxying - Disable the proxy mode");
326//        }
327
328        // Check that all required info are set
329        return dep.getSpecification() != null;
330    }

332    /**
333     * Check if we have to set the dependency specification with the given class name.
334     * @param dep : dependency to check
335     * @param className : class name
336     * @param error : set to true to throw an error if the set dependency specification and the given specification are different.
337     * @throws ConfigurationException : the specification class cannot be loaded correctly
338     */
339    private void setSpecification(Dependency dep, String className, boolean error) throws ConfigurationException {
340        if (className == null) {
341            // No found type (list and vector)
342            if (dep.getSpecification() == null) {
343                if (error) {
344                        String id = dep.getId();
345                        if (id == null) {
346                                id = dep.getField();
347                                if (id == null) {
348                                        id = Integer.toString(dep.getConstructorParameterIndex());
349                                }
350                        }
351                    throw new ConfigurationException("Cannot discover the required specification for " + id);
352                } else {
353                    // If the specification is different, warn that we will override it.
354                    info("Cannot discover the required specification for " + dep.getField());
355                }
356            }
357        } else { // In all other case, className is not null.
358            if (dep.getSpecification() == null || !dep.getSpecification().getName().equals(className)) {
359                if (dep.getSpecification() != null) {
360                    if (error) {
361                        throw new ConfigurationException("A required service is not correct : the discovered type ["
362                            + className
363                            + "] and the specified (or already discovered)  service interface ["
364                            + dep.getSpecification().getName()
365                            + "] are not the same");
366                    } else {
367                        // If the specification is different, warn that we will override it.
368                        warn("["
369                            + getInstanceManager().getInstanceName()
370                            + "] The field type ["
371                            + className
372                            + "] and the required service interface ["
373                            + dep.getSpecification()
374                            + "] are not the same");
375                    }
376                }
377
378                Bundle bundle = getInstanceManager().getContext().getBundle();
379                try {
380                    dep.setSpecification(bundle.loadClass(className));
381                } catch (ClassNotFoundException e) {
382                    throw new ConfigurationException("The required service interface (" + className
383                            + ") cannot be loaded from bundle " + bundle.getBundleId(), e);
384                }
385            }
386        }
387    }

389    /**
390     * Configure the handler.
391     * @param componentMetadata : the component type metadata
392     * @param configuration : the instance configuration
393     * @throws ConfigurationException : one dependency metadata is not correct.
394     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
395     */
396    public void configure(Element componentMetadata, Dictionary configuration) throws ConfigurationException {
397        PojoMetadata manipulation = getFactory().getPojoMetadata();
398        boolean atLeastOneField = false;
399
400        // Create the dependency according to the component metadata
401        Element[] deps = componentMetadata.getElements("Requires");
402
403        // Get instance filters.
404        Dictionary filtersConfiguration = getRequiresFilters(configuration.get("requires.filters"));
405        Dictionary fromConfiguration = (Dictionary) configuration.get("requires.from");
406
407        for (int i = 0; deps != null && i < deps.length; i++) {
408            // Create the dependency metadata
409            final Element dependencyElement = deps[i];
410
411            String field = dependencyElement.getAttribute("field");
412            String serviceSpecification = getServiceSpecificationAttribute(dependencyElement);
413            String opt = dependencyElement.getAttribute("optional");
414            boolean optional = opt != null && opt.equalsIgnoreCase("true");
415            String defaultImpl = dependencyElement.getAttribute("default-implementation");
416
417            String agg = dependencyElement.getAttribute("aggregate");
418            boolean aggregate = agg != null && agg.equalsIgnoreCase("true");
419
420            String identity = dependencyElement.getAttribute("id");
421
422            String nul = dependencyElement.getAttribute("nullable");
423            boolean nullable = nul == null || nul.equalsIgnoreCase("true");
424
425            boolean isProxy = isProxy(dependencyElement);
426
427            BundleContext context = getFacetedBundleContext(dependencyElement);
428
429            String filter = computeFilter(dependencyElement, filtersConfiguration, fromConfiguration, aggregate, identity);
430            Filter fil = createAndCheckFilter(filter);
431
432            Class spec = null;
433            if (serviceSpecification != null) {
434                spec = DependencyMetadataHelper.loadSpecification(serviceSpecification, getInstanceManager().getContext());
435            }
436
437            int policy = DependencyMetadataHelper.getPolicy(dependencyElement);
438            Comparator cmp = DependencyMetadataHelper.getComparator(dependencyElement, getInstanceManager().getGlobalContext());
439
440            Dependency dep = new Dependency(this, field, spec, fil, optional, aggregate, nullable, isProxy, identity, context, policy, cmp, defaultImpl);
441
442            // Look for dependency callback :
443            addCallbacksToDependency(dependencyElement, dep);
444
445            // Add the constructor parameter if needed
446            String paramIndex = dependencyElement.getAttribute("constructor-parameter");
447            if (paramIndex != null) {
448                int index = Integer.parseInt(paramIndex);
449                dep.addConstructorInjection(index);
450            }
451
452            // Check the dependency :
453            if (checkDependency(dep, manipulation)) {
454                m_dependencies.add(dep);
455                if (dep.getField() != null) {
456                    getInstanceManager().register(manipulation.getField(dep.getField()), dep);
457                    atLeastOneField = true;
458                }
459            }
460        }
461
462        if (atLeastOneField) { // Does register only if we have fields
463            MethodMetadata[] methods = manipulation.getMethods();
464            for (MethodMetadata method : methods) {
465                for (Dependency dep : m_dependencies) {
466                    getInstanceManager().register(method, dep);
467                }
468            }
469        }
470
471        m_description = new DependencyHandlerDescription(this, getDependencies()); // Initialize the description.
472
473        manageContextSources(configuration);
474    }

476    /**
477     * Add internal context source to all dependencies.
478     * @param configuration the instance configuration to creates the instance configuration source
479     */
480    private void manageContextSources(Dictionary<String, Object> configuration) {
481        m_instanceConfigurationSource = new InstanceConfigurationSource(configuration);
482        SystemPropertiesSource systemPropertiesSource = new SystemPropertiesSource();
483
484        for (Dependency dependency : m_dependencies) {
485            if (dependency.getFilter() != null) {
486                dependency.getContextSourceManager().addContextSource(m_instanceConfigurationSource);
487                dependency.getContextSourceManager().addContextSource(systemPropertiesSource);
488
489                for (Handler handler : getInstanceManager().getRegisteredHandlers()) {
490                    if (handler instanceof ContextSource) {
491                        dependency.getContextSourceManager().addContextSource((ContextSource) handler);
492                    }
493                }
494            }
495        }
496    }

498    private String computeFilter(Element dependencyElement, Dictionary filtersConfiguration, Dictionary fromConfiguration, boolean aggregate, String identity) {
499        String filter = dependencyElement.getAttribute("filter");
500        // Get instance filter if available
501        if (filtersConfiguration != null && identity != null && filtersConfiguration.get(identity) != null) {
502            filter = (String) filtersConfiguration.get(identity);
503        }
504
505        // Compute the 'from' attribute
506        filter = updateFilterIfFromIsEnabled(fromConfiguration, dependencyElement, filter, aggregate, identity);
507        return filter;
508    }
509
510    private String updateFilterIfFromIsEnabled(Dictionary fromConfiguration, Element dependencyElement, String filter, boolean aggregate, String identity) {
511        String from = dependencyElement.getAttribute("from");
512        if (fromConfiguration != null && identity != null && fromConfiguration.get(identity) != null) {
513            from = (String) fromConfiguration.get(identity);
514        }
515        if (from != null) {
516            String fromFilter = "(|(instance.name=" + from + ")(service.pid=" + from + "))";
517            if (aggregate) {
518                warn("The 'from' attribute is incompatible with aggregate requirements: only one provider will match : " + fromFilter);
519            }
520            if (filter != null) {
521                filter = "(&" + fromFilter + filter + ")"; // Append the two filters
522            } else {
523                filter = fromFilter;
524            }
525        }
526        return filter;
527    }

529    private boolean isProxy(Element dependencyElement) {
530        boolean isProxy = true;
531        String setting = getProxySetting();
532
533        if (setting == null || PROXY_ENABLED.equals(setting)) { // If not set => Enabled
534            isProxy = true;
535        } else if (PROXY_DISABLED.equals(setting)) {
536            isProxy = false;
537        }
538
539        String proxy = dependencyElement.getAttribute("proxy");
540        // If proxy == null, use default value
541        if (proxy != null) {
542            if (proxy.equals("false")) {
543                isProxy = false;
544            } else if (proxy.equals("true")) {
545                if (! isProxy) { // The configuration overrides the system setting
546                    warn("The configuration of a service dependency overrides the proxy mode");
547                }
548                isProxy = true;
549            }
550        }
551        return isProxy;
552    }
553
554    private String getProxySetting() {
555        // Detect proxy default value.
556        String setting = getInstanceManager().getContext().getProperty(PROXY_SETTINGS_PROPERTY);
557
558        // Felix also includes system properties in the bundle context property, however it is not the case of the
559        // other frameworks, so if it's null we should call System.getProperty.
560
561        if (setting == null) {
562            setting = System.getProperty(PROXY_SETTINGS_PROPERTY);
563        }
564        return setting;
565    }

567    private void addCallbacksToDependency(Element dependencyElement, Dependency dep) throws ConfigurationException {
568        Element[] cbs = dependencyElement.getElements("Callback");
569        for (int j = 0; cbs != null && j < cbs.length; j++) {
570            if (!cbs[j].containsAttribute("method") && cbs[j].containsAttribute("type")) {
571                throw new ConfigurationException("Requirement Callback : a dependency callback must contain a method " +
572                        "and a type (bind or unbind) attribute");
573            }
574            String method = cbs[j].getAttribute("method");
575            String type = cbs[j].getAttribute("type");
576
577            int methodType = DependencyCallback.UNBIND;
578            if ("bind".equalsIgnoreCase(type)) {
579                methodType = DependencyCallback.BIND;
580            } else if ("modified".equalsIgnoreCase(type)) {
581                methodType = DependencyCallback.MODIFIED;
582            }
583
584            dep.addDependencyCallback(createDependencyHandler(dep, method, methodType));
585        }
586    }
587
588    protected DependencyCallback createDependencyHandler(final Dependency dep, final String method, final int type) {
589        return new DependencyCallback(dep, method, type);
590    }

592    private Filter createAndCheckFilter(String filter) throws ConfigurationException {
593        Filter fil = null;
594        if (filter != null) {
595            try {
596                fil = getInstanceManager().getContext().createFilter(filter);
597            } catch (InvalidSyntaxException e) {
598                throw new ConfigurationException("A requirement filter is invalid : " + filter, e);
599            }
600        }
601        return fil;
602    }

604    private BundleContext getFacetedBundleContext(Element dep) {
605        String scope = dep.getAttribute("scope");
606        BundleContext context = getInstanceManager().getContext(); // Get the default bundle context.
607        if (scope != null) {
608            // If we are not in a composite, the policy is set to global.
609            if (scope.equalsIgnoreCase("global") || ((((IPojoContext) getInstanceManager().getContext()).getServiceContext()) == null)) {
610                context =
611                        new PolicyServiceContext(getInstanceManager().getGlobalContext(), getInstanceManager().getLocalServiceContext(),
612                                PolicyServiceContext.GLOBAL);
613            } else if (scope.equalsIgnoreCase("composite")) {
614                context =
615                        new PolicyServiceContext(getInstanceManager().getGlobalContext(), getInstanceManager().getLocalServiceContext(),
616                                PolicyServiceContext.LOCAL);
617            } else if (scope.equalsIgnoreCase("composite+global")) {
618                context =
619                        new PolicyServiceContext(getInstanceManager().getGlobalContext(), getInstanceManager().getLocalServiceContext(),
620                                PolicyServiceContext.LOCAL_AND_GLOBAL);
621            }
622        }
623        return context;
624    }

626    private String getServiceSpecificationAttribute(Element dep) {
627        String serviceSpecification = dep.getAttribute("interface");
628        // the 'interface' attribute is deprecated
629        if (serviceSpecification != null) {
630            warn("The 'interface' attribute is deprecated, use the 'specification' attribute instead");
631        } else {
632            serviceSpecification = dep.getAttribute("specification");
633        }
634        return serviceSpecification;
635    }

637    /**
638         * Gets the requires filter configuration from the given object.
639         * The given object must come from the instance configuration.
640         * This method was made to fix FELIX-2688. It supports filter configuration using
641         * an array:
642         * <code>{"myFirstDep", "(property1=value1)", "mySecondDep", "(property2=value2)"});</code>
643         * @param requiresFiltersValue the value contained in the instance
644         * configuration.
645         * @return the dictionary. If the object in already a dictionary, just returns it,
646         * if it's an array, builds the dictionary.
647         * @throws ConfigurationException the dictionary cannot be built
648         */
649        private Dictionary getRequiresFilters(Object requiresFiltersValue)
650                        throws ConfigurationException {
651                if (requiresFiltersValue != null
652                                && requiresFiltersValue.getClass().isArray()) {
653                        String[] filtersArray = (String[]) requiresFiltersValue;
654                        if (filtersArray.length % 2 != 0) {
655                                throw new ConfigurationException(
656                                                "A requirement filter is invalid : "
657                                                                + requiresFiltersValue);
658                        }
659                        Dictionary<String, Object> requiresFilters = new Hashtable<String, Object>();
660                        for (int i = 0; i < filtersArray.length; i += 2) {
661                                requiresFilters.put(filtersArray[i], filtersArray[i + 1]);
662                        }
663                        return requiresFilters;
664                }
665
666                return (Dictionary) requiresFiltersValue;
667        }

669    /**
670     * Handler start method.
671     * @see org.apache.felix.ipojo.Handler#start()
672     */
673    public void start() {
674        // Start the dependencies
675        for (Dependency dep : m_dependencies) {
676            dep.start();
677        }
678        // Check the state
679        m_started = true;
680        setValidity(false);
681        checkContext();
682    }
683
684    /**
685     * Handler stop method.
686     * @see org.apache.felix.ipojo.Handler#stop()
687     */
688    public void stop() {
689        m_started = false;
690        for (Dependency dep : m_dependencies) {
691            dep.stop();
692        }
693    }
