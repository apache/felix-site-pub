020package org.apache.felix.ipojo.manipulator;
021
022import java.io.IOException;
023import java.util.ArrayList;
024import java.util.List;
025
026import org.apache.felix.ipojo.manipulation.InnerClassManipulator;
027import org.apache.felix.ipojo.manipulation.Manipulator;
028
029/**
030 * A {@code ManipulationEngine} is responsible to drive the component's
031 * classes manipulation.
032 *
033 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
034 */
035public class ManipulationEngine {
036
037    /**
038     * List of component types.
039     */
040    private List<ManipulationUnit> m_manipulationUnits = new ArrayList<ManipulationUnit>();
041
042    /**
043     * Error reporting.
044     */
045    private Reporter m_reporter;
046
047    /**
048     * Bytecode store.
049     */
050    private ResourceStore m_store;
051
052    /**
053     * The visitor handling output result.
054     */
055    private ManipulationVisitor m_manipulationVisitor;
056
057    /**
058     * Add information related to a discovered component that will be manipulated.
059     * @param component additional component
060     */
061    public void addManipulationUnit(ManipulationUnit component) {
062        m_manipulationUnits.add(component);
063    }
064
065    public void setManipulationVisitor(ManipulationVisitor manipulationVisitor) {
066        m_manipulationVisitor = manipulationVisitor;
067    }
068
069    /**
070     * @param reporter Feedback reporter.
071     */
072    public void setReporter(Reporter reporter) {
073        m_reporter = reporter;
074    }
075
076    /**
077     * Provides the bytecode store that allows to retrieve bytecode of the
078     * component's related resources (inner classes for example).
079     * @param store Helps to locate bytecode for classes.
080     */
081    public void setResourceStore(ResourceStore store) {
082        m_store = store;
083    }
084
085    /**
086     * Manipulates classes of all the given component's.
087     */
088    public void generate() {
089
090        // Iterates over the list of discovered components
091        // Note that this list includes components from metadata.xml AND from annotations
092
093        for (ManipulationUnit info : m_manipulationUnits) {
094
095            byte[] bytecode;
096            try {
097                bytecode = m_store.read(info.getResourcePath());
098            } catch (IOException e) {
099                m_reporter.error("Cannot find bytecode for class '" + info.getClassName() + "': no bytecode found.");
100                return;
101            }
102
103            // Is the visitor interested in this component ?
104            ManipulationResultVisitor result = m_manipulationVisitor.visitManipulationResult(info.getComponentMetadata());
105
106            if (result != null) {
107                // Should always be the case
108
109                // Manipulate the original bytecode and store the modified one
110                Manipulator manipulator = new Manipulator();
111                try {
112                    byte[] out = manipulator.manipulate(bytecode);
113                    // Call the visitor
114                    result.visitClassStructure(manipulator.getManipulationMetadata());
115                    result.visitManipulatedResource(info.getResourcePath(), out);
116                } catch (IOException e) {
117                    m_reporter.error("Cannot manipulate the class " + info.getClassName() + " : " + e.getMessage());
118                    return;
119                }
120
121
122                // Visit inner classes
123                for (String inner : manipulator.getInnerClasses()) {
124                    // Get the bytecode and start manipulation
125                    String resourcePath = inner + ".class";
126                    String outerClassInternalName = info.getClassName().replace('.', '/');
127                    byte[] innerClassBytecode;
128                    try {
129                        innerClassBytecode = m_store.read(resourcePath);
130                    } catch (IOException e) {
131                        m_reporter.error("Cannot find inner class '" + resourcePath + "'");
132                        return;
133                    }
134
135                    // Manipulate inner class
136                    // Notice that (for performance reason) re-use the class version information
137                    // discovered in the main class instead of re-parsing the inner class to find
138                    // its own class version
139                    try {
140                        InnerClassManipulator innerManipulator = new InnerClassManipulator(outerClassInternalName, manipulator.getFields().keySet());
141                        byte[] manipulated = innerManipulator.manipulate(innerClassBytecode, manipulator.getClassVersion());
142                        // Propagate manipulated resource
143                        result.visitManipulatedResource(resourcePath, manipulated);
144                    } catch (IOException e) {
145                        m_reporter.error("Cannot manipulate inner class '" + resourcePath + "'");
146                        return;
147                    }
148                }
149
150                // All resources have been manipulated for this component
151                result.visitEnd();
152            }
153        }
154    }
155}
