066    private static Binding newGenericTypeBinding() {
067        Binding binding = new Binding();
068        // ElementType is TYPE
069        // Annotation descriptor is matching generic pattern
070        binding.setPredicate(
071                and(
072                        on(ElementType.TYPE),
073                        customAnnotationPattern()
074                ));
075        binding.setFactory(new AnnotationVisitorFactory() {
076            public AnnotationVisitor newAnnotationVisitor(BindingContext context) {
077                Element element = Elements.buildElement(context.getAnnotationType());
078                return new TypeGenericVisitor(context.getWorkbench(), element);
079            }
080        });
081        return binding;
082    }
083
084    private static Predicate customAnnotationPattern() {
085        return pattern("(.*\\.ipojo\\..*)|(.*\\.handler\\..*)").matches();
086    }

143    public static BindingRegistry newBindingRegistry(Reporter reporter, ResourceStore store) {
144
145        AnnotationRegistry ar = new AnnotationRegistry();
146        BindingRegistry registry = new BindingRegistry(reporter);
147        ServiceLoader<Module> loader = ServiceLoader.load(Module.class, classloader());
148
149        // Build each Module and add its contributed Bindings in the registry
150        for (Module module : loader) {
151            module.configure();
152            registry.addBindings(module);
153        }
154
155        // Do not forget the default Bindings
156        registry.getDefaultBindings().addAll(newDefaultBindings(store, ar));
157
158        return registry;
159    }
160
161    public static List<Binding> newDefaultBindings(final ResourceStore store, final AnnotationRegistry ar) {
162        List<Binding> bindings = new ArrayList<Binding>();
163
164        // Register Stereotype binding support first
165        // That allows iPOJO to provide its own stereotyped annotations
166        bindings.add(newStereotypeTypeBinding(store, ar));
167        bindings.add(newStereotypeFieldBinding(store, ar));
168        bindings.add(newStereotypeMethodBinding(store, ar));
169        bindings.add(newStereotypeParameterBinding(store, ar));
170
171        // Then register the generic bindings
172        bindings.add(newGenericTypeBinding());
173        bindings.add(newGenericFieldBinding());
174        bindings.add(newGenericMethodBinding());
175        bindings.add(newGenericParameterBinding());
176        return bindings;
177
178    }

180    private static Binding newStereotypeParameterBinding(final ResourceStore store, final AnnotationRegistry registry) {
181        Binding binding = new Binding();
182        binding.setPredicate(
183                and(
184                        on(ElementType.PARAMETER),
185                        stereotype(store, registry)
186                ));
187        binding.setFactory(new AnnotationVisitorFactory() {
188            public AnnotationVisitor newAnnotationVisitor(BindingContext context) {
189                return new ParameterStereotypeVisitor((MethodVisitor) context.getVisitor(),
190                                                      context.getParameterIndex(),
191                                                      registry.getRecorders(context.getAnnotationType()));
192            }
193        });
194        return binding;
195    }

245    private static Predicate stereotype(final ResourceStore store, final AnnotationRegistry registry) {
246        return new Predicate() {
247            public boolean matches(final BindingContext context) {
248
249                Type type = context.getAnnotationType();
250                if (registry.isUnknown(type)) {
251                    // The given annotation type was never parsed before
252                    try {
253
254                        // Try to read the annotation's byte code
255                        byte[] bytes = store.read(Strings.asResourcePath(context.getAnnotationType().getClassName()));
256                        StereotypeParser parser = new StereotypeParser();
257                        parser.read(bytes);
258                        if (parser.isStereotype()) {
259                            registry.addStereotype(type, parser.getRecorders());
260                            return true;
261                        }
262                        registry.addUnbound(type);
263                        return false;
264                    } catch (IOException e) {
265                        // Cannot load the byte code, assume it's not a stereotype
266                        // TODO print a warning ?
267                        registry.addUnbound(type);
268                        return false;
269                    }
270                }
271                return registry.isStereotype(type);
272            }
273        };
274    }
