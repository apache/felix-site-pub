019package org.apache.felix.ipojo.api;
020
021import org.apache.felix.ipojo.ComponentInstance;
022import org.apache.felix.ipojo.PrimitiveInstanceDescription;
023import org.apache.felix.ipojo.architecture.PropertyDescription;
024import org.apache.felix.ipojo.metadata.Attribute;
025import org.apache.felix.ipojo.metadata.Element;
026
027/**
028 * Allows configuring a configuration property.
029 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
030 */
031public class Property {
032
033    /**
034     * The property name.
035     */
036    private String m_name;
037
038    /**
039     * The property field.
040     */
041    private String m_field;
042
043    /**
044     * The constructor parameter index.
045     */
046    private int m_parameterIndex = -1;
047
048    /**
049     * The property value.
050     */
051    private String m_value;
052
053    /**
054     * The property method.
055     */
056    private String m_method;
057
058    /**
059     * Is the property mandatory.
060     */
061    private boolean m_mandatory;
062
063    /**
064     * Is the property immutable.
065     */
066    private boolean m_immutable;
067
068    /**
069     * Sets the property name.
070     * @param name the property name
071     * @return the current property object
072     */
073    public Property setName(String name) {
074        m_name = name;
075        return this;
076    }
077
078    /**
079     * Sets the property field.
080     * @param name the property field
081     * @return the current property object
082     */
083    public Property setField(String name) {
084        m_field = name;
085        return this;
086    }
087
088    /**
089     * Sets the constructor parameter index of the property.
090     * @param index the parameter index
091     * @return the current property object
092     */
093    public Property setConstructorParameter(int index) {
094        m_parameterIndex = index;
095        return this;
096    }
097
098    /**
099     * Sets the property method.
100     * @param name the property method
101     * @return the current property object
102     */
103    public Property setMethod(String name) {
104        m_method = name;
105        return this;
106    }
107
108    /**
109     * Sets the property value.
110     * @param name the property value
111     * @return the current property object
112     */
113    public Property setValue(String name) {
114        m_value = name;
115        return this;
116    }
117
118    /**
119     * Sets if the property is mandatory.
120     * @param mandatory <code>true</code> if the dependency is mandatory.
121     * @return the current property object
122     */
123    public Property setMandatory(boolean mandatory) {
124        m_mandatory = mandatory;
125        return this;
126    }
127
128    /**
129     * Sets if the property is immutable.
130     * @param immutable <code>true</code> if the dependency is immutable.
131     * @return the current property object
132     */
133    public Property setImmutable(boolean immutable) {
134        m_immutable = immutable;
135        return this;
136    }
137
138    /**
139     * Gets the property element.
140     * @return the property element.
141     */
142    public Element getElement() {
143        ensureValidity();
144        Element element = new Element("property", "");
145        if (m_name != null) {
146            element.addAttribute(new Attribute("name", m_name));
147        }
148        if (m_method != null) {
149            element.addAttribute(new Attribute("method", m_method));
150        }
151        if (m_parameterIndex != -1) {
152                element.addAttribute(new Attribute("constructor-parameter",
153                                Integer.toString(m_parameterIndex)));
154        }
155        if (m_value != null) {
156            element.addAttribute(new Attribute("value", m_value));
157        }
158        if (m_field != null) {
159            element.addAttribute(new Attribute("field", m_field));
160        }
161        if (m_mandatory) {
162            element.addAttribute(new Attribute("mandatory", new Boolean(m_mandatory).toString()));
163        }
164        if (m_immutable) {
165            element.addAttribute(new Attribute("immutable", new Boolean(m_immutable).toString()));
166        }
167        return element;
168    }
169
170    /**
171     * Checks the configuration validity.
172     */
173    private void ensureValidity() {
174        // Two cases
175        // Field or Method
176        if (m_field == null && m_method == null && m_parameterIndex == -1) {
177            throw new IllegalStateException("A property must have either a field or a method " +
178                        "or a constructor parameter");
179        }
180        if (m_immutable && m_value == null) {
181            throw new IllegalStateException("A immutable service property must have a value");
182        }
183    }
184
185    /**
186     * Gets the property description for the current property.
187     * @param instance the component instance on which looking for the property.
188     * @return the property description associated with the current property
189     * or <code>null</code> if not found.
190     */
191    public PropertyDescription getPropertyDescription(ComponentInstance instance) {
192        PrimitiveInstanceDescription desc = (PrimitiveInstanceDescription) instance.getInstanceDescription();
193        PropertyDescription[] props = desc.getProperties();
194
195        for (int i = 0; i < props.length; i++) {
196            if ((m_name != null && m_name.equals(props[i].getName()))
197                    || (m_field != null && m_field.equals(props[i].getName()))) {
198                return props[i];
199            }
200        }
201
202        return null;
203    }
204
205}
