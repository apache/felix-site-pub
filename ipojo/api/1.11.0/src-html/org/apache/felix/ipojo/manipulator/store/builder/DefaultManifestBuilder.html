040/**
041 * A {@code DefaultManifestBuilder} handles the knowledge of iPOJO Manifest building.
042 * It is responsible to update a given Manifest with all gathered (additional)
043 * referenced packages (from the metadata.xml) + other iPOJO specific additions.
044 *
045 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
046 */
047public class DefaultManifestBuilder implements ManifestBuilder {
048
049    /**
050     * Referenced packages (by the composites).
051     */
052    private List<String> m_referredPackages = new ArrayList<String>();
053
054    /**
055     * Computed metadatas from the bundle (XML files + annotations).
056     */
057    private List<Element> m_metadata = new ArrayList<Element>();
058
059    /**
060     * The metadata renderer used to print Elements.
061     */
062    private MetadataRenderer m_renderer;
063
064    /**
065     * Add all given package names in the referred packages list
066     * @param packageNames additional packages
067     */
068    public void addReferredPackage(Set<String> packageNames) {
069        m_referredPackages.addAll(packageNames);
070    }
071
072    /**
073     * Add all given metadata
074     * @param metadatas additional metadata
075     */
076    public void addMetada(Collection<Element> metadatas) {
077        m_metadata.addAll(metadatas);
078    }
079
080    public void setMetadataRenderer(MetadataRenderer renderer) {
081        m_renderer = renderer;
082    }
083
084    /**
085     * Update the given manifest.
086     * @param original original manifest to be modified
087     * @return modified manifest
088     */
089    public Manifest build(final Manifest original) {
090        Attributes att = original.getMainAttributes();
091
092         // Set the imports (add ipojo and handler namespaces
093        setImports(att);
094         // Add iPOJO-Component
095        setPOJOMetadata(att);
096         // Add iPOJO to the creators
097        setCreatedBy(att);
098
099        return original;
100    }
101
102    /**
103     * Add imports to the given manifest attribute list. This method add ipojo imports and handler imports (if needed).
104     * @param att : the manifest attribute list to modify.
105     */
106    private void setImports(Attributes att) {
107        Map<String, Map<String, String>> imports = parseHeader(att.getValue("Import-Package"));
108        Map<String, String> ver = new TreeMap<String, String>();
109        ver.put("version", Constants.getPackageImportClause());
110        if (!imports.containsKey("org.apache.felix.ipojo")) {
111            imports.put("org.apache.felix.ipojo", ver);
112        }
113        if (!imports.containsKey("org.apache.felix.ipojo.architecture")) {
114            imports.put("org.apache.felix.ipojo.architecture", ver);
115        }
116        if (!imports.containsKey("org.osgi.service.cm")) {
117            Map<String, String> verCM = new TreeMap<String, String>();
118            verCM.put("version", "1.2");
119            imports.put("org.osgi.service.cm", verCM);
120        }
121        if (!imports.containsKey("org.osgi.service.log")) {
122            Map<String, String> verCM = new TreeMap<String, String>();
123            verCM.put("version", "1.3");
124            imports.put("org.osgi.service.log", verCM);
125        }
126
127        // Add referred imports from the metadata
128        for (int i = 0; i < m_referredPackages.size(); i++) {
129            String pack = m_referredPackages.get(i);
130            imports.put(pack, new TreeMap<String, String>());
131        }
132
133        // Write imports
134        att.putValue("Import-Package", printClauses(imports, "resolution:"));
135    }
136
137    /**
138     * Add iPOJO-Components to the given manifest attribute list. This method add the
139     * {@literal iPOJO-Components} header and its value (according to the metadata)
140     * to the manifest.
141     * @param att the manifest attribute list to modify.
142     */
143    private void setPOJOMetadata(Attributes att) {
144        StringBuilder meta = new StringBuilder();
145        for (Element metadata : m_metadata) {
146            meta.append(m_renderer.render(metadata));
147        }
148        if (meta.length() != 0) {
149            att.putValue("iPOJO-Components", meta.toString());
150        }
151    }
152
153    /**
154     * Set the create-by in the manifest.
155     * @param att : manifest attribute.
156     */
157    private void setCreatedBy(Attributes att) {
158        String prev = att.getValue("Created-By");
159        if (prev == null) {
160            att.putValue("Created-By", "iPOJO " + Constants.getVersion());
161        } else {
162            if (prev.indexOf("iPOJO") == -1) {
163                // Avoid appending iPOJO several times
164                att.putValue("Created-By", prev + " & iPOJO " + Constants.getVersion());
165            }
166        }
167    }
168
169    /**
170     * Standard OSGi header parser. This parser can handle the format
171     * <pre>
172     * clauses ::= clause ( ',' clause ) +
173     * clause ::= name ( ';' name ) (';' key '=' value )
174     * </pre>
175     * This is mapped to a Map { name => Map { attr|directive => value } }
176     *
177     * @param value String to parse.
178     * @return parsed map.
179     */
180    protected Map<String, Map<String, String>> parseHeader(String value) {
181        if (value == null || value.trim().length() == 0) {
182            return new HashMap<String, Map<String, String>>();
183        }
184
185        Map<String, Map<String, String>> result = new HashMap<String, Map<String, String>>();
186        QuotedTokenizer qt = new QuotedTokenizer(value, ";=,");
187        char del;
188        do {
189            boolean hadAttribute = false;
190            Map<String, String> clause = new HashMap<String, String>();
191            List<String> aliases = new ArrayList<String>();
192            aliases.add(qt.nextToken());
193            del = qt.getSeparator();
194            while (del == ';') {
195                String adname = qt.nextToken();
196                if ((del = qt.getSeparator()) != '=') {
197                    if (hadAttribute) {
198                        throw new IllegalArgumentException("Header contains name field after attribute or directive: " + adname + " from " + value);
199                    }
200                    aliases.add(adname);
201                } else {
202                    String advalue = qt.nextToken();
203                    clause.put(adname, advalue);
204                    del = qt.getSeparator();
205                    hadAttribute = true;
206                }
207            }
208            for (Iterator<String> i = aliases.iterator(); i.hasNext();) {
209                result.put(i.next(), clause);
210            }
211        } while (del == ',');
212        return result;
213    }
214
215    /**
216     * Print a standard Map based OSGi header.
217     *
218     * @param exports : map { name => Map { attribute|directive => value } }
219     * @param allowedDirectives list of allowed directives.
220     * @return the clauses
221     */
222    private String printClauses(Map<String, Map<String, String>> exports, String allowedDirectives) {
223        StringBuffer sb = new StringBuffer();
224        String del = "";
225
226        for (Iterator i = exports.entrySet().iterator(); i.hasNext();) {
227            Map.Entry entry = (Map.Entry) i.next();
228            String name = (String) entry.getKey();
229            Map map = (Map) entry.getValue();
230            sb.append(del);
231            sb.append(name);
232
233            for (Iterator j = map.entrySet().iterator(); j.hasNext();) {
234                Map.Entry entry2 = (Map.Entry) j.next();
235                String key = (String) entry2.getKey();
236
237                // Skip directives we do not recognize
238                if (key.endsWith(":") && allowedDirectives.indexOf(key) < 0) {
239                    continue;
240                }
241
242                String value = (String) entry2.getValue();
243                sb.append(";");
244                sb.append(key);
245                sb.append("=");
246                boolean dirty = value.indexOf(',') >= 0 || value.indexOf(';') >= 0;
247                if (dirty) {
248                    sb.append("\"");
249                }
250                sb.append(value);
251                if (dirty) {
252                    sb.append("\"");
253                }
254            }
255            del = ", ";
256        }
257        return sb.toString();
258    }
