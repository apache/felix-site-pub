022import org.apache.felix.ipojo.extender.internal.Extender;
023import org.apache.felix.ipojo.util.Log;
024import org.apache.felix.ipojo.util.Logger;
025import org.apache.felix.ipojo.util.ServiceLocator;

028import org.osgi.service.cm.ConfigurationException;

033/**
034 * An object tracking configuration from the configuration admin. It delegates to the underlying factories or
035 * component instance the action.
036 * <p/>
037 * This class implements a Configuration Listener, so events are received asynchronously.
038 */
039public class ConfigurationTracker implements ConfigurationListener {
040
041    /**
042     * The tracker instance.
043     */
044    private static ConfigurationTracker m_singleton;
045    private final ServiceRegistration m_registration;
046    private final BundleContext m_context;
047    private final Logger m_logger;
048    private Map<String, IPojoFactory> m_factories = new HashMap<String, IPojoFactory>();
049
050    public ConfigurationTracker() {
051        m_context = Extender.getIPOJOBundleContext();
052        m_logger = new Logger(m_context, "iPOJO Configuration Admin listener", Log.INFO);
053        // register as listener for configurations
054        Dictionary<String, Object> props = new Hashtable<String, Object>();
055        props.put(Constants.SERVICE_DESCRIPTION, "iPOJO Configuration Admin Listener");
056        props.put(Constants.SERVICE_VENDOR, "The Apache Software Foundation");
057        m_registration = m_context.registerService(ConfigurationListener.class.getName(), this, props);
058    }
059
060    public static void initialize() {
061        synchronized (ConfigurationTracker.class) {
062            if (m_singleton == null) {
063                m_singleton = new ConfigurationTracker();
064            }
065        }
066    }
067
068    public static void shutdown() {
069        m_singleton.dispose();
070        m_singleton = null;
071    }
072
073    public static ConfigurationTracker get() {
074        return m_singleton;
075    }
076
077    /**
078     * This method must be called by the iPOJO System itself, and only once.
079     */
080    public synchronized void dispose() {
081        if (m_registration != null) {
082            m_registration.unregister();
083        }
084        m_factories.clear();
085    }
086
087    public synchronized void registerFactory(IPojoFactory factory) {
088        m_factories.put(factory.getFactoryName(), factory);
089
090        ServiceLocator<ConfigurationAdmin> locator = new ServiceLocator<ConfigurationAdmin>(ConfigurationAdmin
091                .class, m_context);
092        final ConfigurationAdmin admin = locator.get();
093        if (admin == null) {
094            return;
095        }
096
097        List<Configuration> configurations = findFactoryConfiguration(admin, factory);
098        for (Configuration configuration : configurations) {
099            try {
100                factory.updated(configuration.getPid(), configuration.getProperties());
101            } catch (ConfigurationException e) {
102                m_logger.log(Log.ERROR, "Cannot reconfigure instance " + configuration.getPid() + " from " +
103                        configuration.getFactoryPid() + " with the configuration : " + configuration.getProperties(),
104                        e);
105            }
106        }
107    }
108
109    public synchronized void instanceCreated(ComponentInstance instance) {
110        ServiceLocator<ConfigurationAdmin> locator = new ServiceLocator<ConfigurationAdmin>(ConfigurationAdmin
111                .class, m_context);
112        final ConfigurationAdmin admin = locator.get();
113        if (admin == null) {
114            return;
115        }
116
117        Configuration configuration = findSingletonConfiguration(admin, instance.getInstanceName());
118        if (configuration != null) {
119            Hashtable<String, Object> conf = copyConfiguration(configuration);
120            if (!conf.containsKey(Factory.INSTANCE_NAME_PROPERTY)) {
121                conf.put(Factory.INSTANCE_NAME_PROPERTY, configuration.getPid());
122            }
123            try {
124                instance.getFactory().reconfigure(conf);
125            } catch (UnacceptableConfiguration unacceptableConfiguration) {
126                m_logger.log(Log.ERROR, "Cannot reconfigure the instance " + configuration.getPid() + " - the " +
127                        "configuration is unacceptable", unacceptableConfiguration);
128            } catch (MissingHandlerException e) {
129                m_logger.log(Log.ERROR, "Cannot reconfigure the instance " + configuration.getPid() + " - factory is " +
130                        "invalid", e);
131            }
132        }
133    }
134
135    public synchronized void unregisterFactory(IPojoFactory factory) {
136        m_factories.remove(factory.getFactoryName());
137    }
138
139    public void configurationEvent(ConfigurationEvent event) {
140        String pid = event.getPid();
141        String factoryPid = event.getFactoryPid();
142
143        if (factoryPid == null) {
144            ComponentInstance instance = retrieveInstance(pid);
145            if (instance != null) {
146                manageConfigurationEventForSingleton(instance, event);
147            }
148        } else {
149            IPojoFactory factory = retrieveFactory(factoryPid);
150            if (factory != null) {
151                manageConfigurationEventForFactory(factory, event);
152            }
153            // Else the factory is unknown, do nothing.
154        }
155
156    }
157
158    private void manageConfigurationEventForFactory(final IPojoFactory factory, final ConfigurationEvent event) {
159        ServiceLocator<ConfigurationAdmin> locator = new ServiceLocator<ConfigurationAdmin>(ConfigurationAdmin
160                .class, m_context);
161
162        switch (event.getType()) {
163            case ConfigurationEvent.CM_DELETED:
164                factory.deleted(event.getPid());
165                break;
166            case ConfigurationEvent.CM_UPDATED:
167                final ConfigurationAdmin admin = locator.get();
168                if (admin == null) {
169                    break;
170                }
171                final Configuration config = getConfiguration(admin, event.getPid(),
172                        factory.getBundleContext().getBundle());
173                if (config != null) {
174                    try {
175                        factory.updated(event.getPid(), config.getProperties());
176                    } catch (org.osgi.service.cm.ConfigurationException e) {
177                        m_logger.log(Log.ERROR, "Cannot reconfigure instance " + event.getPid() + " with the new " +
178                                "configuration " + config.getProperties(), e);
179                    }
180                }
181            default:
182                // To nothing.
183        }
184
185        locator.unget();
186    }
187
188    private void manageConfigurationEventForSingleton(final ComponentInstance instance,
189                                                      final ConfigurationEvent event) {
190        ServiceLocator<ConfigurationAdmin> locator = new ServiceLocator<ConfigurationAdmin>(ConfigurationAdmin
191                .class, m_context);
192
193        switch (event.getType()) {
194            case ConfigurationEvent.CM_DELETED:
195                instance.dispose();
196                break;
197            case ConfigurationEvent.CM_UPDATED:
198                final ConfigurationAdmin admin = locator.get();
199                if (admin == null) {
200                    break;
201                }
202                final Configuration config = getConfiguration(admin, event.getPid(),
203                        instance.getFactory().getBundleContext().getBundle());
204                if (config != null) {
205                    Hashtable<String, Object> conf = copyConfiguration(config);
206                    if (!conf.containsKey(Factory.INSTANCE_NAME_PROPERTY)) {
207                        conf.put(Factory.INSTANCE_NAME_PROPERTY, event.getPid());
208                    }
209                    try {
210                        instance.getFactory().reconfigure(conf);
211                    } catch (UnacceptableConfiguration unacceptableConfiguration) {
212                        m_logger.log(Log.ERROR, "Cannot reconfigure the instance " + event.getPid() + " - the " +
213                                "configuration is unacceptable", unacceptableConfiguration);
214                    } catch (MissingHandlerException e) {
215                        m_logger.log(Log.ERROR, "Cannot reconfigure the instance " + event.getPid() + " - factory is " +
216                                "invalid", e);
217                    }
218                }
219            default:
220                // To nothing.
221        }
222
223        locator.unget();
224    }
225
226    private Hashtable<String, Object> copyConfiguration(Configuration config) {
227        Hashtable<String, Object> conf = new Hashtable<String, Object>();
228        // Copy configuration
229        Enumeration keys = config.getProperties().keys();
230        while (keys.hasMoreElements()) {
231            String key = (String) keys.nextElement();
232            conf.put(key, config.getProperties().get(key));
233        }
234        return conf;
235    }
236
237    private IPojoFactory retrieveFactory(String factoryPid) {
238        synchronized (this) {
239            return m_factories.get(factoryPid);
240        }
241    }
242
243    private ComponentInstance retrieveInstance(String instanceName) {
244        Collection<IPojoFactory> factories;
245        synchronized (this) {
246            factories = m_factories.values();
247        }
248        for (IPojoFactory factory : factories) {
249            ComponentInstance instance = factory.getInstanceByName(instanceName);
250            if (instance != null) {
251                return instance;
252            }
253        }
254        return null;
255    }
256
257    private Configuration getConfiguration(final ConfigurationAdmin admin, final String pid,
258                                           final Bundle bundle) {
259        if (admin == null) {
260            return null;
261        }
262
263        try {
264            // Even if it is possible, we don't build the filter with bundle.location to detect the case where the
265            // configuration exists but can't be managed by iPOJO.
266            final Configuration cfg = admin.getConfiguration(pid);
267            final String bundleLocation = bundle.getLocation();
268            if (cfg.getBundleLocation() == null || bundleLocation.equals(cfg.getBundleLocation())
269                    || m_context.getBundle().getLocation().equals(cfg.getBundleLocation())) {
270                cfg.setBundleLocation(bundleLocation);
271                return cfg;
272            }
273
274            // Multi-location
275            if (cfg.getBundleLocation().startsWith("?")) {
276                if (bundle.hasPermission(new ConfigurationPermission(cfg.getBundleLocation(), "target"))) {
277                    return cfg;
278                }
279            }
280
281            // configuration belongs to another bundle, cannot be used here
282            m_logger.log(Log.ERROR, "Cannot use configuration pid=" + pid + " for bundle "
283                    + bundleLocation + " because it belongs to bundle " + cfg.getBundleLocation());
284        } catch (IOException ioe) {
285            m_logger.log(Log.WARNING, "Failed reading configuration for pid=" + pid, ioe);
286        }
287
288        return null;
289    }
290
291    public List<Configuration> findFactoryConfiguration(final ConfigurationAdmin admin, final IPojoFactory factory) {
292        final String filter = "(service.factoryPid=" + factory.getFactoryName() + ")";
293        return findConfigurations(admin, filter);
294    }
295
296    public Configuration findSingletonConfiguration(final ConfigurationAdmin admin, final String pid) {
297        final String filter = "(service.pid=" + pid + ")";
298        List<Configuration> list = findConfigurations(admin, filter);
299        if (list.isEmpty()) {
300            return null;
301        } else {
302            return list.get(0);
303        }
304    }
305
306    private List<Configuration> findConfigurations(final ConfigurationAdmin admin, final String filter) {
307        List<Configuration> configurations = Collections.emptyList();
308        if (admin == null) {
309            return configurations;
310        }
311
312        try {
313            Configuration[] list = admin.listConfigurations(filter);
314            if (list == null) {
315                return configurations;
316            } else {
317                return Arrays.asList(list);
318            }
319        } catch (InvalidSyntaxException e) {
320            m_logger.log(Log.ERROR, "Invalid Configuration selection filter " + filter, e);
321        } catch (IOException e) {
322            m_logger.log(Log.ERROR, "Error when retrieving configurations for filter=" + filter, e);
323        }
324        return configurations;
325    }
