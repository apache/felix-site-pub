177    /**
178     * Configures the instance manager.
179     * Sets the class name, and the instance name as well as the factory method.
180     * Initializes handlers.
181     * @param metadata the component type metadata
182     * @param configuration the configuration of the instance
183     * @throws ConfigurationException if the metadata are not correct
184     */
185    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
186        m_className = metadata.getAttribute("classname");
187
188        // Add the name
189        m_name = (String) configuration.get(Factory.INSTANCE_NAME_PROPERTY);
190
191        // Check if an object is injected in the instance
192        Object obj = configuration.get("instance.object");
193        if (obj != null) {
194            m_pojoObjects = new ArrayList(1);
195            m_pojoObjects.add(obj);
196        }
197
198        // Get the factory method if presents.
199        m_factoryMethod = (String) metadata.getAttribute("factory-method");
200
201        // Create the standard handlers and add these handlers to the list
202        for (int i = 0; i < m_handlers.length; i++) {
203            m_handlers[i].init(this, metadata, configuration);
204        }
205        
206        // Fix for Felix-3576
207        handleBCInjections();
208
209        // Check that the constructor parameter are continuous.
210        if (m_constructorRegistration != null) {
211            for (int i = 0; i < m_constructorRegistration.size(); i++) {
212                if (! m_constructorRegistration.containsKey(new Integer(i))) {
213                    throw new ConfigurationException("The constructor parameter " + i + " is not managed");
214                }
215            }
216        }
217    }

219    /**
220     * BundleContext injection is not registered with the InstanceManager.
221     * We're iterating through factory's all constructors and register first
222     * BundleContext parameter as constructor injection. So rest of the code
223     * don't have to do anything to handle BundleContext mixed with other
224     * injections.
225     * 
226     * @throws ConfigurationException
227     */
228    private void handleBCInjections() throws ConfigurationException
229    {
230        MethodMetadata[] constructors = getFactory().getPojoMetadata().getConstructors();
231        for(int i=0; i < constructors.length; i++ )
232        {
233                String[] ctorArguments = constructors[i].getMethodArguments();
234                
235                for(int index = 0; index < ctorArguments.length; index++ )
236                {
237                        if(ctorArguments[index].equals(BundleContext.class.getName()))
238                        {
239                                //Check if its used with only other injections.
240                                boolean injectionsConsistent = true;
241                                for(int siblingIndex = 0; siblingIndex < ctorArguments.length; siblingIndex++)
242                                {
243                                        if(siblingIndex == index){
244                                                continue;
245                                        }
246                                        
247                                        String injectionType = ctorArguments[siblingIndex];
248                                        if(m_constructorRegistration.containsKey(new Integer(siblingIndex)))
249                                        {
250                                                ConstructorInjector siblingInjector = 
251                                                                (ConstructorInjector)m_constructorRegistration.get(new Integer(siblingIndex));
252                                                Class injectorClass = siblingInjector.getConstructorParameterType(siblingIndex);
253                                                
254                                                if(injectorClass != null && ! injectorClass.getName().equals(injectionType))
255                                                {
256                                                        injectionsConsistent = false;
257                                                        break;
258                                                }
259                                        }
260                                        else
261                                        {
262                                                injectionsConsistent = false;
263                                                break;
264                                        }
265                                }
266                                
267                                if(injectionsConsistent)
268                                {
269                                        Property contextInjection = 
270                                                new Property("__context", null, null, index, null, 
271                                                                BundleContext.class.getName(), this, null);
272                                
273                                contextInjection.setValue(getContext());                                
274                                register(index, contextInjection);
275                                
276                                // We register the first valid BC injection.
277                                break;
278                                }
279                        }
280                }
281        }
282    }

306    /**
307     * Returns a specified handler.
308     * This method allows cross-handler interactions.
309     * This must does not need a synchronized block as the handler set is constant.
310     * @param name the class name of the handler to find or its qualified name (namespace:name)
311     * @return the handler, or null if not found
312     */
313    public Handler getHandler(String name) {
314        for (int i = 0; i < m_handlers.length; i++) {
315            HandlerFactory fact = (HandlerFactory) m_handlers[i].getHandler().getHandlerManager().getFactory();
316            if (fact.getHandlerName().equals(name)) {
317                return m_handlers[i].getHandler();
318            }
319        }
320        return null;
321    }

323    /**
324     * Gives access to a field value of the first created pojo.
325     * This method processes by analyzing both managed fields and pojo fields (by reflection).
326     * If no pojo were already created try only on managed fields.
327     * @param fieldName the field name.
328     * @return the field value, <code>null</code> is returned if the value is managed and not already set.
329     */
330    public synchronized Object getFieldValue(String fieldName) {
331        if (m_pojoObjects == null) {
332            return getFieldValue(fieldName, null);
333        } else {
334            return getFieldValue(fieldName, m_pojoObjects.get(0)); // Use the first pojo.
335        }
336    }

338    /**
339     * Gives access to a field value to the given created pojo.
340     * This method processes by analyzing both managed fields and pojo fields (by reflection).
341     * If the given pojo is <code>null</code>, tries only on managed fields.
342     * @param fieldName the field name.
343     * @param pojo  the pojo on which computing field value.
344     * @return the field value, <code>null</code> is returned if the value is managed and not already set.
345     */
346    public synchronized Object getFieldValue(String fieldName, Object pojo) {
347        Object setByContainer = null;
348
349        if (m_fields != null) {
350            setByContainer = m_fields.get(fieldName);
351        }
352
353        if (setByContainer == null && pojo != null) { // In the case of no given pojo, return null.
354            // If null either the value was not already set or has the null value.
355            try {
356                Field field = pojo.getClass().getDeclaredField(fieldName);
357                if (!field.isAccessible()) {
358                    field.setAccessible(true);
359                }
360                return field.get(pojo);
361            } catch (SecurityException e) {
362                m_logger.log(Logger.ERROR, "Cannot reflect on field " + fieldName + " to obtain the value : " + e.getMessage());
363            } catch (NoSuchFieldException e) {
364                m_logger.log(Logger.ERROR, "Cannot reflect on field " + fieldName + " to obtain the value : " + e.getMessage());
365            } catch (IllegalArgumentException e) {
366                m_logger.log(Logger.ERROR, "Cannot reflect on field " + fieldName + " to obtain the value : " + e.getMessage());
367            } catch (IllegalAccessException e) {
368                m_logger.log(Logger.ERROR, "Cannot reflect on field " + fieldName + " to obtain the value : " + e.getMessage());
369            }
370            return null;
371        } else {
372            return setByContainer;
373        }
374    }

376    /**
377     * Starts the instance manager.
378     * This method activates plugged handlers,
379     * and computes the initial instance state.
380     */
381    public void start() {
382        synchronized (this) {
383            if (m_state != STOPPED) { // Instance already started
384                return;
385            } else {
386                m_state = -2; // Temporary state.
387            }
388        }
389
390        // Plug handler descriptions
391        Handler[] handlers = getRegisteredHandlers();
392        for (int i = 0; i < handlers.length; i++) {
393            m_description.addHandler(handlers[i].getDescription());
394        }
395
396        for (int i = 0; i < m_handlers.length; i++) {
397            m_handlers[i].addInstanceStateListener(this);
398            try {
399                m_handlers[i].start();
400            } catch (IllegalStateException e) {
401                m_logger.log(Logger.ERROR, e.getMessage());
402                stop();
403                throw e;
404            }
405        }
406
407        // Is an object already contained (i.e. injected)
408        if (m_pojoObjects != null && ! m_pojoObjects.isEmpty()) {
409            managedInjectedObject();
410        }
411
412        for (int i = 0; i < m_handlers.length; i++) {
413            if (m_handlers[i].getState() != VALID) {
414                setState(INVALID);
415                return;
416            }
417        }
418        setState(VALID);
419    }

421    /**
422     * Stops the instance manager.
423     * This methods sets the instance state to {@link ComponentInstance#STOPPED},
424     * disables attached handlers, and notifies listeners ({@link InstanceStateListener})
425     * of the instance stopping process.
426     */
427    public void stop() {
428        List listeners = null;
429        synchronized (this) {
430            if (m_state == STOPPED) { // Instance already stopped
431                return;
432            }
433            m_stateQueue.clear();
434            m_inTransition = false;
435        }
436
437        setState(INVALID); // Must be called outside a synchronized block.
438
439        // Stop all the handlers
440        for (int i = m_handlers.length - 1; i > -1; i--) {
441            m_handlers[i].removeInstanceStateListener(this);
442            m_handlers[i].stop();
443        }
444
445        synchronized (this) {
446            m_state = STOPPED;
447            if (m_listeners != null) {
448                listeners = new ArrayList(m_listeners); // Stack confinement
449            }
450            m_pojoObjects = null;
451        }
452
453        if (listeners != null) {
454            for (int i = 0; i < listeners.size(); i++) {
455                ((InstanceStateListener) listeners.get(i)).stateChanged(this, STOPPED);
456            }
457        }
458    }

460    /**
461     * Disposes the instance.
462     * This method does the following process:
463     * <li>Stop the instance is not {@link ComponentInstance#STOPPED}</li>
464     * <li>Notifies listeners {@link InstanceStateListener} of the destruction</li>
465     * <li>Disposes attached handlers</li>
466     * <li>Clears structures</li>
467     * @see org.apache.felix.ipojo.ComponentInstance#dispose()
468     */
469    public void dispose() {
470        List listeners = null;
471        int state; // Will be confined in stack.
472        synchronized (this) {
473            state = m_state; // Stack confinement
474            if (m_listeners != null) {
475                listeners = new ArrayList(m_listeners); // Stack confinement
476            }
477            m_listeners = null;
478        }
479
480        if (state > STOPPED) { // Valid or invalid
481            stop(); // Does not hold the lock.
482        }
483
484        synchronized (this) {
485            m_state = DISPOSED;
486        }
487
488        for (int i = 0; listeners != null && i < listeners.size(); i++) {
489            ((InstanceStateListener) listeners.get(i)).stateChanged(this, DISPOSED);
490        }
491
492        for (int i = m_handlers.length - 1; i > -1; i--) {
493            m_handlers[i].dispose();
494        }
495
496        synchronized (this) {
497            m_factory.disposed(this);
498            m_clazz = null;
499            // Do not clean registration map, so injection still works
500            // after disposal for late callbacks.
501        }
502    }

504    /**
505     * Sets the state of the component instance.
506     * If the state changes, calls the {@link PrimitiveHandler#stateChanged(int)} method on the attached handlers.
507     * This method has a reentrant mechanism. If in the flow of the first call the method is called another times,
508     * the second call is stored and executed after the first one finished.
509     * @param state the new state
510     */
511    public void setState(int state) {
512        int originalState = -2;
513        List listeners = null;
514        synchronized (this) {
515            if (m_inTransition) {
516                m_stateQueue.add(new Integer(state));
517                return;
518            }
519
520            if (m_state != state) {
521                m_inTransition = true;
522                originalState = m_state; // Stack confinement.
523                m_state = state;
524                if (m_listeners != null) {
525                    listeners = new ArrayList(m_listeners); // Stack confinement.
526                }
527            }
528        }
529
530        // This section can be executed only by one thread at the same time. The m_inTransition pseudo semaphore block access to this section.
531        if (m_inTransition) { // Check that we are really changing.
532            if (state > originalState) {
533                // The state increases (Stopped = > IV, IV => V) => invoke handlers from the higher priority to the lower
534                try {
535                    for (int i = 0; i < m_handlers.length; i++) {
536                        m_handlers[i].getHandler().stateChanged(state);
537                    }
538                } catch (IllegalStateException e) {
539                    // When an illegal state exception happens, the instance manager must be stopped immediately.
540                    stop();
541                    m_logger.log(Logger.ERROR, e.getMessage(), e);
542                    return;
543                }
544            } else {
545                // The state decreases (V => IV, IV = > Stopped, Stopped => Disposed)
546                try {
547                    for (int i = m_handlers.length - 1; i > -1; i--) {
548                        m_handlers[i].getHandler().stateChanged(state);
549                    }
550                } catch (IllegalStateException e) {
551                    // When an illegal state exception happens, the instance manager must be stopped immediately.
552                    stop();
553                    m_logger.log(Logger.ERROR, e.getMessage());
554                    return;
555                }
556            }
557        }
558
559        if (listeners != null) {
560            for (int i = 0; i < listeners.size(); i++) {
561                ((InstanceStateListener) listeners.get(i)).stateChanged(this, state);
562            }
563        }
564
565        synchronized (this) {
566            m_inTransition = false;
567            if (!m_stateQueue.isEmpty()) {
568                int newState = ((Integer) (m_stateQueue.remove(0))).intValue();
569                setState(newState);
570            }
571        }
572    }

635     * Loads the manipulated class.
636     */
637    private void load() {
638        try {
639            m_clazz = m_factory.loadClass(m_className);
640        } catch (ClassNotFoundException e) {
641            m_logger.log(Logger.ERROR, "[" + m_name + "] Class not found during the loading phase : " + e.getMessage(), e);

643            return;
644        }

658    /**
659     * Creates a POJO objects.
660     * This method is not synchronized and does not require any locks.
661     * If a {@link InstanceManager#m_factoryMethod} is specified,
662     * this method called this static method to creates the object.
663     * Otherwise, the methods uses the regular constructor.
664     * All those methods can receive the {@link BundleContext} in
665     * argument.
666     * @return the created object or <code>null</code> if an error
667     * occurs during the creation.
668     */
669    protected Object createObject() {
670        if (m_clazz == null) {
671            load();
672        }
673
674        // The following code doesn't need to be synchronized as is deal only with immutable fields.
675        Object instance = null;
676        if (m_factoryMethod == null) {
677            // No factory-method, we use the constructor.
678            try {
679                // Try to find the correct constructor.
680                if (m_constructorRegistration != null) {
681                    // Initialize the injected values and types
682                    // We have the IM first.
683                    Object[] values = new Object[m_constructorRegistration.size() + 1];
684                    Class[] types = new Class[m_constructorRegistration.size() + 1];
685                    values[0] = this;
686                    types[0] = InstanceManager.class;
687
688                    // Iterate over the constructor injector
689                    for (int i = 0; i < m_constructorRegistration.size(); i++) {
690                        ConstructorInjector injector = (ConstructorInjector)
691                            m_constructorRegistration.get(new Integer(i));
692                        Object v = injector.getConstructorParameter(i);
693                        if (v != null) {
694                            values[i + 1] = v;
695                            Class t = injector.getConstructorParameterType(i);
696                            if (t == null) {
697                                t = v.getClass();
698                            }
699                            types[i + 1] = t;
700                        }
701                    }
702                    // Find the constructor.
703                    Constructor cst = m_clazz.getDeclaredConstructor(types);
704                    if (! cst.isAccessible()) {
705                        cst.setAccessible(true);
706                    }
707                    String methodId = MethodMetadata.computeMethodId(cst);
708                    onEntry(null, methodId,  values);
709                    instance = cst.newInstance(values);
710                    onExit(instance, methodId, instance);
711                } else {
712                    // Old semantic
713                    // Try to find if there is a constructor with a bundle context as parameter :
714                    try {
715                        Constructor cst = m_clazz.getDeclaredConstructor(new Class[] { InstanceManager.class, BundleContext.class });
716                        if (! cst.isAccessible()) {
717                            cst.setAccessible(true);
718                        }
719                        Object[] args = new Object[] { this, m_context };
720                        onEntry(null, MethodMetadata.BC_CONSTRUCTOR_ID,  new Object[] {m_context});
721                        instance = cst.newInstance(args);
722                        onExit(instance, MethodMetadata.BC_CONSTRUCTOR_ID, instance);
723                    } catch (NoSuchMethodException e) {
724                        // Create an instance if no instance are already created with <init>()BundleContext
725                        if (instance == null) {
726                            Constructor cst = m_clazz.getDeclaredConstructor(new Class[] { InstanceManager.class });
727                            if (! cst.isAccessible()) {
728                                cst.setAccessible(true);
729                            }
730                            Object[] args = new Object[] {this};
731                            onEntry(null, MethodMetadata.EMPTY_CONSTRUCTOR_ID, new Object[0]);
732                            instance = cst.newInstance(args);
733                            onExit(instance, MethodMetadata.EMPTY_CONSTRUCTOR_ID, instance);
734                        }
735                    }
736                }
737
738            } catch (IllegalAccessException e) {
739                m_logger.log(Logger.ERROR,
740                                          "[" + m_name + "] createInstance -> The POJO constructor is not accessible : " + e.getMessage(), e);
741                stop();
742                throw new RuntimeException("Cannot create a POJO instance, the POJO constructor is not accessible", e);
743            } catch (SecurityException e) {
744                m_logger.log(
745                                          Logger.ERROR,
746                                          "["
747                                                  + m_name
748                                                  + "] createInstance -> The POJO constructor is not accessible (security reason) : "
749                                                  + e.getMessage(), e);
750                stop();
751                throw new RuntimeException("Cannot create a POJO instance, the POJO constructor is not accessible", e);
752            } catch (InvocationTargetException e) {
753                m_logger.log(
754                                          Logger.ERROR,
755                                          "["
756                                                  + m_name
757                                                  + "] createInstance -> Cannot invoke the constructor method - the constructor throws an exception : "
758                                                  + e.getTargetException().getMessage(), e.getTargetException());
759                onError(null, m_className, e.getTargetException());
760                stop();
761                throw new RuntimeException("Cannot create a POJO instance, the POJO constructor has thrown an exception", e.getTargetException());
762            } catch (NoSuchMethodException e) {
763                m_logger.log(Logger.ERROR,
764                                          "[" + m_name + "] createInstance -> Cannot invoke the constructor (method not found) : " + e.getMessage(), e);
765                stop();
766                throw new RuntimeException("Cannot create a POJO instance, the POJO constructor cannot be found", e);
767            } catch (Throwable e) {
768                // Catch every other possible error and runtime exception.
769                m_logger.log(Logger.ERROR,
770                        "[" + m_name + "] createInstance -> The POJO constructor invocation failed : " + e.getMessage(), e);
771                stop();
772                throw new RuntimeException("Cannot create a POJO instance, the POJO constructor invocation has thrown an exception", e);
773            }
774        } else {
775            try {
776                // Build the pojo object with the factory-method.
777                Method factory = null;
778                // Try with the bundle context
779                try {
780                    factory = m_clazz.getDeclaredMethod(m_factoryMethod, new Class[] { BundleContext.class });
781                    if (! factory.isAccessible()) {
782                        factory.setAccessible(true);
783                    }
784                    Object[] args = new Object[] { m_context };
785                    onEntry(null, m_className, args);
786                    instance = factory.invoke(null, new Object[] { m_context });
787                } catch (NoSuchMethodException e1) {
788                    // Try without the bundle context
789                    try {
790                        factory = m_clazz.getDeclaredMethod(m_factoryMethod, new Class[0]);
791                        if (! factory.isAccessible()) {
792                            factory.setAccessible(true);
793                        }
794                        Object[] args = new Object[0];
795                        onEntry(null, m_className, args);
796                        instance = factory.invoke(null, args);
797                    } catch (NoSuchMethodException e2) {
798                        // Error : factory-method not found
799                        m_logger.log(
800                                                  Logger.ERROR,
801                                                  "["
802                                                          + m_name
803                                                          + "] createInstance -> Cannot invoke the factory-method (method not found) : "
804                                                          + e2.getMessage(), e2);
805                        stop();
806                        throw new RuntimeException("Cannot create a POJO instance, the factory-method cannot be found", e2);
807                    }
808                }
809
810                // Now call the setInstanceManager method.
811                // Find declaring super class.
812                Class declaringClass = instance.getClass();
813                Method method = null;
814                while (declaringClass != null && method == null) {
815                    try {
816                        method = declaringClass.getDeclaredMethod("_setInstanceManager",
817                                new Class[] { InstanceManager.class });
818                    } catch (NoSuchMethodException e) {
819                        //Do nothing
820                    }
821
822                    declaringClass = declaringClass.getSuperclass();
823                }
824
825                if (method == null) {
826                    // Error : _setInstanceManager method is missing
827                    m_logger
828                            .log(
829                                 Logger.ERROR,
830                                 "["
831                                         + m_name
832                                         + "] createInstance -> Cannot invoke the factory-method (the _setInstanceManager method does not exist");
833                    stop();
834                    throw new RuntimeException("Cannot create a POJO instance, the factory-method cannot be found");
835                }
836
837                if (!method.isAccessible()) {
838                    method.setAccessible(true);
839                }
840                method.invoke(instance, new Object[] { this });
841                onExit(null, m_className, instance);
842
843            } catch (InvocationTargetException e) {
844                // Error : invocation failed
845                m_logger.log(Logger.ERROR,
846                                          "[" + m_name + "] createInstance -> The factory-method throws an exception : " + e.getTargetException(), e.getTargetException());
847                onError(null, m_className, e.getTargetException());
848                stop();
849                throw new RuntimeException("Cannot create a POJO instance, the factory-method has thrown an exception", e.getTargetException());
850            } catch (Throwable e) {
851                // Catch every other possible error and runtime exception.
852                m_logger.log(Logger.ERROR,
853                        "[" + m_name + "] createInstance -> The factory-method invocation failed : " + e.getMessage(), e);
854                stop();
855                throw new RuntimeException("Cannot create a POJO instance, the factory-method invocation has thrown an exception", e);
856            }
857        }
858        return instance;
859    }

1125    /**
1126     * This method is called by the manipulated class each time that a GETFIELD instruction is executed.
1127     * The method asks to each attached handler monitoring this field which value need
1128     * to be injected (i.e. returned) by invoking the {@link PrimitiveHandler#onGet(Object, String, Object)}
1129     * method. If the field value changes, this method call the {@link PrimitiveHandler#onSet(Object, String, Object)}
1130     * method on each field interceptor monitoring the field in order to advertize the new value.
1131     * @param pojo the pojo object on which the field was get
1132     * @param fieldName the field name on which the GETFIELD instruction is called
1133     * @return the value decided by the last asked handler (throws a warning if two fields decide two different values)
1134     */
1135    public Object  onGet(Object pojo, String fieldName) {
1136        Object initialValue = null;
1137        synchronized (this) { // Stack confinement.
1138            initialValue = m_fields.get(fieldName);
1139        }
1140        Object result = initialValue;
1141        boolean hasChanged = false;
1142        // Get the list of registered handlers
1143        FieldInterceptor[] list = (FieldInterceptor[]) m_fieldRegistration.get(fieldName); // Immutable list.
1144        for (int i = 0; list != null && i < list.length; i++) {
1145            // Call onGet outside of a synchronized block.
1146            Object handlerResult = list[i].onGet(pojo, fieldName, initialValue);
1147            if (handlerResult == initialValue) {
1148                continue; // Non-binding case (default implementation).
1149            } else {
1150                if (result != initialValue) {
1151                    //TODO analyze impact of removing conflict detection
1152                    if ((handlerResult != null && !handlerResult.equals(result)) || (result != null && handlerResult == null)) {
1153                        m_logger.log(
1154                                                  Logger.WARNING,
1155                                                  "A conflict was detected on the injection of "
1156                                                          + fieldName);
1157                    }
1158                }
1159                result = handlerResult;
1160                hasChanged = true;
1161            }
1162        }
1163        if (hasChanged) {
1164            // A change occurs => notify the change
1165            //TODO consider just changing the reference, however multiple thread can be an issue
1166            synchronized (this) {
1167                m_fields.put(fieldName, result);
1168            }
1169            // Call onset outside of a synchronized block.
1170            for (int i = 0; list != null && i < list.length; i++) {
1171                list[i].onSet(pojo, fieldName, result);
1172            }
1173        }
1174        return result;
1175    }

1250    /**
1251     * Computes the {@link Method} object from the given id.
1252     * Once computes, a map is used as a cache to avoid to recompute for
1253     * the same id.
1254     * @param methodId the method id
1255     * @return the method object or <code>null</code> if the method cannot be found.
1256     */
1257    private Member getMethodById(String methodId) {
1258        // Used a synchronized map.
1259        Member member = (Member) m_methods.get(methodId);
1260        if (! m_methods.containsKey(methodId) && m_clazz != null) {
1261            // Is it a inner class method
1262            if (methodId.contains("___")) { // Mark to detect a inner class method.
1263                String[] split = methodId.split("___");
1264                if (split.length != 2) {
1265                    m_logger.log(Logger.INFO, "A methodID cannot be associated with a method from the POJO class: " + methodId);
1266                    return null;
1267                } else {
1268                    String innerClassName = split[0];
1269                    methodId = split[1];
1270
1271                    // We can't find the member objects from anonymous methods, identified by their numeric name
1272                    // Just escaping in this case.
1273                    if (innerClassName.matches("-?\\d+")) {
1274                        m_methods.put(methodId, null);
1275                        return null;
1276                    }
1277
1278                    for (Class c : m_clazz.getDeclaredClasses()) {
1279                        if (innerClassName.equals(c.getSimpleName())) {
1280                            Method[] mets = c.getDeclaredMethods();
1281                            for (Method met : mets) {
1282                                if (MethodMetadata.computeMethodId(met).equals(methodId)) {
1283                                    // Store the new methodId
1284                                    m_methods.put(methodId, met);
1285                                    return met;
1286                                }
1287                            }
1288                        }
1289                        m_logger.log(Logger.INFO, "Cannot find the member associated to " + methodId + " - reason: " +
1290                                "cannot find the class " + innerClassName + " declared in " + m_clazz.getName());
1291                    }
1292                }
1293            }
1294
1295
1296            // First try on methods.
1297            Method[] mets = m_clazz.getDeclaredMethods();
1298            for (int i = 0; i < mets.length; i++) {
1299                if (MethodMetadata.computeMethodId(mets[i]).equals(methodId)) {
1300                    // Store the new methodId
1301                    m_methods.put(methodId, mets[i]);
1302                    return mets[i];
1303                }
1304            }
1305
1306            // If not found, it is a constructor, return the constructor object in this case.
1307            if (methodId.startsWith(MethodMetadata.CONSTRUCTOR_PREFIX)) {
1308                Constructor[] constructors = m_clazz.getDeclaredConstructors();
1309                for (int i = 0; i < constructors.length; i++) {
1310                    // Check if the constructor was not already computed. If not, compute the Id and check.
1311                    if (MethodMetadata.computeMethodId(constructors[i]).equals(methodId)) {
1312                        // Store the new methodId
1313                        m_methods.put(methodId, constructors[i]);
1314                        return constructors[i];
1315                    }
1316                }
1317            }
1318
1319            // Should not happen
1320            m_logger.log(Logger.INFO, "A methodID cannot be associated with a method from the POJO class: " + methodId);
1321            return null;
1322        } else {
1323            return member;
1324        }
1325    }

1327    /**
1328     * This method is called by the manipulated class each time that a PUTFIELD instruction is executed.
1329     * The method calls the {@link PrimitiveHandler#onSet(Object, String, Object)} method on each field
1330     * interceptors monitoring this field.
1331     * This method can be invoked with a <code>null</code> pojo argument when the changes comes from another
1332     * handler.
1333     * @param pojo the pojo object on which the field was set
1334     * @param fieldName the field name on which the PUTFIELD instruction is called
1335     * @param objectValue the new value of the field
1336     */
1337    public void onSet(final Object pojo, final String fieldName, final Object objectValue) {
1338        synchronized (this) {
1339            // First, store the new value.
1340            // This must be done in a synchronized block to avoid
1341            // concurrent modification
1342            m_fields.put(fieldName, objectValue);
1343        }
1344        // The registrations cannot be modified, so we can directly access
1345        // the interceptor list.
1346        FieldInterceptor[] list = (FieldInterceptor[]) m_fieldRegistration
1347                .get(fieldName);
1348        for (int i = 0; list != null && i < list.length; i++) {
1349             // The callback must be call outside the synchronization block.
1350            list[i].onSet(pojo, fieldName, objectValue);
1351        }
1352    }

1396    /**
1397     * Reconfigures the current instance.
1398     * Reconfiguring an instance means re-injecting a new
1399     * instance configuration. Some properties are immutable
1400     * such as the instance name.
1401     * This methods calls the {@link PrimitiveHandler#reconfigure(Dictionary)}
1402     * methods on each attached handler, and then recompute the instance
1403     * state. Note that the reconfiguration process does not deactivate the
1404     * instance.
1405     * @param configuration the new configuration to push
1406     * @see org.apache.felix.ipojo.ComponentInstance#reconfigure(java.util.Dictionary)
1407     */
1408    public void reconfigure(Dictionary configuration) {
1409         m_logger.log(Logger.INFO, "Reconfiguring " + getInstanceName());
1410        for (int i = 0; i < m_handlers.length; i++) {
1411            m_handlers[i].getHandler().reconfigure(configuration);
1412        }
1413        // We synchronized the state computation.
1414        synchronized (this) {
1415            if (m_state == STOPPED) {
1416                m_logger.log(Logger.INFO, "Instance stopped during reconfiguration - Try to restart");
1417                start();
1418            } else if (m_state == INVALID) {
1419                m_logger.log(Logger.INFO, "Instance invalid during reconfiguration - Recompute state");
1420                // Try to revalidate the instance after reconfiguration
1421                for (int i = 0; i < m_handlers.length; i++) {
1422                    if (m_handlers[i].getState() != VALID) {
1423                        return;
1424                    }
1425                }
1426                setState(VALID);
1427            }
1428        }
1429    }

1441    /**
1442     * State Change listener callback.
1443     * This method is called every time that a plugged handler becomes valid or invalid.
1444     * This method computes the new instance state and applies it (by calling the
1445     * {@link InstanceManager#setState(int)} method.
1446     * @param instance the handler becoming valid or invalid
1447     * @param newState the new state of the handler
1448     * @see org.apache.felix.ipojo.InstanceStateListener#stateChanged(org.apache.felix.ipojo.ComponentInstance, int)
1449     */
1450    public void stateChanged(ComponentInstance instance, int newState) {
1451        int state;
1452        synchronized (this) {
1453            if (m_state <= STOPPED) {
1454                return;
1455            } else {
1456                state = m_state; // Stack confinement
1457            }
1458        }
1459
1460        // Update the component state if necessary
1461        if (newState == INVALID && state == VALID) {
1462            // Need to update the state to UNRESOLVED
1463            setState(INVALID);
1464            return;
1465        }
1466        if (newState == VALID && state == INVALID) {
1467            // An handler becomes valid => check if all handlers are valid
1468            for (int i = 0; i < m_handlers.length; i++) {
1469                if (m_handlers[i].getState() != VALID) {
1470                    return;
1471                }
1472            }
1473            setState(VALID);
1474            return;
1475        }
1476    }
