039/**
040 * This class is handling the transformations between the base service set and the selected service set.
041 * It handles the matching services and the selected service set.
042 * As this class is tied to the dependency model, it reuses the same locks objects.

104    /**
105     * Creates the service reference manager.
106     *
107     * @param dep        the dependency
108     * @param filter     the filter
109     * @param comparator the comparator
110     */
111    public ServiceReferenceManager(DependencyModel dep, Filter filter, Comparator<ServiceReference> comparator) {
112        m_dependency = dep;
113        m_filter = filter;
114        // The Filter based service tracking interceptor needs to be created every time even if the filter is null.
115        // This arises from the potential re-implementation of the match method in the dependency implementation.
116        // It must be the last interceptor as the chain ends on the filter matching. (FELIX-4199)
117        m_trackingInterceptors.addLast(new FilterBasedServiceTrackingInterceptor(m_filter));
118
119        if (comparator != null) {
120            m_comparator = comparator;
121            m_rankingInterceptor = new ComparatorBasedServiceRankingInterceptor(comparator);
122        } else {
123            m_rankingInterceptor = new EmptyBasedServiceRankingInterceptor();
124        }
125    }

127    public void open() {
128        m_trackingInterceptorTracker = new Tracker(m_dependency.getBundleContext(),
129                ServiceTrackingInterceptor.class.getName(),
130                new TrackerCustomizer() {
131
132                    public boolean addingService(ServiceReference reference) {
133                        return DependencyProperties.match(reference, m_dependency);
134                    }
135
136                    public void addedService(ServiceReference reference) {
137                        ServiceTrackingInterceptor interceptor = (ServiceTrackingInterceptor) m_trackingInterceptorTracker
138                                .getService(reference);
139
140                        if (interceptor != null) {
141                            addTrackingInterceptor(interceptor);
142                        } else {
143                            m_dependency.getComponentInstance().getFactory().getLogger().log(Log.ERROR,
144                                    "Cannot retrieve the interceptor object from service reference " + reference
145                                            .getProperty(Constants.SERVICE_ID) + " - " + reference.getProperty
146                                            (Factory.INSTANCE_NAME_PROPERTY));
147                        }
148                    }
149
150                    public void modifiedService(ServiceReference reference, Object service) {
151                        // Not supported yet.
152                        // TODO it would be nice to support the modification of the interceptor TARGET property.
153                    }
154
155                    public void removedService(ServiceReference reference, Object service) {
156                        if (service != null && service instanceof ServiceTrackingInterceptor &&
157                                m_trackingInterceptors.contains(service)
158                        ) {
159                            removeTrackingInterceptor((ServiceTrackingInterceptor) service);
160                        }
161                    }
162                });
163
164        m_trackingInterceptorTracker.open();
165
166        // Initialize the service interceptor tracker.
167        m_rankingInterceptorTracker = new Tracker(m_dependency.getBundleContext(), ServiceRankingInterceptor.class.getName(),
168                new TrackerCustomizer() {
169
170                    public boolean addingService(ServiceReference reference) {
171                        return DependencyProperties.match(reference, m_dependency);
172                    }
173
174                    public void addedService(ServiceReference reference) {
175                        ServiceRankingInterceptor interceptor = (ServiceRankingInterceptor) m_rankingInterceptorTracker
176                                .getService(reference);
177                        if (interceptor != null) {
178                            setRankingInterceptor(interceptor);
179                        } else {
180                            m_dependency.getComponentInstance().getFactory().getLogger().log(Log.ERROR,
181                                    "Cannot retrieve the interceptor object from service reference " + reference
182                                            .getProperty(Constants.SERVICE_ID) + " - " + reference.getProperty
183                                            (Factory.INSTANCE_NAME_PROPERTY));
184                        }
185                    }
186
187                    public void modifiedService(ServiceReference reference, Object service) {
188                        // Not supported yet.
189                        // TODO it would be nice to support the modification of the interceptor TARGET property.
190                    }
191
192                    public void removedService(ServiceReference reference, Object service) {
193                        if (service == m_rankingInterceptor) {
194                            m_rankingInterceptor.close(m_dependency);
195                            // Do we have another one ?
196                            ServiceReference anotherReference = m_rankingInterceptorTracker.getServiceReference();
197                            if (anotherReference != null) {
198                                ServiceRankingInterceptor interceptor = (ServiceRankingInterceptor) m_rankingInterceptorTracker
199                                        .getService(anotherReference);
200                                if (interceptor != null) setRankingInterceptor(interceptor);
201                            } else if (m_comparator != null) {
202                                // If we have a comparator, we restore the comparator.
203                                setComparator(m_comparator);
204                            } else {
205                                // If we have neither comparator nor interceptor use an empty interceptor.
206                                setRankingInterceptor(new EmptyBasedServiceRankingInterceptor());
207                            }
208                        }
209                    }
210                });
211        m_rankingInterceptorTracker.open();
212
213        m_bindingInterceptorTracker = new Tracker(m_dependency.getBundleContext(),
214                ServiceBindingInterceptor.class.getName(),
215                new TrackerCustomizer() {
216
217                    public boolean addingService(ServiceReference reference) {
218                        return DependencyProperties.match(reference, m_dependency);
219                    }
220
221                    public void addedService(ServiceReference reference) {
222                        ServiceBindingInterceptor interceptor = (ServiceBindingInterceptor) m_bindingInterceptorTracker
223                                .getService(reference);
224                        if (interceptor != null) {
225                            addBindingInterceptor(interceptor);
226                        } else {
227                            m_dependency.getComponentInstance().getFactory().getLogger().log(Log.ERROR,
228                                    "Cannot retrieve the interceptor object from service reference " + reference
229                                            .getProperty(Constants.SERVICE_ID) + " - " + reference.getProperty
230                                            (Factory.INSTANCE_NAME_PROPERTY));
231                        }
232                    }
233
234                    public void modifiedService(ServiceReference reference, Object service) {
235                        // Not supported.
236                    }
237
238                    public void removedService(ServiceReference reference, Object service) {
239                        if (service != null && service instanceof ServiceBindingInterceptor &&
240                                m_bindingInterceptors.contains(service)
241                                ) {
242                            removeBindingInterceptor((ServiceBindingInterceptor) service);
243                        }
244                    }
245                }
246        );
247        m_bindingInterceptorTracker.open();
248    }

250    private void addTrackingInterceptor(ServiceTrackingInterceptor interceptor) {
251        // A new interceptor arrives. Insert it at the beginning of the list.
252        ChangeSet changeset;
253        try {
254            m_dependency.acquireWriteLockIfNotHeld();
255            m_trackingInterceptors.addFirst(interceptor);
256            interceptor.open(m_dependency);
257            changeset = computeChangesInMatchingServices();
258        } finally {
259            m_dependency.releaseWriteLockIfHeld();
260        }
261        m_dependency.onChange(changeset);
262    }
263
264    private void removeTrackingInterceptor(ServiceTrackingInterceptor interceptor) {
265        ChangeSet changeset;
266        try {
267            m_dependency.acquireWriteLockIfNotHeld();
268            m_trackingInterceptors.remove(interceptor);
269            interceptor.close(m_dependency);
270            changeset = computeChangesInMatchingServices();
271        } finally {
272            m_dependency.releaseWriteLockIfHeld();
273        }
274        m_dependency.onChange(changeset);
275    }
276
277    private void addBindingInterceptor(ServiceBindingInterceptor interceptor) {
278        // A new interceptor arrives, open it.
279        // Binding interceptor cannot modify existing bindings.
280        try {
281            m_dependency.acquireWriteLockIfNotHeld();
282            m_bindingInterceptors.add(interceptor);
283            interceptor.open(m_dependency);
284        } finally {
285            m_dependency.releaseWriteLockIfHeld();
286        }
287    }
288
289    private void removeBindingInterceptor(ServiceBindingInterceptor interceptor) {
290        try {
291            m_dependency.acquireWriteLockIfNotHeld();
292            m_bindingInterceptors.remove(interceptor);
293            interceptor.close(m_dependency);
294        } finally {
295            m_dependency.releaseWriteLockIfHeld();
296        }
297    }

299    public Object weavingServiceBinding(DependencyModel.ServiceBindingHolder sbh) {
300        Object svc = sbh.service;
301        try {
302            m_dependency.acquireReadLockIfNotHeld();
303            for (ServiceBindingInterceptor interceptor : m_bindingInterceptors) {
304                // Interceptor are not allowed to return null.
305                svc = interceptor.getService(m_dependency, sbh.reference, svc);
306            }
307        } finally {
308            m_dependency.releaseReadLockIfHeld();
309        }
310        return svc;
311    }
312
313    public void unweavingServiceBinding(DependencyModel.ServiceBindingHolder sbh) {
314        try {
315            m_dependency.acquireReadLockIfNotHeld();
316            for (ServiceBindingInterceptor interceptor : m_bindingInterceptors) {
317                interceptor.ungetService(m_dependency, sbh.reference);
318            }
319        } finally {
320            m_dependency.releaseReadLockIfHeld();
321        }
322    }

324    private ChangeSet computeChangesInMatchingServices() {
325        if (m_dependency.getTracker() == null || m_dependency.getTracker().getServiceReferences() == null) {
326            // Tracker closed, no problem
327            return new ChangeSet(Collections.<ServiceReference>emptyList(),
328                    Collections.<ServiceReference>emptyList(),
329                    Collections.<ServiceReference>emptyList(),
330                    null,
331                    null,
332                    null,
333                    null);
334        }
335        // The set of interceptor has changed.
336        try {
337            m_dependency.acquireWriteLockIfNotHeld();
338            // The tracker is open, we must recheck all services.
339            ServiceReference oldBest = getFirstService();
340            // Recompute the matching services.
341            m_matchingReferences.clear();
342            final List<ServiceReference> serviceReferencesList = m_dependency.getTracker().getServiceReferencesList();
343            if (serviceReferencesList != null) {
344                for (ServiceReference reference : serviceReferencesList) {
345                    TransformedServiceReference ref = new TransformedServiceReferenceImpl(reference);
346                    ref = accept(ref);
347                    if (ref != null) {
348                        m_matchingReferences.put(reference, ref);
349                    }
350                }
351            }
352
353            // We have the new matching set.
354            List<ServiceReference> beforeRanking = getSelectedServices();
355
356            final List<ServiceReference> allServices = getMatchingServices();
357            List<ServiceReference> references;
358            if (allServices.isEmpty()) {
359                references = Collections.emptyList();
360            } else {
361                references = m_rankingInterceptor.getServiceReferences(m_dependency, allServices);
362            }
363
364            RankingResult result = computeDifferences(beforeRanking, references);
365            m_selectedReferences = result.selected;
366
367            ServiceReference newFirst = getFirstService();
368            ServiceReference modified = null;
369            if (ServiceReferenceUtils.haveSameServiceId(oldBest, newFirst) && ServiceReferenceUtils
370                    .haveSameProperties(oldBest, newFirst)) {
371                modified = newFirst;
372            }
373            return new ChangeSet(getSelectedServices(), result.departures, result.arrivals, oldBest, getFirstService(),
374                    null, modified);
375        } finally {
376            m_dependency.releaseWriteLockIfHeld();
377        }
378    }

398    public ServiceReference getFirstService() {
399        try {
400            m_dependency.acquireReadLockIfNotHeld();
401            if (m_selectedReferences.isEmpty()) {
402                return null;
403            }
404            return m_selectedReferences.get(0);
405        } finally {

411        try {
412            m_dependency.acquireReadLockIfNotHeld();

419    public void reset() {
420        try {
421            m_dependency.acquireWriteLockIfNotHeld();
422            m_rankingInterceptor.close(m_dependency);
423            for (ServiceTrackingInterceptor interceptor : m_trackingInterceptors) {
424                interceptor.close(m_dependency);
425            }
426            m_trackingInterceptors.clear();
427            m_matchingReferences.clear();
428            m_selectedReferences = new ArrayList<TransformedServiceReference>();
429        } finally {
430            m_dependency.releaseWriteLockIfHeld();
431        }
432
433    }

435    public boolean addingService(ServiceReference reference) {
436        // We accept all service references except if we are frozen or broken. In these case, just ignore everything.
437
438        // We are doing two tests, we must get the read lock
439        try {
440            m_dependency.acquireReadLockIfNotHeld();
441            return !(m_dependency.getState() == DependencyModel.BROKEN || m_dependency.isFrozen());
442        } finally {
443            m_dependency.releaseReadLockIfHeld();
444        }
445    }

447    /**
448     * Checks if the given reference is accepted.
449     * This method is called when holding the write lock on the dependency.
450     *
451     * @param reference the reference
452     * @param <S> the service interface
453     * @return the transformed reference, null if rejected
454     */
455    private <S> TransformedServiceReference<S> accept(TransformedServiceReference<S> reference) {
456        TransformedServiceReference<S> accumulator = reference;
457        for (ServiceTrackingInterceptor interceptor : m_trackingInterceptors) {
458            TransformedServiceReference<S> accepted = interceptor.accept(m_dependency,
459                    m_dependency.getBundleContext(), accumulator);
460            if (accepted != null) {
461                accumulator = accepted;
462            } else {
463                // refused by an interceptor
464                m_dependency.getComponentInstance().getFactory().getLogger().log(Log.INFO,
465                        "The service reference " + reference.getProperty(Constants.SERVICE_ID) + " was rejected by " +
466                                "interceptor " + interceptor);
467                return null;
468            }
469        }
470
471        return accumulator;
472    }

474    public void addedService(ServiceReference reference) {
475        // A service was added to the tracker.
476
477        // First, check is the tracking interceptors are accepting it.
478        // The transformed reference is creates and check outside of the protected region.
479        TransformedServiceReference ref = new TransformedServiceReferenceImpl(reference);
480
481        boolean match;
482        try {
483            m_dependency.acquireWriteLockIfNotHeld();
484            ref = accept(ref);
485            if (ref != null) {
486                m_matchingReferences.put(reference, ref);
487            }
488            match = ref != null;
489        } finally {
490            m_dependency.releaseWriteLockIfHeld();
491        }
492
493        if (match) {
494            // Callback invoked outside of locks.
495            // The called method is taking the write lock anyway.
496            onNewMatchingService(ref);
497            m_dependency.notifyListeners(ARRIVAL, ref, null);
498        }
499    }
500
501    private void onNewMatchingService(TransformedServiceReference reference) {
502        ServiceReference oldFirst;
503        RankingResult result;
504        try {
505            m_dependency.acquireWriteLockIfNotHeld();
506            // We store the currently 'first' service reference.
507            oldFirst = getFirstService();
508
509            // We apply our ranking strategy.
510            result = applyRankingOnArrival(reference);
511            // Set the selected services.
512            m_selectedReferences = result.selected;
513        } finally {
514            m_dependency.releaseWriteLockIfHeld();
515        }
516        // Fire the event (outside from the synchronized region)
517        fireUpdate(getSelectedServices(), result.departures, result.arrivals, oldFirst,
518                getFirstService(), null, null);
519    }

521    private void onModificationOfAMatchingService(TransformedServiceReference reference, Object service) {
522        ServiceReference oldFirst;
523        RankingResult result;
524        try {
525            m_dependency.acquireWriteLockIfNotHeld();
526            // We store the currently 'first' service reference.
527            oldFirst = getFirstService();
528
529            // We apply our ranking strategy.
530            result = applyRankingOnModification(reference);
531            // Set the selected services.
532            m_selectedReferences = result.selected;
533        } finally {
534            m_dependency.releaseWriteLockIfHeld();
535        }
536        // Fire the event (outside from the synchronized region)
537        fireUpdate(getSelectedServices(), result.departures, result.arrivals, oldFirst,
538                getFirstService(), service, reference);
539    }

541    private RankingResult applyRankingOnModification(ServiceReference reference) {
542        // TODO we are holding the lock here.
543        List<ServiceReference> beforeRanking = getSelectedServices();
544        List<ServiceReference> references = m_rankingInterceptor.onServiceModified(m_dependency, getMatchingServices(),
545                reference);
546        return computeDifferences(beforeRanking, references);
547    }

549    private void fireUpdate(List<ServiceReference> selectedServices, List<ServiceReference> departures,
550                            List<ServiceReference> arrivals, ServiceReference oldFirst,
551                            ServiceReference firstService, Object service, ServiceReference modified) {
552        ChangeSet set = new ChangeSet(selectedServices, departures, arrivals, oldFirst, firstService, service, modified);
553        m_dependency.onChange(set);
554    }

556    private RankingResult applyRankingOnArrival(ServiceReference ref) {
557        // TODO we are holding the lock here.
558        List<ServiceReference> beforeRanking = getSelectedServices();
559        List<ServiceReference> references = m_rankingInterceptor.onServiceArrival(m_dependency, getMatchingServices(),
560                ref);
561        // compute the differences
562        return computeDifferences(beforeRanking, references);
563
564    }
565
566    private RankingResult applyRankingOnDeparture(ServiceReference ref) {
567        // TODO we are holding the lock here.
568        List<ServiceReference> beforeRanking = getSelectedServices();
569        List<ServiceReference> references = m_rankingInterceptor.onServiceDeparture(m_dependency, getMatchingServices(),
570                ref);
571        return computeDifferences(beforeRanking, references);
572    }
573
574    private RankingResult computeDifferences(List<ServiceReference> beforeRanking, List<ServiceReference> ranked) {
575        // compute the differences
576        List<ServiceReference> departures = new ArrayList<ServiceReference>();
577        List<ServiceReference> arrivals = new ArrayList<ServiceReference>();
578        // All references that are no more in the set are considered as leaving services.
579        for (ServiceReference old : beforeRanking) {
580            if (!ServiceReferenceUtils.containsReferenceById(ranked, old)) {
581                departures.add(old);
582            }
583        }
584        // All references that are in `references` but not in `beforeRanking` are new services
585        for (ServiceReference newRef : ranked) {
586            if (!ServiceReferenceUtils.containsReferenceById(beforeRanking, newRef)) {
587                arrivals.add(newRef);
588            }
589        }
590
591        return new RankingResult(departures, arrivals, ranked);
592    }

594    public void modifiedService(ServiceReference reference, Object service) {
595        // We are handling a modified event, we have three case to handle
596        // 1) the service was matching and does not match anymore -> it's a departure.
597        // 2) the service was not matching and matches -> it's an arrival
598        // 3) the service was matching and still matches -> it's a modification.
599
600        // The dependency event to send
601        DependencyEventType eventType = null;
602        ServiceReference<?> eventRef = null;
603
604        try {
605            m_dependency.acquireWriteLockIfNotHeld();
606
607            if (m_matchingReferences.containsKey(reference)) {
608                // do we still accept the reference
609                TransformedServiceReference initial = m_matchingReferences.get(reference);
610                TransformedServiceReference accepted = new TransformedServiceReferenceImpl(reference);
611                accepted = accept(accepted);
612                if (accepted == null) {
613                    // case 1
614                    m_matchingReferences.remove(reference);
615                    onDepartureOfAMatchingService(initial, service);
616                    eventType = DEPARTURE;
617                    eventRef = initial;
618                } else {
619                    // Do we have a real change
620                    if (!ServiceReferenceUtils.haveSameProperties(initial, accepted)) {
621                        // case 3
622                        m_matchingReferences.put(reference, accepted);
623                        onModificationOfAMatchingService(accepted, service);
624                        eventType = MODIFIED;
625                        eventRef = accepted;
626                    }
627                }
628            } else {
629                // Base does not contain the service, let's try to add it.
630                TransformedServiceReference transformed = new TransformedServiceReferenceImpl(reference);
631                transformed = accept(transformed);
632                if (transformed != null) {
633                    // case 2
634                    m_matchingReferences.put(reference, transformed);
635                    onNewMatchingService(transformed);
636                    eventType = ARRIVAL;
637                    eventRef = transformed;
638                }
639            }
640        } finally {
641            m_dependency.releaseWriteLockIfHeld();
642        }
643
644        if (eventType != null) {
645            m_dependency.notifyListeners(eventType, eventRef, service);
646        }
647    }
648
649    public void onDepartureOfAMatchingService(TransformedServiceReference reference, Object service) {
650        ServiceReference oldFirst;
651        RankingResult result = null;
652        try {
653            m_dependency.acquireWriteLockIfNotHeld();
654            // We store the currently 'first' service reference.
655            oldFirst = getFirstService();
656            // We apply our ranking strategy.
657            result = applyRankingOnDeparture(reference);
658            // Set the selected services.
659            m_selectedReferences = result.selected;
660        } finally {
661            m_dependency.releaseWriteLockIfHeld();
662        }
663        // Fire the event (outside from the synchronized region)
664        fireUpdate(getSelectedServices(), result.departures, result.arrivals, oldFirst,
665                getFirstService(), service, null);
666    }

668    public void removedService(ServiceReference reference, Object service) {
669        // A service is leaving
670        // 1 - the service was in the matching set => real departure
671        // 2 - the service was not in the matching set => nothing do do.
672
673        TransformedServiceReference initial = null;
674        try {
675            m_dependency.acquireWriteLockIfNotHeld();
676            initial = m_matchingReferences.remove(reference);
677            if (initial != null) {
678                // Case 1
679                onDepartureOfAMatchingService(initial, service);
680            }
681            // else case 2.
682        } finally {
683            m_dependency.releaseWriteLockIfHeld();
684        }
685
686        // Call the listeners outside the locked region.
687        if (initial != null) {
688            m_dependency.notifyListeners(DEPARTURE, initial, service);
689        }
690
691    }

693    /**
694     * A new filter is set.
695     * We have to recompute the set of matching services.
696     *
697     * @param filter  the new filter
698     * @param tracker the tracker
699     */
700    public ChangeSet setFilter(Filter filter, Tracker tracker) {
701        try {
702            m_dependency.acquireWriteLockIfNotHeld();
703            m_filter = filter;
704
705            if (!m_trackingInterceptors.isEmpty()) {
706                ServiceTrackingInterceptor interceptor = m_trackingInterceptors.getLast();
707                if (interceptor != null && interceptor instanceof FilterBasedServiceTrackingInterceptor) {
708                    // Remove it first.
709                    m_trackingInterceptors.removeLast();
710                }
711            }
712
713            if (m_filter != null) {
714                // Add the new one.
715                ServiceTrackingInterceptor newInterceptor = new FilterBasedServiceTrackingInterceptor(m_filter);
716                m_trackingInterceptors.addLast(newInterceptor);
717            }
718
719            if (tracker == null) {
720                // Tracker closed, no problem
721                return new ChangeSet(Collections.<ServiceReference>emptyList(),
722                        Collections.<ServiceReference>emptyList(),
723                        Collections.<ServiceReference>emptyList(),
724                        null,
725                        null,
726                        null,
727                        null);
728            } else {
729                // The tracker is open, we must recheck all services.
730                ServiceReference oldBest = getFirstService();
731
732                // Recompute the matching services.
733                m_matchingReferences.clear();
734                final List<ServiceReference> serviceReferencesList = tracker.getServiceReferencesList();
735                if (serviceReferencesList != null) {
736                    for (ServiceReference reference : serviceReferencesList) {
737                        TransformedServiceReference ref = new TransformedServiceReferenceImpl(reference);
738                        ref = accept(ref);
739                        if (ref != null) {
740                            m_matchingReferences.put(reference, ref);
741                        }
742                    }
743                }
744
745                // We have the new matching set.
746
747                List<ServiceReference> beforeRanking = getSelectedServices();
748
749                final List<ServiceReference> allServices = getMatchingServices();
750                List<ServiceReference> references;
751                if (allServices.isEmpty()) {
752                    references = Collections.emptyList();
753                } else {
754                    references = m_rankingInterceptor.getServiceReferences(m_dependency, allServices);
755                }
756
757                RankingResult result = computeDifferences(beforeRanking, references);
758                m_selectedReferences = result.selected;
759                return new ChangeSet(getSelectedServices(), result.departures, result.arrivals, oldBest, getFirstService(),
760                        null, null);
761            }
762        } finally {
763            m_dependency.releaseWriteLockIfHeld();
764        }
765    }

809    public void setRankingInterceptor(ServiceRankingInterceptor interceptor) {
810        m_dependency.getComponentInstance().getFactory().getLogger().log(Log.INFO, "Dependency " + m_dependency.getId
811                () + " is getting a new ranking interceptor : " + interceptor);
812        ChangeSet changeSet;
813        try {
814            m_dependency.acquireWriteLockIfNotHeld();
815            ServiceReference oldBest = getFirstService();
816            List<ServiceReference> beforeRanking = getSelectedServices();
817            m_rankingInterceptor = interceptor;
818            m_rankingInterceptor.open(m_dependency);
819
820            final List<ServiceReference> allServices = getMatchingServices();
821            List<ServiceReference> references = Collections.emptyList();
822            if (!allServices.isEmpty()) {
823                references = m_rankingInterceptor.getServiceReferences(m_dependency, allServices);
824            }
825            RankingResult result = computeDifferences(beforeRanking, references);
826            m_selectedReferences = result.selected;
827            changeSet = new ChangeSet(getSelectedServices(), result.departures, result.arrivals, oldBest,
828                    getFirstService(), null, null);
829        } finally {
830            m_dependency.releaseWriteLockIfHeld();
831        }
832        // Calling onChange outside of the lock.
833        m_dependency.onChange(changeSet);
834    }

876     * Gets the list of tracking interceptors attached to the current service dependency.
877     * @return the list of service references of the tracking interceptors participating to the resolution of the
878     * current service dependency. An empty list is returned is there are no participating interceptors.
879     * @since 1.11.0
880     */
881    public List<ServiceReference> getTrackingInterceptorReferences() {
882        try {
883            m_dependency.acquireReadLockIfNotHeld();
884            if (m_trackingInterceptorTracker != null) {
885                List<ServiceReference> refs = m_trackingInterceptorTracker.getUsedServiceReferences();
886                if (refs != null) {
887                    return refs;
888                }
889            }
890            return Collections.emptyList();
891        } finally {
892            m_dependency.releaseReadLockIfHeld();
893        }
894    }

918     * Gets the service reference of the currently attached ranking interceptor. As only one ranking interceptor can
919     * be attached at a point on time, this is not a list but only one reference.
920     * @return the service reference of the ranking interceptor participating to the resolution of the current
921     * service dependency. {@code null} if no (external) ranking interceptor is currently attached.
922     * @since 1.11.0
923     */
924    public ServiceReference getRankingInterceptorReference() {
925        try {
926            m_dependency.acquireReadLockIfNotHeld();
927            if (m_rankingInterceptorTracker != null) {
928                List<ServiceReference> references = m_rankingInterceptorTracker.getUsedServiceReferences();
929                if (references != null  && ! references.isEmpty()) {
930                    return references.get(0);
931                }
932            }
933            return null;
934        } finally {
935            m_dependency.releaseReadLockIfHeld();
936        }
