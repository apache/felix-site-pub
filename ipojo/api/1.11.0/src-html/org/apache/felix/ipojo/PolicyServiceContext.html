040/**
041 * The policy service context is a service context aiming to resolve service dependencies
042 * inside different service context according to a policy.
043 * So, the policy service context behavior follows one of the three following policy:
044 * <li> Local : services are only resolved in the local service context.</li>
045 * <li> Global : services are only resolved in the global context (hte OSGi one)</li>
046 * <li> Local and Global : services are resolved inside the local context and inside
047 * the global context</li>
048 *    
049 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
050 */
051public class PolicyServiceContext implements ServiceContext {
052    
053    /**
054     * Resolving policy, resolves services only in the composite
055     * context (local).
056     * This policy is the default one for services inherited from
057     * service-level dependencies.
058     */
059    public static final int LOCAL = 0;
060    
061    /**
062     * Resolving policy, resolves services only in the composite
063     * (local) and in the global context.
064     * This policy is the default one for implementation dependency.
065     */
066    public static final int LOCAL_AND_GLOBAL = 1;
067    
068    /**
069     * Resolving policy, resolves services inside the global context
070     * only.
071     */
072    public static final int GLOBAL = 2;
073    
074    /**
075     * The global service registry.
076     * Targets the OSGi service registry.
077     */
078    public BundleContext m_global;
079    
080    /**
081     * The local (Composite) Service Registry.
082     */
083    public ServiceContext m_local;
084    
085    /**
086     * The resolving policy to use to resolve
087     * dependencies.
088     */
089    private int m_policy = LOCAL_AND_GLOBAL;
090    
091    
092    /**
093     * Creates a PolicyServiceContext.
094     * If the local context is null, sets the policy to
095     * {@link PolicyServiceContext#GLOBAL}, else use the 
096     * given policy.
097     * @param global the global bundle context
098     * @param local the parent (local) service context
099     * @param policy the resolution policy
100     */
101    public PolicyServiceContext(BundleContext global, ServiceContext local, int policy) {
102        m_global = global;
103        m_local = local;
104        if (m_local == null) {
105            m_policy = GLOBAL;
106        } else {
107            m_policy = policy;
108        }
109    }
110
111    /**
112     * Adds a service listener according to the policy.
113     * Be aware, that the listener can be registered both in the local and in the global context
114     * if the {@link PolicyServiceContext#LOCAL_AND_GLOBAL} is used.
115     * @param listener the listener to add
116     * @param filter the LDAP filter
117     * @throws InvalidSyntaxException if the filter is malformed.
118     * @see org.apache.felix.ipojo.ServiceContext#addServiceListener(org.osgi.framework.ServiceListener, java.lang.String)
119     */
120    public void addServiceListener(ServiceListener listener, String filter) throws InvalidSyntaxException {
121        if (m_policy == LOCAL || m_policy == LOCAL_AND_GLOBAL) {
122            m_local.addServiceListener(listener, filter);
123        }
124        if (m_policy == GLOBAL || m_policy == LOCAL_AND_GLOBAL) {
125            m_global.addServiceListener(listener, filter);
126        }
127
128    }

143        }

146    /**
147     * Gets all service references.
148     * These references are found inside the local registry, global registry or both according to the policy.
149     * The returned array can contain service references from both context.
150     * @param clazz the required service specification.
151     * @param filter the LDAP filter
152     * @return the array of service reference, <code>null</code> if no service available
153     * @throws InvalidSyntaxException if the LDAP filter is malformed 
154     * @see org.apache.felix.ipojo.ServiceContext#getAllServiceReferences(java.lang.String, java.lang.String)
155     */
156    public ServiceReference[] getAllServiceReferences(String clazz,
157            String filter) throws InvalidSyntaxException {
158        switch (m_policy) {
159            case LOCAL:
160                return m_local.getAllServiceReferences(clazz, filter);
161            case GLOBAL:
162                return m_global.getAllServiceReferences(clazz, filter);
163            case LOCAL_AND_GLOBAL:
164                ServiceReference[] refLocal = m_local.getAllServiceReferences(clazz, filter);
165                ServiceReference[] refGlobal = m_global.getAllServiceReferences(clazz, filter);
166                return computeServiceReferencesFromBoth(refLocal, refGlobal);
167            default:
168                return null;
169        }
170    }
171
172    /**
173     * Gets the service object for the given references.
174     * The service is get inside the context according to the policy.
175     * @param ref the service reference
176     * @return the service object
177     * @see org.apache.felix.ipojo.ServiceContext#getService(org.osgi.framework.ServiceReference)
178     */
179    public Object getService(ServiceReference ref) {
180        if (ref instanceof TransformedServiceReference) {
181            ref = ((TransformedServiceReference) ref).getWrappedReference();
182        }
183        switch(m_policy) { // NOPMD No break needed as we return in each branch.
184            case LOCAL:
185                // The reference comes from the local scope
186                return m_local.getService(ref);
187            case GLOBAL:
188                // The reference comes from the global registry
189                return m_global.getService(ref);
190            case LOCAL_AND_GLOBAL:
191                if (ref instanceof org.apache.felix.ipojo.context.ServiceReferenceImpl) {
192                    return m_local.getService(ref);
193                } else {
194                    return m_global.getService(ref);
195                }
196            default : 
197                return null;
198        }
199    }

201    /**
202     * Gets a service reference for the required service specification.
203     * The service is looked inside the context according to the policy.
204     * @param clazz the required service specification
205     * @return a service reference or <code>null</code> if no matching service available
206     * @see org.apache.felix.ipojo.ServiceContext#getServiceReference(java.lang.String)
207     */
208    public ServiceReference getServiceReference(String clazz) {
209        switch (m_policy) { // NOPMD No break needed as we return in each branch.
210            case LOCAL:
211                return m_local.getServiceReference(clazz);
212            case GLOBAL:
213                return m_global.getServiceReference(clazz);
214            case LOCAL_AND_GLOBAL:
215                ServiceReference refLocal = m_local.getServiceReference(clazz);
216                if (refLocal == null) {
217                    return m_global.getServiceReference(clazz);
218                } else {
219                    return refLocal;
220                }
221            default:
222                return null;
223        }
224    }

226    /**
227     * Gets a service reference.
228     * This method is delegated on PolicyServiceContext#getServiceReference(String)
229     * @param sClass the service interface class
230     * @param <S> the service interface class
231     * @return the service reference of <code>null</code> if not found
232     * @see PolicyServiceContext#getServiceReference(String)
233     */
234    public <S> ServiceReference<S> getServiceReference(Class<S> sClass) {
235        return (ServiceReference<S>) getServiceReference(sClass.getName());
236    }

238    /**
239     * Gets a collection of service references
240     * @param clazz the service interface class
241     * @param filter the filter
242     * @param <S> the service interface class
243     * @return the set of service reference
244     * @throws InvalidSyntaxException the filter is invalid
245     * @see PolicyServiceContext#getServiceReferences(String, String)
246     */
247    public <S> Collection<ServiceReference<S>> getServiceReferences(Class<S> clazz, String filter) throws InvalidSyntaxException {
248         ServiceReference<S>[] refs =
249            (ServiceReference<S>[]) getServiceReferences(clazz.getName(), filter);
250        return (refs == null)
251            ? Collections.EMPTY_LIST
252            : (Collection<ServiceReference<S>>) Arrays.asList(refs);
253    }

255    /**
256     * Get a service reference for the required service specification.
257     * The services are looked inside the context according to the policy.
258     * @param clazz the required service specification
259     * @param filter the LDAP filter
260     * @return a service reference array or <code>null</code> if not consistent service available
261     * @throws InvalidSyntaxException if the LDAP filter is malformed 
262     * @see org.apache.felix.ipojo.ServiceContext#getServiceReference(java.lang.String)
263     */
264    public ServiceReference[] getServiceReferences(String clazz, String filter) throws InvalidSyntaxException {
265        switch (m_policy) {
266            case LOCAL:
267                return m_local.getServiceReferences(clazz, filter);
268            case GLOBAL:
269                return m_global.getServiceReferences(clazz, filter);
270            case LOCAL_AND_GLOBAL:
271                ServiceReference[] refLocal = m_local.getServiceReferences(clazz, filter);
272                ServiceReference[] refGlobal = m_global.getServiceReferences(clazz, filter);
273                return computeServiceReferencesFromBoth(refLocal, refGlobal);
274            default:
275                return null;
276        }
277
278    }
279    
280    /**
281     * Computes the service reference array from the two given set of service references
282     * according to the policy.
283     * @param refLocal the set of local references
284     * @param refGlobal the set of global references
285     * @return the set of service references
286     */
287    private ServiceReference[] computeServiceReferencesFromBoth(ServiceReference[] refLocal, ServiceReference[] refGlobal) {
288        if (refLocal == null) {
289            return refGlobal; // If refGlobal is null, return null, else return refGlobal
290        } else if (refGlobal == null) { // refLocal != null && refGlobal == null
291            return refLocal;
292        } else { // Both refGlobal and refLocal are not null
293            ServiceReference[] refs = new ServiceReference[refLocal.length + refGlobal.length];
294            System.arraycopy(refLocal, 0, refs, 0, refLocal.length);
295            System.arraycopy(refGlobal, 0, refs, refLocal.length, refGlobal.length);
296            return refs;
297        }        
298    }
