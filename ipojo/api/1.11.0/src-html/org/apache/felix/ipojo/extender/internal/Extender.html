044/**
045 * iPOJO main activator.
046 */
047public class Extender implements BundleActivator {
048    public static final String BOOTSTRAP_QUEUE_DEBUG_PROPERTY = "org.apache.felix.ipojo.extender.BootstrapQueueDebug";
049    /**
050     * Enables the iPOJO internal dispatcher.
051     * This internal dispatcher helps the OSGi framework to support large
052     * scale applications. The internal dispatcher is disabled by default.
053     */
054    static boolean DISPATCHER_ENABLED = true;
055
056    /**
057     * Disables the iPOJO asynchronous processing.
058     * When set to false, the bundles are processed in the listener thread
059     * making iPOJO usable on Google App Engine. By default, the processing
060     * is asynchronous.
061     */
062    static boolean SYNCHRONOUS_PROCESSING_ENABLED = false;
063
064    /**
065     * Property allowing to set if the internal dispatcher is enabled or disabled.
066     * Possible value are either {@literal true} or {@literal false}.
067     */
068    private static final String ENABLING_DISPATCHER = "ipojo.internal.dispatcher";
069
070    /**
071     * Property allowing to disable the asynchronous process (and so enables the
072     * synchronous processing).
073     * Possible value are either {@literal true} or {@literal false}.
074     */
075    private static final String SYNCHRONOUS_PROCESSING = "ipojo.processing.synchronous";
076
077    /**
078     * The Bundle Context of the iPOJO Core bundle.
079     */
080    private static BundleContext m_context;
081
082    /**
083     * The iPOJO Extender logger.
084     */
085    private Logger m_logger;
086
087    /**
088     * The iPOJO Bundle.
089     */
090    private Bundle m_bundle;
091
092    /**
093     * The chained processor containing all the true bundle processor.
094     */
095    private ChainedBundleProcessor m_processor;
096
097    /**
098     * Binds Instances to Factories to Extensions.
099     */
100    private DeclarationLinker m_linker;
101
102    private LifecycleQueueService m_queueService;
103
104    /**
105     * Track ACTIVE bundles.
106     */
107    private BundleTracker m_tracker;
108
109    /**
110     * The iPOJO bundle is starting.
111     * This method configures the iPOJO system (internal dispatcher and bundle processing). Then it initiates the
112     * bundle processing.
113     * <p/>
114     * To optimize the processing, we process the iPOJO bundle first.
115     *
116     * @param context the iPOJO's bundle bundle context
117     * @throws Exception something terrible happen during startup
118     */
119    public void start(BundleContext context) throws Exception {
120        m_context = context;
121        m_bundle = context.getBundle();
122
123        m_logger = new Logger(m_context, "IPOJO-Main-Extender");
124
125        enablingDispatcher(context, m_logger);
126        enablingSynchronousProcessing(context, m_logger);
127
128        // Create the dispatcher only if required.
129        if (DISPATCHER_ENABLED) {
130            EventDispatcher.create(context);
131        }
132
133        // Initialize ConfigurationTracker
134        ConfigurationTracker.initialize();
135
136        // Initialize the queue event proxy if wanted
137        ReplayQueueEventProxy proxy = null;
138        if (Boolean.getBoolean(BOOTSTRAP_QUEUE_DEBUG_PROPERTY)) {
139            proxy = new ReplayQueueEventProxy();
140            context.registerService(QueueEventProxy.class, proxy, null);
141        }
142
143        BundleProcessor extensionBundleProcessor = new ExtensionBundleProcessor(m_logger);
144        BundleProcessor componentsProcessor = new ComponentsBundleProcessor(m_logger);
145        BundleProcessor configurationProcessor = new ConfigurationProcessor(m_logger);
146        if (SYNCHRONOUS_PROCESSING_ENABLED) {
147            m_queueService = new EnforcedQueueService(
148                    new HeaderPreferenceSelection(),
149                    new SynchronousQueueService(context),
150                    Preference.SYNC,
151                    m_logger);
152
153            // If required, add the event proxy
154            if (proxy != null) {
155                m_queueService.addQueueListener(proxy);
156            }
157        } else {
158            // Build a thread factory that will groups extender's thread together
159            ThreadFactory threadFactory = new GroupThreadFactory(new ThreadGroup("iPOJO Extender"));
160            threadFactory = new NamingThreadFactory(threadFactory);
161            threadFactory = new PrefixedThreadFactory(threadFactory, "[iPOJO] ");
162
163            // Create the queue services
164            SynchronousQueueService sync = new SynchronousQueueService(context);
165            ExecutorQueueService async = new ExecutorQueueService(context,
166                                                                  Integer.getInteger(ExecutorQueueService.THREADPOOL_SIZE_PROPERTY,
167                                                                                     1), // default to 1 if no system property is set
168                                                                  threadFactory);
169            m_queueService = new PreferenceQueueService(new HeaderPreferenceSelection(), sync, async);
170
171            extensionBundleProcessor = new QueuingActivationProcessor(extensionBundleProcessor, m_queueService);
172            componentsProcessor = new QueuingActivationProcessor(componentsProcessor, m_queueService);
173            configurationProcessor = new QueuingActivationProcessor(configurationProcessor, m_queueService);
174
175            // If required, add the event proxy to both real services
176            if (proxy != null) {
177                sync.addQueueListener(proxy);
178                async.addQueueListener(proxy);
179            }
180
181        }
182        m_queueService.start();
183
184        // Start linking
185        m_linker = new DeclarationLinker(context, m_queueService);
186        m_linker.start();
187
188        m_processor = ChainedBundleProcessor.create(extensionBundleProcessor, componentsProcessor, configurationProcessor);
189
190        m_processor.start();
191
192        // Begin by initializing core handlers
193        m_processor.activate(m_bundle);
194
195        m_tracker = new BundleTracker(context, Bundle.ACTIVE, new BundleTrackerCustomizer() {
196            public Object addingBundle(final Bundle bundle, final BundleEvent event) {
197                if (bundle.getBundleId() == m_bundle.getBundleId()) {
198                    // Not interested in our own bundle
199                    return null;
200                }
201                m_processor.activate(bundle);
202                return bundle;
203            }
204
205            public void modifiedBundle(final Bundle bundle, final BundleEvent event, final Object object) {}
206
207            public void removedBundle(final Bundle bundle, final BundleEvent event, final Object object) {
208                m_processor.deactivate(bundle);
209            }
210        });
211
212        m_tracker.open();
213
214        m_logger.log(Logger.INFO, "iPOJO Main Extender started");
215    }
216
217    /**
218     * The iPOJO bundle is stopping.
219     *
220     * @param context the bundle context
221     * @throws Exception something terrible happen
222     */
223    public void stop(BundleContext context) throws Exception {
224        m_tracker.close();
225
226        m_processor.deactivate(m_bundle);
227
228        //Shutdown ConfigurationTracker
229        ConfigurationTracker.shutdown();
230
231        m_processor.stop();
232
233        if (DISPATCHER_ENABLED) {
234            EventDispatcher.dispose();
235        }
236
237        m_linker.stop();
238        m_queueService.stop();
239
240        m_logger.log(Logger.INFO, "iPOJO Main Extender stopped");
241        m_context = null;
242    }
243
244    /**
245     * Gets iPOJO bundle context.
246     *
247     * @return the iPOJO Bundle Context
248     */
249    public static BundleContext getIPOJOBundleContext() {
250        return m_context;
251    }

253    /**
254     * Enables or disables the internal dispatcher, so sets the
255     * {@link Extender#DISPATCHER_ENABLED} flag.
256     * This method checks if the {@link Extender#ENABLING_DISPATCHER}
257     * property is set to {@literal true}. Otherwise, the internal
258     * dispatcher is disabled. The property can be set as a system
259     * property ({@literal ipojo.internal.dispatcher}) or inside the
260     * iPOJO bundle manifest ({@literal ipojo-internal-dispatcher}).
261     *
262     * @param context the bundle context.
263     * @param logger  the logger to indicates if the internal dispatcher is set.
264     */
265    private static void enablingDispatcher(BundleContext context, Logger logger) {
266        // First check in the framework and in the system properties
267        String flag = context.getProperty(ENABLING_DISPATCHER);
268
269        // If null, look in bundle manifest
270        if (flag == null) {
271            String key = ENABLING_DISPATCHER.replace('.', '-');
272            flag = (String) context.getBundle().getHeaders().get(key);
273        }
274
275        if (flag != null) {
276            if (flag.equalsIgnoreCase("true")) {
277                Extender.DISPATCHER_ENABLED = true;
278                logger.log(Logger.INFO, "iPOJO Internal Event Dispatcher enables");
279                return;
280            }
281        }
282
283        // Either l is null, or the specified value was false
284        Extender.DISPATCHER_ENABLED = false;
285        logger.log(Logger.INFO, "iPOJO Internal Event Dispatcher disables");
286
287    }
288
289    /**
290     * Enables or disables the asynchronous processing, so sets the
291     * {@link Extender#SYNCHRONOUS_PROCESSING_ENABLED} flag.
292     * Disabling asynchronous processing avoids iPOJO to create a new
293     * thread to process bundles. So, iPOJO can be used on the
294     * Google App Engine.
295     * This method checks if the {@link Extender#SYNCHRONOUS_PROCESSING}
296     * property is set to {@literal true}. Otherwise, asynchronous processing
297     * is used (default). The property can be set as a system
298     * property ({@literal ipojo.processing.synchronous}) or inside the
299     * iPOJO bundle manifest.
300     *
301     * @param context the bundle context.
302     * @param logger  the logger to indicates if the internal dispatcher is set.
303     */
304    private static void enablingSynchronousProcessing(BundleContext context, Logger logger) {
305        String flag = context.getProperty(SYNCHRONOUS_PROCESSING);
306
307        // If null, look in bundle manifest
308        if (flag == null) {
309            String key = SYNCHRONOUS_PROCESSING.replace('.', '-');
310            flag = (String) context.getBundle().getHeaders().get(key);
311        }
312
313        if (flag != null) {
314            if (flag.equalsIgnoreCase("true")) {
315                Extender.SYNCHRONOUS_PROCESSING_ENABLED = true;
316                logger.log(Logger.INFO, "iPOJO Asynchronous processing disabled");
317                return;
318            }
319        }
320
321        // Either l is null, or the specified value was false
322        Extender.SYNCHRONOUS_PROCESSING_ENABLED = false;
323        logger.log(Logger.INFO, "iPOJO synchronous processing disabled");
324
325    }
