020package org.apache.felix.ipojo.manipulator.metadata.annotation.model.parser;
021
022import org.apache.felix.ipojo.manipulator.metadata.annotation.model.AnnotationType;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.model.parser.replay.AnnotationVisitorPlayback;
024import org.objectweb.asm.AnnotationVisitor;
025import org.objectweb.asm.ClassVisitor;
026import org.objectweb.asm.Type;
027import org.objectweb.asm.commons.EmptyVisitor;
028
029/**
030 * User: guillaume
031 * Date: 01/07/13
032 * Time: 15:59
033 */
034public class AnnotationTypeVisitor extends EmptyVisitor implements ClassVisitor {
035
036    private AnnotationType annotationType;
037
038    @Override
039    public void visit(final int version, final int access, final String name, final String signature, final String superName, final String[] interfaces) {
040        annotationType = new AnnotationType(Type.getObjectType(name));
041    }
042
043    @Override
044    public AnnotationVisitor visitAnnotation(final String desc, final boolean visible) {
045        // Build annotations of this annotation type
046        AnnotationVisitorPlayback playback = new AnnotationVisitorPlayback(desc, visible);
047        annotationType.getPlaybacks().add(playback);
048        return playback;
049    }
050
051    public AnnotationType getAnnotationType() {
052        return annotationType;
053    }
054
055    // Note: if we override visitMethod here, we could get the annotation's default values.
056}
