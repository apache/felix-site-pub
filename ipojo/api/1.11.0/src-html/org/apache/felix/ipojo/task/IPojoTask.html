019package org.apache.felix.ipojo.task;
020
021import java.io.File;
022
023import org.apache.felix.ipojo.manipulator.Pojoization;
024import org.apache.tools.ant.BuildException;
025import org.apache.tools.ant.Project;
026import org.apache.tools.ant.Task;
027
028/**
029 * iPOJO Ant Task. This Ant task manipulates an input bundle.
030 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
031 */
032public class IPojoTask extends Task {
033
034    /** Metadata file. */
035    private File m_metadata;
036
037    /** Input bundle. */
038    private File m_input;
039
040    /** Output bundle. */
041    private File m_output;
042
043    /** Input directory. */
044    private File m_directory;
045
046    /** Input manifest. */
047    private File m_manifest;
048
049    /** Flag describing if we need to ignore annotation of not. */
050    private boolean m_ignoreAnnotations = false;
051
052    /**
053     * Flag describing if we need or not use local XSD files
054     * (i.e. use the {@link org.apache.felix.ipojo.xml.parser.SchemaResolver} or not).
055     * If <code>true</code> the local XSD are not used.
056     */
057    private boolean m_ignoreLocalXSD = false;
058
059    /**
060     * Set the metadata file.
061     * @param meta : the metadata file.
062     */
063    public void setMetadata(File meta) {
064        m_metadata = meta;
065    }
066
067    /**
068     * Set the manifest file.
069     * @param manifest : the manifest file.
070     */
071    public void setManifest(File manifest) {
072        m_manifest = manifest;
073    }
074
075    /**
076     * Set the input bundle.
077     * @param in : the input bundle
078     */
079    public void setInput(File in) {
080        m_input = in;
081    }
082
083    /**
084     * Set the input directory.
085     * @param dir : the input directory
086     */
087    public void setDir(File dir) {
088        m_directory  = dir;
089    }
090
091    /**
092     * Set the output bundle.
093     * @param out : the output bundle
094     */
095    public void setOutput(File out) {
096        m_output = out;
097    }
098
099    /**
100     * Set if we need to ignore annotations or not.
101     * @param flag : true if we need to ignore annotations.
102     */
103    public void setIgnoreAnnotations(boolean flag) {
104        m_ignoreAnnotations = flag;
105    }
106
107    /**
108     * Set if we need to use embedded XSD files or not.
109     * @param flag : true if we need to ignore embedded XSD files.
110     */
111    public void setIgnoreEmbeddedSchemas(boolean flag) {
112        m_ignoreLocalXSD = flag;
113    }
114
115    /**
116     * Execute the Ant Task.
117     * @see org.apache.tools.ant.Task#execute()
118     */
119    public void execute() {
120
121        if (m_input == null  && m_directory == null) {
122            throw new BuildException("Neither input bundle nor directory specified");
123        }
124
125        if (m_input != null && !m_input.exists()) {
126            throw new BuildException("The input bundle " + m_input.getAbsolutePath() + " does not exist");
127        }
128
129        if (m_directory != null && !m_directory.exists()) {
130            throw new BuildException("The input directory " + m_directory.getAbsolutePath() + " does not exist");
131        }
132        if (m_directory != null && !m_directory.isDirectory()) {
133            throw new BuildException("The input directory " + m_directory.getAbsolutePath() + " is not a directory");
134        }
135
136
137        if (m_input != null) {
138            log("Input bundle file : " + m_input.getAbsolutePath());
139        } else {
140            log("Input directory : " + m_directory.getAbsolutePath());
141        }
142
143        if (m_manifest != null) {
144            if (m_input != null) {
145                throw new BuildException("The manifest location cannot be used when manipulating an existing bundle");
146            }
147            if (! m_manifest.exists()) {
148                throw new BuildException("The manifest file " + m_manifest.getAbsolutePath() + " does not exist");
149            }
150        }
151
152        // Get metadata file
153        if (m_metadata == null) {
154            m_metadata = new File("./metadata.xml");
155            if (!m_metadata.exists()) {
156             // Verify if annotations are ignored
157                if (m_ignoreAnnotations) {
158                    log("No metadata file found & annotations ignored : nothing to do");
159                    return;
160                } else {
161                    log("No metadata file found - trying to use only annotations");
162                    m_metadata = null;
163                }
164            } else {
165                log("Metadata file : " + m_metadata.getAbsolutePath());
166            }
167        } else {
168            // Metadata file is specified, check existence
169            if (!m_metadata.exists()) {
170                throw new BuildException("No metadata file found - the file " + m_metadata.getAbsolutePath() + " does not exist");
171            } else {
172                if (m_metadata.isDirectory()) {
173                        log("Metadata directory : " + m_metadata.getAbsolutePath());
174                } else {
175                        log("Metadata file : " + m_metadata.getAbsolutePath());
176                }
177            }
178        }
179
180        initializeSaxDriver();
181
182
183        log("Start manipulation");
184
185        if (m_input != null) { // Prepare output file
186            if (m_output == null) {
187                m_output = new File("./_out.jar");
188            }
189            if (m_output.exists()) {
190                boolean r = m_output.delete();
191                if (!r) { throw new BuildException("The file " + m_output.getAbsolutePath() + " cannot be deleted"); }
192            }
193        }
194
195        AntReporter reporter = new AntReporter(getProject());
196        Pojoization pojo = new Pojoization(reporter);
197        if (m_ignoreAnnotations) {
198            pojo.disableAnnotationProcessing();
199        }
200        if (! m_ignoreLocalXSD) {
201            pojo.setUseLocalXSD();
202        }
203        if (m_input != null) {
204            pojo.pojoization(m_input, m_output, m_metadata);
205        } else {
206            pojo.directoryPojoization(m_directory, m_metadata, m_manifest);
207        }
208        for (int i = 0; i < reporter.getWarnings().size(); i++) {
209            log((String) reporter.getWarnings().get(i), Project.MSG_WARN);
210        }
211        if (reporter.getErrors().size() > 0) {
212            throw new BuildException((String) reporter.getErrors().get(0));
213        }
214
215        if (m_input != null) {
216            String out;
217            if (m_output.getName().equals("_out.jar")) {
218                if (m_input.delete()) {
219                    if (! m_output.renameTo(m_input)) {
220                        log("Cannot rename the output jar to " + m_input.getAbsolutePath(), Project.MSG_WARN);
221                    }
222                } else {
223                    log("Cannot delete the input file : " + m_input.getAbsolutePath(), Project.MSG_WARN);
224                }
225                out = m_input.getAbsolutePath();
226            } else {
227                out = m_output.getAbsolutePath();
228            }
229
230            log("Bundle manipulation - SUCCESS");
231            log("Output file : " + out);
232        } else {
233            log("Manipulation - SUCCESS");
234            log("Output files : " + m_directory.getAbsolutePath());
235            if (m_manifest != null) {
236                log("Manifest : " + m_manifest.getAbsolutePath());
237            }
238
239        }
240
241    }
242
243    /**
244     * If Ant runs with Java 1.4, we should use the embedded Xerces.
245     * To achieve that, we set the org.xml.sax.driver property.
246     * Otherwise, the JVM sets the org.xml.sax.driver property.
247     */
248    private void initializeSaxDriver() {
249        String version = (String) System.getProperty("java.vm.version");
250        if (version.startsWith("1.4")) {
251            System.setProperty("org.xml.sax.driver", "org.apache.xerces.parsers.SAXParser");
252        }
253    }
254
255}
