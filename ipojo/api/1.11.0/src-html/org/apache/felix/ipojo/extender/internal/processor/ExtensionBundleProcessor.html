020package org.apache.felix.ipojo.extender.internal.processor;
021
022import org.apache.felix.ipojo.IPojoFactory;
023import org.apache.felix.ipojo.extender.internal.BundleProcessor;
024import org.apache.felix.ipojo.extender.internal.builder.ReflectiveFactoryBuilder;
025import org.apache.felix.ipojo.extender.internal.declaration.DefaultExtensionDeclaration;
026import org.apache.felix.ipojo.metadata.Element;
027import org.apache.felix.ipojo.parser.ParseUtils;
028import org.apache.felix.ipojo.util.Log;
029import org.apache.felix.ipojo.util.Logger;
030import org.osgi.framework.Bundle;
031import org.osgi.framework.BundleContext;
032
033import java.util.*;
034
035/**
036 * Bundle processor handling the {@link #IPOJO_EXTENSION} header.
037 */
038public class ExtensionBundleProcessor implements BundleProcessor {
039
040    /**
041     * iPOJO Extension declaration header.
042     */
043    public static final String IPOJO_EXTENSION = "IPOJO-Extension";
044
045    /**
046     * Logger.
047     */
048    private final Log m_logger;
049
050    /**
051     * The map storing the association between bundles and the list of extension declaration.
052     */
053    private Map<Bundle, List<DefaultExtensionDeclaration>> m_extensions = new HashMap<Bundle, List<DefaultExtensionDeclaration>>();
054
055    /**
056     * Creates the processor.
057     *
058     * @param logger the logger
059     */
060    public ExtensionBundleProcessor(Log logger) {
061        m_logger = logger;
062    }
063
064    /**
065     * A bundle is starting.
066     *
067     * @param bundle the bundle
068     */
069    public void activate(Bundle bundle) {
070        Dictionary dict = bundle.getHeaders();
071        // Check for abstract factory type
072        String extension = (String) dict.get(IPOJO_EXTENSION);
073        if (extension != null) {
074            activateExtensions(bundle, extension);
075        }
076    }
077
078    /**
079     * A bundle is stopping.
080     *
081     * @param bundle the bundle
082     */
083    public void deactivate(Bundle bundle) {
084        List<DefaultExtensionDeclaration> declarations = m_extensions.get(bundle);
085        if (declarations != null) {
086            for (DefaultExtensionDeclaration declaration : declarations) {
087                declaration.stop();
088            }
089            m_extensions.remove(bundle);
090        }
091    }
092
093    /**
094     * iPOJO is starting.
095     * Nothing to do.
096     */
097    public void start() {
098        // Nothing to do
099    }
100
101    /**
102     * iPOJO is stopping.
103     * We clean up all extension in the reverse order of their installation.
104     */
105    public void stop() {
106        // Construct a new instance to avoid ConcurrentModificationException since deactivate also change the extensions
107        // list
108        // Ignored, for a simple ordered shutdown, use ReverseBundleProcessor
109    }
110
111    /**
112     * Parses an IPOJO-Extension manifest header and then creates
113     * iPOJO extensions (factory types).
114     *
115     * @param bundle the bundle containing the header.
116     * @param header the header to parse.
117     */
118    private void activateExtensions(Bundle bundle, String header) {
119        String[] extensions = ParseUtils.split(header, ",");
120        for (int i = 0; extensions != null && i < extensions.length; i++) {
121            String[] segments = ParseUtils.split(extensions[i], ":");
122
123            /*
124             * Get the fully qualified type name.
125             * type = [namespace] name
126             */
127            String[] nameparts = ParseUtils.split(segments[0].trim(), " \t");
128            String type = nameparts.length == 1 ? nameparts[0] : nameparts[0] + ":" + nameparts[1];
129            String classname = segments[1];
130
131            Class<? extends IPojoFactory> clazz;
132            try {
133                clazz = bundle.loadClass(classname).asSubclass(IPojoFactory.class);
134            } catch (ClassNotFoundException e) {
135                String message = String.format("Cannot load class '%s' from bundle %s (%s) for extension '%s'",
136                                               classname,
137                                               bundle.getSymbolicName(),
138                                               bundle.getVersion(),
139                                               type);
140                m_logger.log(Logger.ERROR, message, e);
141                return;
142            }
143
144            try {
145                ReflectiveFactoryBuilder builder = new ReflectiveFactoryBuilder(clazz.getConstructor(BundleContext.class, Element.class));
146                DefaultExtensionDeclaration declaration = new DefaultExtensionDeclaration(bundle.getBundleContext(), builder, type);
147
148                getBundleDeclarations(bundle).add(declaration);
149
150                declaration.start();
151
152                m_logger.log(Logger.DEBUG, "New factory type available: " + type);
153            } catch (NoSuchMethodException e) {
154                m_logger.log(Logger.ERROR,
155                        String.format("Extension '%s' is missing the required (BundleContext, Element) public " +
156                                "constructor", clazz.getName()));
157            }
158        }
159    }
160
161    /**
162     * Gets the list of declaration for the given method.
163     *
164     * @param bundle the bundle
165     * @return the list of extension declaration associated to the given bundle, <code>null</code> otherwise.
166     */
167    private List<DefaultExtensionDeclaration> getBundleDeclarations(Bundle bundle) {
168        List<DefaultExtensionDeclaration> declarations = m_extensions.get(bundle);
169        if (declarations == null) {
170            declarations = new ArrayList<DefaultExtensionDeclaration>();
171            m_extensions.put(bundle, declarations);
172        }
173        return declarations;
174    }
175
176}
