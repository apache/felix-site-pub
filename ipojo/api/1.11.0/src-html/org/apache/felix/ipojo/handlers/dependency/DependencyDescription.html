019package org.apache.felix.ipojo.handlers.dependency;
020
021import java.util.Comparator;
022import java.util.List;
023
024import org.apache.felix.ipojo.util.DependencyModelListener;
025import org.osgi.framework.Filter;
026import org.osgi.framework.ServiceReference;
027
028/**
029 * Service Dependency Description.
030 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
031 */
032public class DependencyDescription {
033    /**
034     * The described dependency.
035     */
036    private final Dependency m_dependency;
037
038    /**
039     * Creates a dependency description.
040     * @param dep the described dependency
041     */
042    public DependencyDescription(Dependency dep) {
043        m_dependency = dep;
044    }
045
046    public boolean isMultiple() { return m_dependency.isAggregate(); }
047
048    public boolean isOptional() { return m_dependency.isOptional(); }
049
050    public String getFilter() { return m_dependency.getFilter(); }
051
052    public String getInterface() { return m_dependency.getSpecification().getName(); }
053
054    public int getState() { return m_dependency.getState(); }
055    
056    public String getId() { return m_dependency.getId(); }
057    
058    public boolean isProxy() { return m_dependency.isProxy(); }
059
060    
061    /**
062     * Gets <code>true</code> if the dependency uses Nullable objects.
063     * @return true if the dependency is optional and supports nullable object.
064     */
065    public boolean supportsNullable() { return m_dependency.supportsNullable(); }
066    
067    public String getDefaultImplementation() { return m_dependency.getDefaultImplementation(); }
068    
069    public int getPolicy() { return m_dependency.getBindingPolicy(); }
070    
071    public String getComparator() { return m_dependency.getComparator(); }
072    
073    public boolean isFrozen() { return m_dependency.isFrozen(); }
074
075    /**
076     * Gets the service reference list.
077     * @return the list of matching service reference,
078     * <code>null</code> if no service reference.
079     */
080    public List<ServiceReference> getServiceReferences() { return m_dependency.getServiceReferencesAsList(); }
081
082    /**
083     * Gets the service reference if only one service reference is used.
084     * @return the ServiceReference (only if the cardinality could be 1),
085     * or <code>null</code> if no service reference.
086     */
087    public ServiceReference getServiceReference() { 
088        List list = getServiceReferences();
089        if (list == null) {
090            return null;
091        } else {
092            return (ServiceReference) list.get(0);
093        }
094    }
095
096    /**
097     * Gets the used service set.
098     * @return the list [service reference] containing the used services,
099     * <code>null</code> if no providers are used
100     */
101    public List<ServiceReference> getUsedServices() { return m_dependency.getUsedServiceReferences(); }
102    
103    /**
104     * Sets the dependency comparator.
105     * The reference set will be sort at the next usage.
106     * @param cmp the comparator
107     */
108    public void setComparator(Comparator cmp) {
109        m_dependency.setComparator(cmp);
110    }
111    
112    /**
113     * Sets the dependency filter.
114     * @param filter the new LDAP filter
115     */
116    public void setFilter(Filter filter) {
117        m_dependency.setFilter(filter);
118    }
119    
120    /**
121     * Sets the dependency cardinality.
122     * @param isAgg if <code>true</code> sets the dependency to aggregate,
123     * if <code>false</code> sets the dependency to scalar.
124     */
125    public void setAggregate(boolean isAgg) {
126        m_dependency.setAggregate(isAgg);
127    }
128    
129    /**
130     * Sets the dependency optionality.
131     * @param isOpt if <code>true</code> sets the dependency to optional,
132     * if <code>false</code> sets the dependency to mandatory.
133     */
134    public void setOptional(boolean isOpt) {
135        m_dependency.setOptionality(isOpt);
136    }
137
138    /**
139     * Gets the required service specification name.
140     * @return the required service specification class name.
141     */
142    public String getSpecification() {
143        return m_dependency.getSpecification().getName();
144    }
145
146    /**
147     * Gets the described dependency.
148     * @return the dependency.
149     */
150    public Dependency getDependency() {
151        return m_dependency;
152    }
153
154    /**
155     * Add the given listener to the dependency model's list of listeners.
156     *
157     * @param listener the {@code DependencyModelListener} object to be added
158     * @throws NullPointerException if {@code listener} is {@code null}
159     */
160    public void addListener(DependencyModelListener listener) {
161        m_dependency.addListener(listener);
162    }
163
164    /**
165     * Remove the given listener from the dependency model's list of listeners.
166     *
167     * @param listener the {@code DependencyModelListener} object to be removed
168     * @throws NullPointerException if {@code listener} is {@code null}
169     * @throws java.util.NoSuchElementException if {@code listener} wasn't present in the dependency model's list of listeners
170     */
171    public void removeListener(DependencyModelListener listener) {
172        m_dependency.removeListener(listener);
173    }
174}
