020package org.apache.felix.ipojo.manipulator.metadata.annotation.registry;

022import static java.util.Collections.emptyList;
023import static java.util.Collections.unmodifiableList;
024import static org.apache.felix.ipojo.manipulator.spi.helper.Predicates.alwaysTrue;
025
026import java.io.IOException;
027import java.util.ArrayList;
028import java.util.List;
029
030import org.apache.felix.ipojo.annotations.HandlerBinding;
031import org.apache.felix.ipojo.annotations.Ignore;
032import org.apache.felix.ipojo.annotations.Stereotype;
033import org.apache.felix.ipojo.manipulator.Reporter;
034import org.apache.felix.ipojo.manipulator.ResourceStore;
035import org.apache.felix.ipojo.manipulator.metadata.annotation.model.AnnotationType;
036import org.apache.felix.ipojo.manipulator.metadata.annotation.model.discovery.ChainedAnnotationDiscovery;
037import org.apache.felix.ipojo.manipulator.metadata.annotation.model.discovery.HandlerBindingDiscovery;
038import org.apache.felix.ipojo.manipulator.metadata.annotation.model.discovery.IgnoredDiscovery;
039import org.apache.felix.ipojo.manipulator.metadata.annotation.model.discovery.StereotypeDiscovery;
040import org.apache.felix.ipojo.manipulator.metadata.annotation.model.parser.AnnotationParser;
041import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.GenericVisitorFactory;
042import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.ignore.NullBinding;
043import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.stereotype.StereotypeVisitorFactory;
044import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util.Elements;
045import org.apache.felix.ipojo.metadata.Element;
046import org.objectweb.asm.Type;

048/**
049 * The {@link MetaAnnotationBindingRegistry} is a registry that tries to complete its list
050 * of bindings when an unknown one is detected.
051 * It uses the given {@link ResourceStore} to parse the annotation's type and find if
052 * it's annotated with @{@link org.apache.felix.ipojo.annotations.Stereotype},
053 * @{@link org.apache.felix.ipojo.annotations.HandlerBinding} or @{@link org.apache.felix.ipojo.annotations.Ignore}
054 */
055public class MetaAnnotationBindingRegistry extends CompletableBindingRegistry {
056
057    private ResourceStore m_store;
058    private Reporter m_reporter;
059
060    public MetaAnnotationBindingRegistry(final BindingRegistry delegate, final Reporter reporter, final ResourceStore store) {
061        super(delegate, reporter);
062        this.m_reporter = reporter;
063        this.m_store = store;
064        addBindings(nullBindingsForMetaAnnotations());
065    }
066
067    protected Iterable<Binding> nullBindingsForMetaAnnotations() {
068        // Do not re-apply meta-annotations
069        ArrayList<Binding> bindings = new ArrayList<Binding>();
070        bindings.add(new NullBinding(Type.getType(Stereotype.class)));
071        bindings.add(new NullBinding(Type.getType(HandlerBinding.class)));
072        bindings.add(new NullBinding(Type.getType(Ignore.class)));
073        return bindings;
074    }
075
076    @Override
077    protected List<Binding> createBindings(final Type type) {
078
079        // Parse the annotation
080        byte[] bytes;
081        try {
082            bytes = m_store.read(type.getInternalName().concat(".class"));
083        } catch (IOException e) {
084            // Annotation type cannot be read
085            m_reporter.trace("Could not read bytecode for @%s", type.getClassName());
086            return emptyList();
087        }
088        AnnotationParser parser = new AnnotationParser();
089        AnnotationType annotationType = parser.read(bytes);
090
091        // Search meta-annotations
092        ChainedAnnotationDiscovery chain = new ChainedAnnotationDiscovery();
093        StereotypeDiscovery stereotypeDiscovery = new StereotypeDiscovery();
094        HandlerBindingDiscovery handlerBindingDiscovery = new HandlerBindingDiscovery();
095        IgnoredDiscovery ignoredDiscovery = new IgnoredDiscovery();
096        chain.getDiscoveries().add(stereotypeDiscovery);
097        chain.getDiscoveries().add(handlerBindingDiscovery);
098        chain.getDiscoveries().add(ignoredDiscovery);
099
100        annotationType.traverse(chain);
101
102        // Produced Bindings
103        List<Binding> bindings = new ArrayList<Binding>();
104
105        // @Stereotype support
106        if (stereotypeDiscovery.isStereotype()) {
107            m_reporter.trace("@Stereotype detected: @%s", type.getClassName());
108            Binding binding = new Binding();
109            binding.setAnnotationType(type);
110            binding.setPredicate(alwaysTrue());
111            binding.setFactory(new StereotypeVisitorFactory(annotationType));
112
113            bindings.add(binding);
114        }
115
116        // @HandlerBinding support
117        if (handlerBindingDiscovery.isHandlerBinding()) {
118
119            m_reporter.trace("@HandlerBinding detected: @%s", type.getClassName());
120            Binding binding = new Binding();
121            binding.setAnnotationType(type);
122            binding.setPredicate(alwaysTrue());
123            final Element element = buildElement(handlerBindingDiscovery, type);
124            binding.setFactory(new GenericVisitorFactory(element.getName(), element.getNameSpace()));
125
126            bindings.add(binding);
127        }
128
129        // Its IMPORTANT that the @Ignore is processed last since it removes existing bindings
130        if (ignoredDiscovery.isIgnore()) {
131            m_reporter.trace("@Ignore detected: @%s", type.getClassName());
132            Binding binding = new NullBinding(type);
133
134            bindings.clear();
135            bindings.add(binding);
136            bindings = unmodifiableList(bindings); // just in case of ...
137        }
138
139        return bindings;
140
141    }
142
143    private Element buildElement(final HandlerBindingDiscovery handler, final Type type) {
144        Element element;
145        if ((handler.getNamespace() == null) &&
146                (handler.getValue() == null)) {
147            // No attributes specified, use annotation type as element's source
148            element = Elements.buildElement(type);
149        } else if ((handler.getNamespace() == null) &&
150                (handler.getValue() != null)) {
151            // Namespace attribute is omitted
152            element = Elements.buildElement(handler.getValue());
153        } else {
154            element = Elements.buildElement(handler.getNamespace(),
155                                            handler.getValue());
156        }
157        return element;
158    }
