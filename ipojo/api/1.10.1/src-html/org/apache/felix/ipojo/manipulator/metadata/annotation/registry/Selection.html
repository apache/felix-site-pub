020package org.apache.felix.ipojo.manipulator.metadata.annotation.registry;
021
022import org.apache.felix.ipojo.manipulator.Reporter;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.ComponentWorkbench;
024import org.apache.felix.ipojo.manipulator.spi.BindingContext;
025import org.objectweb.asm.AnnotationVisitor;
026import org.objectweb.asm.ClassVisitor;
027import org.objectweb.asm.FieldVisitor;
028import org.objectweb.asm.MethodVisitor;
029import org.objectweb.asm.Type;
030import org.objectweb.asm.tree.ClassNode;
031import org.objectweb.asm.tree.FieldNode;
032import org.objectweb.asm.tree.MemberNode;
033import org.objectweb.asm.tree.MethodNode;
034
035import java.lang.annotation.ElementType;
036import java.util.ArrayList;
037import java.util.Iterator;
038import java.util.List;
039
040/**
041 * A {@link Selection} is used to select a subset of all supported {@link AnnotationVisitor}.
042 * It's a query DSL.
043 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
044 */
045public class Selection implements Iterable<AnnotationVisitor> {
046
047    private BindingRegistry registry;
048    private ComponentWorkbench workbench;
049    private Reporter reporter;
050    private MemberNode node;
051    private int index = -1;
052    private String annotation;
053    private ElementType elementType = null;
054    private Object visitor;
055
056    public Selection(BindingRegistry registry, ComponentWorkbench workbench, Reporter reporter) {
057        this.registry = registry;
058        this.workbench = workbench;
059        this.reporter = reporter;
060    }
061
062    public Selection field(FieldVisitor visitor, FieldNode node) {
063        this.visitor = visitor;
064        this.node = node;
065        this.elementType = ElementType.FIELD;
066        return this;
067    }
068
069    public Selection method(MethodVisitor visitor, MethodNode node) {
070        this.visitor = visitor;
071        this.node = node;
072        this.elementType = ElementType.METHOD;
073        return this;
074    }
075
076    public Selection type(ClassVisitor visitor, ClassNode node) {
077        this.visitor = visitor;
078        this.node = node;
079        this.elementType = ElementType.TYPE;
080        return this;
081    }
082
083    public Selection parameter(MethodVisitor visitor, MethodNode node, int index) {
084        this.visitor = visitor;
085        this.index = index;
086        this.node = node;
087        this.elementType = ElementType.PARAMETER;
088        return this;
089    }
090
091    public Selection annotatedWith(String desc) {
092        this.annotation = desc;
093        return this;
094    }
095
096    public AnnotationVisitor get() {
097        Iterator<AnnotationVisitor> i = iterator();
098        if (iterator().hasNext()) {
099            return i.next();
100        }
101        return null;
102    }
103
104    public Iterator<AnnotationVisitor> iterator() {
105
106        List<AnnotationVisitor> visitors = new ArrayList<AnnotationVisitor>();
107
108        BindingContext context = new BindingContext(workbench, reporter, Type.getType(annotation), node, elementType, index, visitor);
109        List<Binding> predicates = registry.getBindings(annotation);
110
111        if (predicates != null && !predicates.isEmpty()) {
112            collectMatchingVisitors(predicates, context, visitors);
113        }
114
115        if (visitors.isEmpty() && !registry.getDefaultBindings().isEmpty()) {
116            collectMatchingVisitors(registry.getDefaultBindings(), context, visitors);
117        }
118
119
120        return visitors.iterator();
121    }
122
123    private void collectMatchingVisitors(List<Binding> bindings, BindingContext context, List<AnnotationVisitor> visitors) {
124        for (Binding binding : bindings) {
125            if (binding.getPredicate().matches(context)) {
126                AnnotationVisitor visitor = binding.getFactory().newAnnotationVisitor(context);
127                if (visitor != null) {
128                    visitors.add(visitor);
129                }
130            }
131        }
132    }
133
134}
