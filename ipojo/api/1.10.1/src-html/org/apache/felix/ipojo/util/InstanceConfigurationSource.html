020package org.apache.felix.ipojo.util;
021
022import org.apache.felix.ipojo.ContextListener;
023import org.apache.felix.ipojo.ContextSource;
024
025import java.util.*;
026
027/**
028 * A context source giving access to system properties.
029 */
030public class InstanceConfigurationSource implements ContextSource {
031
032    /**
033     * The instance configuration.
034     */
035    private Dictionary<String,Object> m_configuration;
036
037    /**
038     * Listener list.
039     */
040    private List<ContextListener> m_listeners = new ArrayList<ContextListener>();
041
042    public InstanceConfigurationSource(Dictionary<String, Object> configuration) {
043        m_configuration = configuration;
044    }
045
046    /**
047     * The instance is reconfigured.
048     * Changes are computed by comparing the old and new configuration.
049     * @param newConfiguration the new instance configuration
050     */
051    public void reconfigure(Dictionary<String, Object> newConfiguration) {
052        // Copy the current configuration and update the field.
053        Hashtable<String, Object> configuration = new Hashtable<String, Object>();
054        Enumeration<String> enumeration = m_configuration.keys();
055        while (enumeration.hasMoreElements()) {
056            final String key = enumeration.nextElement();
057            configuration.put(key, m_configuration.get(key));
058        }
059        // We now have a copy of the current configuration.
060        // We must set the field to the new configuration, as updates are going to be fired.
061        // so we must have reflected the updates already.
062        m_configuration = newConfiguration;
063
064        // We use the same loop to find lost and updated properties.
065        enumeration = configuration.keys();
066        while (enumeration.hasMoreElements()) {
067            final String key = enumeration.nextElement();
068            final Object newValue = newConfiguration.get(key);
069            final Object oldValue = configuration.get(key);
070            if (newValue == null)  {
071                // If we don't have the property anymore, notify the departure.
072                fireUpdate(key, null);
073            } else {
074                // The new configuration still has a value, is it the same ?
075                if (! newValue.equals(oldValue)) {
076                    fireUpdate(key, newValue);
077                }
078            }
079        }
080
081        // Do we have new properties ?
082        enumeration = newConfiguration.keys();
083        while (enumeration.hasMoreElements()) {
084            final String key = enumeration.nextElement();
085            if (configuration.get(key) == null) {
086                Object newValue = newConfiguration.get(key);
087                fireUpdate(key, newValue);
088            }
089        }
090    }
091
092    private void fireUpdate(String key, Object newValue) {
093        for (ContextListener listener : m_listeners) {
094            listener.update(this, key, newValue);
095        }
096    }
097
098    public Object getProperty(String property) {
099        return m_configuration.get(property);
100    }
101
102    public Dictionary getContext() {
103        return m_configuration;
104    }
105
106    public void registerContextListener(ContextListener listener, String[] properties) {
107        if (! m_listeners.contains(listener)) {
108            m_listeners.add(listener);
109        }
110    }
111
112    public void unregisterContextListener(ContextListener listener) {
113        m_listeners.remove(listener);
114    }
115}
