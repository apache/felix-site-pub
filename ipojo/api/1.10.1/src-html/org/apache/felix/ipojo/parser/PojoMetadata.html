019package org.apache.felix.ipojo.parser;
020
021import org.apache.felix.ipojo.ConfigurationException;
022import org.apache.felix.ipojo.metadata.Element;
023
024/**
025 * Manipulation Metadata allows getting information about the implementation class
026 * without using reflection such as implemented interfaces, super class,
027 * methods and fields.
028 * This method allows getting object to register {@link org.apache.felix.ipojo.FieldInterceptor} and
029 * {@link org.apache.felix.ipojo.MethodInterceptor}.
030 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
031 */
032public class PojoMetadata {
033
034    /**
035     * The list of implemented interfaces.
036     */
037    private String[] m_interfaces = new String[0];
038
039    /**
040     * The list of fields.
041     */
042    private FieldMetadata[] m_fields = new FieldMetadata[0];
043
044    /**
045     * The list of methods.
046     */
047    private MethodMetadata[] m_methods = new MethodMetadata[0];
048
049    /**
050     * The Super class (if <code>null</code> for {@link Object}).
051     */
052    private String m_super;
053
054
055    /**
056     * Creates Pojo metadata.
057     * Manipulation Metadata object are created from component type metadata by
058     * parsing manipulation metadata.
059     * @param metadata the component type metadata
060     * @throws ConfigurationException if the manipulation metadata cannot be found
061     */
062    public PojoMetadata(Element metadata) throws ConfigurationException {
063        Element[] elems = metadata.getElements("manipulation", "");
064        if (elems == null) {
065            throw new ConfigurationException("The component " + metadata.getAttribute("classname") + " has no manipulation metadata");
066        }
067        Element manip = elems[0];
068        m_super = manip.getAttribute("super");
069        Element[] fields = manip.getElements("field");
070        for (int i = 0; fields != null && i < fields.length; i++) {
071            FieldMetadata field = new FieldMetadata(fields[i]);
072            addField(field);
073        }
074        Element[] methods = manip.getElements("method");
075        for (int i = 0; methods != null && i < methods.length; i++) {
076            MethodMetadata method = new MethodMetadata(methods[i]);
077            addMethod(method);
078        }
079        Element[] itfs = manip.getElements("interface");
080        for (int i = 0; itfs != null && i < itfs.length; i++) {
081            addInterface(itfs[i].getAttribute("name"));
082        }
083    }
084
085    public MethodMetadata[] getMethods() { return m_methods; }
086
087    public FieldMetadata[] getFields() { return m_fields; }
088
089    public String[] getInterfaces() { return m_interfaces; }
090
091    /**
092     * Gets the field metadata for the given name.
093     * @param name : the name of the field
094     * @return the corresponding field metadata or <code>null</code> if not found
095     */
096    public FieldMetadata getField(String name) {
097        for (int i = 0; i < m_fields.length; i++) {
098            if (m_fields[i].getFieldName().equalsIgnoreCase(name)) { return m_fields[i]; }
099        }
100        return null;
101    }
102
103    /**
104     * Gets the field metadata for the given name and type.
105     * @param name : the name of the field
106     * @param type : the type of the field
107     * @return the corresponding field metadata or <code>null</code> if not found
108     */
109    public FieldMetadata getField(String name, String type) {
110        for (int i = 0; i < m_fields.length; i++) {
111            if (m_fields[i].getFieldName().equalsIgnoreCase(name) && m_fields[i].getFieldType().equalsIgnoreCase(type)) { return m_fields[i]; }
112        }
113        return null;
114    }
115
116    /**
117     * Checks if the given interface name is implemented.
118     * This methods checks on interface directly implemented
119     * by the implementation class.
120     * @param itf the interface to check.
121     * @return <code>true</code> if the implementation class implements
122     * the given interface.
123     */
124    public boolean isInterfaceImplemented(String itf) {
125        for (int i = 0; i < m_interfaces.length; i++) {
126            if (m_interfaces[i].equals(itf)) { return true; }
127        }
128        return false;
129    }
130
131    /**
132     * Gets the MethodMetadata corresponding to the method
133     * (contained in the implementation class) with
134     * the given name.
135     * If several methods match, the first one is returned.
136     * @param name the name of the method to find.
137     * @return the method metadata object or <code>null</code> if not found
138     */
139    public MethodMetadata getMethod(String name) {
140        for (int i = 0; i < m_methods.length; i++) {
141            if (m_methods[i].getMethodName().equalsIgnoreCase(name)) { return m_methods[i]; }
142        }
143        return null;
144    }
145
146    /**
147     * Gets the MethodMetadata list corresponding to the method
148     * (contained in the implementation class) to given name.
149     * All methods contained in the implementation class matching
150     * with the name are in the returned list.
151     * @param name the name of the method to look for.
152     * @return the Method Metadata array or an empty array if not found
153     */
154    public MethodMetadata[] getMethods(String name) {
155        MethodMetadata[] mms = new MethodMetadata[0];
156        for (int i = 0; i < m_methods.length; i++) {
157            if (m_methods[i].getMethodName().equalsIgnoreCase(name)) {
158                if (mms.length > 0) {
159                    MethodMetadata[] newInstances = new MethodMetadata[mms.length + 1];
160                    System.arraycopy(mms, 0, newInstances, 0, mms.length);
161                    newInstances[mms.length] = m_methods[i];
162                    mms = newInstances;
163                } else {
164                    mms = new MethodMetadata[] { m_methods[i] };
165                }
166            }
167        }
168        return mms;
169    }
170
171    /**
172     * Gets the MethodMetadata list corresponding to the constructors
173     * (contained in the implementation class).
174     * @return the Method Metadata array or an empty array if not found
175     */
176    public MethodMetadata[] getConstructors() {
177        return getMethods("$init");
178    }
179
180    /**
181     * Gets the MethodMetadata corresponding to the method
182     * (contained in the implementation class) to given name
183     * and argument types.
184     * @param name the name of the method to look for.
185     * @param types the array of the argument types of the method
186     * @return the Method Metadata or <code>null</code> if not found
187     */
188    public MethodMetadata getMethod(String name, String[] types) {
189        for (int i = 0; i < m_methods.length; i++) {
190            if (m_methods[i].getMethodName().equalsIgnoreCase(name) && m_methods[i].getMethodArguments().length == types.length) {
191                int argIndex = 0;
192                for (; argIndex < types.length; argIndex++) {
193                    if (! types[argIndex].equals(m_methods[i].getMethodArguments()[argIndex])) {
194                        break;
195                    }
196                }
197                if (argIndex == types.length) { return m_methods[i]; } // No mismatch detected.
198            }
199        }
200        return null;
201    }
202
203    /**
204     * Gets the constructor corresponding to the given argument types.
205     * @param types the argument types
206     * @return the matching constructor or <code>null</code> if not found.
207     */
208    public MethodMetadata getConstructor(String[] types) {
209        return getMethod("$init", types); // Constructors are named $init in the manipulation metadata
210    }
211
212     /**
213     * Adds a method to the list.
214     * This method is used during the creation of the {@link PojoMetadata}
215     * object.
216     * @param method the Method Metadata to add.
217     */
218    private void addMethod(MethodMetadata method) {
219        if (m_methods.length > 0) {
220            MethodMetadata[] newInstances = new MethodMetadata[m_methods.length + 1];
221            System.arraycopy(m_methods, 0, newInstances, 0, m_methods.length);
222            newInstances[m_methods.length] = method;
223            m_methods = newInstances;
224        } else {
225            m_methods = new MethodMetadata[] { method };
226        }
227    }
228
229     /**
230     * Adds a field to the list.
231     * This method is used during the creation of the {@link PojoMetadata}
232     * object.
233     * @param field the Field Metadata to add.
234     */
235    private void addField(FieldMetadata field) {
236        if (m_fields.length > 0) {
237            FieldMetadata[] newInstances = new FieldMetadata[m_fields.length + 1];
238            System.arraycopy(m_fields, 0, newInstances, 0, m_fields.length);
239            newInstances[m_fields.length] = field;
240            m_fields = newInstances;
241        } else {
242            m_fields = new FieldMetadata[] { field };
243        }
244    }
245
246    /**
247     * Adds the interface to the list.
248     * This method is used during the creation of the {@link PojoMetadata}
249     * object.
250     * @param itf the interface name to add.
251     */
252    private void addInterface(String itf) {
253        if (m_interfaces.length > 0) {
254            String[] newInstances = new String[m_interfaces.length + 1];
255            System.arraycopy(m_interfaces, 0, newInstances, 0, m_interfaces.length);
256            newInstances[m_interfaces.length] = itf;
257            m_interfaces = newInstances;
258        } else {
259            m_interfaces = new String[] { itf };
260        }
261    }
262
263    public String getSuperClass() {
264        return m_super;
265    }
266
