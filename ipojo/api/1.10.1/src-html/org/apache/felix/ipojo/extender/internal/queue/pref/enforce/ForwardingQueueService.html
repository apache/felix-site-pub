020package org.apache.felix.ipojo.extender.internal.queue.pref.enforce;
021
022import org.apache.felix.ipojo.extender.internal.LifecycleQueueService;
023import org.apache.felix.ipojo.extender.queue.Callback;
024import org.apache.felix.ipojo.extender.queue.JobInfo;
025
026import java.util.List;
027import java.util.concurrent.Callable;
028import java.util.concurrent.Future;
029
030/**
031 * A queue service implementation delegating to a {@link LifecycleQueueService}.
032 */
033public abstract class ForwardingQueueService implements LifecycleQueueService {
034
035    protected abstract LifecycleQueueService delegate();
036
037    public void start() {
038        delegate().start();
039    }
040
041    public void stop() {
042        delegate().stop();
043    }
044
045    public int getFinished() {
046        return delegate().getFinished();
047    }
048
049    public int getWaiters() {
050        return delegate().getWaiters();
051    }
052
053    public int getCurrents() {
054        return delegate().getCurrents();
055    }
056
057    public List<JobInfo> getWaitersInfo() {
058        return delegate().getWaitersInfo();
059    }
060
061    public <T> Future<T> submit(Callable<T> callable, Callback<T> callback, String description) {
062        return delegate().submit(callable, callback, description);
063    }
064
065    public <T> Future<T> submit(Callable<T> callable, String description) {
066        return delegate().submit(callable, description);
067    }
068
069    public <T> Future<T> submit(Callable<T> callable) {
070        return delegate().submit(callable);
071    }
072}
