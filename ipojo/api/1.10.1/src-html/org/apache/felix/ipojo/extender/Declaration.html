020package org.apache.felix.ipojo.extender;
021
022/**
023 * A declaration is a creation instruction of an entity (Component type, Factory, Instance...).
024 * All declarations are exposed as services. <em>Processors</em> are tracking the adequate declaration type and create
025 * the entity.
026 * <p/>
027 * Declaration can be <em>bound</em> or <em>unbound</em> whether they are fulfilled. When they are unbound,
028 * a message or/and an error can be set.
029 */
030public interface Declaration {
031    /**
032     * Gets the declaration status.
033     *
034     * @return the current status. As Status are immutable, it returns a new object every time.
035     */
036    Status getStatus();
037
038    /**
039     * Marks the declaration bound.
040     */
041    void bind();
042
043    /**
044     * Unbinds the declaration.
045     *
046     * @param message an explanation
047     */
048    void unbind(String message);
049
050    /**
051     * Unbinds the declaration
052     *
053     * @param message   an explanation
054     * @param throwable an error
055     */
056    void unbind(String message, Throwable throwable);
057}
