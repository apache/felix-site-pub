019package org.apache.felix.ipojo.api;
020
021import java.util.Dictionary;
022import java.util.Properties;
023
024import org.apache.felix.ipojo.ComponentInstance;
025import org.apache.felix.ipojo.ConfigurationException;
026import org.apache.felix.ipojo.MissingHandlerException;
027import org.apache.felix.ipojo.UnacceptableConfiguration;
028
029/**
030 * Allows defining a primitive component type that create an unique
031 * instance when created. The factory is set to private by default.
032 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
033 */
034public class SingletonComponentType extends PrimitiveComponentType {
035    
036    /**
037     * The POJO object to inject through the
038     * instance configuration.
039     */
040    private Object m_pojo;
041    
042    /**
043     * Creates a SingletonComponentType.
044     * This type is set to private by default.
045     */
046    public SingletonComponentType() {
047        setPublic(false);
048    }
049    
050    /**
051     * Sets the pojo object used by the instance.
052     * The object must be compatible with the 
053     * implementation class. 
054     * @param obj the object.
055     * @return the current singleton component type.
056     */
057    public SingletonComponentType setObject(Object obj) {
058        m_pojo = obj;
059        return this;
060    }
061    
062    /**
063     * Starts the component type and creates the singleton
064     * instance. This method has to be called in place of the
065     * {@link PrimitiveComponentType#start()} and the
066     * {@link PrimitiveComponentType#createInstance()}  methods.
067     * @return the created component instance.
068     * @throws ConfigurationException occurs if the type description is
069     * incorrect
070     * @throws MissingHandlerException occurs if a handler is not available 
071     * @throws UnacceptableConfiguration  occurs if the configuration is not 
072     * acceptable by the instance  
073     * @see org.apache.felix.ipojo.api.ComponentType#start()
074     * @see PrimitiveComponentType#createInstance()
075     */
076    public ComponentInstance create() throws UnacceptableConfiguration, MissingHandlerException, ConfigurationException {
077        start();
078        if (m_pojo != null) {
079            Dictionary dict = new Properties();
080            dict.put("instance.object", m_pojo);
081            return createInstance(dict);
082        } else {
083            return createInstance();
084        }
085    }
086    
087    /**
088     * Starts the component type and creates the singleton
089     * instance. This method has to be called in place of the
090     * {@link PrimitiveComponentType#start()} and the
091     * {@link PrimitiveComponentType#createInstance()}  methods.
092     * @param conf the instance configuration
093     * @return the created component instance
094     * @throws ConfigurationException occurs if the type description is
095     * incorrect
096     * @throws MissingHandlerException occurs if a handler is not available 
097     * @throws UnacceptableConfiguration  occurs if the configuration is not 
098     * acceptable by the instance  
099     * @see org.apache.felix.ipojo.api.ComponentType#start()
100     * @see PrimitiveComponentType#createInstance()
101     */
102    public ComponentInstance create(Dictionary conf) throws UnacceptableConfiguration, MissingHandlerException, ConfigurationException {
103        start();
104        if (m_pojo != null) {
105            conf.put("instance.object", m_pojo);
106        }
107        return createInstance(conf);
108    }
109    
110    /**
111     * Starts the component type and creates the singleton
112     * instance. This method has to be called in place of the
113     * {@link PrimitiveComponentType#start()} and the
114     * {@link PrimitiveComponentType#createInstance()}  methods.
115     * @param name  the name of the instance to create. This parameter will
116     * be used as the <code>instance.name</code> property.
117     * @return the created component instance.
118     * @throws ConfigurationException occurs if the type description is
119     * incorrect
120     * @throws MissingHandlerException occurs if a handler is not available 
121     * @throws UnacceptableConfiguration  occurs if the configuration is not 
122     * acceptable by the instance  
123     * @see org.apache.felix.ipojo.api.ComponentType#start()
124     * @see PrimitiveComponentType#createInstance()
125     */
126    public ComponentInstance create(String name) throws UnacceptableConfiguration, MissingHandlerException, ConfigurationException {
127        start();
128        if (m_pojo != null) {
129            Dictionary dict = new Properties();
130            dict.put("instance.name", name);
131            dict.put("instance.object", m_pojo);
132            return createInstance(dict);
133        } else {
134            return createInstance(name);
135        }
136    }
137    
138   
139
140}
