038/**
039 * The component factory manages component instance objects. This management
040 * consists to create and manage component instances build with the current
041 * component factory. If the factory is public a {@see Factory} service is exposed.
042 *
043 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
044 * @see IPojoFactory
045 * @see TrackerCustomizer
046 */
047public class ComponentFactory extends IPojoFactory implements TrackerCustomizer {
048
049    /**
050     * System property set to automatically attach primitive handlers to primitive
051     * component types.
052     * The value is a String parsed as a list (comma separated). Each element is
053     * the fully qualified name of the handler <code>namespace:name</code>.
054     */
055    public static final String HANDLER_AUTO_PRIMITIVE = "org.apache.felix.ipojo.handler.auto.primitive";
056    /**
057     * The tracker used to track required handler factories.
058     * Immutable once set.

060    protected Tracker m_tracker;

067     * The component implementation class.

070    private byte[] m_clazz;

134    /**
135     * Allows a factory to check if the given element is well-formed.
136     * A component factory metadata is correct if they contain the 'classname' attribute.
137     * As this method is called from the (single-threaded) constructor, no synchronization is needed.
138     *
139     * @param element the metadata describing the component
140     * @throws ConfigurationException if the element describing the factory is malformed.
141     */
142    public void check(Element element) throws ConfigurationException {
143        m_classname = element.getAttribute("classname");
144        if (m_classname == null) {
145            throw new ConfigurationException("A component needs a class name : " + element);
146        }
147        m_manipulation = new PojoMetadata(m_componentMetadata);
148    }

161    /**
162     * Creates a primitive instance.
163     * This method is called when holding the lock.
164     *
165     * @param config   the instance configuration
166     * @param context  the service context (null if the instance has to be created in the global space).
167     * @param handlers the handlers to attach to the instance
168     * @return the created instance
169     * @throws org.apache.felix.ipojo.ConfigurationException
170     *          if the configuration process failed.
171     * @see org.apache.felix.ipojo.IPojoFactory#createInstance(java.util.Dictionary, org.apache.felix.ipojo.IPojoContext, org.apache.felix.ipojo.HandlerManager[])
172     */
173    public ComponentInstance createInstance(Dictionary config, IPojoContext context, HandlerManager[] handlers) throws org.apache.felix.ipojo.ConfigurationException {
174        InstanceManager instance = new InstanceManager(this, context, handlers);
175
176        try {
177            instance.configure(m_componentMetadata, config);
178            instance.start();
179            return instance;
180        } catch (ConfigurationException e) {
181            // An exception occurs while executing the configure or start
182            // methods, the instance is stopped so the architecture service is still published and so we can debug
183            // the issue.
184            instance.stop();
185            throw e;
186        } catch (Throwable e) { // All others exception are handled here.
187            // As for the previous case, the instance is stopped.
188            instance.stop();
189            m_logger.log(Logger.ERROR, e.getMessage(), e);
190            throw new ConfigurationException(e.getMessage(), e);
191        }
192
193    }

195    /**
196     * Defines a class.
197     * This method needs to be synchronized to avoid that the classloader
198     * is created twice.
199     * This method delegates the <code>define</code> method invocation to the
200     * factory classloader.
201     *
202     * @param name   the qualified name of the class
203     * @param clazz  the byte array of the class
204     * @param domain the protection domain of the class
205     * @return the defined class object
206     */
207    public synchronized Class<? extends Object> defineClass(String name, byte[] clazz, ProtectionDomain domain) {
208        if (!m_useFactoryClassloader) {
209            m_logger.log(Log.WARNING, "A class definition was required even without the factory classloader enabled");
210        }
211
212        if (m_classLoader == null) {
213            m_classLoader = new FactoryClassloader(this);
214        }
215        return m_classLoader.defineClass(name, clazz, domain);
216    }
217
218    /**
219     * Loads a class. This method checks if the class
220     * to load is the implementation class or not.
221     * If it is, the factory classloader is used, else
222     * the {@link Bundle#loadClass(String)} is called.
223     *
224     * The implementation class is loaded using the factory classloader only if the factory classloader was enabled
225     *
226     * @param className the name of the class to load
227     * @return the resulting Class object
228     * @throws ClassNotFoundException if the class is not found
229     * @see #setUseFactoryClassloader(boolean)
230     */
231    public Class loadClass(String className) throws ClassNotFoundException {
232        if (m_useFactoryClassloader && m_clazz != null && m_classname.equals(className)) {  // Immutable fields.
233            return defineClass(className, m_clazz, null);
234        }
235        return m_context.getBundle().loadClass(className);
236    }

238    /**
239     * Starts the factory.
240     * This method is not called when holding the monitor lock.
241     */
242    public void starting() {
243        if (m_tracker == null) {
244            if (m_requiredHandlers.size() != 0) {
245                try {
246                    String filter = "(&(" + Handler.HANDLER_TYPE_PROPERTY + "=" + PrimitiveHandler.HANDLER_TYPE + ")" + "(factory.state=1)" + ")";
247                    m_tracker = new Tracker(m_context, m_context.createFilter(filter), this);
248                    m_tracker.open();
249                } catch (InvalidSyntaxException e) {
250                    m_logger.log(Logger.ERROR, "A factory filter is not valid: " + e.getMessage()); //Holding the lock should not be an issue here.
251                    stop();
252                }
253            }
254        }
255        // Else, the tracking has already started.
256    }
257
258    /**
259     * Stops all the instance managers.
260     * This method is called when holding the lock.
261     */
262    public void stopping() {
263        if (m_tracker != null) {
264            m_tracker.close();
265            m_tracker = null;
266        }
267    }

269    /**
270     * Computes the factory name. The factory name is computed from
271     * the 'name' and 'classname' attributes.
272     * This method does not manipulate any non-immutable fields,
273     * so does not need to be synchronized.
274     *
275     * @return the factory name.
276     */
277    public String getFactoryName() {
278        String name = m_componentMetadata.getAttribute("name");
279        if (name == null) {
280            // No factory name, use the classname (mandatory attribute)
281            name = m_componentMetadata.getAttribute("classname");
282        }
283        return name;
284    }

286    /**
287     * Computes required handlers.
288     * This method does not manipulate any non-immutable fields,
289     * so does not need to be synchronized.
290     * This method checks the {@link ComponentFactory#HANDLER_AUTO_PRIMITIVE}
291     * system property to add the listed handlers to the required handler set.
292     *
293     * @return the required handler list.
294     */
295    public List<RequiredHandler> getRequiredHandlerList() {
296        List<RequiredHandler> list = new ArrayList<RequiredHandler>();
297        Element[] elems = m_componentMetadata.getElements();
298        for (Element current : elems) {
299            if (!"manipulation".equals(current.getName())) { // Remove the manipulation element
300                RequiredHandler req = new RequiredHandler(current.getName(), current.getNameSpace());
301                if (!list.contains(req)) {
302                    list.add(req);
303                }
304            }
305        }
306
307        // Add architecture if architecture != 'false'
308        String arch = m_componentMetadata.getAttribute("architecture");
309        if (arch == null || arch.equalsIgnoreCase("true")) {
310            list.add(new RequiredHandler("architecture", null));
311        }
312
313
314        // Determine if the component must be immediate.
315        // A component becomes immediate if it doesn't provide a service,
316        // and does not specified that the component is not immediate.
317        if (m_componentMetadata.getElements("provides") == null) {
318            String imm = m_componentMetadata.getAttribute("immediate");
319            if (imm == null) { // immediate not specified, set the immediate attribute to true
320                getLogger().log(
321                        Logger.INFO,
322                        "The component type " + getFactoryName()
323                                + " becomes immediate");
324                m_componentMetadata.addAttribute(new Attribute("immediate",
325                        "true"));
326            }
327        }
328
329        // Add lifecycle callback if immediate = true
330        RequiredHandler reqCallback = new RequiredHandler("callback", null);
331        String imm = m_componentMetadata.getAttribute("immediate");
332        if (!list.contains(reqCallback) && imm != null && imm.equalsIgnoreCase("true")) {
333            list.add(reqCallback);
334        }
335
336        // Manage auto attached handler.
337        String v = System.getProperty(HANDLER_AUTO_PRIMITIVE);
338        if (v != null && v.length() != 0) {
339            String[] hs = ParseUtils.split(v, ",");
340            for (String h1 : hs) {
341                String h = h1.trim();
342                String[] segments = ParseUtils.split(h, ":");
343                RequiredHandler rq = null;
344                if (segments.length == 2) { // External handler
345                    rq = new RequiredHandler(segments[1], segments[0]);
346                } else if (segments.length == 1) { // Core handler
347                    rq = new RequiredHandler(segments[1], null);
348                } // Others case are ignored.
349
350                if (rq != null) {
351                    // Check it's not already contained
352                    if (!list.contains(rq)) {
353                        list.add(rq);
354                    }
355                }
356            }
357        }
358
359        return list;
360    }

362    /**
363     * This method is called when a new handler factory is detected.
364     * Test if the factory can be used or not.
365     * This method need to be synchronized as it accesses to the content
366     * of required handlers.
367     *
368     * @param reference the new service reference.
369     * @return <code>true</code> if the given factory reference matches with a required handler.
370     * @see org.apache.felix.ipojo.util.TrackerCustomizer#addingService(org.osgi.framework.ServiceReference)
371     */
372    public synchronized boolean addingService(ServiceReference reference) {
373        for (int i = 0; i < m_requiredHandlers.size(); i++) {
374            RequiredHandler req = (RequiredHandler) m_requiredHandlers.get(i);
375            if (req.getReference() == null && match(req, reference)) {
376                int oldP = req.getLevel();
377                req.setReference(reference);
378                // If the priority has changed, sort the list.
379                if (oldP != req.getLevel()) {
380                    // Manipulate the list.
381                    Collections.sort(m_requiredHandlers);
382                }
383                return true;
384            }
385        }
386        return false;
387    }

389    /**
390     * This method is called when a matching service has been added to the tracker,
391     * we can no compute the factory state. This method is synchronized to avoid
392     * concurrent calls to method modifying the factory state.
393     *
394     * @param reference the added service reference.
395     * @see org.apache.felix.ipojo.util.TrackerCustomizer#addedService(org.osgi.framework.ServiceReference)
396     */
397    public synchronized void addedService(ServiceReference reference) {
398        if (m_state == INVALID) {
399            computeFactoryState();
400        }
401    }

403    /**
404     * This method is called when a used handler factory disappears.
405     * This method is synchronized to avoid concurrent calls to method modifying
406     * the factory state.
407     *
408     * @param reference the leaving service reference.
409     * @param service   the handler factory object.
410     * @see org.apache.felix.ipojo.util.TrackerCustomizer#removedService(org.osgi.framework.ServiceReference, java.lang.Object)
411     */
412    public synchronized void removedService(ServiceReference reference, Object service) {
413        // Look for the implied reference and invalid the handler identifier
414        for (Object m_requiredHandler : m_requiredHandlers) {
415            RequiredHandler req = (RequiredHandler) m_requiredHandler;
416            if (reference.equals(req.getReference())) {
417                req.unRef(); // This method will unget the service.
418                computeFactoryState();
419                return; // The factory can be used only once.
420            }
421        }
422    }
