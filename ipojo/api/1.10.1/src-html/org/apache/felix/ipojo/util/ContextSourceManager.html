020package org.apache.felix.ipojo.util;
021
022import org.apache.felix.ipojo.ContextListener;
023import org.apache.felix.ipojo.ContextSource;
024import org.osgi.framework.BundleContext;
025import org.osgi.framework.InvalidSyntaxException;
026import org.osgi.framework.ServiceReference;
027
028import java.util.ArrayList;
029import java.util.Arrays;
030import java.util.Dictionary;
031import java.util.List;
032
033/**
034 * This class manages context-source management.
035 *
036 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
037 */
038public class ContextSourceManager implements ContextListener {
039
040    /**
041     * The variable prefix
042     */
043    public static final String VARIABLE_START = "${";
044
045    /**
046     * The variable end
047     */
048    public static final char VARIABLE_END = '}';
049
050    /**
051     * List of characters forbidden in variable names
052     */
053    public static final List<Character> FORBIDDEN_CHARACTERS = Arrays.asList(
054            '(', ')', '&', '|', '!', '=', '*', ' ');
055
056    /**
057     * Managed dependency.
058     */
059    private DependencyModel m_dependency;
060    /**
061     * List of monitored context sources.
062     */
063    private List<ContextSource> m_sources = new ArrayList<ContextSource>(1);
064    /**
065     * Variables contained in the original filter.
066     */
067    private List<String> m_variables;
068    /**
069     * Original filter (containing variables).
070     */
071    private String m_filter;
072    /**
073     * Bundle context.
074     */
075    private BundleContext m_context;
076    /**
077     * The Context-Source service tracker.
078     */
079    private Tracker m_tracker;
080
081    /**
082     * Creates the context source manager.
083     *
084     * @param dependency the dependency model on which this manager is plugged.
085     */
086    public ContextSourceManager(DependencyModel dependency) throws InvalidSyntaxException {
087        m_filter = dependency.getFilter();
088        m_variables = extractVariablesFromFilter(m_filter);
089        m_dependency = dependency;
090        m_context = dependency.getComponentInstance().getContext();
091    }
092
093    /**
094     * This method substitutes ${var} substring by values stored in a map.
095     *
096     * @param str    : string with variables
097     * @param values : dictionary containing the variable name and the value.
098     * @return resulted string
099     */
100    public static String substitute(String str, Dictionary values) {
101        int len = str.length();
102        StringBuilder builder = new StringBuilder(len);
103
104        int prev = 0;
105        int start = str.indexOf("${");
106        int end = str.indexOf('}', start);
107        while (start != -1 && end != -1) {
108            String key = str.substring(start + 2, end);
109            Object value = values.get(key);
110            if (value == null) {
111                builder.append(str.substring(prev, end + 1));
112            } else {
113                builder.append(str.substring(prev, start));
114                builder.append(value);
115            }
116            prev = end + 1;
117            if (prev >= str.length()) {
118                break;
119            }
120
121            start = str.indexOf("${", prev);
122            if (start != -1) {
123                end = str.indexOf('}', start);
124            }
125        }
126
127        builder.append(str.substring(prev));
128
129        return builder.toString();
130    }
131
132    /**
133     * Extracts the variables (${name}) from the given filter.
134     *
135     * @param filter : string form of the filter.
136     * @return the list of found properties.
137     * @throws InvalidSyntaxException thrown when the variables are not consistent.
138     */
139    public static List<String> extractVariablesFromFilter(String filter) throws InvalidSyntaxException {
140        List<String> variables = new ArrayList<String>();
141        int prev;
142        int start = filter.indexOf(VARIABLE_START);
143        int end = filter.indexOf(VARIABLE_END, start);
144        while (start != -1) {
145            // Unfinished variable
146            if (end == -1) {
147                throw new InvalidSyntaxException("The filter contains an unfinished variable", filter);
148            }
149
150            String key = filter.substring(start + VARIABLE_START.length(), end);
151
152            // Error detection :
153            // Empty variable.
154            if (key.length() == 0) {
155                throw new InvalidSyntaxException("The filter variable '${}' is not a valid " +
156                        "variable", filter);
157            }
158            // Variable with spaces.
159            Character forbidden = containsForbiddenCharacter(key);
160            if (forbidden != null) {
161                throw new InvalidSyntaxException("The filter variable '${" + key + "}' contains a forbidden " +
162                        "character : '" + forbidden + "'", filter);
163            }
164
165
166            variables.add(key);
167            prev = end + 1;
168            if (prev >= filter.length()) {
169                break;
170            }
171
172            start = filter.indexOf(VARIABLE_START, prev);
173            if (start != -1) {
174                end = filter.indexOf(VARIABLE_END, start);
175            }
176        }
177
178        return variables;
179    }
180
181    private static Character containsForbiddenCharacter(String key) {
182        for (Character character : FORBIDDEN_CHARACTERS) {
183            if (key.indexOf(character) != -1) {
184                return character;
185            }
186        }
187        // Safe key.
188        return null;
189    }
190
191    /**
192     * Start the context management.
193     */
194    public void start() {
195        if (m_tracker == null) {
196            m_tracker = new Tracker(m_context, ContextSource.class.getName(), new SourceTracker());
197        }
198        m_tracker.open();
199        computeFilter();
200    }
201
202    /**
203     * Stop the context management.
204     */
205    public void stop() {
206        if (m_tracker != null) {
207            m_tracker.close();
208            m_tracker = null;
209        }
210        // Reinitialize the filter.
211        setFilter(m_filter);
212        for (ContextSource source : m_sources) {
213            source.unregisterContextListener(this);
214        }
215        m_sources.clear();
216    }
217
218    /**
219     * Set the filter of the managed dependency.
220     *
221     * @param filter : the new filter to apply
222     */
223    private void setFilter(String filter) {
224        if (!filter.equals(m_dependency.getFilter())) {
225            // Reconfigure
226            try {
227                m_dependency.setFilter(m_context.createFilter(filter));
228            } catch (InvalidSyntaxException e) {
229                throw new IllegalStateException("A context filter is invalid : " + filter, e);
230            }
231        }
232    }
233
234    /**
235     * Compute the new filter.
236     */
237    private void computeFilter() {
238        String fil = m_filter; // Gets a copy.
239        synchronized (this) {
240            for (ContextSource m_source : m_sources) {
241                Dictionary props = m_source.getContext();
242                fil = substitute(fil, props); //NOPMD
243            }
244        }
245        if (!fil.equals(m_dependency.getFilter())) {
246            setFilter(fil);
247        }
248    }
249
250    /**
251     * A context source has modified a monitored property.
252     *
253     * @param source   : source
254     * @param property : modified property
255     * @param value    : new value.
256     * @see org.apache.felix.ipojo.ContextListener#update(org.apache.felix.ipojo.ContextSource, String, Object)
257     */
258    public synchronized void update(ContextSource source, String property, Object value) {
259        computeFilter();
260    }
261
262    /**
263     * A context source appears.
264     *
265     * @param source : new context source.
266     */
267    public void addContextSource(ContextSource source) {
268        m_sources.add(source);
269        computeFilter();
270        source.registerContextListener(this, m_variables.toArray(new String[m_variables.size()]));
271    }
272
273    /**
274     * A context source disappears.
275     *
276     * @param source : leaving context source.
277     */
278    public void removeContextSource(ContextSource source) {
279        source.unregisterContextListener(this);
280        m_sources.remove(source);
281        computeFilter();
282    }
283
284    private class SourceTracker implements TrackerCustomizer {
285        /**
286         * A new context-source was added.
287         * This method inject the context-source object in the source manager.
288         *
289         * @param reference : service reference.
290         * @see TrackerCustomizer#addedService(org.osgi.framework.ServiceReference)
291         */
292        public void addedService(ServiceReference reference) {
293            addContextSource((ContextSource) m_tracker.getService(reference));
294        }
295
296        /**
297         * A new context-source is adding in the tracker..
298         *
299         * @param reference : service reference
300         * @return true.
301         * @see TrackerCustomizer#addingService(org.osgi.framework.ServiceReference)
302         */
303        public boolean addingService(ServiceReference reference) {
304            return true;
305        }
306
307        /**
308         * A used context-source is modified.
309         *
310         * @param reference : service reference.
311         * @param service   : service object.
312         * @see TrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, Object)
313         */
314        public void modifiedService(ServiceReference reference, Object service) {
315            // Nothing to do.
316        }
317
318        /**
319         * A used context-source disappears.
320         * This method notify the Source Manager in order to manage this departure.
321         *
322         * @param reference : service reference.
323         * @param service   : service object.
324         * @see TrackerCustomizer#modifiedService(org.osgi.framework.ServiceReference, Object)
325         */
326        public void removedService(ServiceReference reference, Object service) {
327            removeContextSource((ContextSource) service);
328        }
329
330    }
