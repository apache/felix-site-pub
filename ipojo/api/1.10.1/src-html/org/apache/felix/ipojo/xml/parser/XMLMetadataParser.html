077    /**
078     * End of an element.
079     * @param namespaceURI : element namespace
080     * @param localName : local name
081     * @param qName : qualified name
082     * @throws SAXException : occurs when the element is malformed
083     * @see org.xml.sax.ContentHandler#endElement(java.lang.String, java.lang.String, java.lang.String)
084     */
085    public void endElement(String namespaceURI, String localName, String qName) throws SAXException {
086        // Get the last element of the list
087        Element lastElement = removeLastElement();
088
089        // The name is consistent
090        // Add this element last element with if it is not the root
091        if (m_elements.length != 0) {
092            Element newQueue = m_elements[m_elements.length - 1];
093            newQueue.addElement(lastElement);
094        } else {
095            // It is the last element
096            addElement(lastElement);
097        }
098
099    }

165    /**
166     * Start an element.
167     * @param namespaceURI : element namespace.
168     * @param localName : local element.
169     * @param qName : qualified name.
170     * @param atts : attribute
171     * @throws SAXException : occurs if the element cannot be parsed correctly.
172     * @see org.xml.sax.ContentHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
173     */
174    public void startElement(String namespaceURI, String localName, String qName, Attributes atts) throws SAXException {
175        String namespace = namespaceURI;
176        if (namespaceURI != null
177                && (namespaceURI.equalsIgnoreCase("org.apache.felix.ipojo") || namespaceURI.equalsIgnoreCase("org.apache.felix.ipojo.composite"))) {
178            namespace = null; // Remove the 'org.apache.felix.ipojo' namespace
179        }
180
181        Element elem = new Element(localName, namespace);
182        for (int i = 0; i < atts.getLength(); i++) {
183            String name = (String) atts.getLocalName(i);
184            String ns = (String) atts.getURI(i);
185            String value = (String) atts.getValue(i);
186            Attribute att = new Attribute(name, ns, value);
187            elem.addAttribute(att);
188        }
189
190        addElement(elem);
191
192    }

206    /**
207     * Add an element.
208     * @param elem : the element to add
209     */
210    private void addElement(Element elem) {
211        for (int i = 0; (m_elements != null) && (i < m_elements.length); i++) {
212            if (m_elements[i] == elem) {
213                return;
214            }
215        }
216
217        if (m_elements != null) {
218            Element[] newElementsList = new Element[m_elements.length + 1];
219            System.arraycopy(m_elements, 0, newElementsList, 0, m_elements.length);
220            newElementsList[m_elements.length] = elem;
221            m_elements = newElementsList;
222        } else {
223            m_elements = new Element[] { elem };
224        }
225    }
226
227    /**
228     * Remove an element.
229     * @return : the removed element.
230     */
231    private Element removeLastElement() {
232        int idx;
233        idx = m_elements.length - 1;
234        Element last = m_elements[idx];
235        if (idx >= 0) {
236            if ((m_elements.length - 1) == 0) {
237                // It is the last element of the list;
238                m_elements = new Element[0];
239            } else {
240                // Remove the last element of the list :
241                Element[] newElementsList = new Element[m_elements.length - 1];
242                System.arraycopy(m_elements, 0, newElementsList, 0, idx);
243                m_elements = newElementsList;
244            }
245        }
246        return last;
247    }

250    /**
251     * An error occurs during the XML-Schema checking.
252     * This method propagates the error except if the error concerns
253     * no XML-Schemas are used   (<code>cvc-elt.1</code>).
254     * @param saxparseexception the checking error
255     * @throws SAXException the propagated exception
256     * @see org.xml.sax.ErrorHandler#error(org.xml.sax.SAXParseException)
257     */
258    public void error(SAXParseException saxparseexception) throws SAXException {
259        if (saxparseexception.getMessage().contains("cvc-elt.1")) {
260            return; // Do not throw an exception when no schema defined.
261        }
262        throw saxparseexception;
263    }
