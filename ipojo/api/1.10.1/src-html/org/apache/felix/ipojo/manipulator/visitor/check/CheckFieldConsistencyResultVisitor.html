020package org.apache.felix.ipojo.manipulator.visitor.check;
021
022import org.apache.felix.ipojo.manipulator.ManipulationResultVisitor;
023import org.apache.felix.ipojo.manipulator.Reporter;
024import org.apache.felix.ipojo.manipulator.util.Metadatas;
025import org.apache.felix.ipojo.manipulator.visitor.ManipulationResultAdapter;
026import org.apache.felix.ipojo.metadata.Element;
027
028import java.util.ArrayList;
029import java.util.List;
030
031/**
032 * This visitor checks that field referenced in the metadata are present in the bytecode.
033 *
034 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
035 */
036public class CheckFieldConsistencyResultVisitor extends ManipulationResultAdapter {
037
038    /**
039     * Component's metadata.
040     */
041    private Element m_metadata;
042
043    /**
044     * Reporter for errors.
045     */
046    private Reporter m_reporter;
047
048    public CheckFieldConsistencyResultVisitor(ManipulationResultVisitor visitor) {
049        super(visitor);
050    }
051
052    public void setMetadata(Element metadata) {
053        this.m_metadata = metadata;
054    }
055
056    public void setReporter(Reporter reporter) {
057        this.m_reporter = reporter;
058    }
059
060    public void visitClassStructure(Element structure) {
061
062        List<String> fieldsInStructure = new ArrayList<String>();
063        collectStructuralFields(fieldsInStructure, structure);
064
065        List<String> fieldsInMetadata = new ArrayList<String>();
066        Metadatas.findFields(fieldsInMetadata, m_metadata);
067
068        checkReferredFieldsAreInStructure(fieldsInMetadata, fieldsInStructure);
069
070        // Do this at the end because the writer insert the manipulation
071        // metadata inside the component Element
072        // So to avoid duplicate find, we need to execute this code at the end
073        super.visitClassStructure(structure);
074
075    }
076
077    private void collectStructuralFields(List<String> fieldsInStructure, Element structure) {
078        Element[] fields = structure.getElements("field");
079        if (fields != null) {
080            for (Element field : fields) {
081                fieldsInStructure.add(field.getAttribute("name"));
082            }
083        }
084    }
085
086    /**
087     * Detects missing fields.
088     * If a referenced field does not exist in the class
089     * the method throws an error breaking the build process.
090     * @param fieldsInMetadata
091     * @param fieldsInStructure
092     */
093    private void checkReferredFieldsAreInStructure(List<String> fieldsInMetadata, List<String> fieldsInStructure) {
094        // Then, try to find each referred field in the given field map
095        for (String fieldName : fieldsInMetadata) {
096            if (!fieldsInStructure.contains(fieldName)) {
097                m_reporter.error("The field " + fieldName + " is referenced in the "
098                        + "metadata but does not exist in the " + Metadatas.getComponentType(m_metadata)
099                        + " class");
100            }
101        }
102    }
103
104}
