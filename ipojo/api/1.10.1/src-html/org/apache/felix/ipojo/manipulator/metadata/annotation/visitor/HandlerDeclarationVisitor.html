020package org.apache.felix.ipojo.manipulator.metadata.annotation.visitor;
021
022import org.apache.felix.ipojo.manipulator.Reporter;
023import org.apache.felix.ipojo.manipulator.metadata.annotation.ComponentWorkbench;
024import org.apache.felix.ipojo.metadata.Attribute;
025import org.apache.felix.ipojo.metadata.Element;
026import org.objectweb.asm.AnnotationVisitor;
027import org.objectweb.asm.commons.EmptyVisitor;
028import org.w3c.dom.Attr;
029import org.w3c.dom.Document;
030import org.w3c.dom.NamedNodeMap;
031import org.w3c.dom.NodeList;
032
033import javax.xml.parsers.DocumentBuilder;
034import java.io.ByteArrayInputStream;
035import java.io.IOException;
036import java.io.InputStream;
037
038/**
039 * Parse the @HandlerDeclaration annotation.
040 * @see org.apache.felix.ipojo.annotations.HandlerDeclaration
041 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
042 */
043public class HandlerDeclarationVisitor extends EmptyVisitor implements AnnotationVisitor {
044
045    /**
046     * XML accepted by the handler.
047     */
048    private String m_value;
049
050    private DocumentBuilder builder;
051
052    private ComponentWorkbench workbench;
053    private Reporter reporter;
054
055    public HandlerDeclarationVisitor(ComponentWorkbench workbench, DocumentBuilder builder, Reporter reporter) {
056        this.workbench = workbench;
057        this.builder = builder;
058        this.reporter = reporter;
059    }
060
061    /**
062     * Parses the value attribute.
063     * @param name 'value'
064     * @param value the value
065     * @see org.objectweb.asm.commons.EmptyVisitor#visit(java.lang.String, java.lang.Object)
066     */
067    public void visit(String name, Object value) {
068        // there is only a 'value' attribute
069        this.m_value = (String) value;
070    }
071
072    /**
073     * End of the visit.
074     * Builds the XML document.
075     * @see org.objectweb.asm.commons.EmptyVisitor#visitEnd()
076     */
077    public void visitEnd() {
078        // The value is an XML document
079        InputStream is = new ByteArrayInputStream(m_value.getBytes());
080        Document document = null;
081        try {
082            document = builder.parse(is);
083            Element e = convertDOMElements(document.getDocumentElement());
084            workbench.getElements().put(e, null);
085        } catch (Exception e) {
086            reporter.warn("Cannot convert {} to iPOJO Elements.", m_value, e);
087        } finally {
088            try {
089                is.close();
090            } catch (IOException e) {
091                reporter.trace("Cannot close correctly the value input stream ({}).", m_value, e);
092            }
093        }
094    }
095
096    /**
097     * Converts recursively the given XML Element into an iPOJO Element.
098     * @param xmlElement DOM Element to be converted
099     */
100    private static Element convertDOMElements(final org.w3c.dom.Element xmlElement) {
101
102        // Create an equivalent iPOJO element
103        Element converted = transformElement(xmlElement);
104
105        convertDOMElements(converted, xmlElement);
106
107        return converted;
108    }
109
110    /**
111     * Converts recursively the given XML Element into an iPOJO Element.
112     * @param root iPOJO root Element
113     * @param xmlElement DOM Element to be converted
114     */
115    private static void convertDOMElements(final Element root,
116                                           final org.w3c.dom.Element xmlElement) {
117
118        // Convert attributes if any
119        if (xmlElement.hasAttributes()) {
120            NamedNodeMap attributes = xmlElement.getAttributes();
121            for (int i = 0; i < attributes.getLength(); i++) {
122                Attr attr = (Attr) attributes.item(i);
123                if (!"xmlns".equals(attr.getPrefix())) {
124                    root.addAttribute(transformAttribute(attr));
125                }
126            }
127        }
128
129        // Convert child elements if any
130        if (xmlElement.hasChildNodes()) {
131            NodeList children = xmlElement.getChildNodes();
132            for (int i = 0; i < children.getLength(); i++) {
133
134                // Create an equivalent iPOJO element
135                org.w3c.dom.Element child = (org.w3c.dom.Element) children.item(i);
136                Element converted = transformElement(child);
137
138                // Add converted element as a root's child
139                root.addElement(converted);
140
141                // Recursive call
142                convertDOMElements(converted, child);
143            }
144        }
145
146    }
147
148    private static Attribute transformAttribute(Attr attr) {
149        return new Attribute(attr.getLocalName(),
150                attr.getNamespaceURI(),
151                attr.getValue());
152    }
153
154    private static Element transformElement(org.w3c.dom.Element xmlElement) {
155        return new Element(xmlElement.getLocalName(), xmlElement.getNamespaceURI());
156    }
157
158}
