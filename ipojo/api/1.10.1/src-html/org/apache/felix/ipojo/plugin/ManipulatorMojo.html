019package org.apache.felix.ipojo.plugin;
020
021import java.io.ByteArrayInputStream;
022import java.io.File;
023import java.io.InputStream;
024import java.util.Arrays;
025import java.util.List;
026
027import org.apache.felix.ipojo.manipulator.Pojoization;
028import org.apache.felix.ipojo.manipulator.Reporter;
029import org.apache.maven.artifact.Artifact;
030import org.apache.maven.plugin.AbstractMojo;
031import org.apache.maven.plugin.MojoExecutionException;
032import org.apache.maven.project.MavenProject;
033import org.apache.maven.project.MavenProjectHelper;
034
035/**
036 * Packages an OSGi jar "bundle" as an "iPOJO bundle".
037 *
038 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
039 * @version $Rev$, $Date$
040 * @goal ipojo-bundle
041 * @phase package
042 * @requiresDependencyResolution runtime
043 * @description manipulate an OSGi bundle jar to build an iPOJO bundle
044 * @threadSafe
045 */
046public class ManipulatorMojo extends AbstractMojo {
047
048    /**
049     * The directory for the generated JAR.
050     *
051     * @parameter expression="${project.build.directory}"
052     * @required
053     */
054    private String m_buildDirectory;
055
056    /**
057     * The directory containing generated classes.
058     *
059     * @parameter expression="${project.build.outputDirectory}"
060     * @required
061     * @readonly
062     */
063    private File m_outputDirectory;
064
065    /**
066     * Location of the metadata file or iPOJO metadata configuration.
067     * @parameter alias="metadata"
068     */
069    private String m_metadata;
070
071    /**
072     * If set, the manipulated jar will be attached to the project as a separate artifact.
073     *
074     * @parameter alias="classifier" expression="${ipojo.classifier}"
075     */
076    private String m_classifier;
077
078    /**
079     * If set, select the manipulated artifact using this classifier.
080     *
081     * @parameter alias="input-classifier"
082     */
083    private String m_inputClassifier;
084
085    /**
086     * The Maven project.
087     *
088     * @parameter expression="${project}"
089     * @required
090     * @readonly
091     */
092    private MavenProject m_project;
093
094    /**
095     * Used for attaching new artifacts.
096     * @component
097     * @required
098     */
099    private MavenProjectHelper m_helper;
100
101    /**
102     * Project types which this plugin supports.
103     * @parameter
104     */
105    private List<String> m_supportedProjectTypes = Arrays.asList(new String[]{"bundle", "jar", "war"});
106
107    /**
108     * Ignore annotations parameter.
109     * @parameter alias="ignoreAnnotations" default-value="false"
110     */
111    private boolean m_ignoreAnnotations;
112
113    /**
114     * Ignore embedded XSD parameter.
115     * @parameter alias="IgnoreEmbeddedSchemas" default-value="false"
116     */
117    private boolean m_ignoreEmbeddedXSD;
118
119    private boolean isXML() {
120        return m_metadata != null && (m_metadata.indexOf('<') > -1);
121    }
122
123    /**
124     * Execute method : this method launches the pojoization.
125     * @throws MojoExecutionException : an exception occurs during the manipulation.
126     * @see org.apache.maven.plugin.AbstractMojo#execute()
127     */
128    public void execute() throws MojoExecutionException {
129        // ignore project types not supported, useful when the plugin is configured in the parent pom
130        if (!this.m_supportedProjectTypes.contains(m_project.getArtifact().getType())) {
131            this.getLog().debug("Ignoring project "
132                    + m_project.getArtifact() + " : type " + m_project.getArtifact().getType()
133                    + " is not supported by iPOJO plugin, supported types are " + this.m_supportedProjectTypes);
134            return;
135        }
136
137        initializeSaxDriver();
138
139        getLog().info("Start bundle manipulation");
140
141        // Get metadata
142        // Check if metadata are contained in the configuration
143        File metadata = null; // Metadata File or directory containing the metadata files.
144        InputStream is = null; //Use if contained in the configuration
145
146        if (isXML()) {
147            is = new ByteArrayInputStream(m_metadata.getBytes());
148        } else {
149            // If the metadata is not set,
150            // first check if ./src/main/ipojo exists, if so look into it.
151            if (m_metadata == null) {
152                File m = new File(m_project.getBasedir(), "src/main/ipojo");
153                if (m.isDirectory()) {
154                    metadata = m;
155                    getLog().info("Metadata directory : " + metadata.getAbsolutePath());
156                } else {
157                    // Else check target/classes/metadata.xml
158                    File meta = new File(m_outputDirectory + File.separator + "metadata.xml");
159                    if (! meta.exists()) {
160                        // If it still does not exist, try ./metadata.xml
161                        meta = new File(m_project.getBasedir() + File.separator + "metadata.xml");
162                    }
163
164                    if (meta.exists()) {
165                        metadata = meta;
166                        getLog().info("Metadata file : " + metadata.getAbsolutePath());
167                    }
168
169                    // No metadata.
170                }
171            } else {
172                // metadata path set.
173                File m = new File(m_project.getBasedir(), m_metadata);
174                if (! m.exists()) {
175                    throw new MojoExecutionException("The metadata file does not exist : " + m.getAbsolutePath());
176                }
177                metadata = m;
178                if (m.isDirectory()) {
179                    getLog().info("Metadata directory : " + metadata.getAbsolutePath());
180                } else {
181                    getLog().info("Metadata file : " + metadata.getAbsolutePath());
182                }
183            }
184
185            if (metadata == null) {
186                // Verify if annotations are ignored
187                if (m_ignoreAnnotations) {
188                    getLog().info("No metadata file found - ignoring annotations");
189                    return;
190                } else {
191                    getLog().info("No metadata file found - trying to use only annotations");
192                }
193            }
194        }
195
196        // Get input bundle, we use the already create artifact.
197        File in = null;
198        if (m_inputClassifier == null) {
199            in = m_project.getArtifact().getFile();
200            getLog().info("Input Bundle File : " + in.getAbsolutePath());
201            if (! in.exists()) {
202                throw new MojoExecutionException("The specified bundle file does not exist : " + in.getAbsolutePath());
203            }
204        } else {
205            // Look from attached artifacts.
206            @SuppressWarnings("unchecked")
207            List<Artifact> attached = m_project.getAttachedArtifacts();
208            for (int i = 0; in == null  && attached != null  && i < attached.size(); i++) {
209                Artifact artifact = attached.get(i);
210                if (artifact.hasClassifier()  && m_inputClassifier.equals(artifact.getClassifier())) {
211                    in = artifact.getFile();
212                }
213            }
214
215            if (in == null) {
216                throw new MojoExecutionException("Cannot find the file to manipulate, " +
217                        "no attached artifact with classifier " + m_inputClassifier);
218            }
219
220            getLog().info("Input Bundle File : " + in.getAbsolutePath());
221            if (! in.exists()) {
222                throw new MojoExecutionException("The specified bundle file does not exist : " + in.getAbsolutePath());
223            }
224        }
225
226        File out = new File(m_buildDirectory + File.separator + "_out.jar");
227
228        Reporter reporter = new MavenReporter(getLog());
229        Pojoization pojo = new Pojoization(reporter);
230        if (m_ignoreAnnotations) { pojo.disableAnnotationProcessing(); }
231        if (!m_ignoreEmbeddedXSD) { pojo.setUseLocalXSD(); }
232
233        // Executes the pojoization.
234        if (is == null) {
235            if (metadata == null) { // No metadata.
236                pojo.pojoization(in, out, (File) null); // Only annotations
237            } else {
238                pojo.pojoization(in, out, metadata); // Metadata set
239            }
240        } else  { // In-Pom metadata.
241            pojo.pojoization(in, out, is);
242        }
243
244        for (int i = 0; i < reporter.getWarnings().size(); i++) {
245            getLog().warn((String) reporter.getWarnings().get(i));
246        }
247        if (reporter.getErrors().size() > 0) {
248            throw new MojoExecutionException((String) reporter.getErrors().get(0));
249        }
250
251        if (m_classifier != null) {
252            // The user want to attach the resulting jar
253            // Do not delete in File
254            m_helper.attachArtifact(m_project, "jar", m_classifier, out);
255        } else {
256            // Usual behavior
257            if (in.delete()) {
258                if (! out.renameTo(in)) {
259                    getLog().warn("Cannot rename the manipulated jar file");
260                }
261            } else {
262                getLog().warn("Cannot delete the input jar file");
263            }
264        }
265        getLog().info("Bundle manipulation - SUCCESS");
266    }
267
268    /**
269     * If Maven runs with Java 1.4, we should use the Maven Xerces.
270     * To achieve that, we set the org.xml.sax.driver property.
271     * Otherwise, the JVM sets the org.xml.sax.driver property.
272     */
273    private void initializeSaxDriver() {
274        String version = (String) System.getProperty("java.vm.version");
275        if (version.startsWith("1.4")) {
276            getLog().info("Set the Sax driver to org.apache.xerces.parsers.SAXParser");
277            System.setProperty("org.xml.sax.driver", "org.apache.xerces.parsers.SAXParser");
278        }
279
280    }
