040/**
041 * Composite Instance Handler.
042 * This handler allows creating an instance inside a composite.
043 * This instance is determine by its type and a configuration.
044 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
045 */
046public class InstanceHandler extends CompositeHandler implements InstanceStateListener {

048    /**
049     * Internal context.
050     */
051    private ServiceContext m_scope;

053    /**
054     * Available factories.

059     * Handler description.

064    /**
065     * This structure aims to manage a configuration. It stores all necessary
066     * information to create an instance and to track the factory.
067     */
068    class ManagedConfiguration {
069        /**
070         * Configuration of the instance to create.
071         */
072        private Dictionary m_configuration;
073
074        /**
075         * Factory name.
076         */
077        private String m_factoryName;
078
079        /**
080         * Created instance.
081         */
082        private ComponentInstance m_instance;
083
084        /**
085         * Desired Factory (can be the classname).
086         */
087        private String m_desiredFactory;
088
089        /**
090         * Constructor.
091         *
092         * @param conf : the configuration to create.
093         */
094        ManagedConfiguration(Dictionary conf) {
095            m_configuration = conf;
096            m_desiredFactory = (String) conf.get("component");
097        }
098
099        /**
100         * Return the managed configuration.
101         * @return the configuration.
102         */
103        protected Dictionary getConfiguration() {
104            return m_configuration;
105        }

107        /**
108         * Return the used factory name.
109         * @return the factory name
110         */
111        protected String getFactory() {
112            return m_factoryName;
113        }
114
115        protected String getNeededFactoryName() {
116            return m_desiredFactory;
117        }
118
119        /**
120         * Return the created instance.
121         * @return the instance (or null if no instance are created).
122         */
123        protected ComponentInstance getInstance() {
124            return m_instance;

127        /**
128         * Set the factory name.
129         *
130         * @param name : the factory name.

137         * Set the instance object.

143        }

146    /**
147     * Configurations to create and maintains.
148     */
149    private ManagedConfiguration[] m_configurations = new ManagedConfiguration[0];
150
151    /**
152     * Create an instance using the given factory and the given configuration.
153     *
154     * @param fact : the factory name to used.
155     * @param config : the configuration.
156     */
157    private void createInstance(Factory fact, ManagedConfiguration config) {
158        Dictionary conf = config.getConfiguration();
159        try {
160            config.setInstance(fact.createComponentInstance(conf, m_scope));
161            config.setFactory(fact.getName());
162            config.getInstance().addInstanceStateListener(this);
163        } catch (UnacceptableConfiguration e) {
164            error("A factory is available for the configuration but the configuration is not acceptable", e);
165        } catch (MissingHandlerException e) {
166            error("The instance creation has failed, at least one handler is missing", e);
167        } catch (ConfigurationException e) {
168            error("The instance creation has failed, an error during the configuration has occured", e);
169        }
170    }
171
172    /**
173     * A new valid factory appears.
174     * @param factory : factory.
175     */
176    public void bindFactory(Factory factory) {
177        boolean implicated = false;
178        String factName = factory.getName();
179        String className = factory.getComponentDescription().getClassName();
180        for (int i = 0; i < m_configurations.length; i++) {
181            if (m_configurations[i].getInstance() == null
182                    && (m_configurations[i].getNeededFactoryName().equals(factName)
183                            || m_configurations[i].getNeededFactoryName().equals(className))) {
184                createInstance(factory, m_configurations[i]);
185                implicated = true;
186            }
187        }
188        if (implicated && ! getValidity()) {
189            checkValidity();
190        }
191    }
192
193    /**
194     * An existing factory disappears or becomes invalid.
195     * @param factory : factory
196     */
197    public void unbindFactory(Factory factory) {
198        boolean implicated = false;
199        for (int i = 0; i < m_configurations.length; i++) {
200            if (m_configurations[i].getInstance() != null
201                    && m_configurations[i].getFactory().equals(factory.getName())) {
202                m_configurations[i].setInstance(null);
203                m_configurations[i].setFactory(null);
204                implicated = true;
205            }
206        }
207        if (implicated && getValidity()) {
208            checkValidity();
209        }
210    }
211
212    /**
213     * Stop all created instances.
214     */
215    public synchronized void stop() {
216        for (int i = 0; i < m_configurations.length; i++) {
217            if (m_configurations[i].getInstance() != null) {
218                m_configurations[i].getInstance().removeInstanceStateListener(this);
219                if (m_configurations[i].getInstance().getState() != ComponentInstance.DISPOSED) {
220                    m_configurations[i].getInstance().dispose();
221                }
222            }
223            m_configurations[i].setInstance(null);
224            m_configurations[i].setFactory(null);
225        }
226        m_configurations = new ManagedConfiguration[0];
227    }

229    /**
230     * Configure method.
231     * @param metadata : component type metadata.
232     * @param configuration : instance configuration.
233     * @throws ConfigurationException : occurs an instance cannot be parsed correctly.
234     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
235     */
236    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
237        m_scope = getCompositeManager().getServiceContext();
238
239        // Prepare the configuration to append.
240        Properties toAppend = new Properties();
241        Enumeration keys = configuration.keys();
242        while(keys.hasMoreElements()) {
243            String key = (String) keys.nextElement();
244            if (! (key.equals("instance.name")
245                    || key.equals("component"))) { // Remove instance.name and component
246                toAppend.put(key, configuration.get(key));
247            }
248        }
249
250        Element[] instances = metadata.getElements("instance");
251        m_configurations = new ManagedConfiguration[instances.length];
252        for (int i = 0; i < instances.length; i++) {
253            Properties conf = null;
254            try {
255                conf = parseInstance(instances[i]);
256            } catch (ParseException e) {
257                error("An instance cannot be parsed correctly", e);
258                throw new ConfigurationException("An instance cannot be parsed correctly", e);
259            }
260
261            Properties instanceConfiguration = new Properties();
262            instanceConfiguration.putAll(conf);
263            instanceConfiguration.putAll(toAppend);
264            m_configurations[i] = new ManagedConfiguration(instanceConfiguration);
265        }
266
267        m_description = new InstanceHandlerDescription(this, m_configurations);
268    }

270    /**
271     * Parse an Element to get a dictionary.
272     * @param instance : the Element describing an instance.
273     * @return : the resulting dictionary
274     * @throws ParseException : occurs when a configuration cannot be parse correctly.
275     */
276    public static Properties parseInstance(Element instance) throws ParseException {
277        Properties dict = new Properties();
278        String name = instance.getAttribute("name");
279        if (name != null) {
280            dict.put("instance.name", name);
281        }
282
283        String comp = instance.getAttribute("component");
284        if (comp == null) {
285            throw new ParseException("An instance does not have the 'component' attribute");
286        } else {
287            dict.put("component", comp);
288        }
289
290        Element[] props = instance.getElements("property");
291        for (int i = 0; props != null && i < props.length; i++) {
292            parseProperty(props[i], dict);
293        }
294
295        return dict;
296    }
297
298    /**
299     * Parse a property.
300     * @param prop : the current element to parse
301     * @param dict : the dictionary to populate
302     * @throws ParseException : occurs if the property cannot be parsed correctly
303     */
304    public static void parseProperty(Element prop, Dictionary dict) throws ParseException {
305        // Check that the property has a name
306        String name = prop.getAttribute("name");
307        String value = prop.getAttribute("value");
308        if (name == null) { throw new ParseException("A property does not have the 'name' attribute"); }
309        // Final case : the property element has a 'value' attribute
310        if (value == null) {
311            // Recursive case
312            // Check if there is 'property' element
313            Element[] subProps = prop.getElements("property");
314            if (subProps == null) { throw new ParseException("A complex property must have at least one 'property' sub-element"); }
315            Dictionary dict2 = new Properties();
316            for (int i = 0; i < subProps.length; i++) {
317                parseProperty(subProps[i], dict2);
318                dict.put(prop.getAttribute("name"), dict2);
319            }
320        } else {
321            dict.put(name, value);
322        }
323    }

325    /**
326     * Start method.
327     * @see org.apache.felix.ipojo.Handler#start()
328     */
329    public void start() {
330        for (int j = 0; j < m_factories.length; j++) {
331            String factName = m_factories[j].getName();
332            String className = m_factories[j].getClassName();
333            for (int i = 0; i < m_configurations.length; i++) {
334                if (m_configurations[i].getInstance() == null && (m_configurations[i].getNeededFactoryName().equals(factName) || m_configurations[i].getNeededFactoryName().equals(className))) {
335                    createInstance(m_factories[j], m_configurations[i]);
336                }
337            }
338        }
339        checkValidity();
340    }
341
342    /**
343     * Check handler validity.
344     * The method updates the validity of the handler.
345     */
346    private void checkValidity() {
347        for (int i = 0; i < m_configurations.length; i++) {
348            if (m_configurations[i].getInstance() == null || m_configurations[i].getInstance().getState() != ComponentInstance.VALID) {
349                setValidity(false);
350                return;
351            }
352        }
353        setValidity(true);
354    }

356    /**
357     *  Instance state listener.
358     *  This method listens when managed instance states change.
359     *  @param instance : instance
360     *  @param newState : the now state of the given instance
361     *  @see org.apache.felix.ipojo.InstanceStateListener#stateChanged(org.apache.felix.ipojo.ComponentInstance, int)
362     */
363    public void stateChanged(ComponentInstance instance, int newState) {
364        switch (newState) {
365            case ComponentInstance.DISPOSED:
366                break;  // Should not happen
367            case ComponentInstance.STOPPED:
368                break;  // Should not happen
369            case ComponentInstance.VALID:
370                if (!getValidity()) {
371                    checkValidity();
372                }
373                break;
374            case ComponentInstance.INVALID:
375                if (getValidity()) {
376                    checkValidity();
377                }
378                break;
379            default:
380                break;
381
382        }
383    }

385    /**
386     * Method returning an instance object of the given component type.
387     * This method must be called only on 'primitive' type.
388     * @param type : type.
389     * @return an instance object or null if not found.
390     */
391    public Object getObjectFromInstance(String type) {
392        for (int i = 0; i < m_configurations.length; i++) {
393            if (m_configurations[i].getInstance() != null && type.equals(m_configurations[i].getFactory())) {
394                if (m_configurations[i].getInstance().getState() == ComponentInstance.VALID) {
395                    return ((InstanceManager) m_configurations[i].getInstance()).getPojoObject();
396                } else {
397                    error("An object cannot be get from the instance of the type " + type + ": invalid instance" +
398                            m_configurations[i].getInstance().getInstanceDescription().getDescription());
399                    return null;
400                }
401            }
402        }
403        return null;
404    }

415    /**
416     * Get the list of used component type.
417     * @return the list containing the used component type
418     */
419    public List getUsedType() {
420        List result = new ArrayList();
421        for (int i = 0; i < m_configurations.length; i++) {
422            result.add(m_configurations[i].getConfiguration().get("component"));
423        }
424        return result;
425    }

427    /**
428     * The composite is reconfigured, we check if we have a property to change.
429     * We reconfigure all contained instances.
430     * @param configuration the new instance configuration
431     */
432    public void reconfigure(Dictionary configuration) {
433        for (int i = 0; i < m_configurations.length; i++) {
434            if (m_configurations[i].getInstance() != null) {
435                info("Reconfiguring " + m_configurations[i].getInstance().getInstanceName());
436                m_configurations[i].getInstance().reconfigure(configuration);
437            }
438        }
