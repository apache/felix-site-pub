019package org.apache.felix.ipojo.context;
020
021import java.util.Dictionary;
022
023import org.apache.felix.ipojo.ComponentInstance;
024import org.osgi.framework.Bundle;
025import org.osgi.framework.Constants;
026import org.osgi.framework.ServiceReference;
027
028/**
029 * Internal service reference implementation. This class is used for in the
030 * composition.
031 * 
032 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
033 */
034public class ServiceReferenceImpl implements ServiceReference {
035
036    /**
037     * Service Registration attached to the service reference.
038     */
039    private ServiceRegistrationImpl m_registration = null;
040    
041    /**
042     * Component Instance.
043     */
044    private ComponentInstance m_cm;
045
046    /**
047     * Constructor.
048     * 
049     * @param instance : component instance publishing the service.
050     * @param ref : registration attached to this service reference.
051     */
052    public ServiceReferenceImpl(ComponentInstance instance, ServiceRegistrationImpl ref) {
053        m_registration = ref;
054        m_cm = instance;
055    }
056
057    /**
058     * Not supported in composite.
059     * @return null
060     * @see org.osgi.framework.ServiceReference#getBundle()
061     */
062    public Bundle getBundle() {
063        return m_cm.getContext().getBundle();
064    }
065
066    /**
067     * Get the service registration for this reference.
068     * @return the service registration for this service reference.
069     */
070    public ServiceRegistrationImpl getServiceRegistration() {
071        return m_registration;
072    }
073
074
075    /**
076     * Get a property value.
077     * @param name : the key of the required property.
078     * @return the property value or null if no property for the given key.
079     * @see org.osgi.framework.ServiceReference#getProperty(java.lang.String)
080     */
081    public Object getProperty(String name) {
082        return m_registration.getProperty(name);
083    }
084
085    /**
086     * Get the String arrays of service property keys.
087     * @return : the list of property keys.
088     * @see org.osgi.framework.ServiceReference#getPropertyKeys()
089     */
090    public String[] getPropertyKeys() {
091        return m_registration.getPropertyKeys();
092    }
093    
094    public Dictionary getProperties() {
095        return m_registration.getProperties();
096    }
097
098
099    /**
100     * Unsupported Operation inside composite.
101     * @return bundles using this reference.
102     * @see org.osgi.framework.ServiceReference#getUsingBundles()
103     */
104    public Bundle[] getUsingBundles() {
105        throw new UnsupportedOperationException("getUsingBundles is not supported in service context");
106    }
107
108    /**
109     * Check if the current service reference is assignable to the given bundle.
110     * @param arg0 : the bundle to check
111     * @param arg1 : the class name to check.
112     * @return true in the case of composite
113     * @see org.osgi.framework.ServiceReference#isAssignableTo(org.osgi.framework.Bundle, java.lang.String)
114     */
115    public boolean isAssignableTo(Bundle arg0, String arg1) {
116        return true;
117    }
118
119    /**
120     * Service Reference compare method.
121     * @param reference the service reference
122     * @return this methods is not yet supported, and throws an
123     *         {@link UnsupportedOperationException}.
124     * @see org.osgi.framework.ServiceReference#compareTo(java.lang.Object)
125     */
126    public int compareTo(Object reference) {
127
128        ServiceReference other = (ServiceReference) reference;
129
130        Long id = (Long) getProperty(Constants.SERVICE_ID);
131        Long otherId = (Long) other.getProperty(Constants.SERVICE_ID);
132
133        if (id.equals(otherId)) {
134            return 0; // same service
135        }
136
137        Integer rank = (Integer) getProperty(Constants.SERVICE_RANKING);
138        Integer otherRank = (Integer) other
139                .getProperty(Constants.SERVICE_RANKING);
140
141        // If no rank, then spec says it defaults to zero.
142        rank = (rank == null) ? new Integer(0) : rank;
143        otherRank = (otherRank == null) ? new Integer(0) : otherRank;
144
145        // Sort by rank in ascending order.
146        if (rank.compareTo(otherRank) < 0) {
147            return -1; // lower rank
148        } else if (rank.compareTo(otherRank) > 0) {
149            return 1; // higher rank
150        }
151
152        // If ranks are equal, then sort by service id in descending order.
153        return (id.compareTo(otherId) < 0) ? 1 : -1;
154    }
155
156}
