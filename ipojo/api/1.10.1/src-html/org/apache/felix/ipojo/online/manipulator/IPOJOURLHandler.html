019package org.apache.felix.ipojo.online.manipulator;
020
021import org.apache.felix.ipojo.manipulator.Pojoization;
022import org.osgi.framework.BundleContext;
023import org.osgi.service.url.URLStreamHandlerService;
024
025import java.io.File;
026import java.io.FileOutputStream;
027import java.io.IOException;
028import java.io.InputStream;
029import java.net.MalformedURLException;
030import java.net.URL;
031import java.net.URLConnection;
032import java.util.jar.JarEntry;
033import java.util.jar.JarFile;
034
035/**
036 * iPOJO URL Handler allowing installation time manipulation.
037 * When a bundle is installed with the <code>ipojo:</code> URL
038 * prefix, the bundle is downloaded and manipulated by this
039 * handler.
040 * The metadata.xml file can either be provided inside the bundle (root,
041 * or in META-INF) or given in the URL:
042 * ipojo:URL_BUNDLE!URL_METADATA.
043 *
044 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
045 */
046public class IPOJOURLHandler
047        extends org.osgi.service.url.AbstractURLStreamHandlerService implements URLStreamHandlerService {
048
049    /**
050     * The bundle context.
051     */
052    private BundleContext m_context;
053    /**
054     * The directory storing bundles.
055     */
056    private File m_temp;
057
058    /**
059     * Creates a IPOJOURLHandler.
060     * Gets the bundle context and create the working
061     * directory.
062     *
063     * @param bc the bundle context
064     */
065    public IPOJOURLHandler(BundleContext bc) {
066        m_context = bc;
067        m_temp = m_context.getDataFile("temp");
068        if (!m_temp.exists()) {
069            m_temp.mkdir();
070        }
071    }
072
073    /**
074     * Stops the URL handler:
075     * Deletes the working directory.
076     */
077    public void stop() {
078        File[] files = m_temp.listFiles();
079        if (files != null) {
080            for (int i = 0; i < files.length; i++) {
081                files[i].delete();
082            }
083        }
084        m_temp.delete();
085    }
086
087    /**
088     * Opens a connection using the ipojo url handler.
089     * This methods parses the URL and manipulate the given bundle.
090     *
091     * @param url the url.
092     * @return the URL connection on the manipulated bundle
093     * @throws java.io.IOException occurs when the bundle cannot be either downloaded, or manipulated or
094     *                             installed correctly.
095     * @see org.osgi.service.url.AbstractURLStreamHandlerService#openConnection(java.net.URL)
096     */
097    public URLConnection openConnection(URL url) throws IOException {
098        System.out.println("Processing URL : " + url);
099
100        // Parse the url:
101        String full = url.toExternalForm();
102        // Remote ipojo://
103        if (full.startsWith("ipojo:")) {
104            full = full.substring(6);
105        }
106        // Remove '/' or '//'
107        while (full.startsWith("/")) {
108            full = full.substring(1);
109        }
110
111        full = full.trim();
112
113        // Now full is like : URL,URL or URL
114        String[] urls = full.split("!");
115        URL bundleURL = null;
116        URL metadataURL = null;
117        if (urls.length == 1) {
118            // URL form
119            System.out.println("Extracted URL : " + urls[0]);
120            bundleURL = new URL(urls[0]);
121        } else if (urls.length == 2) {
122            // URL,URL form
123            bundleURL = new URL(urls[0]);
124            metadataURL = new URL(urls[1]);
125        } else {
126            throw new MalformedURLException("The iPOJO url is not formatted correctly, ipojo:bundle_url[!metadata_url] expected");
127        }
128
129        File bundle = File.createTempFile("ipojo_", ".jar", m_temp);
130        save(bundleURL, bundle);
131        File metadata = null;
132        if (metadataURL != null) {
133            metadata = File.createTempFile("ipojo_", ".xml", m_temp);
134            save(metadataURL, metadata);
135        } else {
136            // Check that the metadata are in the jar file
137            JarFile jar = new JarFile(bundle);
138            metadata = findMetadata(jar);
139        }
140
141        // Pojoization
142        Pojoization pojoizator = new Pojoization();
143        File out = new File(m_temp, bundle.getName() + "-ipojo.jar");
144        System.out.println("Pojoization " + bundle.exists() + " - " + metadata.exists());
145        try {
146            pojoizator.pojoization(bundle, out, metadata);
147        } catch (Exception e) {
148            if (!pojoizator.getErrors().isEmpty()) {
149                throw new IOException("Errors occured during the manipulation : " + pojoizator.getErrors());
150            }
151            e.printStackTrace();
152            throw new RuntimeException(e.getMessage());
153        }
154
155        if (!pojoizator.getErrors().isEmpty()) {
156            throw new IOException("Errors occured during the manipulation : " + pojoizator.getErrors());
157        }
158        if (!pojoizator.getWarnings().isEmpty()) {
159            System.err.println("Warnings occured during the manipulation : " + pojoizator.getWarnings());
160        }
161
162        System.out.println("Manipulation done : " + out.exists());
163
164        // Cleanup
165        bundle.delete();
166        if (metadata != null) {
167            metadata.delete();
168        }
169        out.deleteOnExit();
170        // Returns the URL Connection
171        return out.toURI().toURL().openConnection();
172
173
174    }
175
176    /**
177     * Downloads the content pointed by the given url to
178     * the given file.
179     *
180     * @param url  the url
181     * @param file the file
182     * @throws java.io.IOException occurs if the content cannot be read
183     *                             and save inside the file
184     */
185    private void save(URL url, File file) throws IOException {
186        InputStream is = url.openStream();
187        save(is, file);
188    }
189
190    /**
191     * Saves the content of the input stream to the given file.
192     *
193     * @param is   the input stream to read
194     * @param file the file
195     * @throws java.io.IOException occurs if the content cannot be read
196     *                             and save inside the file
197     */
198    private void save(InputStream is, File file) throws IOException {
199        FileOutputStream writer = new FileOutputStream(file);
200        int cc = 0;
201        do {
202            int i = is.read();
203            if (i == -1) {
204                break;
205            }
206            cc++;
207            writer.write(i);
208        } while (true);
209        System.out.println(cc + " bytes copied");
210        is.close();
211        writer.close();
212    }
213
214    /**
215     * Looks for the metadata.xml file in the jar file.
216     * Two locations are checked:
217     * <ol>
218     * <li>the root of the jar file</li>
219     * <li>the META-INF directory</li>
220     * </ol>
221     *
222     * @param jar the jar file
223     * @return the founded file or <code>null</code> if not found.
224     * @throws java.io.IOException occurs when the Jar file cannot be read.
225     */
226    private File findMetadata(JarFile jar) throws IOException {
227        JarEntry je = jar.getJarEntry("metadata.xml");
228        if (je == null) {
229            je = jar.getJarEntry("META-INF/metadata.xml");
230        }
231
232        if (je == null) {
233            System.out.println("Metadata file not found, use annotations only.");
234            return null; // Not Found, use annotation only
235        } else {
236            System.out.println("Metadata file found: " + je.getName());
237            File metadata = File.createTempFile("ipojo_", ".xml", m_temp);
238            save(jar.getInputStream(je), metadata);
239            System.out.println("Metadata file saved to " + metadata.getAbsolutePath());
240            return metadata;
241        }
242
243    }
