020package org.apache.felix.ipojo.dependency.impl;
021
022import org.apache.felix.ipojo.context.ServiceReferenceImpl;
023import org.osgi.framework.Constants;
024import org.osgi.framework.Filter;
025import org.osgi.framework.ServiceReference;
026
027import java.util.List;
028
029/**
030 * Some utility methods to handle service references.
031 */
032public class ServiceReferenceUtils {
033    /**
034     * Checks if the given service reference match the current filter.
035     * This method aims to avoid calling {@link org.osgi.framework.Filter#match(org.osgi.framework.ServiceReference)}
036     * method when manipulating a composite reference. In fact, this method thrown
037     * a {@link ClassCastException} on Equinox.
038     *
039     * @param ref the service reference to check.
040     * @return <code>true</code> if the service reference matches.
041     */
042    public static boolean match(Filter filter, ServiceReference ref) {
043        boolean match = true;
044        if (filter != null) {
045            if (ref instanceof ServiceReferenceImpl) {
046                // Can't use the match(ref) as it throw a class cast exception on Equinox.
047                //noinspection unchecked
048                match = filter.match(((ServiceReferenceImpl) ref).getProperties());
049            } else { // Non composite reference.
050                match = filter.match(ref);
051            }
052        }
053        return match;
054    }
055
056    /**
057     * Checks whether a list of service references contains a reference with the same {@literal service.id} as the
058     * given reference.
059     * @param references the list of reference
060     * @param ref the reference
061     * @return {@literal true} if references contains a reference with the same service.id as ref.
062     */
063    public static boolean containsReferenceById(List<? extends ServiceReference> references, ServiceReference ref) {
064        return getServiceReferenceById(references, ref) != null;
065    }
066
067    /**
068     * Gets a service reference with the same service.id as the given reference from the given list.
069     * @param references the list of references
070     * @param ref the reference
071     * @return the service reference from references having the same service.id as ref. {@literal null} if there is
072     * no such reference in the list.
073     */
074    public static ServiceReference getServiceReferenceById(List<? extends ServiceReference> references,
075                                                           ServiceReference ref) {
076        Object id = ref.getProperty(Constants.SERVICE_ID);
077        for (ServiceReference reference : references) {
078            if (reference.getProperty(Constants.SERVICE_ID).equals(id)) {
079                return reference;
080            }
081        }
082        return null;
083    }
084
085    /**
086     * Checks whether the two references has the same properties and their value are equals.
087     * @param ref1 first reference
088     * @param ref2 second reference
089     * @return {@literal true} if the two references have the same properties and their values are equals.
090     */
091    public static boolean haveSameProperties(ServiceReference ref1, ServiceReference ref2) {
092        if (ref2 == null  && ref1 == null) {
093            return true;
094        }
095
096        if ((ref1 == null) || (ref2 == null)) {
097            return false;
098        }
099
100        String[] keys = ref2.getPropertyKeys();
101
102        if (ref2.getPropertyKeys().length != keys.length) {
103            return false;
104        }
105
106        for (String key : keys) {
107            if (! ref2.getProperty(key).equals(ref1.getProperty(key))) {
108                return false;
109            }
110        }
111
112        return true;
113
114
115    }
116
117    /**
118     * Checks whether two service references have the same service id.
119     * @param ref1 first reference
120     * @param ref2 second reference
121     * @return {@literal true} if the two references have the same service.id, {@literal false} otherwise.
122     */
123    public static boolean haveSameServiceId(ServiceReference ref1, ServiceReference ref2) {
124        return !(ref1 == null || ref2 == null)
125                && ref1.getProperty(Constants.SERVICE_ID).equals(ref2.getProperty(Constants.SERVICE_ID));
126    }
127}
