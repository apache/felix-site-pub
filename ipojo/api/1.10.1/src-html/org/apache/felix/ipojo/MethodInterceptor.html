019package org.apache.felix.ipojo;
020
021import java.lang.reflect.Constructor;
022import java.lang.reflect.Member;
023import java.lang.reflect.Method;
024
025/**
026* Method interceptor.
027* A class implementing this interface is able to be notified of method invocations (
028* i.e. entries, exits, and errors).
029* The listener needs to be register on the instance manager with the
030* {@link InstanceManager#register(org.apache.felix.ipojo.parser.MethodMetadata, MethodInterceptor)}
031* method.
032* Events are sent before the method entry (onEntry), after the method returns (onExit),
033* when an error is thrown by the method (onError), and before the after either a returns or an error (onFinally).
034*
035* Instead of a {@link Method} object, the callbacks received a {@link Member} object which can be either a {@link Method}
036* or a {@link Constructor}.
037* @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
038*/
039public interface MethodInterceptor {
040
041    /**
042     * This method is called when a thread enters in a method.
043     * The given argument array is created from the method argument.
044     * @param pojo the pojo on which the method is called.
045     * @param method the invoked method.
046     * @param args the arguments array.
047     */
048    void onEntry(Object pojo, Member method, Object[] args);
049
050    /**
051     * This method is called when the execution exits a method :
052     * before a <code>return</code>.
053     * If the given returned object is <code>null</code>, either the method is
054     * <code>void</code>, or it returns <code>null</code>.
055     * This method must not modify the returned object.
056     * @param pojo the pojo on which the method exits.
057     * @param method the exiting method.
058     * @param returnedObj the the returned object (boxed for primitive type)
059     */
060    void onExit(Object pojo, Member method, Object returnedObj);
061
062    /**
063     * This method is called when the execution throws an exception in the given
064     * method.
065     * @param pojo the pojo on which the method was accessed.
066     * @param method the invoked method.
067     * @param throwable the thrown exception
068     */
069    void onError(Object pojo, Member method, Throwable throwable);
070
071    /**
072     * This method is called when the execution of a method is going to terminate :
073     * just before to throw an exception or before to return.
074     * (onError or onExit was already called).
075     * @param pojo the pojo on which the method was accessed.
076     * @param method the invoked method.
077     */
078    void onFinally(Object pojo, Member method);
079
080}
