019package org.apache.felix.ipojo.handlers.dependency;
020
021
022import java.lang.reflect.Method;
023import java.lang.reflect.Modifier;
024
025import org.objectweb.asm.ClassWriter;
026import org.objectweb.asm.Label;
027import org.objectweb.asm.MethodVisitor;
028import org.objectweb.asm.Opcodes;
029import org.objectweb.asm.Type;
030
031/**
032 * Generates proxy class delegating operation invocations thanks to a
033 * a dependency.
034 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
035 */
036public class ProxyGenerator implements Opcodes {
037
038    /**
039     * The dependency name.
040     */
041    private static final String DEPENDENCY = "m_dependency";
042
043    /**
044     * The dependency descriptor.
045     */
046    private static final String DEPENDENCY_DESC = Type.getDescriptor(Dependency.class);
047
048    /**
049     * Dependency internal class name.
050     */
051    private static final String DEPENDENCY_INTERNAL_NAME = "org/apache/felix/ipojo/handlers/dependency/Dependency";
052
053    /**
054     * Gets the internal names of the given class objects.
055     * @param classes the classes
056     * @return the array containing internal names of the given class array.
057     */
058    private static String[] getInternalClassNames(Class[] classes) {
059        final String[] names = new String[classes.length];
060        for (int i = 0; i < names.length; i++) {
061            names[i] = Type.getInternalName(classes[i]);
062        }
063        return names;
064    }
065
066    /**
067     * Generates a proxy class.
068     * @param spec the proxied service specification
069     * @return the byte[] for the generated proxy class.
070     */
071    public static byte[] dumpProxy(Class spec) {
072        ClassWriter cw = new ClassWriter(ClassWriter.COMPUTE_MAXS);
073        String internalClassName = Type.getInternalName(spec); // Specification class internal name.
074        String[] itfs = new String[0];
075        String parent = "java/lang/Object";
076        if (spec.isInterface()) {
077                itfs = new String[] {internalClassName};  // Implemented interface.
078        } else {
079                parent = internalClassName;
080        }
081        String className = internalClassName + "$$Proxy"; // Unique name.
082
083        // Turn around the VM changes (FELIX-2716) about java.* classes.
084        if (className.startsWith("java/")) {
085                className = "$" + className;
086        }
087
088        Method[] methods = spec.getMethods(); // Method to delegate
089
090        cw.visit(Opcodes.V1_3, Opcodes.ACC_PUBLIC | Opcodes.ACC_FINAL, className, null, parent, itfs);
091        addDependencyField(cw);
092
093        // We try to call super() on the parent, however this should not be used as proxing does work only for interface.
094        generateConstructor(cw, className, parent);
095
096        // For each method, create the delegator code.
097        for (int i = 0; i < methods.length; i++) {
098            if ((methods[i].getModifiers() & (Modifier.STATIC | Modifier.FINAL)) == 0) {
099                generateDelegator(cw, methods[i], className, internalClassName);
100            }
101        }
102
103        cw.visitEnd();
104
105        return cw.toByteArray();
106
107    }
108
109    /**
110     * Generates a delegated method.
111     * @param cw the class writer
112     * @param method the method object to delegate
113     * @param className the generated class name
114     * @param itfName the internal specification class name
115     */
116    private static void generateDelegator(ClassWriter cw, Method method,
117            String className, String itfName) {
118        String methodName = method.getName();
119        String desc = Type.getMethodDescriptor(method);
120        String[] exceptions = getInternalClassNames(method.getExceptionTypes());
121        int modifiers = method.getModifiers()
122                & ~(Modifier.ABSTRACT | Modifier.NATIVE | Modifier.SYNCHRONIZED);
123        Type[] types = Type.getArgumentTypes(method);
124
125        int freeRoom = 1;
126        for (int t = 0; t < types.length; t++) {
127            freeRoom = freeRoom + types[t].getSize();
128        }
129
130        MethodVisitor mv = cw.visitMethod(modifiers, methodName, desc, null,
131                exceptions);
132        mv.visitCode();
133
134        mv.visitVarInsn(ALOAD, 0);
135        mv.visitFieldInsn(GETFIELD, className, DEPENDENCY, DEPENDENCY_DESC);  // The dependency is on the stack.
136        mv.visitMethodInsn(INVOKEVIRTUAL, DEPENDENCY_INTERNAL_NAME, "getService", // Call getService
137                "()Ljava/lang/Object;"); // The service object is on the stack.
138        int varSvc = freeRoom;
139        freeRoom = freeRoom + 1; // Object Reference.
140        mv.visitVarInsn(ASTORE, varSvc); // Store the service object.
141
142        Label notNull = new Label();
143        Label isNull = new Label();
144        mv.visitVarInsn(ALOAD, varSvc); // Load the service
145        mv.visitJumpInsn(IFNONNULL, notNull); // If not null go to not null
146        // Null branch - throw the exception
147        mv.visitLabel(isNull);
148        mv.visitTypeInsn(NEW, "java/lang/RuntimeException");
149        mv.visitInsn(DUP);
150        mv.visitLdcInsn("No service available");
151        mv.visitMethodInsn(INVOKESPECIAL, "java/lang/RuntimeException", "<init>", "(Ljava/lang/String;)V");
152        mv.visitInsn(ATHROW);
153        // End of the null branch
154
155        // Not null, go one the execution
156        mv.visitLabel(notNull);
157
158        // Invoke the method on the service object.
159        mv.visitVarInsn(ALOAD, varSvc);
160        // Push argument on the stack.
161        int i = 1; // Arguments. (non static method)
162        for (int t = 0; t < types.length; t++) {
163            mv.visitVarInsn(types[t].getOpcode(ILOAD), i);
164            i = i + types[t].getSize();
165        }
166        // Invocation
167        mv.visitMethodInsn(INVOKEINTERFACE, itfName, methodName, desc);
168
169        // Return the result
170        Type returnType = Type.getReturnType(desc);
171        if (returnType.getSort() != Type.VOID) {
172            mv.visitInsn(returnType.getOpcode(IRETURN));
173        } else {
174            mv.visitInsn(RETURN);
175        }
176
177        // End of the method.
178        mv.visitMaxs(0, 0);
179        mv.visitEnd();
180    }
181
182    /**
183     * Generates the constructors. The constructor receives a dependency
184     * and set the {@link ProxyGenerator#DEPENDENCY} field.
185     * @param cw the class writer
186     * @param className the generated class name.
187     */
188    private static void generateConstructor(ClassWriter cw, String className, String parent) {
189        MethodVisitor mv = cw.visitMethod(Opcodes.ACC_PUBLIC, "<init>", '(' + DEPENDENCY_DESC + ")V", null, null);
190        mv.visitCode();
191
192        mv.visitVarInsn(ALOAD, 0); // Load this
193        mv.visitInsn(DUP); // Dup
194        mv.visitMethodInsn(INVOKESPECIAL, parent, "<init>", "()V"); // Call  super
195        mv.visitVarInsn(ALOAD, 1); // Load the argument
196        mv.visitFieldInsn(PUTFIELD, className, DEPENDENCY, DEPENDENCY_DESC); // Assign the dependency field
197        mv.visitInsn(RETURN); // Return void
198
199        mv.visitMaxs(0, 0);
200        mv.visitEnd();
201    }
202
203    /**
204     * Adds the dependency field {@link ProxyGenerator#DEPENDENCY}.
205     * @param cw the class writer
206     */
207    private static void addDependencyField(ClassWriter cw) {
208        cw.visitField(Opcodes.ACC_FINAL, DEPENDENCY, DEPENDENCY_DESC, null, null);
209        cw.visitEnd();
210    }
