019package org.apache.felix.ipojo.api.composite;
020
021import java.util.ArrayList;
022import java.util.Dictionary;
023import java.util.Enumeration;
024import java.util.Iterator;
025import java.util.List;
026import java.util.Map;
027import java.util.Map.Entry;
028import java.util.Set;
029import java.util.Vector;
030
031import org.apache.felix.ipojo.api.HandlerConfiguration;
032import org.apache.felix.ipojo.metadata.Attribute;
033import org.apache.felix.ipojo.metadata.Element;
034
035/**
036 * Allows defining a contained instance. A contained instance is like
037 * an instance declaration but the instance is created in the composite.
038 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
039 */
040public class Instance implements HandlerConfiguration {
041    
042    /**
043     * Targeted component type.
044     */
045    private String m_type;
046    
047    /**
048     * Configuration.
049     */
050    private List m_conf = new ArrayList();
051
052    /**
053     * Creates a Instance.
054     * @param type the targeted type.
055     */
056    public Instance(String type) {
057        m_type = type;
058    }
059
060    /**
061     * Adds the string property.
062     * @param name property name
063     * @param value property value
064     * @return the current instance
065     */
066    public Instance addProperty(String name, String value) {
067        Element elem = new Element("property", "");
068        m_conf.add(elem);
069        elem.addAttribute(new Attribute("name", name));
070        elem.addAttribute(new Attribute("value", value));
071        return this;
072    }
073
074    /**
075     * Adds a list property.
076     * @param name property name
077     * @param values the list
078     * @return the current instance
079     */
080    public Instance addProperty(String name, List values) {
081        Element elem = new Element("property", "");
082        elem.addAttribute(new Attribute("name", name));
083        elem.addAttribute(new Attribute("type", "list"));
084
085        m_conf.add(elem);
086
087        for (int i = 0; i < values.size(); i++) {
088            Object obj = values.get(i);
089            Element e = new Element("property", "");
090            elem.addElement(e);
091            if (obj instanceof String) {
092                e.addAttribute(new Attribute("value", obj.toString()));
093            } else {
094                // TODO
095                throw new UnsupportedOperationException(
096                        "Complex properties are not supported yet");
097            }
098        }
099
100        return this;
101    }
102
103    /**
104     * Adds an array property.
105     * @param name the property name
106     * @param values the array
107     * @return the current instance
108     */
109    public Instance addProperty(String name, String[] values) {
110        Element elem = new Element("property", "");
111        elem.addAttribute(new Attribute("name", name));
112        elem.addAttribute(new Attribute("type", "array"));
113
114        m_conf.add(elem);
115
116        for (int i = 0; i < values.length; i++) {
117            Object obj = values[i];
118            Element e = new Element("property", "");
119            elem.addElement(e);
120            e.addAttribute(new Attribute("value", obj.toString()));
121        }
122
123        return this;
124    }
125
126    /**
127     * Adds a vector property.
128     * @param name the property name
129     * @param values the vector
130     * @return the current instance
131     */
132    public Instance addProperty(String name, Vector values) {
133        Element elem = new Element("property", "");
134        elem.addAttribute(new Attribute("name", name));
135        elem.addAttribute(new Attribute("type", "vector"));
136
137        m_conf.add(elem);
138
139        for (int i = 0; i < values.size(); i++) {
140            Object obj = values.get(i);
141            Element e = new Element("property", "");
142            elem.addElement(e);
143            if (obj instanceof String) {
144                e.addAttribute(new Attribute("value", obj.toString()));
145            } else {
146                // TODO
147                throw new UnsupportedOperationException(
148                        "Complex properties are not supported yet");
149            }
150        }
151
152        return this;
153    }
154
155    /**
156     * Adds a map property.
157     * @param name the property name
158     * @param values the map
159     * @return the current instance
160     */
161    public Instance addProperty(String name, Map values) {
162        Element elem = new Element("property", "");
163        elem.addAttribute(new Attribute("name", name));
164        elem.addAttribute(new Attribute("type", "map"));
165
166        m_conf.add(elem);
167        Set entries = values.entrySet();
168        Iterator it = entries.iterator();
169        while (it.hasNext()) {
170            Map.Entry entry = (Entry) it.next();
171            Element e = new Element("property", "");
172            elem.addElement(e);
173
174            String n = (String) entry.getKey();
175            Object v = entry.getValue();
176            if (v instanceof String) {
177                e.addAttribute(new Attribute("name", n));
178                e.addAttribute(new Attribute("value", v.toString()));
179            } else {
180                // TODO
181                throw new UnsupportedOperationException(
182                        "Complex properties are not supported yet");
183            }
184        }
185
186        return this;
187    }
188
189    /**
190     * Adds a dictionary property.
191     * @param name the property name
192     * @param values the dictionary
193     * @return the current instance
194     */
195    public Instance addProperty(String name, Dictionary values) {
196        Element elem = new Element("property", "");
197        elem.addAttribute(new Attribute("name", name));
198        elem.addAttribute(new Attribute("type", "dictionary"));
199
200        m_conf.add(elem);
201        Enumeration e = values.keys();
202        while (e.hasMoreElements()) {
203            Element el = new Element("property", "");
204            elem.addElement(el);
205
206            String n = (String) e.nextElement();
207            Object v = values.get(n);
208            if (v instanceof String) {
209                el.addAttribute(new Attribute("name", n));
210                el.addAttribute(new Attribute("value", v.toString()));
211            } else {
212                // TODO
213                throw new UnsupportedOperationException(
214                        "Complex properties are not supported yet");
215            }
216        }
217
218        return this;
219    }
220
221    /**
222     * Ensures the validity of the instance description.
223     */
224    private void ensureValidity() {
225        if (m_type == null) {
226            throw new IllegalStateException(
227                    "Invalid containted instance configuration : the component type is not set");
228        }
229    }
230
231    /**
232     * Gets the instance description in the Element-Attribute form.
233     * @return the root Element of the instance description 
234     * @see org.apache.felix.ipojo.api.HandlerConfiguration#getElement()
235     */
236    public Element getElement() {
237        ensureValidity();
238        Element instance = new Element("instance", "");
239        instance.addAttribute(new Attribute("component", m_type));
240        for (int i = 0; i < m_conf.size(); i++) {
241            Element elem = (Element) m_conf.get(i);
242            instance.addElement(elem);
243        }
244        return instance;
245    }
246}
