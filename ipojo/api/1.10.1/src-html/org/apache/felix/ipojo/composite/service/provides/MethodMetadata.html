019package org.apache.felix.ipojo.composite.service.provides;
020
021import java.lang.reflect.Method;
022
023/**
024 * Information on Method for the composition.
025 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
026 */
027public class MethodMetadata {
028
029    /**
030     * ONE POLICY.
031     */
032    public static final int ONE_POLICY = 1;
033
034    /**
035     * ALL POLICY. 
036     */
037    public static final int ALL_POLICY = 2;
038
039    /**
040     * Method Object.
041     */
042    private Method m_method;
043    
044    /**
045     * Delegation field.
046     */
047    private FieldMetadata m_delegation;
048
049    /**
050     * Delegation policy (default = ONE).
051     */
052    private int m_policy = ONE_POLICY;
053
054    /**
055     * Constructor.
056     * @param method : method object.
057     */
058    public MethodMetadata(Method method) {
059        m_method = method;
060    }
061
062    public Method getMethod() {
063        return m_method;
064    }
065
066    public void setDelegation(FieldMetadata field) {
067        m_delegation = field;
068    }
069
070    public FieldMetadata getDelegation() {
071        return m_delegation;
072    }
073    
074    /**
075     * Equals method.
076     * This method check if two MethodMetadata are equals or if the current MemethodMetadata is equals with a Method object. 
077     * @param object : object.
078     * @return true if the current object and the given object are equals.
079     * @see java.lang.Object#equals(java.lang.Object)
080     */
081    public boolean equals(Object object) {
082        if (object instanceof MethodMetadata) {
083            Method met = ((MethodMetadata) object).getMethod();
084            return equals(met);
085        }
086        
087        if (object instanceof Method) {
088            Method met = (Method) object;
089            if (! met.getName().equals(m_method.getName()) || met.getParameterTypes().length != m_method.getParameterTypes().length) {
090                return false;
091            }
092
093            for (int i = 0; i < m_method.getParameterTypes().length; i++) {
094                if (!m_method.getParameterTypes()[i].getName().equals(met.getParameterTypes()[i].getName())) {
095                    return false;
096                }
097            }
098
099            return true;
100        }
101        
102        return false;
103    }
104    
105    /**
106     * Hash code method.
107     * @return the parent hash code.
108     * @see java.lang.Object#hashCode()
109     */
110    public int hashCode() {
111        return super.hashCode();
112    }
113
114    public int getPolicy() {
115        return m_policy;
116    }
117
118    /**
119     * Activate the all policy for this method.
120     */
121    public void setAllPolicy() {
122        m_policy = ALL_POLICY;
123    }
124    
125    /**
126     * Check if the method can throw UnsupportedOperationException.
127     * @return true if the method has declared the UnsupportedOperationException.
128     */
129    protected boolean throwsUnsupportedOperationException() {
130        for (int i = 0; i < m_method.getExceptionTypes().length; i++) {
131            if (m_method.getExceptionTypes()[i].getName().equals(UnsupportedOperationException.class.getName())) {
132                return true;
133            }
134        }
135        return false;
136    }
137
138}
