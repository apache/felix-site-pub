044/**
045 * iPOJO Arch command giving information about the current
046 * system architecture. This is a Gogo command.

111    /**
112     * Displays iPOJO instances.
113     */
114    @Descriptor("Display iPOJO instances")
115    public void instances() {
116        StringBuilder buffer = new StringBuilder();
117        for (Architecture m_arch : m_archs) {
118            InstanceDescription instance = m_arch.getInstanceDescription();
119            if (instance.getState() == ComponentInstance.VALID) {
120                buffer.append(format("Instance %s -> valid%n", instance.getName()));
121            }
122            if (instance.getState() == ComponentInstance.INVALID) {
123                buffer.append(format("Instance %s -> invalid%n", instance.getName()));
124            }
125            if (instance.getState() == ComponentInstance.STOPPED) {
126                buffer.append(format("Instance %s -> stopped%n", instance.getName()));
127            }
128        }
129
130        for (InstanceDeclaration instance : m_instances) {
131            // Only print unbound instances (others already printed above)
132            if (!instance.getStatus().isBound()) {
133                buffer.append(format("Instance %s of type %s is not bound.%n",
134                        name(instance.getConfiguration()),
135                        instance.getConfiguration().get("component")));
136                buffer.append(format("  Reason: %s", instance.getStatus().getMessage()));
137                buffer.append("\n");
138            }
139        }
140        
141        if (buffer.length() == 0) {
142            buffer.append("No instances \n");
143        }
144        
145        System.out.println(buffer.toString());   
146    }
147
148    private String name(Dictionary<String, Object> configuration) {
149        String name = (String) configuration.get("instance.name");
150        if (name == null) {
151            name = "unnamed";
152        }
153        return name;
154    }

156    /**
157     * Displays the architecture of a specific instance.
158     * @param instance the instance name
159     */
160    @Descriptor("Display the architecture of a specific instance")
161    public void instance(@Descriptor("target instance name") String instance) {
162
163        StringBuilder sb = new StringBuilder();
164
165        for (Architecture m_arch : m_archs) {
166            InstanceDescription id = m_arch.getInstanceDescription();
167            if (id.getName().equalsIgnoreCase(instance)) {
168                sb.append(id.getDescription());
169                sb.append('\n');
170            }
171        }
172
173        for (InstanceDeclaration instanceDeclaration : m_instances) {
174            if (!instanceDeclaration.getStatus().isBound()) {
175                if (instance.equals(name(instanceDeclaration.getConfiguration()))) {
176                    sb.append(format("InstanceDeclaration %s not bound to its factory%n", instance));
177                    sb.append(format(" type: %s%n", instanceDeclaration.getComponentName()));
178                    sb.append(format(" reason: %s%n", instanceDeclaration.getStatus().getMessage()));
179                    Throwable throwable = instanceDeclaration.getStatus().getThrowable();
180                    if (throwable != null) {
181                        ByteArrayOutputStream os = new ByteArrayOutputStream();
182                        throwable.printStackTrace(new PrintStream(os));
183                        sb.append(" throwable: ");
184                        sb.append(os.toString());
185                    }
186                }
187            }
188        }
189
190        if (sb.length() == 0) {
191            System.err.printf("Instance named '%s' not found", instance);
192        } else {
193            System.out.print(sb);
194        }
195
196    }

209    /**
210     * Displays the information about a specific factory.
211     * Note that factory name are not unique, so all matching
212     * factories are displayed.
213     * @param name the factory name
214     */
215    @Descriptor("Display the information about a specific factory")
216    public void factory(@Descriptor("target factory name") String name) {
217
218        List<Factory> factories = new ArrayList<Factory>();
219        List<TypeDeclaration> types = new ArrayList<TypeDeclaration>();
220
221        // Looking for public factories
222        for (Factory factory : m_factories) {
223            if (factory.getName().equalsIgnoreCase(name)) {
224                factories.add(factory);
225            }
226        }
227
228        // Looking for all unbound or private bound types
229        for (TypeDeclaration type : m_types) {
230            if (name.equalsIgnoreCase(type.getComponentName())) {
231                // (Public + Unbound) or private types have no exported factories
232                if (!type.isPublic() || (!type.getStatus().isBound() && type.isPublic())) {
233                    types.add(type);
234                }
235            }
236        }
237
238        if (factories.isEmpty() && types.isEmpty()) {
239            System.err.println("Factory " + name + " not found");
240            return;
241        }
242
243        // Display found factories and types
244        for (Factory factory : factories) {
245            System.out.println(factory.getComponentDescription());
246        }
247        for (TypeDeclaration type : types) {
248            if (!type.getStatus().isBound()) {
249                // Unbound: maybe private or public type
250                System.out.printf("Factory %s is not bound%n", type.getComponentName());
251                System.out.printf("  reason: %s%n", type.getStatus().getMessage());
252                Throwable throwable = type.getStatus().getThrowable();
253                if (throwable != null) {
254                    System.out.print("  throwable: ");
255                    throwable.printStackTrace(System.out);
256                }
257            } else {
258                // Bound, this is only a private factory
259                System.out.printf("Factory %s is bound - Private%n", type.getComponentName());
260            }
261        }
262
263    }

273    /**
274     * Displays the list of public iPOJO factories.
275     */
276    @Descriptor("Display iPOJO factories")
277    public void factories() {
278        StringBuilder buffer = new StringBuilder();
279        for (Factory m_factory : m_factories) {
280            if (m_factory.getMissingHandlers().size() == 0) {
281                buffer.append(format("Factory %s (VALID)%n", m_factory.getName()));
282            } else {
283                buffer.append(format("Factory %s (INVALID: %s)%n",
284                        m_factory.getName(),
285                        m_factory.getMissingHandlers()));
286            }
287        }
288
289        for (TypeDeclaration type : m_types) {
290            if (!type.isPublic()) {
291                // Private factories: always display them
292                // Cannot display much more than presence/absence since the TypeDeclaration API does not
293                // give access to the underlying Factory or description (if valid)
294                if (type.getStatus().isBound()) {
295                    buffer.append(format("Factory %s (UNKNOWN) - Private%n", type.getComponentName()));
296                } else {
297                    // Unbound type means that required extension is not available
298                    // We'll say that the factory is INVALID even if in reality it's not even instantiated
299                    buffer.append(format("Factory %s (INVALID) - Private%n", type.getComponentName()));
300                    buffer.append(format("  -> %s", type.getStatus().getMessage()));
301                }
302            } else {
303                if (!type.getStatus().isBound()) {
304                    buffer.append(format("Factory %s is not bound%n", type.getComponentName()));
305                    buffer.append(format("  -> %s%n", type.getStatus().getMessage()));
306                }
307            }
308        }
309        
310        if (buffer.length() == 0) {
311            buffer.append("No factories \n");
312        }
313        
314        System.out.println(buffer.toString());
315    }

317    /**
318     * Displays the list of available handlers.
319     */
320    @Descriptor("Display iPOJO handlers")
321    public void handlers() {
322        PrintStream out = System.out;
323        for (HandlerFactory m_handler : m_handlers) {
324            String name = m_handler.getHandlerName();
325            if ("composite".equals(m_handler.getType())) {
326                name = name + " [composite]";
327            }
328            if (m_handler.getMissingHandlers().size() == 0) {
329                out.println("Handler " + name + " (VALID)");
330            } else {
331                out.println("Handler " + name + " (INVALID : " + m_handler.getMissingHandlers() + ")");
332            }
333        }
334
335        for (TypeDeclaration type : m_types) {
336            if (!type.getStatus().isBound()) {
337                out.println("HandlerFactory " + type.getComponentName() + " is not bound");
338                out.println("  -> " + type.getStatus().getMessage());
339            }
340        }
341    }

352        }
