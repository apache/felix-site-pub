019package org.apache.felix.ipojo.api;
020
021import org.apache.felix.ipojo.ComponentInstance;
022import org.apache.felix.ipojo.PrimitiveInstanceDescription;
023import org.apache.felix.ipojo.handlers.providedservice.ProvidedServiceDescription;
024import org.apache.felix.ipojo.metadata.Attribute;
025import org.apache.felix.ipojo.metadata.Element;
026
027/**
028 * Allows configuring a service property.
029 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
030 */
031public class ServiceProperty {
032    
033    /**
034     * The property name.
035     */
036    private String m_name;
037    
038    /**
039     * The property field. 
040     */
041    private String m_field;
042    
043    /**
044     * The property type. 
045     */
046    private String m_type;
047    
048    /**
049     * The property value. 
050     */
051    private String m_value;
052    
053    /**
054     * Is the property mandatory. 
055     */
056    private boolean m_mandatory;
057    
058    /**
059     * Is the property immutable. 
060     */
061    private boolean m_immutable;
062    
063    /**
064     * Sets the property name.
065     * @param name the property name
066     * @return the current property object
067     */
068    public ServiceProperty setName(String name) {
069        m_name = name;
070        return this;
071    }
072    
073    /**
074     * Sets the property field.
075     * @param name the property field
076     * @return the current property object
077     */
078    public ServiceProperty setField(String name) {
079        m_field = name;
080        return this;
081    }
082    
083    /**
084     * Sets the property type.
085     * @param name the property type
086     * @return the current property object
087     */
088    public ServiceProperty setType(String name) {
089        m_type = name;
090        return this;
091    }
092    
093    /**
094     * Sets the property value.
095     * @param name the property value
096     * @return the current property object
097     */
098    public ServiceProperty setValue(String name) {
099        m_value = name;
100        return this;
101    }
102    
103    /**
104     * Sets if the property is immutable.
105     * @param immutable <code>true</code> if the dependency is immutable.
106     * @return the current property object
107     */
108    public ServiceProperty setImmutable(boolean immutable) {
109        m_immutable = immutable;
110        return this;
111    }
112    
113    /**
114     * Sets if the property is mandatory.
115     * @param mandatory <code>true</code> if the dependency is mandatory.
116     * @return the current property object
117     */
118    public ServiceProperty setMandatory(boolean mandatory) {
119        m_mandatory = mandatory;
120        return this;
121    }
122    
123    
124    /**
125     * Gets the 'property' element.
126     * @return the element describing the current property.
127     */
128    public Element getElement() {
129        ensureValidity();
130        Element element = new Element("property", "");
131        if (m_name != null) {
132            element.addAttribute(new Attribute("name", m_name));
133        }
134        if (m_type != null) {
135            element.addAttribute(new Attribute("type", m_type));
136        }
137        if (m_value != null) {
138            element.addAttribute(new Attribute("value", m_value));
139        }
140        if (m_field != null) {
141            element.addAttribute(new Attribute("field", m_field));
142        }
143        if (m_mandatory) {
144            element.addAttribute(new Attribute("mandatory", new Boolean(m_mandatory).toString()));
145        }
146        if (m_immutable) {
147            element.addAttribute(new Attribute("immutable", new Boolean(m_immutable).toString()));
148        }
149        return element;
150    }
151
152    /**
153     * Checks the configuration validity.
154     */
155    private void ensureValidity() {
156        // Two cases
157        // Field or type
158        if (m_field == null && m_type == null) {
159            throw new IllegalStateException("A service property must have either a field or a type");
160        }
161        if (m_immutable && m_value == null) {
162            throw new IllegalStateException("A immutable service property must have a value");
163        }
164    }
165    
166    /**
167     * Gets the property value of the current property
168     * on the given instance.
169     * @param instance the instance on which looking for 
170     * the property value
171     * @return the property value or <code>null</code>
172     * if not found.
173     */
174    public Object getPropertyValue(ComponentInstance instance) {
175        PrimitiveInstanceDescription desc = (PrimitiveInstanceDescription) instance.getInstanceDescription();
176        ProvidedServiceDescription[] pss = desc.getProvidedServices();
177        for (int i = 0; i < pss.length; i++) {
178            // Check with the name
179            if (m_name != null && pss[i].getProperties().containsKey(m_name)) {
180                return pss[i].getProperties().get(m_name);
181            }
182            // Check with the field
183            if (m_field != null && pss[i].getProperties().containsKey(m_field)) {
184                return pss[i].getProperties().get(m_field);
185            }
186        }
187        // Not found.
188        return null;
189    }
190
191
192}
