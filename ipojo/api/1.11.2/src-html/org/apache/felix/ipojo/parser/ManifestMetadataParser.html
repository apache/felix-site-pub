032/**
033 * The Manifest Metadata parser reads a manifest file and builds
034 * the iPOJO metadata ({@link Element} / {@link Attribute} ) structure.
035 *
036 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
037 */
038public class ManifestMetadataParser {

040    /**
041     * The element list.
042     * Contains the element found in the parsed header.
043     */
044    private Element[] m_elements = new Element[0];
045
046    /**
047     * Gets the array of component type metadata.
048     * @return the component metadata (composite & component).
049     * An empty array is returned if no component type declaration.
050     * @throws ParseException if a parsing error occurs
051     */
052    public Element[] getComponentsMetadata() throws ParseException {
053        Element[] elems = m_elements[0].getElements();
054        List list = new ArrayList();
055        for (int i = 0; i < elems.length; i++) {
056            if (!"instance".equals(elems[i].getName())) {
057                list.add(elems[i]);
058            }
059        }
060        return (Element[]) list.toArray(new Element[list.size()]);
061    }

063    /**
064     * Gets the array of instance configuration described in the metadata.
065     * @return the instances list or <code>null</code> if no instance configuration.
066     * @throws ParseException if the metadata cannot be parsed successfully
067     */
068    public Dictionary[] getInstances() throws ParseException {
069        Element[] configs = m_elements[0].getElements("instance");
070        if (configs == null) {
071            return null;
072        }
073        Dictionary[] dicts = new Dictionary[configs.length];
074        for (int i = 0; i < configs.length; i++) {
075            dicts[i] = parseInstance(configs[i]);
076        }
077        return dicts;
078    }
079
080    /**
081     * Parses an Element to create an instance configuration dictionary.
082     * The 'name' attribute of the instance declaration is mapped
083     * to the 'instance.name' property.
084     * @param instance the Element describing an instance.
085     * @return the resulting dictionary
086     * @throws ParseException if a configuration cannot be parse correctly.
087     */
088    private Dictionary parseInstance(Element instance) throws ParseException {
089        Dictionary dict = new Properties();
090        String name = instance.getAttribute("name");
091        String comp = instance.getAttribute("component");
092        String version = instance.getAttribute("version");
093
094        if (name != null) {
095            dict.put(Factory.INSTANCE_NAME_PROPERTY, instance.getAttribute("name"));
096        }
097
098        if (comp == null) {
099            throw new ParseException("An instance does not have the 'component' attribute");
100        }
101
102        dict.put("component", comp);
103
104        if (version != null) {
105            dict.put(Factory.FACTORY_VERSION_PROPERTY, version);
106        }
107
108        Element[] props = instance.getElements("property");
109
110        for (int i = 0; props != null && i < props.length; i++) {
111            parseProperty(props[i], dict);
112        }
113
114        return dict;
115    }

117    /**
118     * Parses an instance property.
119     * This method is recursive to handle complex properties.
120     * @param prop the current element to parse
121     * @param dict the dictionary to populate (the future instance configuration)
122     * @throws ParseException if the property cannot be parsed correctly
123     */
124    private void parseProperty(Element prop, Dictionary dict) throws ParseException {
125        // Check that the property has a name
126        String name = prop.getAttribute("name");
127        String value = prop.getAttribute("value");
128        if (name == null) {
129            throw new ParseException("A property does not have the 'name' attribute: " + prop);
130        }
131
132        //case : the property element has no 'value' attribute
133        if (value == null) {
134            // Recursive case
135            // Get the type of the structure to create
136            String type = prop.getAttribute("type");
137            if (type == null || type.equalsIgnoreCase("dictionary")) {
138                dict.put(name, parseDictionary(prop));
139            } else if (type.equalsIgnoreCase("map")) {
140                dict.put(name, parseMap(prop));
141            } else if (type.equalsIgnoreCase("list")) {
142                dict.put(name, parseList(prop));
143            } else if (type.equalsIgnoreCase("array")) {
144                List list = parseList(prop);
145                boolean isString = true;
146                for (int i = 0; isString && i < list.size(); i++) {
147                    isString = list.get(i) instanceof String;
148                }
149                Object[] obj = null;
150                if (isString) {
151                    obj = new String[list.size()];
152                } else {
153                    obj = new Object[list.size()];
154                }
155                // Transform the list to array
156                dict.put(name, list.toArray(obj));
157            }
158        } else {
159            dict.put(prop.getAttribute("name"), prop.getAttribute("value"));
160        }
161    }

163    /**
164     * Parses a complex property.
165     * This property will be built as a {@link Dictionary}.
166     * @param prop the Element to parse.
167     * @return the resulting dictionary
168     * @throws ParseException if an internal property is incorrect.
169     */
170    private Dictionary parseDictionary(Element prop) throws ParseException {
171     // Check if there is 'property' elements
172        Element[] subProps = prop.getElements("property");
173        if (subProps != null) {
174            Dictionary dict2 = new Properties();
175            for (int i = 0; i < subProps.length; i++) {
176                parseProperty(subProps[i], dict2);
177            }
178            return dict2;
179        } else {
180            // If the no sub-properties, inject an empty dictionary.
181            return new Properties();
182        }
183    }
184
185    /**
186     * Parses a complex property.
187     * This property will be built as a {@link Map}.
188     * The used {@link Map} implementation is {@link HashMap}.
189     * @param prop the property to parse
190     * @return the resulting Map
191     * @throws ParseException if an internal property is incorrect.
192     */
193    private Map parseMap(Element prop) throws ParseException {
194        // Check if there is 'property' elements
195        Element[] subProps = prop.getElements("property");
196        if (subProps != null) {
197            Map map = new HashMap(); // Create an hashmap to store elements.
198            for (int i = 0; i < subProps.length; i++) {
199                parseProperty(subProps[i], map);
200            }
201            return map;
202        } else { // if not inject an empty map
203            return new HashMap(0);
204        }
205    }
206
207    /**
208     * Parses a complex property. This property will be built as a {@link List}.
209     * The used {@link List} implementation is {@link ArrayList}.
210     * The order of elements is kept.
211     * @param prop the property to parse
212     * @return the resulting List
213     * @throws ParseException if an internal property is incorrect.
214     */
215    private List parseList(Element prop) throws ParseException {
216        Element[] subProps = prop.getElements("property");
217        if (subProps != null) {
218            List list = new ArrayList(subProps.length); // Create a list to store elements.
219            for (int i = 0; i < subProps.length; i++) {
220                parseAnonymousProperty(subProps[i], list); // Anonymous properties.
221            }
222            return list;
223        } else {
224            // If no sub-properties, inject an empty list.
225            return new ArrayList(0);
226        }
227    }

229    /**
230     * Parse a property.
231     * This methods handles complex properties.
232     * @param prop the current element to parse
233     * @param map the map to populate
234     * @throws ParseException if the property cannot be parsed correctly
235     */
236    private void parseProperty(Element prop, Map map) throws ParseException {
237        // Check that the property has a name
238        String name = prop.getAttribute("name");
239        String value = prop.getAttribute("value");
240        if (name == null) {
241            throw new ParseException(
242                    "A property does not have the 'name' attribute");
243        }
244        // case : the property element has no 'value' attribute
245        if (value == null) {
246            // Recursive case
247            // Get the type of the structure to create
248            String type = prop.getAttribute("type");
249            if (type == null || type.equalsIgnoreCase("dictionary")) {
250                map.put(name, parseDictionary(prop));
251            } else if (type.equalsIgnoreCase("map")) {
252                map.put(name, parseMap(prop));
253            } else if (type.equalsIgnoreCase("list")) {
254                map.put(name, parseList(prop));
255            } else if (type.equalsIgnoreCase("array")) {
256                List list = parseList(prop);
257                boolean isString = true;
258                for (int i = 0; isString && i < list.size(); i++) {
259                    isString = list.get(i) instanceof String;
260                }
261                Object[] obj = null;
262                if (isString) {
263                    obj = new String[list.size()];
264                } else {
265                    obj = new Object[list.size()];
266                }
267                map.put(name, list.toArray(obj)); // Transform the list to array
268            }
269        } else {
270            map.put(prop.getAttribute("name"), prop.getAttribute("value"));
271        }
272    }

274    /**
275     * Parse an anonymous property.
276     * An anonymous property is a property with no name.
277     * An anonymous property can be simple (just a value) or complex (i.e. a map, a dictionary
278     * a list or an array).
279     * @param prop the property to parse
280     * @param list the list to populate with the resulting property
281     * @throws ParseException if an internal property cannot be parse correctly
282     */
283    private void parseAnonymousProperty(Element prop, List list) throws ParseException {
284        // Check that the property has a name
285        String name = prop.getAttribute("name");
286        String value = prop.getAttribute("value");
287        if (name != null) {
288            throw new ParseException("Anonymous property expected in a list or an array");
289        }
290        //case : the property element has no 'value' attribute
291        if (value == null) {
292            // Recursive case
293
294            // Get the type of the structure to create
295            String type = prop.getAttribute("type");
296            if (type == null || type.equalsIgnoreCase("dictionary")) {
297                // Check if there is 'property' elements
298                Element[] subProps = prop.getElements("property");
299                if (subProps != null) {
300                    Dictionary dict2 = new Properties();
301                    for (int i = 0; i < subProps.length; i++) {
302                        parseProperty(subProps[i], dict2);
303                    }
304                    list.add(dict2);
305                } else {
306                    // If the no sub-properties, inject an empty dictionary.
307                    list.add(new Properties());
308                }
309            } else if (type.equalsIgnoreCase("map")) {
310                // Check if there is 'property' elements
311                Element[] subProps = prop.getElements("property");
312                if (subProps != null) {
313                    Map map2 = new HashMap(); // Create an hashmap to store elements.
314                    for (int i = 0; i < subProps.length; i++) {
315                        parseProperty(subProps[i], map2);
316                    }
317                    list.add(map2);
318                } else { // if not inject an empty map
319                    list.add(new HashMap(0));
320                }
321            }  else if (type.equalsIgnoreCase("list")) {
322                Element[] subProps = prop.getElements("property");
323                if (subProps != null) {
324                    // Create a list to store elements.
325                    List list2 = new ArrayList(subProps.length);
326                    for (int i = 0; i < subProps.length; i++) {
327                        parseAnonymousProperty(subProps[i], list2); // Anonymous properties
328                    }
329                    list.add(list2);
330                } else {
331                    // If no sub-properties, inject an empty list.
332                    list.add(new ArrayList(0));
333                }
334            } else if (type.equalsIgnoreCase("array")) {
335                // Check sub-props.
336                Element[] subProps = prop.getElements("property");
337                if (subProps != null) {
338                    List list2 = new ArrayList(subProps.length); // Use list as
339                                                                 // pivot type
340                    for (int i = 0; i < subProps.length; i++) {
341                        parseAnonymousProperty(subProps[i], list2);
342                    }
343                    list.add(list.toArray(new Object[list.size()])); // Transform
344                                                                     // the list
345                                                                     // to array
346                } else {
347                    list.add(new Element[0]); // Insert an empty Element array.
348                }
349            }
350        } else {
351            list.add(prop.getAttribute("value"));
352        }
353
354    }

356    /**
357     * Adds an element to the {@link ManifestMetadataParser#m_elements} list.
358     * @param elem the the element to add
359     */
360    private void addElement(Element elem) {
361        if (m_elements == null) {
362            m_elements = new Element[] { elem };
363        } else {
364            Element[] newElementsList = new Element[m_elements.length + 1];
365            System.arraycopy(m_elements, 0, newElementsList, 0, m_elements.length);
366            newElementsList[m_elements.length] = elem;
367            m_elements = newElementsList;
368        }
369    }
370
371    /**
372     * Removes an element from the {@link ManifestMetadataParser#m_elements} list.
373     * @return an element to remove
374     */
375    private Element removeLastElement() {
376        int idx = -1;
377        idx = m_elements.length - 1;
378        Element last = m_elements[idx];
379        if (idx >= 0) {
380            if ((m_elements.length - 1) == 0) {
381                // It is the last element of the list;
382                m_elements = new Element[0];
383            } else {
384                // Remove the last element of the list :
385                Element[] newElementsList = new Element[m_elements.length - 1];
386                System.arraycopy(m_elements, 0, newElementsList, 0, idx);
387                m_elements = newElementsList;
388            }
389        }
390        return last;
391    }

393    /**
394     * Looks for the <code>iPOJO-Components</code> header
395     * in the given dictionary. Then, initializes the
396     * {@link ManifestMetadataParser#m_elements} list (adds the
397     * <code>iPOJO</code> root element) and parses the contained
398     * component type declarations and instance configurations.
399     * @param dict the given headers of the manifest file
400     * @throws ParseException if any error occurs
401     */
402    public void parse(Dictionary dict) throws ParseException {
403        String componentClassesStr = (String) dict.get("iPOJO-Components");
404        // Add the ipojo element inside the element list
405        addElement(new Element("iPOJO", ""));
406        parseElements(componentClassesStr.trim());
407    }

409    /**
410     * Parses the given header, initialized the
411     * {@link ManifestMetadataParser#m_elements} list
412     * (adds the <code>iPOJO</code> element) and parses
413     * contained component type declarations and instance configurations.
414     * @param header the given header of the manifest file
415     * @throws ParseException if any error occurs
416     */
417    public void parseHeader(String header) throws ParseException {
418        // Add the ipojo element inside the element list
419        addElement(new Element("iPOJO", ""));
420        parseElements(header.trim());
421    }

423    /**
424     * Parses the metadata from the string given in argument.
425     * This methods creates a new {@link ManifestMetadataParser} object
426     * and calls the {@link ManifestMetadataParser#parseElements(String)}
427     * method. The parsing must result as a tree (only one root element).
428     * @param metadata the metadata to parse
429     * @return Element the root element resulting of the parsing
430     * @throws ParseException if any error occurs
431     */
432    public static Element parse(String metadata) throws ParseException {
433        ManifestMetadataParser parser = new ManifestMetadataParser();
434        parser.parseElements(metadata);
435        if (parser.m_elements.length != 1) {
436            throw new ParseException("Error in parsing, root element not found : " + metadata);
437        }
438        return parser.m_elements[0];
439    }

441    /**
442     * Parses the metadata from the given header string.
443     * This method creates a new {@link ManifestMetadataParser} object and then
444     * creates the <code>iPOJO</code> root element, parses content elements
445     * (component types and instances declarations), and returns the resulting
446     * {@link Element} / {@link Attribute} structure. The parsed string
447     * must be a tree (only one root element).
448     * @param header the header to parse
449     * @return Element the root element resulting of the parsing
450     * @throws ParseException if any error occurs
451     */
452    public static Element parseHeaderMetadata(String header) throws ParseException {
453        ManifestMetadataParser parser = new ManifestMetadataParser();
454        parser.addElement(new Element("iPOJO", ""));
455        parser.parseElements(header);
456        if (parser.m_elements.length != 1) {
457            throw new ParseException("Error in parsing, root element not found : " + header);
458        }
459        return parser.m_elements[0];
460    }

462    /**
463     * Parses the given string.
464     * This methods populates the {@link ManifestMetadataParser#m_elements}
465     * list.
466     * @param elems the string to parse
467     */
468    private void parseElements(String elems) {
469        char[] string = elems.toCharArray();
470
471        for (int i = 0; i < string.length; i++) {
472            char current = string[i];
473            switch (current) { //NOPMD
474                // Beginning of an attribute.
475                case '$':
476                    StringBuffer attName = new StringBuffer();
477                    StringBuffer attValue = new StringBuffer();
478                    StringBuffer attNs = null;
479                    i++;
480                    current = string[i]; // Increment and get the new current char.
481                    while (current != '=') {
482                        if (current == ':') {
483                            attNs = attName;
484                            attName = new StringBuffer();
485                        } else {
486                            attName.append(current);
487                        }
488                        i++;
489                        current = string[i];
490                    }
491                    i = i + 2; // skip ="
492                    current = string[i];
493                    while (current != '"') {
494                        attValue.append(current);
495                        i++;
496                        current = string[i]; // Increment and get the new current char.
497                    }
498                    i++; // skip "
499                    current = string[i];
500
501                    Attribute att = null;
502                    if (attNs == null) {
503                        att = new Attribute(attName.toString(), attValue.toString());
504                    } else {
505                        att = new Attribute(attName.toString(), attNs.toString(), attValue.toString());
506                    }
507                    m_elements[m_elements.length - 1].addAttribute(att);
508                    break;
509
510                // End of an element
511                case '}':
512                    Element lastElement = removeLastElement();
513                    if (m_elements.length == 0) {
514                        addElement(lastElement);
515                    } else {
516                        Element newQueue = m_elements[m_elements.length - 1];
517                        newQueue.addElement(lastElement);
518                    }
519                    break;
520
521                // Space
522                case ' ':
523                    break; // do nothing;
524
525                // Default case
526                default:
527                    StringBuffer qname = new StringBuffer();
528                    current = string[i];
529                    while (current != ' ') {
530                        qname.append(current);
531                        i++;
532                        current = string[i]; // Increment and get the new current char.
533                    }
534                    // Skip spaces
535                    while (string[i] == ' ') {
536                        i = i + 1;
537                    }
538                    i = i + 1; // skip {
539
540                    Element elem = null;
541                    // Parse the qname
542                    String n = qname.toString();
543                    if (n.indexOf(':') == -1) {
544                        // No namespace
545                        elem = new Element(n, null);
546                    } else {
547                        // The namespace ends on the first ':'
548                        int last = n.lastIndexOf(':');
549                        String ns = n.substring(0, last);
550                        String name = n.substring(last + 1);
551                        elem = new Element(name.toString(), ns.toString());
552                    }
553
554                    addElement(elem);
555
556                    break;
557            }
558        }
559    }
