019package org.apache.felix.ipojo.context;
020
021import java.util.ArrayList;
022import java.util.Dictionary;
023import java.util.Enumeration;
024import java.util.Iterator;
025import java.util.List;
026import java.util.Map;
027import java.util.Properties;
028
029import org.apache.felix.ipojo.ComponentInstance;
030import org.apache.felix.ipojo.InstanceManager;
031import org.osgi.framework.Constants;
032import org.osgi.framework.ServiceFactory;
033import org.osgi.framework.ServiceReference;
034import org.osgi.framework.ServiceRegistration;
035
036/**
037 * Internal service registration implementation. This class is used for in the
038 * composition.
039 * 
040 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
041 */
042public class ServiceRegistrationImpl implements ServiceRegistration {
043
044    /**
045     * Service Registry.
046     */
047    private ServiceRegistry m_registry = null;
048
049    /**
050     * Interfaces associated with the service object.
051     */
052    private String[] m_classes = null;
053
054    /**
055     * Service Id associated with the service object.
056     */
057    private Long m_serviceId = null;
058
059    /**
060     * Service object.
061     */
062    private Object m_svcObj = null;
063
064    /**
065     * Service factory interface.
066     */
067    private ServiceFactory m_factory = null;
068
069    /**
070     * Associated property dictionary.
071     */
072    private Map m_propMap = null;
073
074    /**
075     * Re-usable service reference.
076     */
077    private ServiceReferenceImpl m_ref = null;
078
079    /**
080     * Property Keys List.
081     */
082    private List m_list = new ArrayList();
083
084    /**
085     * Constructor.
086     * 
087     * @param registry : the service registry
088     * @param instance : component instance
089     * @param classes : published interfaces array
090     * @param serviceId : the unique service id
091     * @param svcObj : the service object or the service factory object
092     * @param dict : service properties
093     */
094    public ServiceRegistrationImpl(ServiceRegistry registry, ComponentInstance instance, String[] classes, Long serviceId, Object svcObj, Dictionary dict) {
095        m_registry = registry;
096        m_classes = classes;
097        m_serviceId = serviceId;
098        m_svcObj = svcObj;
099        if (m_svcObj instanceof ServiceFactory) { m_factory = (ServiceFactory) m_svcObj; }
100        initializeProperties(dict);
101
102        // This reference is the "standard" reference for this service and will
103        // always be returned by getReference().
104        // Since all reference to this service are supposed to be equal, we use
105        // the hash code of this reference for
106        // a references to this service in ServiceReference.
107        m_ref = new ServiceReferenceImpl(instance, this);
108    }
109
110    /**
111     * Check if the service registration still valid.
112     * @return true if the service registration is valid.
113     */
114    protected boolean isValid() {
115        return m_svcObj != null;
116    }
117
118    /**
119     * Get the service reference attached with this service registration.
120     * @return the service reference
121     * @see org.osgi.framework.ServiceRegistration#getReference()
122     */
123    public ServiceReference getReference() {
124        return m_ref;
125    }
126
127    /**
128     * Add properties to a service registration.
129     * @param dict : the properties to add
130     * @see org.osgi.framework.ServiceRegistration#setProperties(java.util.Dictionary)
131     */
132    public void setProperties(Dictionary dict) {
133        // Make sure registration is valid.
134        if (!isValid()) {
135            throw new IllegalStateException("The service registration is no longer valid.");
136        }
137        // Set the properties.
138        initializeProperties(dict);
139        // Tell registry about it.
140        m_registry.servicePropertiesModified(this);
141    }
142
143    /**
144     * Unregister the service.
145     * @see org.osgi.framework.ServiceRegistration#unregister()
146     */
147    public void unregister() {
148        if (m_svcObj == null) {
149            throw new IllegalStateException("Service already unregistered.");
150        } else {
151            m_registry.unregisterService(this);
152            m_svcObj = null;
153            m_factory = null;
154        }
155    }
156
157    /**
158     * Look for a property in the service properties.
159     * @param key : property key
160     * @return the object associated with the key or null if the key is not
161     * present.
162     */
163    protected Object getProperty(String key) {
164        return m_propMap.get(key);
165    }
166
167    /**
168     * Get the property keys.
169     * @return the property keys list.
170     */
171    protected String[] getPropertyKeys() {
172        synchronized (m_propMap) {
173            m_list.clear();
174            Iterator iterator = m_propMap.entrySet().iterator();
175            while (iterator.hasNext()) {
176                Map.Entry entry = (Map.Entry) iterator.next();
177                m_list.add(entry.getKey());
178            }
179            return (String[]) m_list.toArray(new String[m_list.size()]);
180        }
181    }
182    
183    /**
184     * Gets the published properties.
185     * @return the dictionary containing each published properties.
186     */
187    protected Dictionary getProperties() {
188        synchronized (m_propMap) {
189            Dictionary dict = new Properties();
190            Iterator keys = m_propMap.keySet().iterator();
191            while (keys.hasNext()) {
192                String key = (String) keys.next();
193                dict.put(key, m_propMap.get(key));
194            }
195            return dict;
196        }
197    }
198
199    /**
200     * Get the service object.
201     * @return the service object. Call the service factory if needed.
202     */
203    protected Object getService() {
204        // If the service object is a service factory, then
205        // let it create the service object.
206        if (m_factory == null) {
207            return m_svcObj;
208        } else {
209            return getFactoryUnchecked();
210        }
211    }
212
213    /**
214     * Initialize properties.
215     * @param dict : service properties to publish.
216     */
217    private void initializeProperties(Dictionary dict) {
218        // Create a case insensitive map.
219        if (m_propMap == null) {
220            m_propMap = new StringMap(false);
221        } else {
222            m_propMap.clear();
223        }
224
225        if (dict != null) {
226            Enumeration keys = dict.keys();
227            while (keys.hasMoreElements()) {
228                Object key = keys.nextElement();
229                m_propMap.put(key, dict.get(key));
230            }
231        }
232        // Add the framework assigned properties.
233        m_propMap.put(Constants.OBJECTCLASS, m_classes);
234        m_propMap.put(Constants.SERVICE_ID, m_serviceId);
235    }
236
237    /**
238     * Get a service object via a service factory.
239     * @return the service object via the service factory invocation.
240     */
241    private Object getFactoryUnchecked() {
242        return m_factory.getService(null, this);
243    }
244
245    /**
246     * Unget a service. (Internal Method)
247     * 
248     * @param instance : component instance using the service.
249     * @param svcObj : the unget service object.
250     */
251    private void ungetFactoryUnchecked(ComponentInstance instance, Object svcObj) {
252        if (instance instanceof InstanceManager) {
253            m_factory.ungetService(((InstanceManager) instance).getContext().getBundle(), this, svcObj);
254        }
255
256    }
257
258    /**
259     * Unget a service.
260     * 
261     * @param instance : component instance using the service.
262     * @param srvObj : the unget service object.
263     */
264    public void ungetService(ComponentInstance instance, Object srvObj) {
265        // If the service object is a service factory, then let is release the
266        // service object.
267        if (m_factory != null) {
268            ungetFactoryUnchecked(instance, srvObj);
269        }
270    }
271
