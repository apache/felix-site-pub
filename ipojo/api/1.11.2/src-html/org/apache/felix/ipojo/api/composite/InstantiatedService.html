023import java.util.Enumeration;
024import java.util.Iterator;

026import java.util.Map;

031import org.apache.felix.ipojo.api.HandlerConfiguration;

036/**
037 * Allows defining an instantiated sub-service. An instantiated sub-
038 * service will be reified by instances publishing the required service. 
039 * Those instances are created from public factories
040 * inside the composite.
041 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
042 */
043public class InstantiatedService implements HandlerConfiguration {
044
045    /**
046     * The required specification.
047     */
048    private String m_specification;
049
050    /**
051     * The LDAP filter of the dependency.
052     */
053    private String m_filter;
054
055    /**
056     * Is the dependency optional?
057     */
058    private boolean m_optional;
059
060    /**
061     * Is the dependency aggregate?
062     */
063    private boolean m_aggregate;
064
065    /**
066     * The dependency binding policy. Default: Dynamic policy.
067     */
068    private int m_policy = DependencyModel.DYNAMIC_BINDING_POLICY;
069
070    /**
071     * The dependency comparator. (used to compare service providers)
072     */
073    private String m_comparator;
074
075    /**
076     * Instance configuration List of Element (Property).
077     */
078    private List m_conf = new ArrayList();
079
080    /**
081     * Gets the dependency metadata.
082     * @return the 'subservice' element describing the current 
083    * instantiated service.
084     */
085    public Element getElement() {
086        ensureValidity();
087
088        Element dep = new Element("subservice", "");
089        dep.addAttribute(new Attribute("action", "instantiate"));
090
091        dep.addAttribute(new Attribute("specification", m_specification));
092
093        if (m_filter != null) {
094            dep.addAttribute(new Attribute("filter", m_filter));
095        }
096        if (m_comparator != null) {
097            dep.addAttribute(new Attribute("comparator", m_comparator));
098        }
099
100        if (m_optional) {
101            dep.addAttribute(new Attribute("optional", "true"));
102        }
103        if (m_aggregate) {
104            dep.addAttribute(new Attribute("aggregate", "true"));
105        }
106
107        if (m_policy == DependencyModel.DYNAMIC_BINDING_POLICY) {
108            dep.addAttribute(new Attribute("policy", "dynamic"));
109        } else if (m_policy == DependencyModel.STATIC_BINDING_POLICY) {
110            dep.addAttribute(new Attribute("policy", "static"));
111        } else if (m_policy == DependencyModel.DYNAMIC_PRIORITY_BINDING_POLICY) {
112            dep.addAttribute(new Attribute("policy", "dynamic-priority"));
113        }
114
115        for (int i = 0; i < m_conf.size(); i++) {
116            Element elem = (Element) m_conf.get(i);
117            dep.addElement(elem);
118        }
119
120        return dep;
121    }
122
123    /**
124     * Adds a string property.
125     * @param name the property name
126     * @param value the property value
127     * @return the current instantiated sub-service
128     */
129    public InstantiatedService addProperty(String name, String value) {
130        Element elem = new Element("property", "");
131        m_conf.add(elem);
132        elem.addAttribute(new Attribute("name", name));
133        elem.addAttribute(new Attribute("value", value));
134        return this;
135    }
136
137    /**
138     * Adds a list property.
139     * @param name the property name
140     * @param values the property value
141     * @return the current instantiated sub-service
142     */
143    public InstantiatedService addProperty(String name, List values) {
144        Element elem = new Element("property", "");
145        elem.addAttribute(new Attribute("name", name));
146        elem.addAttribute(new Attribute("type", "list"));
147
148        m_conf.add(elem);
149
150        for (int i = 0; i < values.size(); i++) {
151            Object obj = values.get(i);
152            Element e = new Element("property", "");
153            elem.addElement(e);
154            if (obj instanceof String) {
155                e.addAttribute(new Attribute("value", obj.toString()));
156            } else {
157                // TODO
158                throw new UnsupportedOperationException(
159                        "Complex properties are not supported yet");
160            }
161        }
162
163        return this;
164    }
165

167     * Adds an array property.
168     * @param name the property name

172    public InstantiatedService addProperty(String name, String[] values) {
173        Element elem = new Element("property", "");
174        elem.addAttribute(new Attribute("name", name));
175        elem.addAttribute(new Attribute("type", "array"));
176

179        for (int i = 0; i < values.length; i++) {

218    /**
219     * Adds a map property.
220     * @param name the property name
221     * @param values the property value
222     * @return the current instantiated sub-service
223     */
224    public InstantiatedService addProperty(String name, Map values) {
225        Element elem = new Element("property", "");
226        elem.addAttribute(new Attribute("name", name));
227        elem.addAttribute(new Attribute("type", "map"));
228
229        m_conf.add(elem);
230        Set entries = values.entrySet();
231        Iterator it = entries.iterator();
232        while (it.hasNext()) {
233            Map.Entry entry = (Entry) it.next();
234            Element e = new Element("property", "");
235            elem.addElement(e);
236
237            String n = (String) entry.getKey();
238            Object v = entry.getValue();
239            if (v instanceof String) {
240                e.addAttribute(new Attribute("name", n));
241                e.addAttribute(new Attribute("value", v.toString()));
242            } else {
243                // TODO
244                throw new UnsupportedOperationException(
245                        "Complex properties are not supported yet");
246            }
247        }
248
249        return this;
250    }
251
252    /**
253     * Adds a dictionary property.
254     * @param name the property name
255     * @param values the property value
256     * @return the current instantiated sub-service
257     */
258    public InstantiatedService addProperty(String name, Dictionary values) {
259        Element elem = new Element("property", "");
260        elem.addAttribute(new Attribute("name", name));
261        elem.addAttribute(new Attribute("type", "dictionary"));
262
263        m_conf.add(elem);
264        Enumeration e = values.keys();
265        while (e.hasMoreElements()) {
266            Element el = new Element("property", "");
267            elem.addElement(el);
268
269            String n = (String) e.nextElement();
270            Object v = values.get(n);
271            if (v instanceof String) {
272                el.addAttribute(new Attribute("name", n));
273                el.addAttribute(new Attribute("value", v.toString()));
274            } else {
275                // TODO
276                throw new UnsupportedOperationException(
277                        "Complex properties are not supported yet");
278            }
279        }
280
281        return this;
282    }

320        m_aggregate = agg;
321        return this;
322    }

324    /**
325     * Sets the dependency binding policy.
326     * @param policy the binding policy
327     * @return the current instantiated sub-service

329    public InstantiatedService setBindingPolicy(int policy) {
330        m_policy = policy;
331        return this;
332    }
333
334    /**
335     * Sets the dependency comparator.
336     * @param cmp the comparator class name
337     * @return the current instantiated sub-service
338     */
339    public InstantiatedService setComparator(String cmp) {
340        m_comparator = cmp;
341        return this;
342    }
343
344    /**
345     * Checks dependency configuration validity.
346     */
347    private void ensureValidity() {
348        // Check specification
349        if (m_specification == null) {
350            throw new IllegalStateException(
351                    "The specification of the instantiated service must be set");
352        }
353
354        // Check binding policy.
355        if (!(m_policy == DependencyModel.DYNAMIC_BINDING_POLICY
356                || m_policy == DependencyModel.STATIC_BINDING_POLICY || m_policy == DependencyModel.DYNAMIC_PRIORITY_BINDING_POLICY)) {
357            throw new IllegalStateException("Unknown binding policy : "
358                    + m_policy);
359        }
360    }
