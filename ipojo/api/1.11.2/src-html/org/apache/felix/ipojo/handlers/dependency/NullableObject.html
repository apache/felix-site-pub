019package org.apache.felix.ipojo.handlers.dependency;
020
021import java.lang.reflect.InvocationHandler;
022import java.lang.reflect.Method;
023
024/**
025 * Default nullable object.
026 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
027 */
028public class NullableObject implements InvocationHandler {
029
030    /**
031     * Default boolean value.
032     */
033    private static final Boolean DEFAULT_BOOLEAN = Boolean.FALSE;
034
035    /**
036     * Default byte value.
037     */
038    private static final Byte DEFAULT_BYTE = new Byte((byte) 0);
039
040    /**
041     * Default short value.
042     */
043    private static final Short DEFAULT_SHORT = new Short((short) 0);
044
045    /**
046     * Default integer value.
047     */
048    private static final Integer DEFAULT_INT = new Integer(0);
049
050    /**
051     * Default long value.
052     */
053    private static final Long DEFAULT_LONG = new Long(0);
054
055    /**
056     * Default float value.
057     */
058    private static final Float DEFAULT_FLOAT = new Float(0.0f);
059
060    /**
061     * Default double value.
062     */
063    private static final Double DEFAULT_DOUBLE = new Double(0.0);
064
065    /**
066     * Invokes a method on this null object. The method will return a default
067     * value without doing anything.
068     * @param proxy : proxy object
069     * @param method : invoked method
070     * @param args : arguments.
071     * @return the returned value.
072     * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
073     */
074    public Object invoke(Object proxy, Method method, Object[] args) {
075        Class returnType = method.getReturnType();
076        if (Boolean.TYPE.equals(returnType)) {
077            return DEFAULT_BOOLEAN;
078        } else if (Byte.TYPE.equals(returnType)) {
079            return DEFAULT_BYTE;
080        } else if (Short.TYPE.equals(returnType)) {
081            return DEFAULT_SHORT;
082        } else if (Integer.TYPE.equals(returnType)) {
083            return DEFAULT_INT;
084        } else if (Long.TYPE.equals(returnType)) {
085            return DEFAULT_LONG;
086        } else if (Float.TYPE.equals(returnType)) {
087            return DEFAULT_FLOAT;
088        } else if (Double.TYPE.equals(returnType)) {
089            return DEFAULT_DOUBLE;
090        } else {
091            return null;
092        }
093    }
094}
