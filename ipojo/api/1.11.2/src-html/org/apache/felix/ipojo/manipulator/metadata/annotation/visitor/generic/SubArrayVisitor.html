020package org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic;
021
022import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util.Elements;
023import org.apache.felix.ipojo.metadata.Attribute;
024import org.apache.felix.ipojo.metadata.Element;
025import org.objectweb.asm.AnnotationVisitor;
026import org.objectweb.asm.Type;
027import org.objectweb.asm.commons.EmptyVisitor;
028
029/**
030 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
031 */
032public class SubArrayVisitor extends EmptyVisitor implements AnnotationVisitor {
033
034    /**
035     * Parent element.
036     */
037    private Element m_elem;
038
039    /**
040     * Attribute name.
041     */
042    private String m_name;
043
044    /**
045     * Attribute value.
046     * (accumulator)
047     */
048    private String m_acc;
049
050    /**
051     * Constructor.
052     * @param elem : element element.
053     * @param name : attribute name.
054     */
055    public SubArrayVisitor(Element elem, String name) {
056        m_elem = elem;
057        m_name = name;
058    }
059
060    /**
061     * Visit a 'simple' element of the visited array.
062     * @param name : null
063     * @param value : element value.
064     * @see org.objectweb.asm.commons.EmptyVisitor#visit(String, Object)
065     */
066    public void visit(String name, Object value) {
067        if (m_acc == null) {
068            if (!(value instanceof Type)) {
069                m_acc = "{" + value.toString();
070            } else {
071                // Attributes of type class need a special handling
072                m_acc = "{" + ((Type) value).getClassName();
073            }
074        } else {
075            if (!(value instanceof Type)) {
076                m_acc = m_acc + "," + value.toString();
077            } else {
078                // Attributes of type class need a special handling
079                m_acc = m_acc + "," + ((Type) value).getClassName();
080            }
081        }
082    }
083
084    /**
085     * Visits an enumeration attribute.
086     * @param name the attribute name
087     * @param desc the enumeration descriptor
088     * @param value the attribute value
089     */
090    public void visitEnum(String name, String desc, String value) {
091        if (m_acc == null) {
092            m_acc = "{" + value;
093        } else {
094            m_acc = m_acc + "," + value;
095        }
096    }
097
098
099    /**
100     * Visit an annotation element of the visited array.
101     * @param name : null
102     * @param desc : annotation to visit
103     * @return the visitor which will visit the annotation
104     * @see org.objectweb.asm.commons.EmptyVisitor#visitAnnotation(String, String)
105     */
106    public AnnotationVisitor visitAnnotation(String name, String desc) {
107        // Sub annotations are map to sub-elements
108        Element elem = Elements.buildElement(Type.getType(desc));
109        m_elem.addElement(elem);
110        return new GenericVisitor(elem);
111    }
112
113    /**
114     * End of the visit.
115     * @see org.objectweb.asm.commons.EmptyVisitor#visitEnd()
116     */
117    public void visitEnd() {
118        if (m_acc != null) {
119            // We have analyzed an attribute
120            m_elem.addAttribute(new Attribute(m_name, m_acc + "}"));
121        }
122    }
123
124}
