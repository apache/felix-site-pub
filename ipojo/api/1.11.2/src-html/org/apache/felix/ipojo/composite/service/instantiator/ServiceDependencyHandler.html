044/**
045 * Service Instantiator Class. This handler allows to instantiate service
046 * instance inside the composition.
047 *
048 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
049 */
050public class ServiceDependencyHandler extends CompositeHandler implements DependencyStateListener {
051
052    /**
053     * List of instances to manage.
054     */
055    private List/* <SvcInstance> */m_instances = new ArrayList();
056
057    /**
058     * List of importers.
059     */
060    private List/* <ServiceImporter> */ m_importers = new ArrayList();
061
062    /**
063     * Flag indicating if the handler has already finished the start method.
064     */
065    private boolean m_isStarted;
066
067    /**
068     * The handler description.
069     */
070    private ServiceInstantiatorDescription m_description;
071
072    /**
073     * Source Managers.
074     */
075    private List m_sources;

078    /**
079     * Create a Service instance object form the given Element.
080     * This method parse the given element and configure the service instance object.
081     * @param service : the Element describing the service instance
082     * @param conf : the configuration from the composite instance
083     * @throws ConfigurationException : the service instance cannot be created correctly
084     */
085    private void createServiceInstance(Element service, Dictionary conf) throws ConfigurationException {
086        // Prepare the configuration to append.
087        Properties toAppend = new Properties();
088        Enumeration keys = conf.keys();
089        while(keys.hasMoreElements()) {
090                String key = (String) keys.nextElement();
091            if (! (key.equals("instance.name")
092                    || key.equals("component"))) { // Remove instance.name and component
093                        toAppend.put(key, conf.get(key));
094                }
095        }
096
097        String spec = service.getAttribute("specification");
098        if (spec == null) {
099            throw new ConfigurationException("Malformed service : the specification attribute is mandatory");
100        }
101        String filter = "(&(!(factory.name=" + getCompositeManager().getFactory().getComponentDescription().getName() + "))(factory.state=1))"; // Cannot reinstantiate yourself
102        String givenFilter = service.getAttribute("filter");
103        if (givenFilter != null) {
104            filter = "(&" + filter + givenFilter + ")"; //NOPMD
105        }
106
107        Filter fil;
108        try {
109            fil = getCompositeManager().getGlobalContext().createFilter(filter);
110        } catch (InvalidSyntaxException e) {
111            throw new ConfigurationException("Malformed filter " + filter, e);
112        }
113
114        Properties prop = new Properties();
115        Element[] props = service.getElements("property");
116        for (int k = 0; props != null && k < props.length; k++) {
117            try {
118                InstanceHandler.parseProperty(props[k], prop);
119            } catch (ParseException e) {
120                throw new ConfigurationException("An instance configuration is invalid", e);
121            }
122        }
123
124        Properties instanceConfiguration = new Properties();
125        instanceConfiguration.putAll(prop);
126        instanceConfiguration.putAll(toAppend);
127
128        String aggregate = service.getAttribute("aggregate");
129        boolean agg = aggregate != null && aggregate.equalsIgnoreCase("true");
130
131        String optional = service.getAttribute("optional");
132        boolean opt = optional != null && optional.equalsIgnoreCase("true");
133
134        int policy = DependencyMetadataHelper.getPolicy(service);
135
136        Comparator cmp = DependencyMetadataHelper.getComparator(service, getCompositeManager().getGlobalContext());
137
138        SvcInstance inst = new SvcInstance(this, spec, instanceConfiguration, agg, opt, fil, cmp, policy);
139        m_instances.add(inst);
140
141        String sources = service.getAttribute("context-source");
142        if (sources != null) {
143            SourceManager source = new SourceManager(sources, filter, inst, getCompositeManager());
144            if (m_sources == null) {
145                m_sources = new ArrayList(1);
146            }
147            m_sources.add(source);
148        }
149    }
150
151    /**
152     * Create a Service importer object from the given Element.
153     * This method parse the given element and configure the service importer object.
154     * @param imp : Element describing the import
155     * @param confFilter : instance filter customization
156     * @throws ConfigurationException : the service importer cannot be created correctly
157     */
158    private void createServiceImport(Element imp, Dictionary confFilter) throws ConfigurationException {
159        boolean optional = false;
160        boolean aggregate = false;
161        String specification = imp.getAttribute("specification");
162
163        if (specification == null) {
164            // Malformed import
165            error("Malformed import: the specification attribute is mandatory");
166            throw new ConfigurationException("Malformed import : the specification attribute is mandatory");
167        } else {
168            String opt = imp.getAttribute("optional");
169            optional = opt != null && opt.equalsIgnoreCase("true");
170
171            String agg = imp.getAttribute("aggregate");
172            aggregate = agg != null && agg.equalsIgnoreCase("true");
173
174            String original = "(&(objectClass=" + specification + ")(!(instance.name=" + getCompositeManager().getInstanceName() + ")))"; // Cannot import yourself
175            String filter = original;
176            String givenFilter = imp.getAttribute("filter");
177            if (givenFilter != null) {
178                filter = "(&" + filter + givenFilter + ")"; //NOPMD
179            }
180
181            String identitity = imp.getAttribute("id");
182
183            String scope = imp.getAttribute("scope");
184            BundleContext context = getCompositeManager().getGlobalContext(); // Get the default bundle context.
185            if (scope != null) {
186                if (scope.equalsIgnoreCase("global")) {
187                    context = new PolicyServiceContext(getCompositeManager().getGlobalContext(), getCompositeManager().getParentServiceContext(), PolicyServiceContext.GLOBAL);
188                } else if (scope.equalsIgnoreCase("composite")) {
189                    context = new PolicyServiceContext(getCompositeManager().getGlobalContext(), getCompositeManager().getParentServiceContext(), PolicyServiceContext.LOCAL);
190                } else if (scope.equalsIgnoreCase("composite+global")) {
191                    context = new PolicyServiceContext(getCompositeManager().getGlobalContext(), getCompositeManager().getParentServiceContext(), PolicyServiceContext.LOCAL_AND_GLOBAL);
192                }
193            }
194
195            // Configure instance filter if available
196            if (confFilter != null && identitity != null && confFilter.get(identitity) != null) {
197                filter = "(&" + original + (String) confFilter.get(identitity) + ")";
198            }
199
200            Filter fil = null;
201            if (filter != null) {
202                try {
203                    fil = getCompositeManager().getGlobalContext().createFilter(filter);
204                } catch (InvalidSyntaxException e) {
205                    throw new ConfigurationException("A required filter " + filter + " is malformed", e);
206                }
207            }
208
209            Comparator cmp = DependencyMetadataHelper.getComparator(imp, getCompositeManager().getGlobalContext());
210            Class spec = DependencyMetadataHelper.loadSpecification(specification, getCompositeManager().getGlobalContext());
211            int policy = DependencyMetadataHelper.getPolicy(imp);
212
213            ServiceImporter importer = new ServiceImporter(spec, fil, aggregate, optional, cmp, policy, context, identitity, this);
214            m_importers.add(importer);
215
216            String sources = imp.getAttribute("context-source");
217            if (sources != null) {
218                SourceManager source = new SourceManager(sources, filter, importer, getCompositeManager());
219                if (m_sources == null) {
220                    m_sources = new ArrayList(1);
221                }
222                m_sources.add(source);
223            }
224
225        }
226    }

228    /**
229     * Configure the handler.
230     * @param metadata : the metadata of the component
231     * @param conf : the instance configuration
232     * @throws ConfigurationException : the specification attribute is missing
233     * @see CompositeHandler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
234     */
235    public void configure(Element metadata, Dictionary conf) throws ConfigurationException {
236        Element[] services = metadata.getElements("subservice");
237        // Get instance filters
238        Dictionary confFilter = null;
239        if (conf.get("requires.filters") != null) {
240            confFilter = (Dictionary) conf.get("requires.filters");
241        }
242
243        for (int i = 0; i < services.length; i++) {
244            String action = services[i].getAttribute("action");
245            if (action == null) {
246                throw new ConfigurationException("The action attribute must be set to 'instantiate' or 'import'");
247            } else if ("instantiate".equalsIgnoreCase(action)) {
248                createServiceInstance(services[i], conf);
249            } else if ("import".equalsIgnoreCase(action)) {
250                createServiceImport(services[i], confFilter);
251            } else {
252                throw new ConfigurationException("Unknown action : " + action);
253            }
254        }
255
256        m_description = new ServiceInstantiatorDescription(this, m_instances, m_importers);
257    }
258
259    /**
260     * Start the service instantiator handler.
261     * Start all created service instance.
262     * @see org.apache.felix.ipojo.composite.CompositeHandler#start()
263     */
264    public void start() {
265        for (int i = 0; m_sources != null && i < m_sources.size(); i++) {
266            SourceManager source = (SourceManager) m_sources.get(i);
267            source.start();
268        }
269
270        for (int i = 0; i < m_importers.size(); i++) {
271            ServiceImporter imp = (ServiceImporter) m_importers.get(i);
272            imp.start();
273        }
274
275        for (int i = 0; i < m_instances.size(); i++) {
276            SvcInstance inst = (SvcInstance) m_instances.get(i);
277            inst.start();
278        }
279
280        isHandlerValid();
281        m_isStarted = true;
282    }
283
284    /**
285     * Check the handler validity.
286     * @see org.apache.felix.ipojo.composite.CompositeHandler#isValid()
287     */
288    private void isHandlerValid() {
289        for (int i = 0; i < m_importers.size(); i++) {
290            ServiceImporter imp = (ServiceImporter) m_importers.get(i);
291            if (imp.getState() != DependencyModel.RESOLVED) {
292                setValidity(false);
293                return;
294            }
295        }
296
297        for (int i = 0; i < m_instances.size(); i++) {
298            SvcInstance inst = (SvcInstance) m_instances.get(i);
299            if (inst.getState() != DependencyModel.RESOLVED) {
300                setValidity(false);
301                return;
302            }
303        }
304
305        setValidity(true);
306    }
307
308    /**
309     * Handler stop method.
310     * Stop all created service instance.
311     * @see org.apache.felix.ipojo.composite.CompositeHandler#stop()
312     */
313    public void stop() {
314        for (int i = 0; m_sources != null && i < m_sources.size(); i++) {
315            SourceManager source = (SourceManager) m_sources.get(i);
316            source.stop();
317        }
318
319        for (int i = 0; i < m_instances.size(); i++) {
320            SvcInstance inst = (SvcInstance) m_instances.get(i);
321            inst.stop();
322        }
323
324        for (int i = 0; i < m_importers.size(); i++) {
325            ServiceImporter imp = (ServiceImporter) m_importers.get(i);
326            imp.stop();
327        }
328
329        m_isStarted = false;
330    }

332    /**
333     * State change callback.
334     * This method is used to freeze the set of used provider if the static binding policy is used.
335     * @param newState : the new state of the underlying instance
336     * @see org.apache.felix.ipojo.Handler#stateChanged(int)
337     */
338    public void stateChanged(int newState) {
339        // If we are becoming valid and started, check if we need to freeze importers.
340        if (m_isStarted && newState == ComponentInstance.VALID) {
341            for (int i = 0; i < m_importers.size(); i++) {
342                ServiceImporter imp = (ServiceImporter) m_importers.get(i);
343                if (imp.getBindingPolicy() == DependencyModel.STATIC_BINDING_POLICY) {
344                    imp.freeze();
345                }
346            }
347
348            for (int i = 0; i < m_instances.size(); i++) {
349                SvcInstance imp = (SvcInstance) m_instances.get(i);
350                if (imp.getBindingPolicy() == DependencyModel.STATIC_BINDING_POLICY) {
351                    imp.freeze();
352                }
353            }
354        }
355    }
356
357    /**
358     * An service instance becomes valid.
359     * @param dep : dependency becoming valid.
360     */
361    public void validate(DependencyModel dep) {
362        if (!getValidity()) {
363            isHandlerValid();
364        }
365    }
366
367    /**
368     * A service instance becomes invalid.
369     * @param dep : dependency becoming valid.
370     */
371    public void invalidate(DependencyModel dep) {
372        if (getValidity()) {
373            isHandlerValid();
374        }
375    }
