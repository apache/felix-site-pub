019package org.apache.felix.ipojo.annotations;
020
021import java.lang.annotation.ElementType;
022import java.lang.annotation.Target;
023
024/**
025 * This annotation declares that the component instances will provide a service.
026 *
027 * <pre>
028 *     {@linkplain org.apache.felix.ipojo.annotations.Component @Component}
029 *     {@code @Provides}
030 *     public class MyComponent implements Service {
031 *         // ...
032 *     }
033 * </pre>
034 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
035*/
036@Target({ElementType.TYPE, ElementType.ANNOTATION_TYPE})
037public @interface Provides {
038
039    /**
040     * Set the provided specifications.
041     * It can be used to force the exposed service to use an implementation class
042     * (not an interface) as specification.
043     * Default : all implemented interfaces
044     *
045     * <pre>
046     *     {@linkplain org.apache.felix.ipojo.annotations.Component @Component}
047     *     {@code @Provides(specifications = AbsComponent.class)}
048     *     public class MyComponent extends AbsComponent {
049     *         // ...
050     *     }
051     * </pre>
052     */
053    Class[] specifications() default { };
054
055    /**
056     * Set the service object creation strategy.
057     * Multiple values are possible: {@literal SINGLETON}, {@literal SERVICE},
058     * {@literal METHOD}, {@literal INSTANCE} or the strategy fully qualified class name:
059     * <ul>
060     *     <li>{@literal SINGLETON}: Default strategy</li>
061     *     <li>{@literal SERVICE}: OSGi Service Factory style, 1 POJO instance per consumer bundle</li>
062     *     <li>{@literal METHOD}: Delegates the creation to the factory-method of the component, method will be called every time the service reference is get.</li>
063     *     <li>{@literal INSTANCE}: Creates one service object per requiring instance</li>
064     *     <li>Any other value is interpreted as the qualified name of a {@code CreationStrategy} implementation</li>
065     * </ul>
066     */
067    String strategy() default "SINGLETON";
068
069    /**
070     * Allows adding static properties to the service.
071     * Nested properties are static service properties, so <b>must</b> contain the name,
072     * value and type as they are not attached to a field (cannot discover type through
073     * introspection).
074     * The array contains {@link StaticServiceProperty} elements.
075     * Default : No service properties
076
077     * <pre>
078     *     {@linkplain org.apache.felix.ipojo.annotations.Component @Component}
079     *     {@code @Provides}(
080     *         properties = {
081     *             {@code @StaticServiceProperty}(name = "size", type = "int", value = "5"),
082     *             {@code @StaticServiceProperty}(name = "name", type = "java.lang.String", value = "OSGi")
083     *         }
084     *     )
085     *     public class MyComponent implements Service {
086     *         // ...
087     *     }
088     * </pre>
089
090     */
091    StaticServiceProperty[] properties() default {};
092}
