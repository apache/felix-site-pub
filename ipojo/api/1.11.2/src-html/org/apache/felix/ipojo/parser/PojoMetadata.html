019package org.apache.felix.ipojo.parser;
020
021import org.apache.felix.ipojo.ConfigurationException;
022import org.apache.felix.ipojo.metadata.Element;
023
024import java.util.*;
025
026/**
027 * Manipulation Metadata allows getting information about the implementation class
028 * without using reflection such as implemented interfaces, super class,
029 * methods and fields.
030 * This method allows getting object to register {@link org.apache.felix.ipojo.FieldInterceptor} and
031 * {@link org.apache.felix.ipojo.MethodInterceptor}.
032 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
033 */
034public class PojoMetadata {
035
036    /**
037     * The list of implemented interfaces.
038     */
039    private String[] m_interfaces = new String[0];
040
041    /**
042     * The list of fields.
043     */
044    private FieldMetadata[] m_fields = new FieldMetadata[0];
045
046    /**
047     * The list of methods.
048     */
049    private List<MethodMetadata> m_methods = new ArrayList<MethodMetadata>();
050
051    /**
052     * The Super class (if <code>null</code> for {@link Object}).
053     */
054    private String m_super;
055
056    /**
057     * The manipulated class name.
058     */
059    private String m_className;
060
061    /**
062     * The inner classes and their methods.
063     */
064    private Map<String, List<MethodMetadata>> m_innerClasses = new HashMap<String, List<MethodMetadata>>();
065
066
067    /**
068     * Creates Pojo metadata.
069     * Manipulation Metadata object are created from component type metadata by
070     * parsing manipulation metadata.
071     * @param metadata the component type metadata
072     * @throws ConfigurationException if the manipulation metadata cannot be found
073     */
074    public PojoMetadata(Element metadata) throws ConfigurationException {
075        Element[] elems = metadata.getElements("manipulation", "");
076        if (elems == null) {
077            throw new ConfigurationException("The component " + metadata.getAttribute("classname") + " has no manipulation metadata");
078        }
079        Element manip = elems[0];
080        m_className = manip.getAttribute("classname");
081        m_super = manip.getAttribute("super");
082
083        Element[] fields = manip.getElements("field");
084        if (fields != null) {
085            for (Element field : fields) {
086                addField(new FieldMetadata(field));
087            }
088        }
089
090        Element[] methods = manip.getElements("method");
091        if (methods != null) {
092            for (Element method : methods) {
093                m_methods.add(new MethodMetadata(method));
094            }
095        }
096
097        Element[] itfs = manip.getElements("interface");
098        if (itfs != null) {
099            for (Element itf : itfs) {
100                addInterface(itf.getAttribute("name"));
101            }
102        }
103
104        Element[] inners = manip.getElements("inner");
105        if (inners != null) {
106            for (Element inner : inners) {
107                String name = inner.getAttribute("name");
108                List<MethodMetadata> list = m_innerClasses.get(name);
109                if (list == null) {
110                    list = new ArrayList<MethodMetadata>();
111                    m_innerClasses.put(name, list);
112                }
113                methods = inner.getElements("method");
114                if (methods != null) {
115                    for (Element m : methods) {
116                        list.add(new MethodMetadata(m));
117                    }
118                }
119            }
120        }
121    }
122
123    public MethodMetadata[] getMethods() { return m_methods.toArray(new MethodMetadata[m_methods.size()]); }
124
125    public FieldMetadata[] getFields() { return m_fields; }
126
127    public String[] getInterfaces() { return m_interfaces; }
128
129    public String getClassName() { return m_className; }
130
131    /**
132     * Gets the inner classes from the manipulated class
133     * @return the list of the inner class names.
134     */
135    public String[] getInnerClasses() {
136        Set<String> classes = m_innerClasses.keySet();
137        return classes.toArray(new String[classes.size()]);
138    }
139
140    /**
141     * Gets the methods from the given inner class.
142     * @param inner the inner class name
143     * @return the list of method, empty if none.
144     */
145    public MethodMetadata[] getMethodsFromInnerClass(String inner) {
146        List<MethodMetadata> methods = m_innerClasses.get(inner);
147        if (inner != null) {
148            return methods.toArray(new MethodMetadata[methods.size()]);
149        } else {
150            return new MethodMetadata[0];
151        }
152    }
153
154    /**
155     * Gets the field metadata for the given name.
156     * @param name : the name of the field
157     * @return the corresponding field metadata or <code>null</code> if not found
158     */
159    public FieldMetadata getField(String name) {
160        for (int i = 0; i < m_fields.length; i++) {
161            if (m_fields[i].getFieldName().equalsIgnoreCase(name)) { return m_fields[i]; }
162        }
163        return null;
164    }
165
166    /**
167     * Gets the field metadata for the given name and type.
168     * @param name : the name of the field
169     * @param type : the type of the field
170     * @return the corresponding field metadata or <code>null</code> if not found
171     */
172    public FieldMetadata getField(String name, String type) {
173        for (int i = 0; i < m_fields.length; i++) {
174            if (m_fields[i].getFieldName().equalsIgnoreCase(name) && m_fields[i].getFieldType().equalsIgnoreCase(type)) { return m_fields[i]; }
175        }
176        return null;
177    }
178
179    /**
180     * Checks if the given interface name is implemented.
181     * This methods checks on interface directly implemented
182     * by the implementation class.
183     * @param itf the interface to check.
184     * @return <code>true</code> if the implementation class implements
185     * the given interface.
186     */
187    public boolean isInterfaceImplemented(String itf) {
188        for (int i = 0; i < m_interfaces.length; i++) {
189            if (m_interfaces[i].equals(itf)) { return true; }
190        }
191        return false;
192    }
193
194    /**
195     * Gets the MethodMetadata corresponding to the method
196     * (contained in the implementation class) with
197     * the given name.
198     * If several methods match, the first one is returned.
199     * @param name the name of the method to find.
200     * @return the method metadata object or <code>null</code> if not found
201     */
202    public MethodMetadata getMethod(String name) {
203        for (MethodMetadata metadata : m_methods) {
204            if (metadata.getMethodName().equalsIgnoreCase(name)) { return metadata; }
205        }
206        return null;
207    }
208
209    /**
210     * Gets the MethodMetadata list corresponding to the method
211     * (contained in the implementation class) to given name.
212     * All methods contained in the implementation class matching
213     * with the name are in the returned list.
214     * @param name the name of the method to look for.
215     * @return the Method Metadata array or an empty array if not found
216     */
217    public MethodMetadata[] getMethods(String name) {
218        List<MethodMetadata> list = new ArrayList<MethodMetadata>();
219        for (MethodMetadata metadata : m_methods) {
220            if (metadata.getMethodName().equalsIgnoreCase(name)) {
221                list.add(metadata);
222            }
223        }
224        return list.toArray(new MethodMetadata[list.size()]);
225    }
226
227    /**
228     * Gets the MethodMetadata list corresponding to the constructors
229     * (contained in the implementation class).
230     * @return the Method Metadata array or an empty array if not found
231     */
232    public MethodMetadata[] getConstructors() {
233        return getMethods("$init");
234    }
235
236    /**
237     * Gets the MethodMetadata corresponding to the method
238     * (contained in the implementation class) to given name
239     * and argument types.
240     * @param name the name of the method to look for.
241     * @param types the array of the argument types of the method
242     * @return the Method Metadata or <code>null</code> if not found
243     */
244    public MethodMetadata getMethod(String name, String[] types) {
245        for (MethodMetadata metadata : m_methods) {
246            if (metadata.getMethodName().equalsIgnoreCase(name) && metadata.getMethodArguments().length == types.length) {
247                int argIndex = 0;
248                for (; argIndex < types.length; argIndex++) {
249                    if (! types[argIndex].equals(metadata.getMethodArguments()[argIndex])) {
250                        break;
251                    }
252                }
253                if (argIndex == types.length) { return metadata; } // No mismatch detected.
254            }
255        }
256        return null;
257    }
258
259    /**
260     * Gets the constructor corresponding to the given argument types.
261     * @param types the argument types
262     * @return the matching constructor or <code>null</code> if not found.
263     */
264    public MethodMetadata getConstructor(String[] types) {
265        return getMethod("$init", types); // Constructors are named $init in the manipulation metadata
266    }
267
268     /**
269     * Adds a field to the list.
270     * This method is used during the creation of the {@link PojoMetadata}
271     * object.
272     * @param field the Field Metadata to add.
273     */
274    private void addField(FieldMetadata field) {
275        if (m_fields.length > 0) {
276            FieldMetadata[] newInstances = new FieldMetadata[m_fields.length + 1];
277            System.arraycopy(m_fields, 0, newInstances, 0, m_fields.length);
278            newInstances[m_fields.length] = field;
279            m_fields = newInstances;
280        } else {
281            m_fields = new FieldMetadata[] { field };
282        }
283    }
284
285    /**
286     * Adds the interface to the list.
287     * This method is used during the creation of the {@link PojoMetadata}
288     * object.
289     * @param itf the interface name to add.
290     */
291    private void addInterface(String itf) {
292        if (m_interfaces.length > 0) {
293            String[] newInstances = new String[m_interfaces.length + 1];
294            System.arraycopy(m_interfaces, 0, newInstances, 0, m_interfaces.length);
295            newInstances[m_interfaces.length] = itf;
296            m_interfaces = newInstances;
297        } else {
298            m_interfaces = new String[] { itf };
299        }
300    }
301
302    public String getSuperClass() {
303        return m_super;
304    }
