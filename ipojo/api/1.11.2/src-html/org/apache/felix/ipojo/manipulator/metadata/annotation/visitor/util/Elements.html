020package org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.util;
021
022import static java.lang.String.format;
023
024import org.apache.felix.ipojo.manipulator.metadata.annotation.ComponentWorkbench;
025import org.apache.felix.ipojo.metadata.Element;
026import org.objectweb.asm.Type;
027
028/**
029 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
030 */
031public class Elements {
032
033    /**
034     * Build the {@link Element} object from the given descriptor.
035     * It splits the annotation's classname in 2 parts (up to the last '.')
036     * first part (package's name) becomes the Element's namespace, and second
037     * part (class simple name) becomes the Element's name.
038     * @param type annotation descriptor
039     * @return the new element
040     */
041    public static Element buildElement(Type type) {
042        String name = type.getClassName();
043        int index = name.lastIndexOf('.');
044        String local = name.substring(index + 1);
045        String namespace = name.substring(0, index);
046        return buildElement(namespace, local);
047    }
048
049    /**
050     * Build an {@link Element} using the provided namespace and local name.
051     */
052    public static Element buildElement(final String namespace, final String name) {
053        return new Element(name, namespace);
054    }
055
056    /**
057     * Build an {@link Element} using the provided binding information.
058     * Expected format is {@literal [namespace:]name} (eg: {@literal com.acme:foo} or {@literal foo} if namespace
059     * by default -org.apache.felix.ipojo- has to be used).
060     * Notice that the ':' character usage in namespace or name part may lead to unexpected results.
061     * In that case, the @HandlerBinding(namespace = "urn:my:namespace", value = "foo:handler") usage is preferred.
062     * @param binding the condensed element name
063     * @return the new element
064     */
065    public static Element buildElement(String binding) {
066        String[] split = binding.split(":");
067        if (split.length == 1) {
068            return buildElement("", binding);
069        }
070        if (split.length > 2) {
071            throw new IllegalArgumentException(
072                    format("@HandlerBinding(\"%s\") is invalid: only 1 ':' char is authorized, please" +
073                           " use the @HandlerBinding(namespace=\"...\", value=\"...\") form instead.",
074                           binding)
075            );
076        }
077        return buildElement(split[0], split[1]);
078    }
079
080    /**
081     * Return the Element named {@literal properties}, creates one if missing.
082     * @param workbench source for search
083     * @return the {@literal properties} Element (never null).
084     */
085    public static Element getPropertiesElement(ComponentWorkbench workbench) {
086        Element properties = workbench.getIds().get("properties");
087        if (properties == null) {
088            properties = buildElement("", "properties");
089            workbench.getIds().put("properties", properties);
090            workbench.getElements().put(properties, null);
091        }
092        return properties;
093    }
094}
