020package org.apache.felix.ipojo.manipulator.store;
021
022import org.apache.felix.ipojo.manipulator.ResourceStore;
023import org.apache.felix.ipojo.manipulator.ResourceVisitor;
024import org.apache.felix.ipojo.manipulator.store.mapper.IdentityResourceMapper;
025import org.apache.felix.ipojo.manipulator.util.Metadatas;
026import org.apache.felix.ipojo.manipulator.util.Streams;
027import org.apache.felix.ipojo.metadata.Element;
028
029import java.io.File;
030import java.io.FileOutputStream;
031import java.io.IOException;
032import java.io.InputStream;
033import java.util.Collections;
034import java.util.List;
035import java.util.Map;
036import java.util.TreeMap;
037import java.util.jar.JarEntry;
038import java.util.jar.JarFile;
039import java.util.jar.JarOutputStream;
040import java.util.jar.Manifest;
041import java.util.zip.ZipEntry;
042
043/**
044 * A {@link JarFileResourceStore} knows how to read and write
045 * resources from (to respectively) a Jar File.
046 *
047 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
048 */
049public class JarFileResourceStore implements ResourceStore {
050
051    /**
052     * Source Jar.
053     */
054    private JarFile m_source;
055
056    /**
057     * Target File.
058     */
059    private File m_target;
060
061    /**
062     * Modified resources.
063     */
064    private Map<String, byte[]> m_content;
065
066    /**
067     * Resource Mapper.
068     */
069    private ResourceMapper m_mapper = new IdentityResourceMapper();
070
071    /**
072     * The builder of the updated manifest.
073     */
074    private ManifestBuilder m_manifestBuilder;
075
076    /**
077     * Original manifest to be updated.
078     */
079    private Manifest m_manifest;
080
081    /**
082     * Construct a {@link JarFileResourceStore} wrapping the given original bundle,
083     * and configured to output in the given target file.
084     * @param source original Bundle
085     * @param target File where the updated Bundle will be outputted
086     * @throws IOException if there is an error retrieving the Manifest from the original JarFile
087     */
088    public JarFileResourceStore(JarFile source, File target) throws IOException {
089        m_source = source;
090        m_target = target;
091
092        // TODO ensure File is not null and not an existing file/directory
093        this.m_target = target;
094        if (source != null) {
095            m_manifest = source.getManifest();
096        } else {
097            m_manifest = new Manifest();
098        }
099        m_content = new TreeMap<String, byte[]>();
100    }
101
102    public void setResourceMapper(ResourceMapper mapper) {
103        this.m_mapper = mapper;
104    }
105
106    public void setManifestBuilder(ManifestBuilder manifestBuilder) {
107        this.m_manifestBuilder = manifestBuilder;
108    }
109
110    public void setManifest(Manifest manifest) {
111        this.m_manifest = manifest;
112    }
113
114    public byte[] read(String path) throws IOException {
115        ZipEntry entry = m_source.getEntry(getInternalPath(path));
116        if (entry == null) {
117            throw new IOException("Jar Entry is not found for class " + path + ".");
118        }
119        return Streams.readBytes(m_source.getInputStream(entry));
120    }
121
122    private String getInternalPath(String path) {
123        return m_mapper.internalize(path);
124    }
125
126    public void accept(ResourceVisitor visitor) {
127        List<JarEntry> entries = Collections.list(m_source.entries());
128        for (JarEntry entry : entries) {
129            String name = entry.getName();
130            if (!name.endsWith("/")) {
131                // Do not visit directories
132                visitor.visit(getExternalName(entry.getName()));
133            }
134        }
135    }
136
137    private String getExternalName(String path) {
138        return m_mapper.externalize(path);
139    }
140
141    public void open() throws IOException {
142        // Nothing to do
143    }
144
145    public void writeMetadata(Element metadata) {
146        m_manifestBuilder.addMetada(Collections.singletonList(metadata));
147        m_manifestBuilder.addReferredPackage(Metadatas.findReferredPackages(metadata));
148    }
149
150    public void write(String resourcePath, byte[] resource) throws IOException {
151        this.m_content.put(getInternalPath(resourcePath), resource);
152    }
153
154    public void close() throws IOException {
155
156        // Update the manifest
157        Manifest updated = m_manifestBuilder.build(m_manifest);
158
159        // Create a new Jar file
160        FileOutputStream fos = new FileOutputStream(m_target);
161        JarOutputStream jos = new JarOutputStream(fos, updated);
162
163        try {
164            // Copy classes and resources
165            List<JarEntry> entries = Collections.list(m_source.entries());
166            for (JarEntry entry : entries) {
167
168                // Ignore some entries (MANIFEST, ...)
169                if (isIgnored(entry)) {
170                    continue;
171                }
172
173                if (isUpdated(entry)) {
174                    // Write newer/updated resource (manipulated classes, ...)
175
176                    JarEntry je = new JarEntry(entry.getName());
177                    byte[] data = m_content.get(getInternalPath(entry.getName()));
178                    jos.putNextEntry(je);
179                    jos.write(data);
180                    jos.closeEntry();
181                } else {
182                    // Copy the resource as-is
183
184                    jos.putNextEntry(entry);
185                    InputStream is = m_source.getInputStream(entry);
186                    try {
187                        Streams.transfer(is, jos);
188                    } finally {
189                        Streams.close(is);
190                    }
191
192                    jos.closeEntry();
193
194                }
195            }
196        } finally {
197            try {
198                m_source.close();
199            } catch (IOException e) {
200                // Ignored
201            }
202            Streams.close(jos, fos);
203
204        }
205    }
206
207    private boolean isUpdated(JarEntry entry) {
208        // Check if that class was manipulated
209        if (entry.getName().endsWith(".class")) {
210            // Need to map this into an external+normalized path
211            String cleaned = getExternalName(entry.getName());
212            return m_content.containsKey(cleaned);
213        } else {
214            return false;
215        }
216    }
217
218    private boolean isIgnored(JarEntry entry) {
219        return "META-INF/MANIFEST.MF".equals(entry.getName());
220    }
221}
