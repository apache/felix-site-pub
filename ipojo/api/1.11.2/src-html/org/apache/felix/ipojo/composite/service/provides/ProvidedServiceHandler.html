090    /**
091     * Initialize the component type.
092     * @param desc : component type description to populate.
093     * @param metadata : component type metadata.
094     * @throws ConfigurationException : metadata are incorrect.
095     * @see org.apache.felix.ipojo.Handler#initializeComponentFactory(org.apache.felix.ipojo.architecture.ComponentTypeDescription, org.apache.felix.ipojo.metadata.Element)
096     */
097    public void initializeComponentFactory(ComponentTypeDescription desc, Element metadata) throws ConfigurationException {
098        Element[] provides = metadata.getElements("provides");
099        for (int i = 0; i < provides.length; i++) {
100            String action = provides[i].getAttribute("action");
101            if (action == null) {
102                throw new ConfigurationException("Invalid composition service providing : no specified action");
103            } else if (action.equalsIgnoreCase("implement")) {
104                String spec = provides[i].getAttribute("specification");
105                if (spec == null) {
106                    throw new ConfigurationException("Malformed provides : the specification attribute is mandatory");
107                } else {
108                    desc.addProvidedServiceSpecification(spec);
109                }
110            } else if (action.equalsIgnoreCase("export")) {
111                String spec = provides[i].getAttribute("specification");
112                if (spec == null) {
113                    throw new ConfigurationException("Malformed exports - Missing the specification attribute");
114                } else {
115                    desc.addProvidedServiceSpecification(spec);
116                }
117            } else {
118                throw new ConfigurationException("Invalid composition service providing : unknown action " + action);
119            }
120        }
121    }

123    /**
124     * Configure the handler.
125     * @param metadata : the metadata of the component
126     * @param configuration : the instance configuration
127     * @throws ConfigurationException  : the exporter cannot be created
128     * @see CompositeHandler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
129     */
130    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
131        m_context = getCompositeManager().getContext();
132
133        Element[] provides = metadata.getElements("provides", "");
134        for (int i = 0; i < provides.length; i++) {
135            String action = provides[i].getAttribute("action");
136            if (action.equalsIgnoreCase("implement")) {
137                ProvidedService svc = new ProvidedService(this, provides[i], Integer.toString(i));
138                m_managedServices.add(svc);
139            } else if (action.equalsIgnoreCase("export")) {
140                boolean optional = false;
141                boolean aggregate = false;
142                String specification = provides[i].getAttribute("specification");
143
144                String filter = "(objectClass=" + specification + ")";
145
146                String opt = provides[i].getAttribute("optional");
147                optional = opt != null && opt.equalsIgnoreCase("true");
148
149                String agg = provides[i].getAttribute("aggregate");
150                aggregate = agg != null && agg.equalsIgnoreCase("true");
151
152                String givenFilter = provides[i].getAttribute("filter");
153                if (givenFilter != null) {
154                    filter = "(&" + filter + givenFilter + ")"; //NOPMD
155                }
156
157                Filter fil = null;
158                try {
159                    fil = m_context.createFilter(filter);
160                } catch (InvalidSyntaxException e) {
161                    throw new ConfigurationException("An exporter filter is invalid " + filter, e);
162                }
163
164                Comparator cmp = DependencyMetadataHelper.getComparator(provides[i], m_context);
165                int policy = DependencyMetadataHelper.getPolicy(provides[i]);
166                Class spec = DependencyMetadataHelper.loadSpecification(specification, m_context);
167
168                ServiceExporter imp = new ServiceExporter(spec, fil, aggregate, optional, cmp, policy, getCompositeManager().getServiceContext(), m_context, this, getCompositeManager());
169                m_exporters.add(imp);
170            } // Others case cannot happen. The test was already made during the factory initialization.
171        }
172        
173        m_description = new ProvidedServiceHandlerDescription(this, m_managedServices, m_exporters);
174
175    }

177    /**
178     * Start method.
179     * Start all managed provided service.
180     * @see org.apache.felix.ipojo.composite.CompositeHandler#start()
181     */
182    public void start() {
183        // Compute imports and instances
184        computeAvailableServices();
185        computeAvailableTypes();
186
187        for (int i = 0; i < m_managedServices.size(); i++) {
188            ProvidedService svc = (ProvidedService) m_managedServices.get(i);
189            try {
190                checkServiceSpecification(svc);
191                svc.start();
192            } catch (CompositionException e) {
193                error("Cannot start the provided service handler", e);
194                setValidity(false);
195                return;
196            }
197        }
198
199        for (int i = 0; i < m_exporters.size(); i++) {
200            ServiceExporter exp = (ServiceExporter) m_exporters.get(i);
201            exp.start();
202        }
203
204        isHandlerValid();
205    }

207    /**
208     * Stop method.
209     * Stop all managed provided service.
210     * @see org.apache.felix.ipojo.composite.CompositeHandler#stop()
211     */
212    public void stop() {
213        for (int i = 0; i < m_managedServices.size(); i++) {
214            ProvidedService svc = (ProvidedService) m_managedServices.get(i);
215            svc.stop();
216        }
217
218        for (int i = 0; i < m_exporters.size(); i++) {
219            ServiceExporter exp = (ServiceExporter) m_exporters.get(i);
220            exp.stop();
221        }
222    }

224    /**
225     * Check the handler validity.
226     * @see org.apache.felix.ipojo.composite.CompositeHandler#isValid()
227     */
228    private void isHandlerValid() {
229        for (int i = 0; i < m_exporters.size(); i++) {
230            ServiceExporter exp = (ServiceExporter) m_exporters.get(i);
231            if (exp.getState() != DependencyModel.RESOLVED) {
232                setValidity(false);
233                return;
234            }
235        }
236
237        setValidity(true);
238    }

240    /**
241     * Handler state changed.
242     * @param state : the new instance state.
243     * @see CompositeHandler#stateChanged(int)
244     */
245    public void stateChanged(int state) {
246        if (state == ComponentInstance.INVALID) {
247            for (int i = 0; i < m_managedServices.size(); i++) {
248                ProvidedService svc = (ProvidedService) m_managedServices.get(i);
249                svc.unregister();
250            }
251            return;
252        }
253
254        // If the new state is VALID => register all the services
255        if (state == ComponentInstance.VALID) {
256            for (int i = 0; i < m_managedServices.size(); i++) {
257                ProvidedService svc = (ProvidedService) m_managedServices.get(i);
258                svc.register();
259            }
260            return;
261        }
262    }

269    public void invalidate(DependencyModel exporter) {

271        if (getValidity()) {
272            setValidity(false);
273        }

279     * 
280     * @param exporter : the implicated exporter.
281     */

283        // An import becomes valid
284        if (!getValidity()) {

286        }

297    /**
298     * Build the list of available specifications.
299     */
300    private void computeAvailableServices() {
301        // Get instantiated services :
302        ServiceDependencyHandler handler = (ServiceDependencyHandler) getHandler(HandlerFactory.IPOJO_NAMESPACE + ":subservice");
303
304        for (int i = 0; handler != null && i < handler.getInstances().size(); i++) {
305            SvcInstance svc = (SvcInstance) handler.getInstances().get(i);
306            String itf = svc.getServiceSpecification();
307            boolean agg = svc.isAggregate();
308            boolean opt = svc.isOptional();
309
310            SpecificationMetadata specMeta = new SpecificationMetadata(itf, m_context, agg, opt, this);
311            m_services.add(specMeta);
312        }
313
314        for (int i = 0; handler != null && i < handler.getRequirements().size(); i++) {
315            ServiceImporter imp = (ServiceImporter) handler.getRequirements().get(i);
316            String itf = imp.getSpecification().getName();
317            boolean agg = imp.isAggregate();
318            boolean opt = imp.isOptional();
319
320            SpecificationMetadata specMeta = new SpecificationMetadata(itf, m_context, agg, opt, this);
321            m_services.add(specMeta);
322        }
323    }

325    /**
326     * Check composite requirement against service specification requirement is available.
327     * @param svc : the provided service to check
328     * @throws CompositionException : occurs if the specification field of the service specification cannot be analyzed correctly.
329     */
330    private void checkServiceSpecification(ProvidedService svc) throws CompositionException {
331        try {
332            Class spec = m_context.getBundle().loadClass(svc.getSpecification());
333            Field specField = spec.getField("specification");
334            Object object = specField.get(null);
335            if (object instanceof String) {
336                Element specification = ManifestMetadataParser.parse((String) object);
337                Element[] reqs = specification.getElements("requires");
338                for (int j = 0; reqs != null && j < reqs.length; j++) {
339                    ServiceImporter imp = getAttachedRequirement(reqs[j]);
340                    if (imp != null) {
341                        // Fix service-level dependency flag
342                        imp.setServiceLevelDependency();
343                    }
344                    checkRequirement(imp, reqs[j]);
345                }
346            } else {
347                error("[" + getCompositeManager().getInstanceName() + "] The specification field of the service specification " + svc.getSpecification() + " needs to be a String");
348                throw new CompositionException("Service Specification checking failed : The specification field of the service specification " + svc.getSpecification() + " needs to be a String");
349            }
350        } catch (NoSuchFieldException e) {
351            return; // No specification field
352        } catch (ClassNotFoundException e) {
353            error("[" + getCompositeManager().getInstanceName() + "] The service specification " + svc.getSpecification() + " cannot be load");
354            throw new CompositionException("The service specification " + svc.getSpecification() + " cannot be loaded", e);
355        } catch (IllegalArgumentException e) {
356            error("[" + getCompositeManager().getInstanceName() + "] The field 'specification' of the service specification " + svc.getSpecification() + " is not accessible : " + e.getMessage());
357            throw new CompositionException("The field 'specification' of the service specification " + svc.getSpecification() + " is not accessible", e);
358        } catch (IllegalAccessException e) {
359            error("[" + getCompositeManager().getInstanceName() + "] The field 'specification' of the service specification " + svc.getSpecification() + " is not accessible : " + e.getMessage());
360            throw new CompositionException("The field 'specification' of the service specification " + svc.getSpecification() + " is not accessible", e);
361        } catch (ParseException e) {
362            error("[" + getCompositeManager().getInstanceName() + "] The field 'specification' of the service specification " + svc.getSpecification() + " does not contain a valid String : " + e.getMessage());
363            throw new CompositionException("The field 'specification' of the service specification " + svc.getSpecification() + " does not contain a valid String", e);
364        }
365    }

367    /**
368     * Look for the implementation (i.e. composite) requirement for the given service-level requirement metadata.
369     * @param element : the service-level requirement metadata
370     * @return the ServiceImporter object, null if not found or if the DependencyHandler is not plugged to the instance
371     */
372    private ServiceImporter getAttachedRequirement(Element element) {
373        ServiceDependencyHandler handler = (ServiceDependencyHandler) getHandler(HandlerFactory.IPOJO_NAMESPACE + ":subservice");
374        if (handler == null) { return null; }
375
376        String identity = element.getAttribute("id");
377        if (identity != null) {
378            // Look for dependency Id
379            for (int i = 0; i < handler.getRequirements().size(); i++) {
380                ServiceImporter imp = (ServiceImporter) handler.getRequirements().get(i);
381                if (imp.getId().equals(identity)) { return imp; }
382            }
383        }
384
385        // If not found or no id, look for a dependency with the same specification
386        String requirement = element.getAttribute("specification");
387        for (int i = 0; i < handler.getRequirements().size(); i++) {
388            ServiceImporter imp = (ServiceImporter) handler.getRequirements().get(i);
389            if (imp.getId().equals(requirement) || imp.getSpecification().getName().equals(requirement)) { return imp; }
390        }
391        return null;
392    }

394    /**
395     * Check the correctness of the composite requirement against the service level dependency.
396     * @param imp : requirement to check
397     * @param elem : service-level dependency metadata
398     * @throws CompositionException : occurs if the requirement does not match with service-level specification requirement
399     */
400    private void checkRequirement(ServiceImporter imp, Element elem) throws CompositionException {
401        String optional = elem.getAttribute("optional");
402        boolean opt = optional != null && optional.equalsIgnoreCase("true");
403
404        String aggregate = elem.getAttribute("aggregate");
405        boolean agg = aggregate != null && aggregate.equalsIgnoreCase("true");
406
407        if (imp == null) {
408            // Add the missing requirement
409            ServiceDependencyHandler handler = (ServiceDependencyHandler) getHandler(HandlerFactory.IPOJO_NAMESPACE + ":subservice");
410            if (handler == null) {
411                // Look for the ServiceDependencyHandler factory
412                HandlerManager handlerManager = null;
413                try {
414                    ServiceReference[] refs = m_context.getServiceReferences(Factory.class.getName(), "(&(handler.name=subservice)(handler.namespace=" + HandlerFactory.IPOJO_NAMESPACE + ")(handler.type=composite))");
415                    Factory factory = (Factory) m_context.getService(refs[0]);
416                    handlerManager = (HandlerManager) factory.createComponentInstance(null, getCompositeManager().getServiceContext());
417                } catch (InvalidSyntaxException e) {
418                    // Should not happen
419                } catch (UnacceptableConfiguration e) {
420                    // Should not happen
421                } catch (MissingHandlerException e) {
422                    // Should not happen
423                } catch (ConfigurationException e) {
424                    // Should not happen
425                }
426                
427                // Add the required handler 
428                try {
429                    handlerManager.init(getCompositeManager(), new Element("composite", ""), new Properties());
430                } catch (ConfigurationException e) {
431                    error("Internal error : cannot configure the Import Handler : " + e.getMessage());
432                    throw new CompositionException("Internal error : cannot configure the Import Handler", e);
433                }
434                handler = (ServiceDependencyHandler) handlerManager.getHandler();
435                getCompositeManager().addCompositeHandler(handlerManager);
436            }
437
438            String spec = elem.getAttribute("specification");
439            String filter = "(&(objectClass=" + spec + ")(!(instance.name=" + getCompositeManager().getInstanceName() + ")))"; // Cannot import yourself
440            String givenFilter = elem.getAttribute("filter");
441            if (givenFilter != null) {
442                filter = "(&" + filter + givenFilter + ")"; //NOPMD
443            }
444
445            BundleContext context = new PolicyServiceContext(getCompositeManager().getGlobalContext(), getCompositeManager().getParentServiceContext(), PolicyServiceContext.GLOBAL);
446
447            Filter fil = null;
448            try {
449                fil = getCompositeManager().getGlobalContext().createFilter(filter);
450            } catch (InvalidSyntaxException e) {
451                throw new CompositionException("A required filter " + filter + " is malformed", e);
452            }
453
454            Class specToImport = null;
455            try {
456                specToImport = getCompositeManager().getGlobalContext().getBundle().loadClass(spec);
457            } catch (ClassNotFoundException e) {
458                throw new CompositionException("A required specification cannot be loaded : " + spec, e);
459            }
460
461            ServiceImporter importer = new ServiceImporter(specToImport, fil, agg, opt, null, DependencyModel.DYNAMIC_BINDING_POLICY, context, null, handler);
462
463            handler.getRequirements().add(importer);
464            SpecificationMetadata specMeta = new SpecificationMetadata(spec, m_context, agg, opt, this);
465            m_services.add(specMeta); // Update the available types
466            return;
467        }
468
469        if (imp.isAggregate() && !agg) {
470            error("[" + getCompositeManager().getInstanceName() + "] The requirement " + elem.getAttribute("specification") + " is aggregate in the implementation and is declared as a simple service-level requirement");
471            throw new CompositionException("The requirement " + elem.getAttribute("specification") + " is aggregate in the implementation and is declared as a simple service-level requirement");
472        }
473
474        String filter = elem.getAttribute("filter");
475        if (filter != null) {
476            String filter2 = imp.getFilter();
477            if (filter2 == null || !filter2.equalsIgnoreCase(filter)) {
478                error("[" + getCompositeManager().getInstanceName() + "] The specification requirement " + elem.getAttribute("specification") + " as not the same filter as declared in the service-level requirement");
479                throw new CompositionException("The specification requirement " + elem.getAttribute("specification") + " as not the same filter as declared in the service-level requirement");
480            }
481        }
482    }

491    private void computeAvailableTypes() {

495        } else {
496            m_types = handler.getUsedType();
