020package org.apache.felix.ipojo.bnd;
021
022import java.io.File;
023import java.util.ArrayList;
024import java.util.List;
025import java.util.Map;
026
027import org.apache.felix.ipojo.manipulator.ManipulationVisitor;
028import org.apache.felix.ipojo.manipulator.Pojoization;
029import org.apache.felix.ipojo.manipulator.ResourceStore;

033import org.apache.felix.ipojo.manipulator.metadata.FileMetadataProvider;
034import org.apache.felix.ipojo.manipulator.visitor.check.CheckFieldConsistencyVisitor;
035import org.apache.felix.ipojo.manipulator.visitor.writer.ManipulatedResourcesWriter;
036import org.apache.felix.ipojo.metadata.Element;
037
038import aQute.bnd.osgi.Analyzer;
039import aQute.bnd.osgi.Resource;
040import aQute.bnd.service.AnalyzerPlugin;
041import aQute.bnd.service.Plugin;
042import aQute.service.reporter.Reporter;

044/**
045 * A {@code BndIpojoPlugin} is ...
046 *
047 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
048 */
049public class PojoizationPlugin implements Plugin, AnalyzerPlugin {
050
051    private static final String PROPERTY_METADATA = "metadata";
052    private static final String PROPERTY_USE_LOCAL_SCHEMAS = "use-local-schemas";
053    private static final String PROPERTY_INCLUDE_EMBED_BUNDLES = "include-embed-bundles";
054
055    private static final String DEFAULT_METADATA = "META-INF/metadata.xml";
056    private static final boolean DEFAULT_USE_LOCAL_SCHEMAS = false;
057    private static final boolean DEFAULT_INCLUDE_EMBED_BUNDLES = false;
058
059    private String m_metadata = DEFAULT_METADATA;
060    private boolean m_useLocalSchemas = DEFAULT_USE_LOCAL_SCHEMAS;
061    private boolean m_includeEmbedBundles = DEFAULT_INCLUDE_EMBED_BUNDLES;
062
063    private Reporter m_reporter;
064
065    public void setProperties(Map<String, String> configuration) {
066
067        // Use metadata file if any
068        if (configuration.containsKey(PROPERTY_METADATA)) {
069            m_metadata = configuration.get(PROPERTY_METADATA);
070        }
071
072        // Use local schemas ?
073        if (configuration.containsKey(PROPERTY_USE_LOCAL_SCHEMAS)) {
074            m_useLocalSchemas = true;
075        }
076
077        // Process embed bundles ?
078        if (configuration.containsKey(PROPERTY_INCLUDE_EMBED_BUNDLES)) {
079            m_includeEmbedBundles = true;
080        }
081    }
082
083    public void setReporter(Reporter reporter) {
084        m_reporter = reporter;
085    }
086
087    public boolean analyzeJar(Analyzer analyzer) throws Exception {
088
089        long start = System.currentTimeMillis();
090
091        // Wraps the Bnd Reporter
092        BndReporter reporter = new BndReporter(this.m_reporter);
093
094        // Build ResourceStore
095        BndJarResourceStore store = new BndJarResourceStore(analyzer, this.m_reporter);
096        store.setIncludeEmbedComponents(m_includeEmbedBundles);
097
098        CompositeMetadataProvider provider = buildMetadataProvider(analyzer, reporter, store);
099
100        // Quick exit
101        CacheableMetadataProvider cache = new CacheableMetadataProvider(provider);
102        if (cache.getMetadatas().isEmpty() && !hasEmbedComponents(analyzer)) {
103            return false;
104        }
105
106        manipulateComponents(reporter, store, cache);
107
108        int nbComponents = findElements(cache.getMetadatas(), "component").size();
109        int nbHandlers = findElements(cache.getMetadatas(), "handler").size();
110        this.m_reporter.trace("iPOJO manipulation performed performed in %s ms (%d components, %d handlers).",
111                               (System.currentTimeMillis() - start),
112                               nbComponents,
113                               nbHandlers);
114
115        // Return true if a new run should be performed after the analyze
116        return false;
117    }
118
119    protected void manipulateComponents(BndReporter reporter, BndJarResourceStore store, CacheableMetadataProvider cache) {
120        Pojoization pojoization = new Pojoization(reporter);
121        pojoization.disableAnnotationProcessing();
122        if (m_useLocalSchemas) {
123            pojoization.setUseLocalXSD();
124        }
125
126        pojoization.pojoization(store, cache, createVisitor(store, reporter));
127    }
128
129    protected CompositeMetadataProvider buildMetadataProvider(Analyzer analyzer, BndReporter reporter, BndJarResourceStore store) {
130        // Build MetadataProvider
131        CompositeMetadataProvider provider = new CompositeMetadataProvider(reporter);
132
133        File file = new File(m_metadata);
134        if (file.exists()) {
135            // Absolute file system resource
136            FileMetadataProvider fmp = new FileMetadataProvider(file, reporter);
137            fmp.setValidateUsingLocalSchemas(m_useLocalSchemas);
138            provider.addMetadataProvider(fmp);
139        } else {
140            // In archive resource
141            Resource resource = analyzer.getJar().getResource(m_metadata);
142            if (resource != null) {
143                ResourceMetadataProvider rmp = new ResourceMetadataProvider(resource, reporter);
144                rmp.setValidateUsingLocalSchemas(m_useLocalSchemas);
145                provider.addMetadataProvider(rmp);
146            }
147        }
148        provider.addMetadataProvider(new AnnotationMetadataProvider(store, reporter));
149        return provider;
150    }
151
152    private boolean hasEmbedComponents(Analyzer analyzer) throws Exception {
153        // We want to process components from embed jars ?
154        return m_includeEmbedBundles && Manifests.hasEmbedComponents(analyzer);
155
156    }
157
158    private List<Element> findElements(List<Element> metadatas, String name) {
159        List<Element> found = new ArrayList<Element>();
160        for (Element element : metadatas) {
161            if (name.equalsIgnoreCase(element.getName())) {
162                found.add(element);
163            }
164        }
165        return found;
166    }
167
168    protected ManipulationVisitor createVisitor(ResourceStore store, BndReporter reporter) {
169        ManipulatedResourcesWriter writer = new ManipulatedResourcesWriter();
170        writer.setReporter(reporter);
171        writer.setResourceStore(store);
172
173        CheckFieldConsistencyVisitor checkFieldConsistencyVisitor = new CheckFieldConsistencyVisitor(writer);
174        checkFieldConsistencyVisitor.setReporter(reporter);
175        return checkFieldConsistencyVisitor;
176    }
177
178}
