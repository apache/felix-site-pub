019package org.apache.felix.ipojo.handlers.providedservice;
020
021import java.util.Dictionary;
022import java.util.Properties;
023
024import org.apache.felix.ipojo.util.Property;
025import org.osgi.framework.ServiceReference;
026
027/**
028 * Provided Service Description.
029 * 
030 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
031 */
032public class ProvidedServiceDescription {
033
034    /**
035     * State : the service is unregistered.
036     */
037    public static final int UNREGISTERED = ProvidedService.UNREGISTERED;
038
039    /**
040     * State : the service is registered.
041     */
042    public static final int REGISTERED = ProvidedService.REGISTERED;
043    
044    /**
045     * The describe provided service.
046     */
047    private final ProvidedService m_ps;
048
049    /**
050     * Constructor.
051     * @param ps the described provided service.
052     */
053    public ProvidedServiceDescription(ProvidedService ps) {
054        m_ps = ps;
055    }
056
057    /**
058     * Gets the represented provided service.
059     * @return the provided service
060     */
061    public ProvidedService getProvidedService() {
062        return m_ps;
063    }
064
065    /**
066     * Gets the list of provided service specifications.
067     * @return the provided contract name.
068     */
069    public String[] getServiceSpecifications() {
070        return m_ps.getServiceSpecifications();
071    }
072
073    /**
074     * Gets the list of properties.
075     * A copy of the actual property set is returned.
076     * @return the properties.
077     */
078    public Properties getProperties() {
079        Properties props = new Properties();
080        org.apache.felix.ipojo.util.Property[] ps = m_ps.getProperties();
081        for (int i = 0; i < ps.length; i++) {
082            if (ps[i].getValue() != Property.NO_VALUE) {
083                props.put(ps[i].getName(), ps[i].getValue());
084            }
085        }
086        return props;
087    }
088    
089    /**
090     * Adds and Updates service properties.
091     * Existing properties are updated. 
092     * New ones are added.
093     * @param props the new properties
094     */
095    public void addProperties(Dictionary props) {
096        m_ps.addProperties(props);
097    }
098    
099    /**
100     * Removes service properties.
101     * @param props the properties to remove
102     */
103    public void removeProperties(Dictionary props) {
104        m_ps.deleteProperties(props);
105    }
106
107    /**
108     * Gets provided service state.
109     * @return the state of the provided service (UNREGISTERED | REGISTERED).
110     */
111    public int getState() {
112        return m_ps.getState();
113    }
114    
115    /**
116     * Gets the controller value as a String.
117     * @return the value
118     */
119    public String getController() {
120        if (m_ps.getControllerBySpecification("ALL") == null) {
121            return null; 
122        } else {
123            return String.valueOf(m_ps.getControllerBySpecification("ALL").getValue());
124        }
125    }
126    
127    /**
128     * Gets the controller value as a String.
129     * @param specification
130     * @return the value
131     */
132    public String getController(String specification) {
133        if (m_ps.getControllerBySpecification(specification) == null) {
134            return null; 
135        } else {
136            return String.valueOf(m_ps.getControllerBySpecification(specification).getValue());
137        }
138    }
139
140    /**
141     * Gets the service reference.
142     * @return the service reference (null if the service is unregistered).
143     */
144    public ServiceReference getServiceReference() {
145        return m_ps.getServiceReference();
146    }
147    
148    /**
149     * Gets the 'main' service object.
150     * @return the 'main' service object or <code>null</code>
151     * if no service object are created.
152     */
153    public Object getService() {
154        Object[] objs = m_ps.getInstanceManager().getPojoObjects();
155        if (objs == null) { 
156            return null;
157        } else {
158            return objs[0];
159        }
160    }
161    
162    public Object[] getServices() {
163        return m_ps.getInstanceManager().getPojoObjects();
164    }
165
166    public int getPolicy() {
167        return m_ps.getPolicy();
168    }
169
170    public Class<? extends CreationStrategy> getCreationStrategy() {
171        return m_ps.getCreationStrategy();
172    }
173
174    /**
175     * Add the given listener to the provided service handler's list of listeners.
176     *
177     * @param listener the {@code ProvidedServiceListener} object to be added
178     * @throws NullPointerException if {@code listener} is {@code null}
179     */
180    public void addListener(ProvidedServiceListener listener) {
181        m_ps.addListener(listener);
182    }
183
184    /**
185     * Remove the given listener from the provided service handler's list of listeners.
186     *
187     * @param listener the {@code ProvidedServiceListener} object to be removed
188     * @throws NullPointerException if {@code listener} is {@code null}
189     * @throws java.util.NoSuchElementException if {@code listener} wasn't present the in provided service handler's list of listeners
190     */
191    public void removeListener(ProvidedServiceListener listener) {
192        m_ps.removeListener(listener);
193    }
194
195}
