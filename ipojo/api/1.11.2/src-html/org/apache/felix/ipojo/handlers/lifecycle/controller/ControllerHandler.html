019package org.apache.felix.ipojo.handlers.lifecycle.controller;
020
021import java.util.Dictionary;
022
023import org.apache.felix.ipojo.ComponentInstance;
024import org.apache.felix.ipojo.ConfigurationException;
025import org.apache.felix.ipojo.InstanceManager;
026import org.apache.felix.ipojo.PrimitiveHandler;
027import org.apache.felix.ipojo.architecture.ComponentTypeDescription;
028import org.apache.felix.ipojo.metadata.Element;
029import org.apache.felix.ipojo.parser.FieldMetadata;
030import org.apache.felix.ipojo.parser.PojoMetadata;
031
032/**
033 * Lifecycle Controller handler.
034 * This handler allow a POJO  to vote for the instance state. By setting a boolean field to true or false, the handler state changed.
035 * @author <a href="mailto:dev@felix.apache.org">Felix Project Team</a>
036 */
037public class ControllerHandler extends PrimitiveHandler {
038
039    /**
040     * Actual handler (i.e. field value) state
041     */
042    private boolean m_state;
043        /**
044         * The controller field name.
045         */
046        private String m_field;
047
048    /**
049     * Configure method.
050     * Look for the first 'controller' element.
051     * @param metadata : metadata
052     * @param configuration : configuration
053     * @throws ConfigurationException : the field attribute is missing or does not exist in the class.
054     * @see org.apache.felix.ipojo.Handler#configure(org.apache.felix.ipojo.metadata.Element, java.util.Dictionary)
055     */
056    public void configure(Element metadata, Dictionary configuration) throws ConfigurationException {
057        Element[] controller = metadata.getElements("controller");
058        m_field = controller[0].getAttribute("field");
059        getInstanceManager().register(new FieldMetadata(m_field, "boolean"), this);
060    }
061
062    /**
063     * Start method.
064     * This methods initializes the controller value.
065     * @see org.apache.felix.ipojo.Handler#start()
066     */
067    public void start() {
068        // On start we should ask for the value
069        Object o = getInstanceManager().getFieldValue(m_field);
070        if (o == null || ! (o instanceof Boolean)) {
071                m_state = true;
072        } else {
073                m_state = ((Boolean) o).booleanValue();
074        }
075
076        if ( ! m_state) {
077                setValidity(false);
078        }
079    }
080
081    /**
082     * Stop method.
083     * Nothing to do.
084     * @see org.apache.felix.ipojo.Handler#stop()
085     */
086    public void stop() {
087        // Nothing to do.
088    }
089
090
091
092    /**
093     * GetterCallback.
094     * @param pojo : the pojo object on which the field is accessed
095     * Return the stored value.
096     * @param field : field name.
097     * @param value : value given by the previous handler.
098     * @return : the handler state.
099     */
100    public Object onGet(Object pojo, String field, Object value) {
101        if (m_state) {
102            return Boolean.TRUE;
103        } else {
104            return Boolean.FALSE;
105        }
106    }
107
108    /**
109     * SetterCallback.
110     * @param pojo : the pojo object on which the field is accessed
111     * Store the new field value & invalidate / validate the handler is required.
112     * @param field : field name.
113     * @param value : new value.
114     */
115    public void onSet(Object pojo, String field, Object value) {
116        if (value instanceof Boolean) {
117            boolean newValue = ((Boolean) value).booleanValue();
118            if (newValue != m_state) {
119                m_state = newValue;
120                if (m_state) {
121                    ((InstanceManager) getHandlerManager()).setState(ComponentInstance.VALID);
122                    setValidity(true);
123                } else {
124                    ((InstanceManager) getHandlerManager()).setState(ComponentInstance.INVALID);
125                    setValidity(false);
126                }
127            }
128        } else {
129            error("Boolean expected for the lifecycle controller");
130            getInstanceManager().stop();
131        }
132    }
133
134    /**
135     * Initialize the component factory.
136     * The controller field is checked to avoid configure check.
137     * @param desc : component description
138     * @param metadata : component type metadata
139     * @throws ConfigurationException : occurs if the controller field is not in the POJO class or is not a boolean.
140     * @see org.apache.felix.ipojo.Handler#initializeComponentFactory(org.apache.felix.ipojo.architecture.ComponentTypeDescription, org.apache.felix.ipojo.metadata.Element)
141     */
142    public void initializeComponentFactory(ComponentTypeDescription desc, Element metadata) throws ConfigurationException {
143        String field = null;
144        Element[] controller = metadata.getElements("controller");
145        // Use only the first controller
146        field = controller[0].getAttribute("field");
147        if (field == null) {
148            throw new ConfigurationException("Lifecycle controller : the controller element needs to contain a field attribute");
149        }
150
151        PojoMetadata method = getFactory().getPojoMetadata();
152        FieldMetadata fieldMetadata = method.getField(field);
153        if (fieldMetadata == null) {
154            throw new ConfigurationException("Lifecycle controller : The field " + field + " does not exist in the implementation class");
155        }
156
157        if (!fieldMetadata.getFieldType().equalsIgnoreCase("boolean")) {
158            throw new ConfigurationException("Lifecycle controller : The field " + field + " must be a boolean (" + fieldMetadata.getFieldType() + " found)");
159        }
160    }
161
162}
