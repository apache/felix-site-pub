022import static java.util.Collections.emptyList;
023import static java.util.Collections.singletonList;
024import static org.apache.felix.ipojo.manipulator.spi.helper.Predicates.alwaysTrue;

026import java.util.List;

029import org.apache.felix.ipojo.manipulator.Reporter;

031import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.MethodGenericVisitor;
032import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.ParameterGenericVisitor;
033import org.apache.felix.ipojo.manipulator.metadata.annotation.visitor.generic.TypeGenericVisitor;

046 * Time: 16:09
047 */
048public class LegacyGenericBindingRegistry extends CompletableBindingRegistry {
049    public static final Pattern CUSTOM_HANDLER_PATTERN = Pattern.compile("(.*\\.ipojo\\..*)|(.*\\.handler\\..*)");
050
051    public LegacyGenericBindingRegistry(final BindingRegistry delegate, final Reporter reporter) {
052        super(delegate, reporter);
053    }
054
055    @Override
056    protected List<Binding> createBindings(final Type type) {
057        if (CUSTOM_HANDLER_PATTERN.matcher(type.getClassName()).matches()) {
058            Binding binding = new Binding();
059            binding.setAnnotationType(type);
060            binding.setPredicate(alwaysTrue());
061            binding.setFactory(new AnnotationVisitorFactory() {
062                // Need to build a new Element instance for each created visitor
063                public AnnotationVisitor newAnnotationVisitor(BindingContext context) {
064                    if (context.getNode() instanceof ClassNode) {
065                        return new TypeGenericVisitor(context.getWorkbench(),
066                                                      Elements.buildElement(type));
067                    } else if (context.getNode() instanceof FieldNode) {
068                        return new FieldGenericVisitor(context.getWorkbench(),
069                                                       Elements.buildElement(type),
070                                                       (FieldNode) context.getNode());
071
072                    } else if ((context.getNode() instanceof MethodNode) &&
073                            (context.getParameterIndex() == BindingContext.NO_INDEX)) {
074                        return new MethodGenericVisitor(context.getWorkbench(),
075                                                        Elements.buildElement(type),
076                                                        (MethodNode) context.getNode());
077                    } else {
078                        // last case: method parameter annotation
079                        return new ParameterGenericVisitor(context.getWorkbench(),
080                                                           Elements.buildElement(type),
081                                                           (MethodNode) context.getNode(),
082                                                           context.getParameterIndex());
083                    }
084                }
085
086                @Override
087                public String toString() {
088                    return "LegacyGenericVisitorFactory";
089                }
090            });
091
092            // Return the produced generic binding
093            return singletonList(binding);
094        }
095
096        return emptyList();
097    }
098}
